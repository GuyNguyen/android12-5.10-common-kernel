  25 /* Limit-check stall timeouts specified at boottime and runtime. */
  26 int rcu_jiffies_till_stall_check(void)
  27 {
  28         int till_stall_check = READ_ONCE(rcu_cpu_stall_timeout);
  29 
  30         /*
  31          * Limit check must be consistent with the Kconfig limits
  32          * for CONFIG_RCU_CPU_STALL_TIMEOUT.
  33          */
  34         if (till_stall_check < 3) {
  35                 WRITE_ONCE(rcu_cpu_stall_timeout, 3);
  36                 till_stall_check = 3;
  37         } else if (till_stall_check > 300) {
  38                 WRITE_ONCE(rcu_cpu_stall_timeout, 300);
  39                 till_stall_check = 300;
  40         }
  41         return till_stall_check * HZ + RCU_STALL_DELAY_DELTA;
  42 }
  43 EXPORT_SYMBOL_GPL(rcu_jiffies_till_stall_check);

  45 /**
  46  * rcu_gp_might_be_stalled - Is it likely that the grace period is stalled?
  47  *
  48  * Returns @true if the current grace period is sufficiently old that
  49  * it is reasonable to assume that it might be stalled.  This can be
  50  * useful when deciding whether to allocate memory to enable RCU-mediated
  51  * freeing on the one hand or just invoking synchronize_rcu() on the other.
  52  * The latter is preferable when the grace period is stalled.
  53  *
  54  * Note that sampling of the .gp_start and .gp_seq fields must be done
  55  * carefully to avoid false positives at the beginnings and ends of
  56  * grace periods.
  57  */
  58 bool rcu_gp_might_be_stalled(void)
  59 {
  60         unsigned long d = rcu_jiffies_till_stall_check() / RCU_STALL_MIGHT_DIV;
  61         unsigned long j = jiffies;
  62 
  63         if (d < RCU_STALL_MIGHT_MIN)
  64                 d = RCU_STALL_MIGHT_MIN;
  65         smp_mb(); // jiffies before .gp_seq to avoid false positives.
  66         if (!rcu_gp_in_progress())
  67                 return false;
  68         // Long delays at this point avoids false positive, but a delay
  69         // of ULONG_MAX/4 jiffies voids your no-false-positive warranty.
  70         smp_mb(); // .gp_seq before second .gp_start
  71         // And ditto here.
  72         return !time_before(j, READ_ONCE(rcu_state.gp_start) + d);
  73 }

 131 /* Start of new grace period, so record stall time (and forcing times). */
 132 static void record_gp_stall_check_time(void)
 133 {
 134         unsigned long j = jiffies;
 135         unsigned long j1;
 136 
 137         WRITE_ONCE(rcu_state.gp_start, j);
 138         j1 = rcu_jiffies_till_stall_check();
 139         smp_mb(); // ->gp_start before ->jiffies_stall and caller's ->gp_seq.
 140         WRITE_ONCE(rcu_state.jiffies_stall, j + j1);
 141         rcu_state.jiffies_resched = j + j1 / 2;
 142         rcu_state.n_force_qs_gpstart = READ_ONCE(rcu_state.n_force_qs);
 143 }

 153 /*
 154  * If too much time has passed in the current grace period, and if
 155  * so configured, go kick the relevant kthreads.
 156  */
 157 static void rcu_stall_kick_kthreads(void)
 158 {
 159         unsigned long j;
 160 
 161         if (!READ_ONCE(rcu_kick_kthreads))
 162                 return;
 163         j = READ_ONCE(rcu_state.jiffies_kick_kthreads);
 164         if (time_after(jiffies, j) && rcu_state.gp_kthread &&
 165             (rcu_gp_in_progress() || READ_ONCE(rcu_state.gp_flags))) {
 166                 WARN_ONCE(1, "Kicking %s grace-period kthread\n",
 167                           rcu_state.name);
 168                 rcu_ftrace_dump(DUMP_ALL);
 169                 wake_up_process(rcu_state.gp_kthread);
 170                 WRITE_ONCE(rcu_state.jiffies_kick_kthreads, j + HZ);
 171         }
 172 }

 234 /*
 235  * Report out the state of a not-running task that is stalling the
 236  * current RCU grace period.
 237  */
 238 static bool check_slow_task(struct task_struct *t, void *arg)
 239 {
 240         struct rcu_stall_chk_rdr *rscrp = arg;
 241 
 242         if (task_curr(t))
 243                 return false; // It is running, so decline to inspect it.
 244         rscrp->nesting = t->rcu_read_lock_nesting;
 245         rscrp->rs = t->rcu_read_unlock_special;
 246         rscrp->on_blkd_list = !list_empty(&t->rcu_node_entry);
 247         return true;
 248 }

 250 /*
 251  * Scan the current list of tasks blocked within RCU read-side critical
 252  * sections, printing out the tid of each of the first few of them.
 253  */
 254 static int rcu_print_task_stall(struct rcu_node *rnp, unsigned long flags)
 255         __releases(rnp->lock)
 256 {
 257         int i = 0;
 258         int ndetected = 0;
 259         struct rcu_stall_chk_rdr rscr;
 260         struct task_struct *t;
 261         struct task_struct *ts[8];
 262 
 263         if (!rcu_preempt_blocked_readers_cgp(rnp))
 264                 return 0;
 265         pr_err("\tTasks blocked on level-%d rcu_node (CPUs %d-%d):",
 266                rnp->level, rnp->grplo, rnp->grphi);
 267         t = list_entry(rnp->gp_tasks->prev,
 268                        struct task_struct, rcu_node_entry);
 269         list_for_each_entry_continue(t, &rnp->blkd_tasks, rcu_node_entry) {
 270                 get_task_struct(t);
 271                 ts[i++] = t;
 272                 if (i >= ARRAY_SIZE(ts))
 273                         break;
 274         }
 275         raw_spin_unlock_irqrestore_rcu_node(rnp, flags);
 276         for (i--; i; i--) {
 277                 t = ts[i];
 278                 if (!try_invoke_on_locked_down_task(t, check_slow_task, &rscr))
 279                         pr_cont(" P%d", t->pid);
 280                 else
 281                         pr_cont(" P%d/%d:%c%c%c%c",
 282                                 t->pid, rscr.nesting,
 283                                 ".b"[rscr.rs.b.blocked],
 284                                 ".q"[rscr.rs.b.need_qs],
 285                                 ".e"[rscr.rs.b.exp_hint],
 286                                 ".l"[rscr.on_blkd_list]);
 287                 put_task_struct(t);
 288                 ndetected++;
 289         }
 290         pr_cont("\n");
 291         return ndetected;
 292 }

 315 /*
 316  * Dump stacks of all tasks running on stalled CPUs.  First try using
 317  * NMIs, but fall back to manual remote stack tracing on architectures
 318  * that don't support NMI-based stack dumps.  The NMI-triggered stack
 319  * traces are more accurate because they are printed by the target CPU.
 320  */
 321 static void rcu_dump_cpu_stacks(void)
 322 {
 323         int cpu;
 324         unsigned long flags;
 325         struct rcu_node *rnp;
 326 
 327         rcu_for_each_leaf_node(rnp) {
 328                 raw_spin_lock_irqsave_rcu_node(rnp, flags);
 329                 for_each_leaf_node_possible_cpu(rnp, cpu)
 330                         if (rnp->qsmask & leaf_node_cpu_bit(rnp, cpu))
 331                                 if (!trigger_single_cpu_backtrace(cpu))
 332                                         dump_cpu_task(cpu);
 333                 raw_spin_unlock_irqrestore_rcu_node(rnp, flags);
 334         }
 335 }

 389 /*
 390  * Print out diagnostic information for the specified stalled CPU.
 391  *
 392  * If the specified CPU is aware of the current RCU grace period, then
 393  * print the number of scheduling clock interrupts the CPU has taken
 394  * during the time that it has been aware.  Otherwise, print the number
 395  * of RCU grace periods that this CPU is ignorant of, for example, "1"
 396  * if the CPU was aware of the previous grace period.
 397  *
 398  * Also print out idle and (if CONFIG_RCU_FAST_NO_HZ) idle-entry info.
 399  */
 400 static void print_cpu_stall_info(int cpu)
 401 {
 402         unsigned long delta;
 403         bool falsepositive;
 404         char fast_no_hz[72];
 405         struct rcu_data *rdp = per_cpu_ptr(&rcu_data, cpu);
 406         char *ticks_title;
 407         unsigned long ticks_value;
 408 
 409         /*
 410          * We could be printing a lot while holding a spinlock.  Avoid
 411          * triggering hard lockup.
 412          */
 413         touch_nmi_watchdog();
 414 
 415         ticks_value = rcu_seq_ctr(rcu_state.gp_seq - rdp->gp_seq);
 416         if (ticks_value) {
 417                 ticks_title = "GPs behind";
 418         } else {
 419                 ticks_title = "ticks this GP";
 420                 ticks_value = rdp->ticks_this_gp;
 421         }
 422         print_cpu_stall_fast_no_hz(fast_no_hz, cpu);
 423         delta = rcu_seq_ctr(rdp->mynode->gp_seq - rdp->rcu_iw_gp_seq);
 424         falsepositive = rcu_is_gp_kthread_starving(NULL) &&
 425                         rcu_dynticks_in_eqs(rcu_dynticks_snap(rdp));
 426         pr_err("\t%d-%c%c%c%c: (%lu %s) idle=%03x/%ld/%#lx softirq=%u/%u fqs=%ld %s%s\n",
 427                cpu,
 428                "O."[!!cpu_online(cpu)],
 429                "o."[!!(rdp->grpmask & rdp->mynode->qsmaskinit)],
 430                "N."[!!(rdp->grpmask & rdp->mynode->qsmaskinitnext)],
 431                !IS_ENABLED(CONFIG_IRQ_WORK) ? '?' :
 432                         rdp->rcu_iw_pending ? (int)min(delta, 9UL) + '0' :
 433                                 "!."[!delta],
 434                ticks_value, ticks_title,
 435                rcu_dynticks_snap(rdp) & 0xfff,
 436                rdp->dynticks_nesting, rdp->dynticks_nmi_nesting,
 437                rdp->softirq_snap, kstat_softirqs_cpu(RCU_SOFTIRQ, cpu),
 438                data_race(rcu_state.n_force_qs) - rcu_state.n_force_qs_gpstart,
 439                fast_no_hz,
 440                falsepositive ? " (false positive?)" : "");
 441 }

 443 /* Complain about starvation of grace-period kthread.  */
 444 static void rcu_check_gp_kthread_starvation(void)
 445 {
 446         struct task_struct *gpk = rcu_state.gp_kthread;
 447         unsigned long j;
 448 
 449         if (rcu_is_gp_kthread_starving(&j)) {
 450                 pr_err("%s kthread starved for %ld jiffies! g%ld f%#x %s(%d) ->state=%#lx ->cpu=%d\n",
 451                        rcu_state.name, j,
 452                        (long)rcu_seq_current(&rcu_state.gp_seq),
 453                        data_race(rcu_state.gp_flags),
 454                        gp_state_getname(rcu_state.gp_state), rcu_state.gp_state,
 455                        gpk ? gpk->state : ~0, gpk ? task_cpu(gpk) : -1);
 456                 if (gpk) {
 457                         pr_err("\tUnless %s kthread gets sufficient CPU time, OOM is now expected behavior.\n", rcu_state.name);
 458                         pr_err("RCU grace-period kthread stack dump:\n");
 459                         sched_show_task(gpk);
 460                         wake_up_process(gpk);
 461                 }
 462         }
 463 }

 465 static void print_other_cpu_stall(unsigned long gp_seq, unsigned long gps)
 466 {
 467         int cpu;
 468         unsigned long flags;
 469         unsigned long gpa;
 470         unsigned long j;
 471         int ndetected = 0;
 472         struct rcu_node *rnp;
 473         long totqlen = 0;
 474 
 475         /* Kick and suppress, if so configured. */
 476         rcu_stall_kick_kthreads();
 477         if (rcu_stall_is_suppressed())
 478                 return;
 479 
 480         /*
 481          * OK, time to rat on our buddy...
 482          * See Documentation/RCU/stallwarn.rst for info on how to debug
 483          * RCU CPU stall warnings.
 484          */
 485         trace_rcu_stall_warning(rcu_state.name, TPS("StallDetected"));
 486         pr_err("INFO: %s detected stalls on CPUs/tasks:\n", rcu_state.name);
 487         rcu_for_each_leaf_node(rnp) {
 488                 raw_spin_lock_irqsave_rcu_node(rnp, flags);
 489                 if (rnp->qsmask != 0) {
 490                         for_each_leaf_node_possible_cpu(rnp, cpu)
 491                                 if (rnp->qsmask & leaf_node_cpu_bit(rnp, cpu)) {
 492                                         print_cpu_stall_info(cpu);
 493                                         ndetected++;
 494                                 }
 495                 }
 496                 ndetected += rcu_print_task_stall(rnp, flags); // Releases rnp->lock.
 497         }
 498 
 499         for_each_possible_cpu(cpu)
 500                 totqlen += rcu_get_n_cbs_cpu(cpu);
 501         pr_cont("\t(detected by %d, t=%ld jiffies, g=%ld, q=%lu)\n",
 502                smp_processor_id(), (long)(jiffies - gps),
 503                (long)rcu_seq_current(&rcu_state.gp_seq), totqlen);
 504         if (ndetected) {
 505                 rcu_dump_cpu_stacks();
 506 
 507                 /* Complain about tasks blocking the grace period. */
 508                 rcu_for_each_leaf_node(rnp)
 509                         rcu_print_detail_task_stall_rnp(rnp);
 510         } else {
 511                 if (rcu_seq_current(&rcu_state.gp_seq) != gp_seq) {
 512                         pr_err("INFO: Stall ended before state dump start\n");
 513                 } else {
 514                         j = jiffies;
 515                         gpa = data_race(rcu_state.gp_activity);
 516                         pr_err("All QSes seen, last %s kthread activity %ld (%ld-%ld), jiffies_till_next_fqs=%ld, root ->qsmask %#lx\n",
 517                                rcu_state.name, j - gpa, j, gpa,
 518                                data_race(jiffies_till_next_fqs),
 519                                rcu_get_root()->qsmask);
 520                 }
 521         }
 522         /* Rewrite if needed in case of slow consoles. */
 523         if (ULONG_CMP_GE(jiffies, READ_ONCE(rcu_state.jiffies_stall)))
 524                 WRITE_ONCE(rcu_state.jiffies_stall,
 525                            jiffies + 3 * rcu_jiffies_till_stall_check() + 3);
 526 
 527         rcu_check_gp_kthread_starvation();
 528 
 529         panic_on_rcu_stall();
 530 
 531         rcu_force_quiescent_state();  /* Kick them all. */
 532 }

 534 static void print_cpu_stall(unsigned long gps)
 535 {
 536         int cpu;
 537         unsigned long flags;
 538         struct rcu_data *rdp = this_cpu_ptr(&rcu_data);
 539         struct rcu_node *rnp = rcu_get_root();
 540         long totqlen = 0;
 541 
 542         /* Kick and suppress, if so configured. */
 543         rcu_stall_kick_kthreads();
 544         if (rcu_stall_is_suppressed())
 545                 return;
 546 
 547         /*
 548          * OK, time to rat on ourselves...
 549          * See Documentation/RCU/stallwarn.rst for info on how to debug
 550          * RCU CPU stall warnings.
 551          */
 552         trace_rcu_stall_warning(rcu_state.name, TPS("SelfDetected"));
 553         pr_err("INFO: %s self-detected stall on CPU\n", rcu_state.name);
 554         raw_spin_lock_irqsave_rcu_node(rdp->mynode, flags);
 555         print_cpu_stall_info(smp_processor_id());
 556         raw_spin_unlock_irqrestore_rcu_node(rdp->mynode, flags);
 557         for_each_possible_cpu(cpu)
 558                 totqlen += rcu_get_n_cbs_cpu(cpu);
 559         pr_cont("\t(t=%lu jiffies g=%ld q=%lu)\n",
 560                 jiffies - gps,
 561                 (long)rcu_seq_current(&rcu_state.gp_seq), totqlen);
 562 
 563         rcu_check_gp_kthread_starvation();
 564 
 565         rcu_dump_cpu_stacks();
 566 
 567         raw_spin_lock_irqsave_rcu_node(rnp, flags);
 568         /* Rewrite if needed in case of slow consoles. */
 569         if (ULONG_CMP_GE(jiffies, READ_ONCE(rcu_state.jiffies_stall)))
 570                 WRITE_ONCE(rcu_state.jiffies_stall,
 571                            jiffies + 3 * rcu_jiffies_till_stall_check() + 3);
 572         raw_spin_unlock_irqrestore_rcu_node(rnp, flags);
 573 
 574         panic_on_rcu_stall();
 575 
 576         /*
 577          * Attempt to revive the RCU machinery by forcing a context switch.
 578          *
 579          * A context switch would normally allow the RCU state machine to make
 580          * progress and it could be we're stuck in kernel space without context
 581          * switches for an entirely unreasonable amount of time.
 582          */
 583         set_tsk_need_resched(current);
 584         set_preempt_need_resched();
 585 }

 587 static void check_cpu_stall(struct rcu_data *rdp)
 588 {
 589         unsigned long gs1;
 590         unsigned long gs2;
 591         unsigned long gps;
 592         unsigned long j;
 593         unsigned long jn;
 594         unsigned long js;
 595         struct rcu_node *rnp;
 596 
 597         if ((rcu_stall_is_suppressed() && !READ_ONCE(rcu_kick_kthreads)) ||
 598             !rcu_gp_in_progress())
 599                 return;
 600         rcu_stall_kick_kthreads();
 601         j = jiffies;
 602 
 603         /*
 604          * Lots of memory barriers to reject false positives.
 605          *
 606          * The idea is to pick up rcu_state.gp_seq, then
 607          * rcu_state.jiffies_stall, then rcu_state.gp_start, and finally
 608          * another copy of rcu_state.gp_seq.  These values are updated in
 609          * the opposite order with memory barriers (or equivalent) during
 610          * grace-period initialization and cleanup.  Now, a false positive
 611          * can occur if we get an new value of rcu_state.gp_start and a old
 612          * value of rcu_state.jiffies_stall.  But given the memory barriers,
 613          * the only way that this can happen is if one grace period ends
 614          * and another starts between these two fetches.  This is detected
 615          * by comparing the second fetch of rcu_state.gp_seq with the
 616          * previous fetch from rcu_state.gp_seq.
 617          *
 618          * Given this check, comparisons of jiffies, rcu_state.jiffies_stall,
 619          * and rcu_state.gp_start suffice to forestall false positives.
 620          */
 621         gs1 = READ_ONCE(rcu_state.gp_seq);
 622         smp_rmb(); /* Pick up ->gp_seq first... */
 623         js = READ_ONCE(rcu_state.jiffies_stall);
 624         smp_rmb(); /* ...then ->jiffies_stall before the rest... */
 625         gps = READ_ONCE(rcu_state.gp_start);
 626         smp_rmb(); /* ...and finally ->gp_start before ->gp_seq again. */
 627         gs2 = READ_ONCE(rcu_state.gp_seq);
 628         if (gs1 != gs2 ||
 629             ULONG_CMP_LT(j, js) ||
 630             ULONG_CMP_GE(gps, js))
 631                 return; /* No stall or GP completed since entering function. */
 632         rnp = rdp->mynode;
 633         jn = jiffies + 3 * rcu_jiffies_till_stall_check() + 3;
 634         if (rcu_gp_in_progress() &&
 635             (READ_ONCE(rnp->qsmask) & rdp->grpmask) &&
 636             cmpxchg(&rcu_state.jiffies_stall, js, jn) == js) {
 637 
 638                 /* We haven't checked in, so go dump stack. */
 639                 print_cpu_stall(gps);
 640                 if (READ_ONCE(rcu_cpu_stall_ftrace_dump))
 641                         rcu_ftrace_dump(DUMP_ALL);
 642 
 643         } else if (rcu_gp_in_progress() &&
 644                    ULONG_CMP_GE(j, js + RCU_STALL_RAT_DELAY) &&
 645                    cmpxchg(&rcu_state.jiffies_stall, js, jn) == js) {
 646 
 647                 /* They had a few time units to dump stack, so complain. */
 648                 print_other_cpu_stall(gs2, gps);
 649                 if (READ_ONCE(rcu_cpu_stall_ftrace_dump))
 650                         rcu_ftrace_dump(DUMP_ALL);
 651         }
 652 }

 659 /*
 660  * Show the state of the grace-period kthreads.
 661  */
 662 void show_rcu_gp_kthreads(void)
 663 {
 664         unsigned long cbs = 0;
 665         int cpu;
 666         unsigned long j;
 667         unsigned long ja;
 668         unsigned long jr;
 669         unsigned long jw;
 670         struct rcu_data *rdp;
 671         struct rcu_node *rnp;
 672         struct task_struct *t = READ_ONCE(rcu_state.gp_kthread);
 673 
 674         j = jiffies;
 675         ja = j - data_race(rcu_state.gp_activity);
 676         jr = j - data_race(rcu_state.gp_req_activity);
 677         jw = j - data_race(rcu_state.gp_wake_time);
 678         pr_info("%s: wait state: %s(%d) ->state: %#lx delta ->gp_activity %lu ->gp_req_activity %lu ->gp_wake_time %lu ->gp_wake_seq %ld ->gp_seq %ld ->gp_seq_needed %ld ->gp_flags %#x\n",
 679                 rcu_state.name, gp_state_getname(rcu_state.gp_state),
 680                 rcu_state.gp_state, t ? t->state : 0x1ffffL,
 681                 ja, jr, jw, (long)data_race(rcu_state.gp_wake_seq),
 682                 (long)data_race(rcu_state.gp_seq),
 683                 (long)data_race(rcu_get_root()->gp_seq_needed),
 684                 data_race(rcu_state.gp_flags));
 685         rcu_for_each_node_breadth_first(rnp) {
 686                 if (ULONG_CMP_GE(READ_ONCE(rcu_state.gp_seq),
 687                                  READ_ONCE(rnp->gp_seq_needed)))
 688                         continue;
 689                 pr_info("\trcu_node %d:%d ->gp_seq %ld ->gp_seq_needed %ld\n",
 690                         rnp->grplo, rnp->grphi, (long)data_race(rnp->gp_seq),
 691                         (long)data_race(rnp->gp_seq_needed));
 692                 if (!rcu_is_leaf_node(rnp))
 693                         continue;
 694                 for_each_leaf_node_possible_cpu(rnp, cpu) {
 695                         rdp = per_cpu_ptr(&rcu_data, cpu);
 696                         if (READ_ONCE(rdp->gpwrap) ||
 697                             ULONG_CMP_GE(READ_ONCE(rcu_state.gp_seq),
 698                                          READ_ONCE(rdp->gp_seq_needed)))
 699                                 continue;
 700                         pr_info("\tcpu %d ->gp_seq_needed %ld\n",
 701                                 cpu, (long)data_race(rdp->gp_seq_needed));
 702                 }
 703         }
 704         for_each_possible_cpu(cpu) {
 705                 rdp = per_cpu_ptr(&rcu_data, cpu);
 706                 cbs += data_race(rdp->n_cbs_invoked);
 707                 if (rcu_segcblist_is_offloaded(&rdp->cblist))
 708                         show_rcu_nocb_state(rdp);
 709         }
 710         pr_info("RCU callbacks invoked since boot: %lu\n", cbs);
 711         show_rcu_tasks_gp_kthreads();
 712 }
 713 EXPORT_SYMBOL_GPL(show_rcu_gp_kthreads);

 715 /*
 716  * This function checks for grace-period requests that fail to motivate
 717  * RCU to come out of its idle mode.
 718  */
 719 static void rcu_check_gp_start_stall(struct rcu_node *rnp, struct rcu_data *rdp,
 720                                      const unsigned long gpssdelay)
 721 {
 722         unsigned long flags;
 723         unsigned long j;
 724         struct rcu_node *rnp_root = rcu_get_root();
 725         static atomic_t warned = ATOMIC_INIT(0);
 726 
 727         if (!IS_ENABLED(CONFIG_PROVE_RCU) || rcu_gp_in_progress() ||
 728             ULONG_CMP_GE(READ_ONCE(rnp_root->gp_seq),
 729                          READ_ONCE(rnp_root->gp_seq_needed)) ||
 730             !smp_load_acquire(&rcu_state.gp_kthread)) // Get stable kthread.
 731                 return;
 732         j = jiffies; /* Expensive access, and in common case don't get here. */
 733         if (time_before(j, READ_ONCE(rcu_state.gp_req_activity) + gpssdelay) ||
 734             time_before(j, READ_ONCE(rcu_state.gp_activity) + gpssdelay) ||
 735             atomic_read(&warned))
 736                 return;
 737 
 738         raw_spin_lock_irqsave_rcu_node(rnp, flags);
 739         j = jiffies;
 740         if (rcu_gp_in_progress() ||
 741             ULONG_CMP_GE(READ_ONCE(rnp_root->gp_seq),
 742                          READ_ONCE(rnp_root->gp_seq_needed)) ||
 743             time_before(j, READ_ONCE(rcu_state.gp_req_activity) + gpssdelay) ||
 744             time_before(j, READ_ONCE(rcu_state.gp_activity) + gpssdelay) ||
 745             atomic_read(&warned)) {
 746                 raw_spin_unlock_irqrestore_rcu_node(rnp, flags);
 747                 return;
 748         }
 749         /* Hold onto the leaf lock to make others see warned==1. */
 750 
 751         if (rnp_root != rnp)
 752                 raw_spin_lock_rcu_node(rnp_root); /* irqs already disabled. */
 753         j = jiffies;
 754         if (rcu_gp_in_progress() ||
 755             ULONG_CMP_GE(READ_ONCE(rnp_root->gp_seq),
 756                          READ_ONCE(rnp_root->gp_seq_needed)) ||
 757             time_before(j, READ_ONCE(rcu_state.gp_req_activity) + gpssdelay) ||
 758             time_before(j, READ_ONCE(rcu_state.gp_activity) + gpssdelay) ||
 759             atomic_xchg(&warned, 1)) {
 760                 if (rnp_root != rnp)
 761                         /* irqs remain disabled. */
 762                         raw_spin_unlock_rcu_node(rnp_root);
 763                 raw_spin_unlock_irqrestore_rcu_node(rnp, flags);
 764                 return;
 765         }
 766         WARN_ON(1);
 767         if (rnp_root != rnp)
 768                 raw_spin_unlock_rcu_node(rnp_root);
 769         raw_spin_unlock_irqrestore_rcu_node(rnp, flags);
 770         show_rcu_gp_kthreads();
 771 }

 773 /*
 774  * Do a forward-progress check for rcutorture.  This is normally invoked
 775  * due to an OOM event.  The argument "j" gives the time period during
 776  * which rcutorture would like progress to have been made.
 777  */
 778 void rcu_fwd_progress_check(unsigned long j)
 779 {
 780         unsigned long cbs;
 781         int cpu;
 782         unsigned long max_cbs = 0;
 783         int max_cpu = -1;
 784         struct rcu_data *rdp;
 785 
 786         if (rcu_gp_in_progress()) {
 787                 pr_info("%s: GP age %lu jiffies\n",
 788                         __func__, jiffies - rcu_state.gp_start);
 789                 show_rcu_gp_kthreads();
 790         } else {
 791                 pr_info("%s: Last GP end %lu jiffies ago\n",
 792                         __func__, jiffies - rcu_state.gp_end);
 793                 preempt_disable();
 794                 rdp = this_cpu_ptr(&rcu_data);
 795                 rcu_check_gp_start_stall(rdp->mynode, rdp, j);
 796                 preempt_enable();
 797         }
 798         for_each_possible_cpu(cpu) {
 799                 cbs = rcu_get_n_cbs_cpu(cpu);
 800                 if (!cbs)
 801                         continue;
 802                 if (max_cpu < 0)
 803                         pr_info("%s: callbacks", __func__);
 804                 pr_cont(" %d: %lu", cpu, cbs);
 805                 if (cbs <= max_cbs)
 806                         continue;
 807                 max_cbs = cbs;
 808                 max_cpu = cpu;
 809         }
 810         if (max_cpu >= 0)
 811                 pr_cont("\n");
 812 }
 813 EXPORT_SYMBOL_GPL(rcu_fwd_progress_check);
