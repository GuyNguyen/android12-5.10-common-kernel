     /*   */
  90 {
  91         u32 value_size = attr->value_size;
  92         struct bpf_stack_map *smap;
  93         struct bpf_map_memory mem;
  94         u64 cost, n_buckets;
  95         int err;
  96 
  97         if (!bpf_capable())
  98                 return ERR_PTR(-EPERM);
  99 
 100         if (attr->map_flags & ~STACK_CREATE_FLAG_MASK)
 101                 return ERR_PTR(-EINVAL);
 102 
 103         /* check sanity of attributes */
 104         if (attr->max_entries == 0 || attr->key_size != 4 ||
 105             value_size < 8 || value_size % 8)
 106                 return ERR_PTR(-EINVAL);
 107 
 108         BUILD_BUG_ON(sizeof(struct bpf_stack_build_id) % sizeof(u64));
 109         if (attr->map_flags & BPF_F_STACK_BUILD_ID) {
 110                 if (value_size % sizeof(struct bpf_stack_build_id) ||
 111                     value_size / sizeof(struct bpf_stack_build_id)
 112                     > sysctl_perf_event_max_stack)
 113                         return ERR_PTR(-EINVAL);
 114         } else if (value_size / 8 > sysctl_perf_event_max_stack)
 115                 return ERR_PTR(-EINVAL);
 116 
 117         /* hash table size must be power of 2 */
 118         n_buckets = roundup_pow_of_two(attr->max_entries);
 119         if (!n_buckets)
 120                 return ERR_PTR(-E2BIG);
 121 
 122         cost = n_buckets * sizeof(struct stack_map_bucket *) + sizeof(*smap);
 123         cost += n_buckets * (value_size + sizeof(struct stack_map_bucket));
 124         err = bpf_map_charge_init(&mem, cost);
 125         if (err)
 126                 return ERR_PTR(err);
 127 
 128         smap = bpf_map_area_alloc(cost, bpf_map_attr_numa_node(attr));
 129         if (!smap) {
 130                 bpf_map_charge_finish(&mem);
 131                 return ERR_PTR(-ENOMEM);
 132         }
 133 
 134         bpf_map_init_from_attr(&smap->map, attr);
 135         smap->map.value_size = value_size;
 136         smap->n_buckets = n_buckets;
 137 
 138         err = get_callchain_buffers(sysctl_perf_event_max_stack);
 139         if (err)
 140                 goto free_charge;
 141 
 142         err = prealloc_elems_and_freelist(smap);
 143         if (err)
 144                 goto put_buffers;
 145 
 146         bpf_map_charge_move(&smap->map.memory, &mem);
 147 
 148         return &smap->map;
 149 
 150 put_buffers:
 151         put_callchain_buffers();
 152 free_charge:
 153         bpf_map_charge_finish(&mem);
 154         bpf_map_area_free(smap);
 155         return ERR_PTR(err);

 159 /*
 160  * Parse build id from the note segment. This logic can be shared between
 161  * 32-bit and 64-bit system, because Elf32_Nhdr and Elf64_Nhdr are
 162  * identical.
 163  */
 164 static inline int stack_map_parse_build_id(void *page_addr,
     /*   */
 165                                            unsigned char *build_id,
 166                                            void *note_start,
 167                                            Elf32_Word note_size)
 168 {
 169         Elf32_Word note_offs = 0, new_offs;
 170 
 171         /* check for overflow */
 172         if (note_start < page_addr || note_start + note_size < note_start)
 173                 return -EINVAL;
 174 
 175         /* only supports note that fits in the first page */
 176         if (note_start + note_size > page_addr + PAGE_SIZE)
 177                 return -EINVAL;
 178 
 179         while (note_offs + sizeof(Elf32_Nhdr) < note_size) {
 180                 Elf32_Nhdr *nhdr = (Elf32_Nhdr *)(note_start + note_offs);
 181 
 182                 if (nhdr->n_type == BPF_BUILD_ID &&
 183                     nhdr->n_namesz == sizeof("GNU") &&
 184                     nhdr->n_descsz > 0 &&
 185                     nhdr->n_descsz <= BPF_BUILD_ID_SIZE) {
 186                         memcpy(build_id,
 187                                note_start + note_offs +
 188                                ALIGN(sizeof("GNU"), 4) + sizeof(Elf32_Nhdr),
 189                                nhdr->n_descsz);
 190                         memset(build_id + nhdr->n_descsz, 0,
 191                                BPF_BUILD_ID_SIZE - nhdr->n_descsz);
 192                         return 0;
 193                 }
 194                 new_offs = note_offs + sizeof(Elf32_Nhdr) +
 195                         ALIGN(nhdr->n_namesz, 4) + ALIGN(nhdr->n_descsz, 4);
 196                 if (new_offs <= note_offs)  /* overflow */
 197                         break;
 198                 note_offs = new_offs;
 199         }
 200         return -EINVAL;
 201 }

 203 /* Parse build ID from 32-bit ELF */
 204 static int stack_map_get_build_id_32(void *page_addr,
     /*   */
 205                                      unsigned char *build_id)
 206 {
 207         Elf32_Ehdr *ehdr = (Elf32_Ehdr *)page_addr;
 208         Elf32_Phdr *phdr;
 209         int i;
 210 
 211         /* only supports phdr that fits in one page */
 212         if (ehdr->e_phnum >
 213             (PAGE_SIZE - sizeof(Elf32_Ehdr)) / sizeof(Elf32_Phdr))
 214                 return -EINVAL;
 215 
 216         phdr = (Elf32_Phdr *)(page_addr + sizeof(Elf32_Ehdr));
 217 
 218         for (i = 0; i < ehdr->e_phnum; ++i) {
 219                 if (phdr[i].p_type == PT_NOTE &&
 220                     !stack_map_parse_build_id(page_addr, build_id,
 221                                               page_addr + phdr[i].p_offset,
 222                                               phdr[i].p_filesz))
 223                         return 0;
 224         }
 225         return -EINVAL;
 226 }

 253 /* Parse build ID of ELF file mapped to vma */
 254 static int stack_map_get_build_id(struct vm_area_struct *vma,
     /*   */
 255                                   unsigned char *build_id)
 256 {
 257         Elf32_Ehdr *ehdr;
 258         struct page *page;
 259         void *page_addr;
 260         int ret;
 261 
 262         /* only works for page backed storage  */
 263         if (!vma->vm_file)
 264                 return -EINVAL;
 265 
 266         page = find_get_page(vma->vm_file->f_mapping, 0);
 267         if (!page)
 268                 return -EFAULT; /* page not mapped */
 269 
 270         ret = -EINVAL;
 271         page_addr = kmap_atomic(page);
 272         ehdr = (Elf32_Ehdr *)page_addr;
 273 
 274         /* compare magic x7f "ELF" */
 275         if (memcmp(ehdr->e_ident, ELFMAG, SELFMAG) != 0)
 276                 goto out;
 277 
 278         /* only support executable file and shared object file */
 279         if (ehdr->e_type != ET_EXEC && ehdr->e_type != ET_DYN)
 280                 goto out;
 281 
 282         if (ehdr->e_ident[EI_CLASS] == ELFCLASS32)
 283                 ret = stack_map_get_build_id_32(page_addr, build_id);
 284         else if (ehdr->e_ident[EI_CLASS] == ELFCLASS64)
 285                 ret = stack_map_get_build_id_64(page_addr, build_id);
 286 out:
 287         kunmap_atomic(page_addr);
 288         put_page(page);
 289         return ret;
 290 }

 292 static void stack_map_get_build_id_offset(struct bpf_stack_build_id *id_offs,
     /*   */
 293                                           u64 *ips, u32 trace_nr, bool user)
 294 {
 295         int i;
 296         struct vm_area_struct *vma;
 297         bool irq_work_busy = false;
 298         struct stack_map_irq_work *work = NULL;
 299 
 300         if (irqs_disabled()) {
 301                 if (!IS_ENABLED(CONFIG_PREEMPT_RT)) {
 302                         work = this_cpu_ptr(&up_read_work);
 303                         if (atomic_read(&work->irq_work.flags) & IRQ_WORK_BUSY) {
 304                                 /* cannot queue more up_read, fallback */
 305                                 irq_work_busy = true;
 306                         }
 307                 } else {
 308                         /*
 309                          * PREEMPT_RT does not allow to trylock mmap sem in
 310                          * interrupt disabled context. Force the fallback code.
 311                          */
 312                         irq_work_busy = true;
 313                 }
 314         }
 315 
 316         /*
 317          * We cannot do up_read() when the irq is disabled, because of
 318          * risk to deadlock with rq_lock. To do build_id lookup when the
 319          * irqs are disabled, we need to run up_read() in irq_work. We use
 320          * a percpu variable to do the irq_work. If the irq_work is
 321          * already used by another lookup, we fall back to report ips.
 322          *
 323          * Same fallback is used for kernel stack (!user) on a stackmap
 324          * with build_id.
 325          */
 326         if (!user || !current || !current->mm || irq_work_busy ||
 327             !mmap_read_trylock_non_owner(current->mm)) {
 328                 /* cannot access current->mm, fall back to ips */
 329                 for (i = 0; i < trace_nr; i++) {
 330                         id_offs[i].status = BPF_STACK_BUILD_ID_IP;
 331                         id_offs[i].ip = ips[i];
 332                         memset(id_offs[i].build_id, 0, BPF_BUILD_ID_SIZE);
 333                 }
 334                 return;
 335         }
 336 
 337         for (i = 0; i < trace_nr; i++) {
 338                 vma = find_vma(current->mm, ips[i]);
 339                 if (!vma || stack_map_get_build_id(vma, id_offs[i].build_id)) {
 340                         /* per entry fall back to ips */
 341                         id_offs[i].status = BPF_STACK_BUILD_ID_IP;
 342                         id_offs[i].ip = ips[i];
 343                         memset(id_offs[i].build_id, 0, BPF_BUILD_ID_SIZE);
 344                         continue;
 345                 }
 346                 id_offs[i].offset = (vma->vm_pgoff << PAGE_SHIFT) + ips[i]
 347                         - vma->vm_start;
 348                 id_offs[i].status = BPF_STACK_BUILD_ID_VALID;
 349         }
 350 
 351         if (!work) {
 352                 mmap_read_unlock_non_owner(current->mm);
 353         } else {
 354                 work->mm = current->mm;
 355                 irq_work_queue(&work->irq_work);
 356         }
 357 }

 393         return NULL;

 397 static long __bpf_get_stackid(struct bpf_map *map,
     /*   */
 398                               struct perf_callchain_entry *trace, u64 flags)
 399 {
 400         struct bpf_stack_map *smap = container_of(map, struct bpf_stack_map, map);
 401         struct stack_map_bucket *bucket, *new_bucket, *old_bucket;
 402         u32 max_depth = map->value_size / stack_map_data_size(map);
 403         /* stack_map_alloc() checks that max_depth <= sysctl_perf_event_max_stack */
 404         u32 init_nr = sysctl_perf_event_max_stack - max_depth;
 405         u32 skip = flags & BPF_F_SKIP_FIELD_MASK;
 406         u32 hash, id, trace_nr, trace_len;
 407         bool user = flags & BPF_F_USER_STACK;
 408         u64 *ips;
 409         bool hash_matches;
 410 
 411         /* get_perf_callchain() guarantees that trace->nr >= init_nr
 412          * and trace-nr <= sysctl_perf_event_max_stack, so trace_nr <= max_depth
 413          */
 414         trace_nr = trace->nr - init_nr;
 415 
 416         if (trace_nr <= skip)
 417                 /* skipping more than usable stack trace */
 418                 return -EFAULT;
 419 
 420         trace_nr -= skip;
 421         trace_len = trace_nr * sizeof(u64);
 422         ips = trace->ip + skip + init_nr;
 423         hash = jhash2((u32 *)ips, trace_len / sizeof(u32), 0);
 424         id = hash & (smap->n_buckets - 1);
 425         bucket = READ_ONCE(smap->buckets[id]);
 426 
 427         hash_matches = bucket && bucket->hash == hash;
 428         /* fast cmp */
 429         if (hash_matches && flags & BPF_F_FAST_STACK_CMP)
 430                 return id;
 431 
 432         if (stack_map_use_build_id(map)) {
 433                 /* for build_id+offset, pop a bucket before slow cmp */
 434                 new_bucket = (struct stack_map_bucket *)
 435                         pcpu_freelist_pop(&smap->freelist);
 436                 if (unlikely(!new_bucket))
 437                         return -ENOMEM;
 438                 new_bucket->nr = trace_nr;
 439                 stack_map_get_build_id_offset(
 440                         (struct bpf_stack_build_id *)new_bucket->data,
 441                         ips, trace_nr, user);
 442                 trace_len = trace_nr * sizeof(struct bpf_stack_build_id);
 443                 if (hash_matches && bucket->nr == trace_nr &&
 444                     memcmp(bucket->data, new_bucket->data, trace_len) == 0) {
 445                         pcpu_freelist_push(&smap->freelist, &new_bucket->fnode);
 446                         return id;
 447                 }
 448                 if (bucket && !(flags & BPF_F_REUSE_STACKID)) {
 449                         pcpu_freelist_push(&smap->freelist, &new_bucket->fnode);
 450                         return -EEXIST;
 451                 }
 452         } else {
 453                 if (hash_matches && bucket->nr == trace_nr &&
 454                     memcmp(bucket->data, ips, trace_len) == 0)
 455                         return id;
 456                 if (bucket && !(flags & BPF_F_REUSE_STACKID))
 457                         return -EEXIST;
 458 
 459                 new_bucket = (struct stack_map_bucket *)
 460                         pcpu_freelist_pop(&smap->freelist);
 461                 if (unlikely(!new_bucket))
 462                         return -ENOMEM;
 463                 memcpy(new_bucket->data, ips, trace_len);
 464         }
 465 
 466         new_bucket->hash = hash;
 467         new_bucket->nr = trace_nr;
 468 
 469         old_bucket = xchg(&smap->buckets[id], new_bucket);
 470         if (old_bucket)
 471                 pcpu_freelist_push(&smap->freelist, &old_bucket->fnode);
 472         return id;
 473 }

 475 BPF_CALL_3(bpf_get_stackid, struct pt_regs *, regs, struct bpf_map *, map,
     /*   */
 476            u64, flags)
 477 {
 478         u32 max_depth = map->value_size / stack_map_data_size(map);
 479         /* stack_map_alloc() checks that max_depth <= sysctl_perf_event_max_stack */
 480         u32 init_nr = sysctl_perf_event_max_stack - max_depth;
 481         bool user = flags & BPF_F_USER_STACK;
 482         struct perf_callchain_entry *trace;
 483         bool kernel = !user;
 484 
 485         if (unlikely(flags & ~(BPF_F_SKIP_FIELD_MASK | BPF_F_USER_STACK |
 486                                BPF_F_FAST_STACK_CMP | BPF_F_REUSE_STACKID)))
 487                 return -EINVAL;
 488 
 489         trace = get_perf_callchain(regs, init_nr, kernel, user,
 490                                    sysctl_perf_event_max_stack, false, false);
 491 
 492         if (unlikely(!trace))
 493                 /* couldn't fetch the stack trace */
 494                 return -EFAULT;
 495 
 496         return __bpf_get_stackid(map, trace, flags);
 497 }

 508 static __u64 count_kernel_ip(struct perf_callchain_entry *trace)
     /*   */
 509 {
 510         __u64 nr_kernel = 0;

 520 BPF_CALL_3(bpf_get_stackid_pe, struct bpf_perf_event_data_kern *, ctx,
     /*   */
 521            struct bpf_map *, map, u64, flags)
 522 {
 523         struct perf_event *event = ctx->event;
 524         struct perf_callchain_entry *trace;
 525         bool kernel, user;
 526         __u64 nr_kernel;
 527         int ret;
 528 
 529         /* perf_sample_data doesn't have callchain, use bpf_get_stackid */
 530         if (!(event->attr.sample_type & __PERF_SAMPLE_CALLCHAIN_EARLY))
 531                 return bpf_get_stackid((unsigned long)(ctx->regs),
 532                                        (unsigned long) map, flags, 0, 0);
 533 
 534         if (unlikely(flags & ~(BPF_F_SKIP_FIELD_MASK | BPF_F_USER_STACK |
 535                                BPF_F_FAST_STACK_CMP | BPF_F_REUSE_STACKID)))
 536                 return -EINVAL;
 537 
 538         user = flags & BPF_F_USER_STACK;
 539         kernel = !user;
 540 
 541         trace = ctx->data->callchain;
 542         if (unlikely(!trace))
 543                 return -EFAULT;
 544 
 545         nr_kernel = count_kernel_ip(trace);
 546 
 547         if (kernel) {
 548                 __u64 nr = trace->nr;
 549 
 550                 trace->nr = nr_kernel;
 551                 ret = __bpf_get_stackid(map, trace, flags);
 552 
 553                 /* restore nr */
 554                 trace->nr = nr;
 555         } else { /* user */
 556                 u64 skip = flags & BPF_F_SKIP_FIELD_MASK;
 557 
 558                 skip += nr_kernel;
 559                 if (skip > BPF_F_SKIP_FIELD_MASK)
 560                         return -EFAULT;
 561 
 562                 flags = (flags & ~BPF_F_SKIP_FIELD_MASK) | skip;
 563                 ret = __bpf_get_stackid(map, trace, flags);
 564         }
 565         return ret;
 566 }

 577 static long __bpf_get_stack(struct pt_regs *regs, struct task_struct *task,
     /*   */
 578                             struct perf_callchain_entry *trace_in,

 580 {
 581         u32 init_nr, trace_nr, copy_len, elem_size, num_elem;
 582         bool user_build_id = flags & BPF_F_USER_BUILD_ID;
 583         u32 skip = flags & BPF_F_SKIP_FIELD_MASK;
 584         bool user = flags & BPF_F_USER_STACK;
 585         struct perf_callchain_entry *trace;
 586         bool kernel = !user;
 587         int err = -EINVAL;
 588         u64 *ips;
 589 
 590         if (unlikely(flags & ~(BPF_F_SKIP_FIELD_MASK | BPF_F_USER_STACK |
 591                                BPF_F_USER_BUILD_ID)))
 592                 goto clear;
 593         if (kernel && user_build_id)
 594                 goto clear;
 595 
 596         elem_size = (user && user_build_id) ? sizeof(struct bpf_stack_build_id)
 597                                             : sizeof(u64);
 598         if (unlikely(size % elem_size))
 599                 goto clear;
 600 
 601         /* cannot get valid user stack for task without user_mode regs */
 602         if (task && user && !user_mode(regs))
 603                 goto err_fault;
 604 
 605         num_elem = size / elem_size;
 606         if (sysctl_perf_event_max_stack < num_elem)
 607                 init_nr = 0;
 608         else
 609                 init_nr = sysctl_perf_event_max_stack - num_elem;
 610 
 611         if (trace_in)
 612                 trace = trace_in;
 613         else if (kernel && task)
 614                 trace = get_callchain_entry_for_task(task, init_nr);
 615         else
 616                 trace = get_perf_callchain(regs, init_nr, kernel, user,
 617                                            sysctl_perf_event_max_stack,
 618                                            false, false);
 619         if (unlikely(!trace))
 620                 goto err_fault;
 621 
 622         trace_nr = trace->nr - init_nr;
 623         if (trace_nr < skip)
 624                 goto err_fault;
 625 
 626         trace_nr -= skip;
 627         trace_nr = (trace_nr <= num_elem) ? trace_nr : num_elem;
 628         copy_len = trace_nr * elem_size;
 629         ips = trace->ip + skip + init_nr;
 630         if (user && user_build_id)
 631                 stack_map_get_build_id_offset(buf, ips, trace_nr, user);
 632         else
 633                 memcpy(buf, ips, copy_len);
 634 
 635         if (size > copy_len)
 636                 memset(buf + copy_len, 0, size - copy_len);
 637         return copy_len;
 638 
 639 err_fault:
 640         err = -EFAULT;
 641 clear:
 642         memset(buf, 0, size);
 643         return err;
 644 }

 691 BPF_CALL_4(bpf_get_stack_pe, struct bpf_perf_event_data_kern *, ctx,
     /*   */
 692            void *, buf, u32, size, u64, flags)
 693 {
 694         struct pt_regs *regs = (struct pt_regs *)(ctx->regs);
 695         struct perf_event *event = ctx->event;
 696         struct perf_callchain_entry *trace;
 697         bool kernel, user;
 698         int err = -EINVAL;
 699         __u64 nr_kernel;
 700 
 701         if (!(event->attr.sample_type & __PERF_SAMPLE_CALLCHAIN_EARLY))
 702                 return __bpf_get_stack(regs, NULL, NULL, buf, size, flags);
 703 
 704         if (unlikely(flags & ~(BPF_F_SKIP_FIELD_MASK | BPF_F_USER_STACK |
 705                                BPF_F_USER_BUILD_ID)))
 706                 goto clear;
 707 
 708         user = flags & BPF_F_USER_STACK;
 709         kernel = !user;
 710 
 711         err = -EFAULT;
 712         trace = ctx->data->callchain;
 713         if (unlikely(!trace))
 714                 goto clear;
 715 
 716         nr_kernel = count_kernel_ip(trace);
 717 
 718         if (kernel) {
 719                 __u64 nr = trace->nr;
 720 
 721                 trace->nr = nr_kernel;
 722                 err = __bpf_get_stack(regs, NULL, trace, buf, size, flags);
 723 
 724                 /* restore nr */
 725                 trace->nr = nr;
 726         } else { /* user */
 727                 u64 skip = flags & BPF_F_SKIP_FIELD_MASK;
 728 
 729                 skip += nr_kernel;
 730                 if (skip > BPF_F_SKIP_FIELD_MASK)
 731                         goto clear;
 732 
 733                 flags = (flags & ~BPF_F_SKIP_FIELD_MASK) | skip;
 734                 err = __bpf_get_stack(regs, NULL, trace, buf, size, flags);
 735         }
 736         return err;
 737 
 738 clear:
 739         memset(buf, 0, size);
 740         return err;
 741 
 742 }

 760 /* Called from syscall */
 761 int bpf_stackmap_copy(struct bpf_map *map, void *key, void *value)
     /*   */
 762 {
 763         struct bpf_stack_map *smap = container_of(map, struct bpf_stack_map, map);
 764         struct stack_map_bucket *bucket, *old_bucket;
 765         u32 id = *(u32 *)key, trace_len;
 766 
 767         if (unlikely(id >= smap->n_buckets))
 768                 return -ENOENT;
 769 
 770         bucket = xchg(&smap->buckets[id], NULL);
 771         if (!bucket)
 772                 return -ENOENT;
 773 
 774         trace_len = bucket->nr * stack_map_data_size(map);
 775         memcpy(value, bucket->data, trace_len);
 776         memset(value + trace_len, 0, map->value_size - trace_len);
 777 
 778         old_bucket = xchg(&smap->buckets[id], bucket);
 779         if (old_bucket)
 780                 pcpu_freelist_push(&smap->freelist, &old_bucket->fnode);
 781         return 0;
 782 }

 784 static int stack_map_get_next_key(struct bpf_map *map, void *key,
     /*   */
 785                                   void *next_key)
 786 {
 787         struct bpf_stack_map *smap = container_of(map,
 788                                                   struct bpf_stack_map, map);
 789         u32 id;
 790 
 791         WARN_ON_ONCE(!rcu_read_lock_held());
 792 
 793         if (!key) {
 794                 id = 0;
 795         } else {
 796                 id = *(u32 *)key;
 797                 if (id >= smap->n_buckets || !smap->buckets[id])
 798                         id = 0;
 799                 else
 800                         id++;
 801         }
 802 
 803         while (id < smap->n_buckets && !smap->buckets[id])
 804                 id++;
 805 
 806         if (id >= smap->n_buckets)
 807                 return -ENOENT;
 808 
 809         *(u32 *)next_key = id;
 810         return 0;
 811 }

 820 static int stack_map_delete_elem(struct bpf_map *map, void *key)
     /*   */
 821 {
 822         struct bpf_stack_map *smap = container_of(map, struct bpf_stack_map, map);
 823         struct stack_map_bucket *old_bucket;
 824         u32 id = *(u32 *)key;
 825 
 826         if (unlikely(id >= smap->n_buckets))
 827                 return -E2BIG;
 828 
 829         old_bucket = xchg(&smap->buckets[id], NULL);
 830         if (old_bucket) {
 831                 pcpu_freelist_push(&smap->freelist, &old_bucket->fnode);
 832                 return 0;
 833         } else {
 834                 return -ENOENT;
 835         }
 836 }
