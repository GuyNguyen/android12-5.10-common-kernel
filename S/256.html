 240 /**
 241  * struct cond_snapshot - conditional snapshot data and callback
 242  *
 243  * The cond_snapshot structure encapsulates a callback function and
 244  * data associated with the snapshot for a given tracing instance.
 245  *
 246  * When a snapshot is taken conditionally, by invoking
 247  * tracing_snapshot_cond(tr, cond_data), the cond_data passed in is
 248  * passed in turn to the cond_snapshot.update() function.  That data
 249  * can be compared by the update() implementation with the cond_data
 250  * contained within the struct cond_snapshot instance associated with
 251  * the trace_array.  Because the tr->max_lock is held throughout the
 252  * update() call, the update() function can directly retrieve the
 253  * cond_snapshot and cond_data associated with the per-instance
 254  * snapshot associated with the trace_array.
 255  *
 256  * The cond_snapshot.update() implementation can save data to be
 257  * associated with the snapshot if it decides to, and returns 'true'
 258  * in that case, or it returns 'false' if the conditional snapshot
 259  * shouldn't be taken.
 260  *
 261  * The cond_snapshot instance is created and associated with the
 262  * user-defined cond_data by tracing_cond_snapshot_enable().
 263  * Likewise, the cond_snapshot instance is destroyed and is no longer
 264  * associated with the trace instance by
 265  * tracing_cond_snapshot_disable().
 266  *
 267  * The method below is required.
 268  *
 269  * @update: When a conditional snapshot is invoked, the update()
 270  *      callback function is invoked with the tr->max_lock held.  The
 271  *      update() implementation signals whether or not to actually
 272  *      take the snapshot, by returning 'true' if so, 'false' if no
 273  *      snapshot should be taken.  Because the max_lock is held for
 274  *      the duration of update(), the implementation is safe to
 275  *      directly retrieved and save any implementation data it needs
 276  *      to in association with the snapshot.
 277  */
 278 struct cond_snapshot {
 279         void                            *cond_data;
 280         cond_update_fn_t                update;
 281 };

 564 /*
 565  * For function tracing recursion:
 566  *  The order of these bits are important.
 567  *
 568  *  When function tracing occurs, the following steps are made:
 569  *   If arch does not support a ftrace feature:
 570  *    call internal function (uses INTERNAL bits) which calls...
 571  *   If callback is registered to the "global" list, the list
 572  *    function is called and recursion checks the GLOBAL bits.
 573  *    then this function calls...
 574  *   The function callback, which can use the FTRACE bits to
 575  *    check for recursion.
 576  *
 577  * Now if the arch does not support a feature, and it calls
 578  * the global list function which calls the ftrace callback
 579  * all three of these steps will do a recursion protection.
 580  * There's no reason to do one if the previous caller already
 581  * did. The recursion that we are protecting against will
 582  * go through the same steps again.
 583  *
 584  * To prevent the multiple recursion checks, if a recursion
 585  * bit is set that is higher than the MAX bit of the current
 586  * check, then we know that the check was made by the previous
 587  * caller, and we can skip the current check.
 588  */
 589 enum {
 590         /* Function recursion bits */
 591         TRACE_FTRACE_BIT,
 592         TRACE_FTRACE_NMI_BIT,
 593         TRACE_FTRACE_IRQ_BIT,
 594         TRACE_FTRACE_SIRQ_BIT,
 595 
 596         /* INTERNAL_BITs must be greater than FTRACE_BITs */
 597         TRACE_INTERNAL_BIT,
 598         TRACE_INTERNAL_NMI_BIT,
 599         TRACE_INTERNAL_IRQ_BIT,
 600         TRACE_INTERNAL_SIRQ_BIT,
 601 
 602         TRACE_BRANCH_BIT,
 603 /*
 604  * Abuse of the trace_recursion.
 605  * As we need a way to maintain state if we are tracing the function
 606  * graph in irq because we want to trace a particular function that
 607  * was called in irq context but we have irq tracing off. Since this
 608  * can only be modified by current, we can reuse trace_recursion.
 609  */
 610         TRACE_IRQ_BIT,
 611 
 612         /* Set if the function is in the set_graph_function file */
 613         TRACE_GRAPH_BIT,
 614 
 615         /*
 616          * In the very unlikely case that an interrupt came in
 617          * at a start of graph tracing, and we want to trace
 618          * the function in that interrupt, the depth can be greater
 619          * than zero, because of the preempted start of a previous
 620          * trace. In an even more unlikely case, depth could be 2
 621          * if a softirq interrupted the start of graph tracing,
 622          * followed by an interrupt preempting a start of graph
 623          * tracing in the softirq, and depth can even be 3
 624          * if an NMI came in at the start of an interrupt function
 625          * that preempted a softirq start of a function that
 626          * preempted normal context!!!! Luckily, it can't be
 627          * greater than 3, so the next two bits are a mask
 628          * of what the depth is when we set TRACE_GRAPH_BIT
 629          */
 630 
 631         TRACE_GRAPH_DEPTH_START_BIT,
 632         TRACE_GRAPH_DEPTH_END_BIT,
 633 
 634         /*
 635          * To implement set_graph_notrace, if this bit is set, we ignore
 636          * function graph tracing of called functions, until the return
 637          * function is called to clear it.
 638          */
 639         TRACE_GRAPH_NOTRACE_BIT,
 640 
 641         /*
 642          * When transitioning between context, the preempt_count() may
 643          * not be correct. Allow for a single recursion to cover this case.
 644          */
 645         TRACE_TRANSITION_BIT,
 646 };

 648 #define trace_recursion_set(bit)        do { (current)->trace_recursion |= (1<<(bit)); } while (0)
 649 #define trace_recursion_clear(bit)      do { (current)->trace_recursion &= ~(1<<(bit)); } while (0)
 650 #define trace_recursion_test(bit)       ((current)->trace_recursion & (1<<(bit)))
 651 
 652 #define trace_recursion_depth() \
 653         (((current)->trace_recursion >> TRACE_GRAPH_DEPTH_START_BIT) & 3)
 654 #define trace_recursion_set_depth(depth) \
 655         do {                                                            \
 656                 current->trace_recursion &=                             \
 657                         ~(3 << TRACE_GRAPH_DEPTH_START_BIT);            \
 658                 current->trace_recursion |=                             \
 659                         ((depth) & 3) << TRACE_GRAPH_DEPTH_START_BIT;   \
 660         } while (0)

 662 #define TRACE_CONTEXT_BITS      4
 663 
 664 #define TRACE_FTRACE_START      TRACE_FTRACE_BIT
 665 #define TRACE_FTRACE_MAX        ((1 << (TRACE_FTRACE_START + TRACE_CONTEXT_BITS)) - 1)
 666 
 667 #define TRACE_LIST_START        TRACE_INTERNAL_BIT
 668 #define TRACE_LIST_MAX          ((1 << (TRACE_LIST_START + TRACE_CONTEXT_BITS)) - 1)

 672 static __always_inline int trace_get_context_bit(void)
     /*   */
 673 {
 674         int bit;
 675 
 676         if (in_interrupt()) {
 677                 if (in_nmi())
 678                         bit = 0;
 679 
 680                 else if (in_irq())
 681                         bit = 1;
 682                 else
 683                         bit = 2;
 684         } else
 685                 bit = 3;
 686 
 687         return bit;
 688 }

 690 static __always_inline int trace_test_and_set_recursion(int start, int max)
     /*   */
 691 {
 692         unsigned int val = current->trace_recursion;
 693         int bit;
 694 
 695         /* A previous recursion check was made */
 696         if ((val & TRACE_CONTEXT_MASK) > max)
 697                 return 0;
 698 
 699         bit = trace_get_context_bit() + start;
 700         if (unlikely(val & (1 << bit))) {
 701                 /*
 702                  * It could be that preempt_count has not been updated during
 703                  * a switch between contexts. Allow for a single recursion.
 704                  */
 705                 bit = TRACE_TRANSITION_BIT;
 706                 if (trace_recursion_test(bit))
 707                         return -1;
 708                 trace_recursion_set(bit);
 709                 barrier();
 710                 return bit + 1;
 711         }
 712 
 713         /* Normal check passed, clear the transition to allow it again */
 714         trace_recursion_clear(TRACE_TRANSITION_BIT);
 715 
 716         val |= 1 << bit;
 717         current->trace_recursion = val;
 718         barrier();
 719 
 720         return bit + 1;
 721 }

 723 static __always_inline void trace_clear_recursion(int bit)
     /*   */
 724 {
 725         unsigned int val = current->trace_recursion;
 726 
 727         if (!bit)
 728                 return;
 729 
 730         bit--;
 731         bit = 1 << bit;
 732         val &= ~bit;
 733 
 734         barrier();
 735         current->trace_recursion = val;
 736 }

1020 static inline int ftrace_graph_addr(struct ftrace_graph_ent *trace)
     /*   */
1021 {
1022         unsigned long addr = trace->func;
1023         int ret = 0;
1024         struct ftrace_hash *hash;
1025 
1026         preempt_disable_notrace();
1027 
1028         /*
1029          * Have to open code "rcu_dereference_sched()" because the
1030          * function graph tracer can be called when RCU is not
1031          * "watching".
1032          * Protected with schedule_on_each_cpu(ftrace_sync)
1033          */
1034         hash = rcu_dereference_protected(ftrace_graph_hash, !preemptible());
1035 
1036         if (ftrace_hash_empty(hash)) {
1037                 ret = 1;
1038                 goto out;
1039         }
1040 
1041         if (ftrace_lookup_ip(hash, addr)) {
1042 
1043                 /*
1044                  * This needs to be cleared on the return functions
1045                  * when the depth is zero.
1046                  */
1047                 trace_recursion_set(TRACE_GRAPH_BIT);
1048                 trace_recursion_set_depth(trace->depth);
1049 
1050                 /*
1051                  * If no irqs are to be traced, but a set_graph_function
1052                  * is set, and called by an interrupt handler, we still
1053                  * want to trace it.
1054                  */
1055                 if (in_irq())
1056                         trace_recursion_set(TRACE_IRQ_BIT);
1057                 else
1058                         trace_recursion_clear(TRACE_IRQ_BIT);
1059                 ret = 1;
1060         }
1061 
1062 out:
1063         preempt_enable_notrace();
1064         return ret;
1065 }

1505 /*
1506  * Helper function for event_trigger_unlock_commit{_regs}().
1507  * If there are event triggers attached to this event that requires
1508  * filtering against its fields, then they will be called as the
1509  * entry already holds the field information of the current event.
1510  *
1511  * It also checks if the event should be discarded or not.
1512  * It is to be discarded if the event is soft disabled and the
1513  * event was only recorded to process triggers, or if the event
1514  * filter is active and this event did not match the filters.
1515  *
1516  * Returns true if the event is discarded, false otherwise.
1517  */
1518 static inline bool
1519 __event_trigger_test_discard(struct trace_event_file *file,
     /*   */
1520                              struct trace_buffer *buffer,
1521                              struct ring_buffer_event *event,
1522                              void *entry,
1523                              enum event_trigger_type *tt)
1524 {
1525         unsigned long eflags = file->flags;
1526 
1527         if (eflags & EVENT_FILE_FL_TRIGGER_COND)
1528                 *tt = event_triggers_call(file, entry, event);
1529 
1530         if (test_bit(EVENT_FILE_FL_SOFT_DISABLED_BIT, &file->flags) ||
1531             (unlikely(file->flags & EVENT_FILE_FL_FILTERED) &&
1532              !filter_match_preds(file->filter, entry))) {
1533                 __trace_event_discard_commit(buffer, event);
1534                 return true;
1535         }
1536 
1537         return false;
1538 }

1788 /**
1789  * struct event_trigger_ops - callbacks for trace event triggers
1790  *
1791  * The methods in this structure provide per-event trigger hooks for
1792  * various trigger operations.
1793  *
1794  * All the methods below, except for @init() and @free(), must be
1795  * implemented.
1796  *
1797  * @func: The trigger 'probe' function called when the triggering
1798  *      event occurs.  The data passed into this callback is the data
1799  *      that was supplied to the event_command @reg() function that
1800  *      registered the trigger (see struct event_command) along with
1801  *      the trace record, rec.
1802  *
1803  * @init: An optional initialization function called for the trigger
1804  *      when the trigger is registered (via the event_command reg()
1805  *      function).  This can be used to perform per-trigger
1806  *      initialization such as incrementing a per-trigger reference
1807  *      count, for instance.  This is usually implemented by the
1808  *      generic utility function @event_trigger_init() (see
1809  *      trace_event_triggers.c).
1810  *
1811  * @free: An optional de-initialization function called for the
1812  *      trigger when the trigger is unregistered (via the
1813  *      event_command @reg() function).  This can be used to perform
1814  *      per-trigger de-initialization such as decrementing a
1815  *      per-trigger reference count and freeing corresponding trigger
1816  *      data, for instance.  This is usually implemented by the
1817  *      generic utility function @event_trigger_free() (see
1818  *      trace_event_triggers.c).
1819  *
1820  * @print: The callback function invoked to have the trigger print
1821  *      itself.  This is usually implemented by a wrapper function
1822  *      that calls the generic utility function @event_trigger_print()
1823  *      (see trace_event_triggers.c).
1824  */

1827                                         void *rec,
1828                                         struct ring_buffer_event *rbe);
1829         int                     (*init)(struct event_trigger_ops *ops,
1830                                         struct event_trigger_data *data);
1831         void                    (*free)(struct event_trigger_ops *ops,
1832                                         struct event_trigger_data *data);
1833         int                     (*print)(struct seq_file *m,
1834                                          struct event_trigger_ops *ops,
1835                                          struct event_trigger_data *data);

1838 /**
1839  * struct event_command - callbacks and data members for event commands
1840  *
1841  * Event commands are invoked by users by writing the command name
1842  * into the 'trigger' file associated with a trace event.  The
1843  * parameters associated with a specific invocation of an event
1844  * command are used to create an event trigger instance, which is
1845  * added to the list of trigger instances associated with that trace
1846  * event.  When the event is hit, the set of triggers associated with
1847  * that event is invoked.
1848  *
1849  * The data members in this structure provide per-event command data
1850  * for various event commands.
1851  *
1852  * All the data members below, except for @post_trigger, must be set
1853  * for each event command.
1854  *
1855  * @name: The unique name that identifies the event command.  This is
1856  *      the name used when setting triggers via trigger files.
1857  *
1858  * @trigger_type: A unique id that identifies the event command
1859  *      'type'.  This value has two purposes, the first to ensure that
1860  *      only one trigger of the same type can be set at a given time
1861  *      for a particular event e.g. it doesn't make sense to have both
1862  *      a traceon and traceoff trigger attached to a single event at
1863  *      the same time, so traceon and traceoff have the same type
1864  *      though they have different names.  The @trigger_type value is
1865  *      also used as a bit value for deferring the actual trigger
1866  *      action until after the current event is finished.  Some
1867  *      commands need to do this if they themselves log to the trace
1868  *      buffer (see the @post_trigger() member below).  @trigger_type
1869  *      values are defined by adding new values to the trigger_type
1870  *      enum in include/linux/trace_events.h.
1871  *
1872  * @flags: See the enum event_command_flags below.
1873  *
1874  * All the methods below, except for @set_filter() and @unreg_all(),
1875  * must be implemented.
1876  *
1877  * @func: The callback function responsible for parsing and
1878  *      registering the trigger written to the 'trigger' file by the
1879  *      user.  It allocates the trigger instance and registers it with
1880  *      the appropriate trace event.  It makes use of the other
1881  *      event_command callback functions to orchestrate this, and is
1882  *      usually implemented by the generic utility function
1883  *      @event_trigger_callback() (see trace_event_triggers.c).
1884  *
1885  * @reg: Adds the trigger to the list of triggers associated with the
1886  *      event, and enables the event trigger itself, after
1887  *      initializing it (via the event_trigger_ops @init() function).
1888  *      This is also where commands can use the @trigger_type value to
1889  *      make the decision as to whether or not multiple instances of
1890  *      the trigger should be allowed.  This is usually implemented by
1891  *      the generic utility function @register_trigger() (see
1892  *      trace_event_triggers.c).
1893  *
1894  * @unreg: Removes the trigger from the list of triggers associated
1895  *      with the event, and disables the event trigger itself, after
1896  *      initializing it (via the event_trigger_ops @free() function).
1897  *      This is usually implemented by the generic utility function
1898  *      @unregister_trigger() (see trace_event_triggers.c).
1899  *
1900  * @unreg_all: An optional function called to remove all the triggers
1901  *      from the list of triggers associated with the event.  Called
1902  *      when a trigger file is opened in truncate mode.
1903  *
1904  * @set_filter: An optional function called to parse and set a filter
1905  *      for the trigger.  If no @set_filter() method is set for the
1906  *      event command, filters set by the user for the command will be
1907  *      ignored.  This is usually implemented by the generic utility
1908  *      function @set_trigger_filter() (see trace_event_triggers.c).
1909  *
1910  * @get_trigger_ops: The callback function invoked to retrieve the
1911  *      event_trigger_ops implementation associated with the command.
1912  */
1913 struct event_command {
1914         struct list_head        list;
1915         char                    *name;
1916         enum event_trigger_type trigger_type;
1917         int                     flags;
1918         int                     (*func)(struct event_command *cmd_ops,
1919                                         struct trace_event_file *file,
1920                                         char *glob, char *cmd, char *params);
1921         int                     (*reg)(char *glob,
1922                                        struct event_trigger_ops *ops,
1923                                        struct event_trigger_data *data,
1924                                        struct trace_event_file *file);
1925         void                    (*unreg)(char *glob,
1926                                          struct event_trigger_ops *ops,
1927                                          struct event_trigger_data *data,
1928                                          struct trace_event_file *file);
1929         void                    (*unreg_all)(struct trace_event_file *file);
1930         int                     (*set_filter)(char *filter_str,
1931                                               struct event_trigger_data *data,
1932                                               struct trace_event_file *file);
1933         struct event_trigger_ops *(*get_trigger_ops)(char *cmd, char *param);
1934 };

1936 /**
1937  * enum event_command_flags - flags for struct event_command
1938  *
1939  * @POST_TRIGGER: A flag that says whether or not this command needs
1940  *      to have its action delayed until after the current event has
1941  *      been closed.  Some triggers need to avoid being invoked while
1942  *      an event is currently in the process of being logged, since
1943  *      the trigger may itself log data into the trace buffer.  Thus
1944  *      we make sure the current event is committed before invoking
1945  *      those triggers.  To do that, the trigger invocation is split
1946  *      in two - the first part checks the filter using the current
1947  *      trace record; if a command has the @post_trigger flag set, it
1948  *      sets a bit for itself in the return value, otherwise it
1949  *      directly invokes the trigger.  Once all commands have been
1950  *      either invoked or set their return flag, the current record is
1951  *      either committed or discarded.  At that point, if any commands
1952  *      have deferred their triggers, those commands are finally
1953  *      invoked following the close of the current event.  In other
1954  *      words, if the event_trigger_ops @func() probe implementation
1955  *      itself logs to the trace buffer, this flag should be set,
1956  *      otherwise it can be left unspecified.
1957  *
1958  * @NEEDS_REC: A flag that says whether or not this command needs
1959  *      access to the trace record in order to perform its function,
1960  *      regardless of whether or not it has a filter associated with
1961  *      it (filters make a trigger require access to the trace record
1962  *      but are not always present).
1963  */
1964 enum event_command_flags {
1965         EVENT_CMD_FL_POST_TRIGGER       = 1,
1966         EVENT_CMD_FL_NEEDS_REC          = 2,
1967 };
1968 
1969 static inline bool event_command_post_trigger(struct event_command *cmd_ops)
     /*   */

1972 }

1974 static inline bool event_command_needs_rec(struct event_command *cmd_ops)
     /*   */
1975 {
1976         return cmd_ops->flags & EVENT_CMD_FL_NEEDS_REC;
