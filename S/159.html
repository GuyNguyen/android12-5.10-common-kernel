  48 static gfp_t dma_direct_optimal_gfp_mask(struct device *dev, u64 dma_mask,
     /*   */
  49                                   u64 *phys_limit)
  50 {
  51         u64 dma_limit = min_not_zero(dma_mask, dev->bus_dma_limit);
  52 
  53         /*
  54          * Optimistically try the zone that the physical address mask falls
  55          * into first.  If that returns memory that isn't actually addressable
  56          * we will fallback to the next lower zone and try again.
  57          *
  58          * Note that GFP_DMA32 and GFP_DMA are no ops without the corresponding
  59          * zones.
  60          */
  61         *phys_limit = dma_to_phys(dev, dma_limit);
  62         if (*phys_limit <= DMA_BIT_MASK(zone_dma_bits))
  63                 return GFP_DMA;
  64         if (*phys_limit <= DMA_BIT_MASK(32))
  65                 return GFP_DMA32;
  66         return 0;
  67 }

  69 static bool dma_coherent_ok(struct device *dev, phys_addr_t phys, size_t size)
     /*   */
  70 {
  71         dma_addr_t dma_addr = phys_to_dma_direct(dev, phys);
  72 
  73         if (dma_addr == DMA_MAPPING_ERROR)
  74                 return false;
  75         return dma_addr + size - 1 <=
  76                 min_not_zero(dev->coherent_dma_mask, dev->bus_dma_limit);

  79 static struct page *__dma_direct_alloc_pages(struct device *dev, size_t size,
     /*   */
  80                 gfp_t gfp)
  81 {
  82         int node = dev_to_node(dev);
  83         struct page *page = NULL;
  84         u64 phys_limit;
  85 
  86         WARN_ON_ONCE(!PAGE_ALIGNED(size));
  87 
  88         gfp |= dma_direct_optimal_gfp_mask(dev, dev->coherent_dma_mask,
  89                                            &phys_limit);
  90         page = dma_alloc_contiguous(dev, size, gfp);
  91         if (page && !dma_coherent_ok(dev, page_to_phys(page), size)) {
  92                 dma_free_contiguous(dev, page, size);
  93                 page = NULL;
  94         }
  95 again:
  96         if (!page)
  97                 page = alloc_pages_node(node, gfp, get_order(size));
  98         if (page && !dma_coherent_ok(dev, page_to_phys(page), size)) {
  99                 dma_free_contiguous(dev, page, size);
 100                 page = NULL;
 101 
 102                 if (IS_ENABLED(CONFIG_ZONE_DMA32) &&
 103                     phys_limit < DMA_BIT_MASK(64) &&
 104                     !(gfp & (GFP_DMA32 | GFP_DMA))) {
 105                         gfp |= GFP_DMA32;
 106                         goto again;
 107                 }
 108 
 109                 if (IS_ENABLED(CONFIG_ZONE_DMA) && !(gfp & GFP_DMA)) {
 110                         gfp = (gfp & ~GFP_DMA32) | GFP_DMA;
 111                         goto again;
 112                 }
 113         }
 114 
 115         return page;
 116 }

 134 void *dma_direct_alloc(struct device *dev, size_t size,
     /*   */
 135                 dma_addr_t *dma_handle, gfp_t gfp, unsigned long attrs)
 136 {
 137         struct page *page;
 138         void *ret;
 139         int err;
 140 
 141         size = PAGE_ALIGN(size);
 142         if (attrs & DMA_ATTR_NO_WARN)
 143                 gfp |= __GFP_NOWARN;
 144 
 145         if ((attrs & DMA_ATTR_NO_KERNEL_MAPPING) &&
 146             !force_dma_unencrypted(dev)) {
 147                 page = __dma_direct_alloc_pages(dev, size, gfp & ~__GFP_ZERO);
 148                 if (!page)
 149                         return NULL;
 150                 /* remove any dirty cache lines on the kernel alias */
 151                 if (!PageHighMem(page))
 152                         arch_dma_prep_coherent(page, size);
 153                 *dma_handle = phys_to_dma_direct(dev, page_to_phys(page));
 154                 /* return the page pointer as the opaque cookie */
 155                 return page;
 156         }
 157 
 158         if (!IS_ENABLED(CONFIG_ARCH_HAS_DMA_SET_UNCACHED) &&
 159             !IS_ENABLED(CONFIG_DMA_DIRECT_REMAP) &&
 160             !dev_is_dma_coherent(dev))
 161                 return arch_dma_alloc(dev, size, dma_handle, gfp, attrs);
 162 
 163         /*
 164          * Remapping or decrypting memory may block. If either is required and
 165          * we can't block, allocate the memory from the atomic pools.
 166          */
 167         if (IS_ENABLED(CONFIG_DMA_COHERENT_POOL) &&
 168             !gfpflags_allow_blocking(gfp) &&
 169             (force_dma_unencrypted(dev) ||
 170              (IS_ENABLED(CONFIG_DMA_DIRECT_REMAP) && !dev_is_dma_coherent(dev))))
 171                 return dma_direct_alloc_from_pool(dev, size, dma_handle, gfp);
 172 
 173         /* we always manually zero the memory once we are done */
 174         page = __dma_direct_alloc_pages(dev, size, gfp & ~__GFP_ZERO);
 175         if (!page)
 176                 return NULL;
 177 
 178         if ((IS_ENABLED(CONFIG_DMA_DIRECT_REMAP) &&
 179              !dev_is_dma_coherent(dev)) ||
 180             (IS_ENABLED(CONFIG_DMA_REMAP) && PageHighMem(page))) {
 181                 /* remove any dirty cache lines on the kernel alias */
 182                 arch_dma_prep_coherent(page, size);
 183 
 184                 /* create a coherent mapping */
 185                 ret = dma_common_contiguous_remap(page, size,
 186                                 dma_pgprot(dev, PAGE_KERNEL, attrs),
 187                                 __builtin_return_address(0));
 188                 if (!ret)
 189                         goto out_free_pages;
 190                 if (force_dma_unencrypted(dev)) {
 191                         err = set_memory_decrypted((unsigned long)ret,
 192                                                    1 << get_order(size));
 193                         if (err)
 194                                 goto out_free_pages;
 195                 }
 196                 memset(ret, 0, size);
 197                 goto done;
 198         }
 199 
 200         if (PageHighMem(page)) {
 201                 /*
 202                  * Depending on the cma= arguments and per-arch setup
 203                  * dma_alloc_contiguous could return highmem pages.
 204                  * Without remapping there is no way to return them here,
 205                  * so log an error and fail.
 206                  */
 207                 dev_info(dev, "Rejecting highmem page from CMA.\n");
 208                 goto out_free_pages;
 209         }
 210 
 211         ret = page_address(page);
 212         if (force_dma_unencrypted(dev)) {
 213                 err = set_memory_decrypted((unsigned long)ret,
 214                                            1 << get_order(size));
 215                 if (err)
 216                         goto out_free_pages;
 217         }
 218 
 219         memset(ret, 0, size);
 220 
 221         if (IS_ENABLED(CONFIG_ARCH_HAS_DMA_SET_UNCACHED) &&
 222             !dev_is_dma_coherent(dev)) {
 223                 arch_dma_prep_coherent(page, size);
 224                 ret = arch_dma_set_uncached(ret, size);
 225                 if (IS_ERR(ret))
 226                         goto out_encrypt_pages;
 227         }
 228 done:
 229         *dma_handle = phys_to_dma_direct(dev, page_to_phys(page));
 230         return ret;
 231 
 232 out_encrypt_pages:
 233         if (force_dma_unencrypted(dev)) {
 234                 err = set_memory_encrypted((unsigned long)page_address(page),
 235                                            1 << get_order(size));
 236                 /* If memory cannot be re-encrypted, it must be leaked */
 237                 if (err)
 238                         return NULL;
 239         }
 240 out_free_pages:
 241         dma_free_contiguous(dev, page, size);
 242         return NULL;
 243 }

 245 void dma_direct_free(struct device *dev, size_t size,
     /*   */
 246                 void *cpu_addr, dma_addr_t dma_addr, unsigned long attrs)
 247 {
 248         unsigned int page_order = get_order(size);
 249 
 250         if ((attrs & DMA_ATTR_NO_KERNEL_MAPPING) &&
 251             !force_dma_unencrypted(dev)) {
 252                 /* cpu_addr is a struct page cookie, not a kernel address */
 253                 dma_free_contiguous(dev, cpu_addr, size);
 254                 return;
 255         }
 256 
 257         if (!IS_ENABLED(CONFIG_ARCH_HAS_DMA_SET_UNCACHED) &&
 258             !IS_ENABLED(CONFIG_DMA_DIRECT_REMAP) &&
 259             !dev_is_dma_coherent(dev)) {
 260                 arch_dma_free(dev, size, cpu_addr, dma_addr, attrs);
 261                 return;
 262         }
 263 
 264         /* If cpu_addr is not from an atomic pool, dma_free_from_pool() fails */
 265         if (IS_ENABLED(CONFIG_DMA_COHERENT_POOL) &&
 266             dma_free_from_pool(dev, cpu_addr, PAGE_ALIGN(size)))
 267                 return;
 268 
 269         if (force_dma_unencrypted(dev))
 270                 set_memory_encrypted((unsigned long)cpu_addr, 1 << page_order);
 271 
 272         if (IS_ENABLED(CONFIG_DMA_REMAP) && is_vmalloc_addr(cpu_addr))
 273                 vunmap(cpu_addr);
 274         else if (IS_ENABLED(CONFIG_ARCH_HAS_DMA_CLEAR_UNCACHED))
 275                 arch_dma_clear_uncached(cpu_addr, size);
 276 
 277         dma_free_contiguous(dev, dma_direct_to_page(dev, dma_addr), size);
 278 }

 280 struct page *dma_direct_alloc_pages(struct device *dev, size_t size,
     /*   */
 281                 dma_addr_t *dma_handle, enum dma_data_direction dir, gfp_t gfp)
 282 {
 283         struct page *page;
 284         void *ret;
 285 
 286         if (IS_ENABLED(CONFIG_DMA_COHERENT_POOL) &&
 287             force_dma_unencrypted(dev) && !gfpflags_allow_blocking(gfp))
 288                 return dma_direct_alloc_from_pool(dev, size, dma_handle, gfp);
 289 
 290         page = __dma_direct_alloc_pages(dev, size, gfp);
 291         if (!page)
 292                 return NULL;
 293         if (PageHighMem(page)) {
 294                 /*
 295                  * Depending on the cma= arguments and per-arch setup
 296                  * dma_alloc_contiguous could return highmem pages.
 297                  * Without remapping there is no way to return them here,
 298                  * so log an error and fail.
 299                  */
 300                 dev_info(dev, "Rejecting highmem page from CMA.\n");
 301                 goto out_free_pages;
 302         }
 303 
 304         ret = page_address(page);
 305         if (force_dma_unencrypted(dev)) {
 306                 if (set_memory_decrypted((unsigned long)ret,
 307                                 1 << get_order(size)))
 308                         goto out_free_pages;
 309         }
 310         memset(ret, 0, size);
 311         *dma_handle = phys_to_dma_direct(dev, page_to_phys(page));
 312         return page;
 313 out_free_pages:
 314         dma_free_contiguous(dev, page, size);
 315         return NULL;
 316 }
 317 EXPORT_SYMBOL_GPL(dma_direct_alloc);

 319 void dma_direct_free_pages(struct device *dev, size_t size,
     /*   */
 320                 struct page *page, dma_addr_t dma_addr,
 321                 enum dma_data_direction dir)
 322 {
 323         unsigned int page_order = get_order(size);
 324         void *vaddr = page_address(page);
 325 
 326         /* If cpu_addr is not from an atomic pool, dma_free_from_pool() fails */
 327         if (IS_ENABLED(CONFIG_DMA_COHERENT_POOL) &&
 328             dma_free_from_pool(dev, vaddr, size))
 329                 return;
 330 
 331         if (force_dma_unencrypted(dev))
 332                 set_memory_encrypted((unsigned long)vaddr, 1 << page_order);
 333 
 334         dma_free_contiguous(dev, page, size);
 335 }

 455 int dma_direct_mmap(struct device *dev, struct vm_area_struct *vma,
     /*   */
 456                 void *cpu_addr, dma_addr_t dma_addr, size_t size,
 457                 unsigned long attrs)
 458 {
 459         unsigned long user_count = vma_pages(vma);
 460         unsigned long count = PAGE_ALIGN(size) >> PAGE_SHIFT;
 461         unsigned long pfn = PHYS_PFN(dma_to_phys(dev, dma_addr));
 462         int ret = -ENXIO;
 463 
 464         vma->vm_page_prot = dma_pgprot(dev, vma->vm_page_prot, attrs);
 465 
 466         if (dma_mmap_from_dev_coherent(dev, vma, cpu_addr, size, &ret))
 467                 return ret;
 468 
 469         if (vma->vm_pgoff >= count || user_count > count - vma->vm_pgoff)
 470                 return -ENXIO;
 471         return remap_pfn_range(vma, vma->vm_start, pfn + vma->vm_pgoff,
 472                         user_count << PAGE_SHIFT, vma->vm_page_prot);
 473 }

 475 int dma_direct_supported(struct device *dev, u64 mask)
     /*   */
 476 {
 477         u64 min_mask = (max_pfn - 1) << PAGE_SHIFT;
 478 
 479         /*
 480          * Because 32-bit DMA masks are so common we expect every architecture
 481          * to be able to satisfy them - either by not supporting more physical
 482          * memory, or by providing a ZONE_DMA32.  If neither is the case, the
 483          * architecture needs to use an IOMMU instead of the direct mapping.
 484          */
 485         if (mask >= DMA_BIT_MASK(32))
 486                 return 1;
 487 
 488         /*
 489          * This check needs to be against the actual bit mask value, so use
 490          * phys_to_dma_unencrypted() here so that the SME encryption mask isn't
 491          * part of the check.
 492          */
 493         if (IS_ENABLED(CONFIG_ZONE_DMA))
 494                 min_mask = min_t(u64, min_mask, DMA_BIT_MASK(zone_dma_bits));
 495         return mask >= phys_to_dma_unencrypted(dev, min_mask);
 496 }

 513 /**
 514  * dma_direct_set_offset - Assign scalar offset for a single DMA range.
 515  * @dev:        device pointer; needed to "own" the alloced memory.
 516  * @cpu_start:  beginning of memory region covered by this offset.
 517  * @dma_start:  beginning of DMA/PCI region covered by this offset.
 518  * @size:       size of the region.
 519  *
 520  * This is for the simple case of a uniform offset which cannot
 521  * be discovered by "dma-ranges".
 522  *
 523  * It returns -ENOMEM if out of memory, -EINVAL if a map
 524  * already exists, 0 otherwise.
 525  *
 526  * Note: any call to this from a driver is a bug.  The mapping needs
 527  * to be described by the device tree or other firmware interfaces.
 528  */
 529 int dma_direct_set_offset(struct device *dev, phys_addr_t cpu_start,
     /*   */
 530                          dma_addr_t dma_start, u64 size)
 531 {
 532         struct bus_dma_region *map;
 533         u64 offset = (u64)cpu_start - (u64)dma_start;
 534 
 535         if (dev->dma_range_map) {
 536                 dev_err(dev, "attempt to add DMA range to existing map\n");
 537                 return -EINVAL;
 538         }
 539 
 540         if (!offset)
 541                 return 0;
 542 
 543         map = kcalloc(2, sizeof(*map), GFP_KERNEL);
 544         if (!map)
 545                 return -ENOMEM;
 546         map[0].cpu_start = cpu_start;
 547         map[0].dma_start = dma_start;
 548         map[0].offset = offset;
 549         map[0].size = size;
 550         dev->dma_range_map = map;
 551         return 0;
 552 }
 553 EXPORT_SYMBOL_GPL(dma_direct_set_offset);
