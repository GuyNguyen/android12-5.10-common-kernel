 142 /**
 143  * get_image_page - Allocate a page for a hibernation image.
 144  * @gfp_mask: GFP mask for the allocation.
 145  * @safe_needed: Get pages that were not used before hibernation (restore only)
 146  *
 147  * During image restoration, for storing the PBE list and the image data, we can
 148  * only use memory pages that do not conflict with the pages used before
 149  * hibernation.  The "unsafe" pages have PageNosaveFree set and we count them
 150  * using allocated_unsafe_pages.
 151  *
 152  * Each allocated image page is marked as PageNosave and PageNosaveFree so that
 153  * swsusp_free() can release it.
 154  */
 155 static void *get_image_page(gfp_t gfp_mask, int safe_needed)
     /*   */
 156 {
 157         void *res;
 158 
 159         res = (void *)get_zeroed_page(gfp_mask);
 160         if (safe_needed)
 161                 while (res && swsusp_page_is_free(virt_to_page(res))) {
 162                         /* The page is unsafe, mark it for swsusp_free() */
 163                         swsusp_set_page_forbidden(virt_to_page(res));
 164                         allocated_unsafe_pages++;
 165                         res = (void *)get_zeroed_page(gfp_mask);
 166                 }
 167         if (res) {
 168                 swsusp_set_page_forbidden(virt_to_page(res));
 169                 swsusp_set_page_free(virt_to_page(res));
 170         }
 171         return res;
 172 }

 245 /*
 246  * struct chain_allocator is used for allocating small objects out of
 247  * a linked list of pages called 'the chain'.
 248  *
 249  * The chain grows each time when there is no room for a new object in
 250  * the current page.  The allocated objects cannot be freed individually.
 251  * It is only possible to free them all at once, by freeing the entire
 252  * chain.
 253  *
 254  * NOTE: The chain allocator may be inefficient if the allocated objects
 255  * are not much smaller than PAGE_SIZE.
 256  */
 257 struct chain_allocator {
 258         struct linked_page *chain;      /* the chain */
 259         unsigned int used_space;        /* total size of objects allocated out
 260                                            of the current page */
 261         gfp_t gfp_mask;         /* mask for allocating pages */
 262         int safe_needed;        /* if set, only "safe" pages are allocated */
 263 };

 265 static void chain_init(struct chain_allocator *ca, gfp_t gfp_mask,
     /*   */
 266                        int safe_needed)
 267 {
 268         ca->chain = NULL;
 269         ca->used_space = LINKED_PAGE_DATA_SIZE;
 270         ca->gfp_mask = gfp_mask;
 271         ca->safe_needed = safe_needed;

 274 static void *chain_alloc(struct chain_allocator *ca, unsigned int size)
     /*   */
 275 {
 276         void *ret;
 277 
 278         if (LINKED_PAGE_DATA_SIZE - ca->used_space < size) {
 279                 struct linked_page *lp;
 280 
 281                 lp = ca->safe_needed ? __get_safe_page(ca->gfp_mask) :
 282                                         get_image_page(ca->gfp_mask, PG_ANY);
 283                 if (!lp)
 284                         return NULL;
 285 
 286                 lp->next = ca->chain;
 287                 ca->chain = lp;
 288                 ca->used_space = 0;
 289         }
 290         ret = ca->chain->data + ca->used_space;
 291         ca->used_space += size;
 292         return ret;
 293 }

 295 /**
 296  * Data types related to memory bitmaps.
 297  *
 298  * Memory bitmap is a structure consiting of many linked lists of
 299  * objects.  The main list's elements are of type struct zone_bitmap
 300  * and each of them corresonds to one zone.  For each zone bitmap
 301  * object there is a list of objects of type struct bm_block that
 302  * represent each blocks of bitmap in which information is stored.
 303  *
 304  * struct memory_bitmap contains a pointer to the main list of zone
 305  * bitmap objects, a struct bm_position used for browsing the bitmap,
 306  * and a pointer to the list of pages used for allocating all of the
 307  * zone bitmap objects and bitmap block objects.
 308  *
 309  * NOTE: It has to be possible to lay out the bitmap in memory
 310  * using only allocations of order 0.  Additionally, the bitmap is
 311  * designed to work with arbitrary number of zones (this is over the
 312  * top for now, but let's avoid making unnecessary assumptions ;-).
 313  *
 314  * struct zone_bitmap contains a pointer to a list of bitmap block
 315  * objects and a pointer to the bitmap block object that has been
 316  * most recently used for setting bits.  Additionally, it contains the
 317  * PFNs that correspond to the start and end of the represented zone.
 318  *
 319  * struct bm_block contains a pointer to the memory page in which
 320  * information is stored (in the form of a block of bitmap)
 321  * It also contains the pfns that correspond to the start and end of
 322  * the represented memory area.
 323  *
 324  * The memory bitmap is organized as a radix tree to guarantee fast random
 325  * access to the bits. There is one radix tree for each zone (as returned
 326  * from create_mem_extents).
 327  *
 328  * One radix tree is represented by one struct mem_zone_bm_rtree. There are
 329  * two linked lists for the nodes of the tree, one for the inner nodes and
 330  * one for the leave nodes. The linked leave nodes are used for fast linear
 331  * access of the memory bitmap.
 332  *
 333  * The struct rtree_node represents one node of the radix tree.
 334  */
 335

 420 /**
 421  * add_rtree_block - Add a new leave node to the radix tree.
 422  *
 423  * The leave nodes need to be allocated in order to keep the leaves
 424  * linked list in order. This is guaranteed by the zone->blocks
 425  * counter.
 426  */
 427 static int add_rtree_block(struct mem_zone_bm_rtree *zone, gfp_t gfp_mask,
     /*   */
 428                            int safe_needed, struct chain_allocator *ca)
 429 {
 430         struct rtree_node *node, *block, **dst;
 431         unsigned int levels_needed, block_nr;
 432         int i;
 433 
 434         block_nr = zone->blocks;
 435         levels_needed = 0;
 436 
 437         /* How many levels do we need for this block nr? */
 438         while (block_nr) {
 439                 levels_needed += 1;
 440                 block_nr >>= BM_RTREE_LEVEL_SHIFT;
 441         }
 442 
 443         /* Make sure the rtree has enough levels */
 444         for (i = zone->levels; i < levels_needed; i++) {
 445                 node = alloc_rtree_node(gfp_mask, safe_needed, ca,
 446                                         &zone->nodes);
 447                 if (!node)
 448                         return -ENOMEM;
 449 
 450                 node->data[0] = (unsigned long)zone->rtree;
 451                 zone->rtree = node;
 452                 zone->levels += 1;
 453         }
 454 
 455         /* Allocate new block */
 456         block = alloc_rtree_node(gfp_mask, safe_needed, ca, &zone->leaves);
 457         if (!block)
 458                 return -ENOMEM;
 459 
 460         /* Now walk the rtree to insert the block */
 461         node = zone->rtree;
 462         dst = &zone->rtree;
 463         block_nr = zone->blocks;
 464         for (i = zone->levels; i > 0; i--) {
 465                 int index;
 466 
 467                 if (!node) {
 468                         node = alloc_rtree_node(gfp_mask, safe_needed, ca,
 469                                                 &zone->nodes);
 470                         if (!node)
 471                                 return -ENOMEM;
 472                         *dst = node;
 473                 }
 474 
 475                 index = block_nr >> ((i - 1) * BM_RTREE_LEVEL_SHIFT);
 476                 index &= BM_RTREE_LEVEL_MASK;
 477                 dst = (struct rtree_node **)&((*dst)->data[index]);
 478                 node = *dst;
 479         }
 480 
 481         zone->blocks += 1;
 482         *dst = block;
 483 
 484         return 0;
 485 }

 565 /**
 566  * free_mem_extents - Free a list of memory extents.
 567  * @list: List of extents to free.
 568  */
 569 static void free_mem_extents(struct list_head *list)
     /*   */
 570 {
 571         struct mem_extent *ext, *aux;
 572 
 573         list_for_each_entry_safe(ext, aux, list, hook) {
 574                 list_del(&ext->hook);
 575                 kfree(ext);
 576         }
 577 }

 579 /**
 580  * create_mem_extents - Create a list of memory extents.
 581  * @list: List to put the extents into.
 582  * @gfp_mask: Mask to use for memory allocations.
 583  *
 584  * The extents represent contiguous ranges of PFNs.
 585  */
 586 static int create_mem_extents(struct list_head *list, gfp_t gfp_mask)
     /*   */
 587 {
 588         struct zone *zone;
 589 
 590         INIT_LIST_HEAD(list);
 591 
 592         for_each_populated_zone(zone) {
 593                 unsigned long zone_start, zone_end;
 594                 struct mem_extent *ext, *cur, *aux;
 595 
 596                 zone_start = zone->zone_start_pfn;
 597                 zone_end = zone_end_pfn(zone);
 598 
 599                 list_for_each_entry(ext, list, hook)
 600                         if (zone_start <= ext->end)
 601                                 break;
 602 
 603                 if (&ext->hook == list || zone_end < ext->start) {
 604                         /* New extent is necessary */
 605                         struct mem_extent *new_ext;
 606 
 607                         new_ext = kzalloc(sizeof(struct mem_extent), gfp_mask);
 608                         if (!new_ext) {
 609                                 free_mem_extents(list);
 610                                 return -ENOMEM;
 611                         }
 612                         new_ext->start = zone_start;
 613                         new_ext->end = zone_end;
 614                         list_add_tail(&new_ext->hook, &ext->hook);
 615                         continue;
 616                 }
 617 
 618                 /* Merge this zone's range of PFNs with the existing one */
 619                 if (zone_start < ext->start)
 620                         ext->start = zone_start;
 621                 if (zone_end > ext->end)
 622                         ext->end = zone_end;
 623 
 624                 /* More merging may be possible */
 625                 cur = ext;
 626                 list_for_each_entry_safe_continue(cur, aux, list, hook) {
 627                         if (zone_end < cur->start)
 628                                 break;
 629                         if (zone_end < cur->end)
 630                                 ext->end = cur->end;
 631                         list_del(&cur->hook);
 632                         kfree(cur);
 633                 }
 634         }
 635 
 636         return 0;
 637 }

 639 /**
 640  * memory_bm_create - Allocate memory for a memory bitmap.
 641  */
 642 static int memory_bm_create(struct memory_bitmap *bm, gfp_t gfp_mask,
     /*   */
 643                             int safe_needed)
 644 {
 645         struct chain_allocator ca;
 646         struct list_head mem_extents;
 647         struct mem_extent *ext;
 648         int error;
 649 
 650         chain_init(&ca, gfp_mask, safe_needed);
 651         INIT_LIST_HEAD(&bm->zones);
 652 
 653         error = create_mem_extents(&mem_extents, gfp_mask);
 654         if (error)
 655                 return error;
 656 
 657         list_for_each_entry(ext, &mem_extents, hook) {
 658                 struct mem_zone_bm_rtree *zone;
 659 
 660                 zone = create_zone_bm_rtree(gfp_mask, safe_needed, &ca,
 661                                             ext->start, ext->end);
 662                 if (!zone) {
 663                         error = -ENOMEM;
 664                         goto Error;
 665                 }
 666                 list_add_tail(&zone->list, &bm->zones);
 667         }
 668 
 669         bm->p_list = ca.chain;
 670         memory_bm_position_reset(bm);
 671  Exit:
 672         free_mem_extents(&mem_extents);
 673         return error;
 674 
 675  Error:
 676         bm->p_list = ca.chain;
 677         memory_bm_free(bm, PG_UNSAFE_CLEAR);
 678         goto Exit;
 679 }

 697 /**
 698  * memory_bm_find_bit - Find the bit for a given PFN in a memory bitmap.
 699  *
 700  * Find the bit in memory bitmap @bm that corresponds to the given PFN.
 701  * The cur.zone, cur.block and cur.node_pfn members of @bm are updated.
 702  *
 703  * Walk the radix tree to find the page containing the bit that represents @pfn
 704  * and return the position of the bit in @addr and @bit_nr.
 705  */
 706 static int memory_bm_find_bit(struct memory_bitmap *bm, unsigned long pfn,
     /*   */
 707                               void **addr, unsigned int *bit_nr)
 708 {
 709         struct mem_zone_bm_rtree *curr, *zone;
 710         struct rtree_node *node;
 711         int i, block_nr;
 712 
 713         zone = bm->cur.zone;
 714 
 715         if (pfn >= zone->start_pfn && pfn < zone->end_pfn)
 716                 goto zone_found;
 717 
 718         zone = NULL;
 719 
 720         /* Find the right zone */
 721         list_for_each_entry(curr, &bm->zones, list) {
 722                 if (pfn >= curr->start_pfn && pfn < curr->end_pfn) {
 723                         zone = curr;
 724                         break;
 725                 }
 726         }
 727 
 728         if (!zone)
 729                 return -EFAULT;
 730 
 731 zone_found:
 732         /*
 733          * We have found the zone. Now walk the radix tree to find the leaf node
 734          * for our PFN.
 735          */
 736 
 737         /*
 738          * If the zone we wish to scan is the current zone and the
 739          * pfn falls into the current node then we do not need to walk
 740          * the tree.
 741          */
 742         node = bm->cur.node;
 743         if (zone == bm->cur.zone &&
 744             ((pfn - zone->start_pfn) & ~BM_BLOCK_MASK) == bm->cur.node_pfn)
 745                 goto node_found;
 746 
 747         node      = zone->rtree;
 748         block_nr  = (pfn - zone->start_pfn) >> BM_BLOCK_SHIFT;
 749 
 750         for (i = zone->levels; i > 0; i--) {
 751                 int index;
 752 
 753                 index = block_nr >> ((i - 1) * BM_RTREE_LEVEL_SHIFT);
 754                 index &= BM_RTREE_LEVEL_MASK;
 755                 BUG_ON(node->data[index] == 0);
 756                 node = (struct rtree_node *)node->data[index];
 757         }
 758 
 759 node_found:
 760         /* Update last position */
 761         bm->cur.zone = zone;
 762         bm->cur.node = node;
 763         bm->cur.node_pfn = (pfn - zone->start_pfn) & ~BM_BLOCK_MASK;
 764 
 765         /* Set return values */
 766         *addr = node->data;
 767         *bit_nr = (pfn - zone->start_pfn) & BM_BLOCK_MASK;
 768 
 769         return 0;
 770 }

 834 /*
 835  * rtree_next_node - Jump to the next leaf node.
 836  *
 837  * Set the position to the beginning of the next node in the
 838  * memory bitmap. This is either the next node in the current
 839  * zone's radix tree or the first node in the radix tree of the
 840  * next zone.
 841  *
 842  * Return true if there is a next node, false otherwise.
 843  */
 844 static bool rtree_next_node(struct memory_bitmap *bm)
     /*   */
 845 {
 846         if (!list_is_last(&bm->cur.node->list, &bm->cur.zone->leaves)) {
 847                 bm->cur.node = list_entry(bm->cur.node->list.next,
 848                                           struct rtree_node, list);
 849                 bm->cur.node_pfn += BM_BITS_PER_BLOCK;
 850                 bm->cur.node_bit  = 0;
 851                 touch_softlockup_watchdog();
 852                 return true;
 853         }
 854 
 855         /* No more nodes, goto next zone */
 856         if (!list_is_last(&bm->cur.zone->list, &bm->zones)) {
 857                 bm->cur.zone = list_entry(bm->cur.zone->list.next,
 858                                   struct mem_zone_bm_rtree, list);
 859                 bm->cur.node = list_entry(bm->cur.zone->leaves.next,
 860                                           struct rtree_node, list);
 861                 bm->cur.node_pfn = 0;
 862                 bm->cur.node_bit = 0;
 863                 return true;
 864         }
 865 
 866         /* No more zones */
 867         return false;
 868 }

 870 /**
 871  * memory_bm_rtree_next_pfn - Find the next set bit in a memory bitmap.
 872  * @bm: Memory bitmap.
 873  *
 874  * Starting from the last returned position this function searches for the next
 875  * set bit in @bm and returns the PFN represented by it.  If no more bits are
 876  * set, BM_END_OF_MAP is returned.
 877  *
 878  * It is required to run memory_bm_position_reset() before the first call to
 879  * this function for the given memory bitmap.
 880  */
 881 static unsigned long memory_bm_next_pfn(struct memory_bitmap *bm)
     /*   */
 882 {
 883         unsigned long bits, pfn, pages;
 884         int bit;
 885 
 886         do {
 887                 pages     = bm->cur.zone->end_pfn - bm->cur.zone->start_pfn;
 888                 bits      = min(pages - bm->cur.node_pfn, BM_BITS_PER_BLOCK);
 889                 bit       = find_next_bit(bm->cur.node->data, bits,
 890                                           bm->cur.node_bit);
 891                 if (bit < bits) {
 892                         pfn = bm->cur.zone->start_pfn + bm->cur.node_pfn + bit;
 893                         bm->cur.node_bit = bit + 1;
 894                         return pfn;
 895                 }
 896         } while (rtree_next_node(bm));
 897 
 898         return BM_END_OF_MAP;
 899 }

 941 /**
 942  * register_nosave_region - Register a region of unsaveable memory.
 943  *
 944  * Register a range of page frames the contents of which should not be saved
 945  * during hibernation (to be used in the early initialization code).
 946  */
 947 void __init __register_nosave_region(unsigned long start_pfn,
     /*   */
 948                                      unsigned long end_pfn, int use_kmalloc)
 949 {
 950         struct nosave_region *region;
 951 
 952         if (start_pfn >= end_pfn)
 953                 return;
 954 
 955         if (!list_empty(&nosave_regions)) {
 956                 /* Try to extend the previous region (they should be sorted) */
 957                 region = list_entry(nosave_regions.prev,
 958                                         struct nosave_region, list);
 959                 if (region->end_pfn == start_pfn) {
 960                         region->end_pfn = end_pfn;
 961                         goto Report;
 962                 }
 963         }
 964         if (use_kmalloc) {
 965                 /* During init, this shouldn't fail */
 966                 region = kmalloc(sizeof(struct nosave_region), GFP_KERNEL);
 967                 BUG_ON(!region);
 968         } else {
 969                 /* This allocation cannot fail */
 970                 region = memblock_alloc(sizeof(struct nosave_region),
 971                                         SMP_CACHE_BYTES);
 972                 if (!region)
 973                         panic("%s: Failed to allocate %zu bytes\n", __func__,
 974                               sizeof(struct nosave_region));
 975         }
 976         region->start_pfn = start_pfn;
 977         region->end_pfn = end_pfn;
 978         list_add_tail(&region->list, &nosave_regions);
 979  Report:
 980         pr_info("Registered nosave memory: [mem %#010llx-%#010llx]\n",
 981                 (unsigned long long) start_pfn << PAGE_SHIFT,
 982                 ((unsigned long long) end_pfn << PAGE_SHIFT) - 1);
 983 }

1072  *
1073  * Create bitmaps needed for marking page frames that should not be saved and
1074  * free page frames.  The forbidden_pages_map and free_pages_map pointers are
1075  * only modified if everything goes well, because we don't want the bits to be
1076  * touched before both bitmaps are set up.
1077  */

     /*   */
1079 {
1080         struct memory_bitmap *bm1, *bm2;
1081         int error = 0;
1082 
1083         if (forbidden_pages_map && free_pages_map)
1084                 return 0;
1085         else
1086                 BUG_ON(forbidden_pages_map || free_pages_map);

1090                 return -ENOMEM;

1177 /**
1178  * snapshot_additional_pages - Estimate the number of extra pages needed.
1179  * @zone: Memory zone to carry out the computation for.
1180  *
1181  * Estimate the number of additional pages needed for setting up a hibernation
1182  * image data structures for @zone (usually, the returned value is greater than
1183  * the exact number).
1184  */
1185 unsigned int snapshot_additional_pages(struct zone *zone)
     /*   */
1186 {
1187         unsigned int rtree, nodes;
1188 
1189         rtree = nodes = DIV_ROUND_UP(zone->spanned_pages, BM_BITS_PER_BLOCK);
1190         rtree += DIV_ROUND_UP(rtree * sizeof(struct rtree_node),
1191                               LINKED_PAGE_DATA_SIZE);
1192         while (nodes > 1) {
1193                 nodes = DIV_ROUND_UP(nodes, BM_ENTRIES_PER_LEVEL);
1194                 rtree += nodes;
1195         }
1196 
1197         return 2 * rtree;
1198 }

1354  * safe_copy_page - Copy a page in a safe way.
1355  *
1356  * Check if the page we are going to copy is marked as present in the kernel
1357  * page tables. This always is the case if CONFIG_DEBUG_PAGEALLOC or
1358  * CONFIG_ARCH_HAS_SET_DIRECT_MAP is not set. In that case kernel_page_present()

     /*   */

1367                 do_copy_page(dst, page_address(s_page));

1379 static void copy_data_page(unsigned long dst_pfn, unsigned long src_pfn)
     /*   */
1380 {
1381         struct page *s_page, *d_page;
1382         void *src, *dst;
1383

1385         d_page = pfn_to_page(dst_pfn);

1466 /**
1467  * swsusp_free - Free pages allocated for hibernation image.
1468  *
1469  * Image pages are alocated before snapshot creation, so they need to be
1470  * released after resume.
1471  */
1472 void swsusp_free(void)
     /*   */
1473 {
1474         unsigned long fb_pfn, fr_pfn;
1475 
1476         if (!forbidden_pages_map || !free_pages_map)
1477                 goto out;
1478 
1479         memory_bm_position_reset(forbidden_pages_map);
1480         memory_bm_position_reset(free_pages_map);
1481 
1482 loop:
1483         fr_pfn = memory_bm_next_pfn(free_pages_map);
1484         fb_pfn = memory_bm_next_pfn(forbidden_pages_map);
1485 
1486         /*
1487          * Find the next bit set in both bitmaps. This is guaranteed to
1488          * terminate when fb_pfn == fr_pfn == BM_END_OF_MAP.
1489          */
1490         do {
1491                 if (fb_pfn < fr_pfn)
1492                         fb_pfn = memory_bm_next_pfn(forbidden_pages_map);
1493                 if (fr_pfn < fb_pfn)
1494                         fr_pfn = memory_bm_next_pfn(free_pages_map);
1495         } while (fb_pfn != fr_pfn);
1496 
1497         if (fr_pfn != BM_END_OF_MAP && pfn_valid(fr_pfn)) {
1498                 struct page *page = pfn_to_page(fr_pfn);
1499 
1500                 memory_bm_clear_current(forbidden_pages_map);
1501                 memory_bm_clear_current(free_pages_map);
1502                 hibernate_restore_unprotect_page(page_address(page));
1503                 __free_page(page);
1504                 goto loop;
1505         }
1506 
1507 out:
1508         nr_copy_pages = 0;
1509         nr_meta_pages = 0;
1510         restore_pblist = NULL;
1511         buffer = NULL;
1512         alloc_normal = 0;
1513         alloc_highmem = 0;
1514         hibernate_restore_protection_end();
1515 }

1601 /**
1602  * free_unnecessary_pages - Release preallocated pages not needed for the image.
1603  */
1604 static unsigned long free_unnecessary_pages(void)
     /*   */
1605 {
1606         unsigned long save, to_free_normal, to_free_highmem, free;
1607 
1608         save = count_data_pages();
1609         if (alloc_normal >= save) {
1610                 to_free_normal = alloc_normal - save;
1611                 save = 0;
1612         } else {
1613                 to_free_normal = 0;
1614                 save -= alloc_normal;
1615         }
1616         save += count_highmem_pages();
1617         if (alloc_highmem >= save) {
1618                 to_free_highmem = alloc_highmem - save;
1619         } else {
1620                 to_free_highmem = 0;
1621                 save -= alloc_highmem;
1622                 if (to_free_normal > save)
1623                         to_free_normal -= save;
1624                 else
1625                         to_free_normal = 0;
1626         }
1627         free = to_free_normal + to_free_highmem;
1628 
1629         memory_bm_position_reset(&copy_bm);
1630 
1631         while (to_free_normal > 0 || to_free_highmem > 0) {
1632                 unsigned long pfn = memory_bm_next_pfn(&copy_bm);
1633                 struct page *page = pfn_to_page(pfn);
1634 
1635                 if (PageHighMem(page)) {
1636                         if (!to_free_highmem)
1637                                 continue;
1638                         to_free_highmem--;
1639                         alloc_highmem--;
1640                 } else {
1641                         if (!to_free_normal)
1642                                 continue;
1643                         to_free_normal--;
1644                         alloc_normal--;
1645                 }
1646                 memory_bm_clear_bit(&copy_bm, pfn);
1647                 swsusp_unset_page_forbidden(page);
1648                 swsusp_unset_page_free(page);
1649                 __free_page(page);
1650         }
1651 
1652         return free;
1653 }

1655 /**
1656  * minimum_image_size - Estimate the minimum acceptable size of an image.
1657  * @saveable: Number of saveable pages in the system.
1658  *
1659  * We want to avoid attempting to free too much memory too hard, so estimate the
1660  * minimum acceptable size of a hibernation image to use as the lower limit for
1661  * preallocating memory.
1662  *
1663  * We assume that the minimum image size should be proportional to
1664  *
1665  * [number of saveable pages] - [number of pages that can be freed in theory]
1666  *
1667  * where the second term is the sum of (1) reclaimable slab pages, (2) active
1668  * and (3) inactive anonymous pages, (4) active and (5) inactive file pages.
1669  */
1670 static unsigned long minimum_image_size(unsigned long saveable)
     /*   */
1671 {
1672         unsigned long size;
1673 
1674         size = global_node_page_state_pages(NR_SLAB_RECLAIMABLE_B)
1675                 + global_node_page_state(NR_ACTIVE_ANON)
1676                 + global_node_page_state(NR_INACTIVE_ANON)
1677                 + global_node_page_state(NR_ACTIVE_FILE)
1678                 + global_node_page_state(NR_INACTIVE_FILE);
1679 
1680         return saveable <= size ? 0 : saveable - size;
1681 }

1683 /**
1684  * hibernate_preallocate_memory - Preallocate memory for hibernation image.
1685  *
1686  * To create a hibernation image it is necessary to make a copy of every page
1687  * frame in use.  We also need a number of page frames to be free during
1688  * hibernation for allocations made while saving the image and for device
1689  * drivers, in case they need to allocate memory from their hibernation
1690  * callbacks (these two numbers are given by PAGES_FOR_IO (which is a rough
1691  * estimate) and reserved_size divided by PAGE_SIZE (which is tunable through
1692  * /sys/power/reserved_size, respectively).  To make this happen, we compute the
1693  * total number of available page frames and allocate at least
1694  *
1695  * ([page frames total] + PAGES_FOR_IO + [metadata pages]) / 2
1696  *  + 2 * DIV_ROUND_UP(reserved_size, PAGE_SIZE)
1697  *
1698  * of them, which corresponds to the maximum size of a hibernation image.
1699  *
1700  * If image_size is set below the number following from the above formula,
1701  * the preallocation of memory is continued until the total number of saveable
1702  * pages in the system is below the requested image size or the minimum
1703  * acceptable image size returned by minimum_image_size(), whichever is greater.
1704  */
1705 int hibernate_preallocate_memory(void)
     /*   */
1706 {
1707         struct zone *zone;
1708         unsigned long saveable, size, max_size, count, highmem, pages = 0;
1709         unsigned long alloc, save_highmem, pages_highmem, avail_normal;
1710         ktime_t start, stop;
1711         int error;
1712 
1713         pr_info("Preallocating image memory\n");
1714         start = ktime_get();
1715 
1716         error = memory_bm_create(&orig_bm, GFP_IMAGE, PG_ANY);
1717         if (error) {
1718                 pr_err("Cannot allocate original bitmap\n");
1719                 goto err_out;
1720         }
1721 
1722         error = memory_bm_create(&copy_bm, GFP_IMAGE, PG_ANY);
1723         if (error) {
1724                 pr_err("Cannot allocate copy bitmap\n");
1725                 goto err_out;
1726         }
1727 
1728         alloc_normal = 0;
1729         alloc_highmem = 0;
1730 
1731         /* Count the number of saveable data pages. */
1732         save_highmem = count_highmem_pages();
1733         saveable = count_data_pages();
1734 
1735         /*
1736          * Compute the total number of page frames we can use (count) and the
1737          * number of pages needed for image metadata (size).
1738          */
1739         count = saveable;
1740         saveable += save_highmem;
1741         highmem = save_highmem;
1742         size = 0;
1743         for_each_populated_zone(zone) {
1744                 size += snapshot_additional_pages(zone);
1745                 if (is_highmem(zone))
1746                         highmem += zone_page_state(zone, NR_FREE_PAGES);
1747                 else
1748                         count += zone_page_state(zone, NR_FREE_PAGES);
1749         }
1750         avail_normal = count;
1751         count += highmem;
1752         count -= totalreserve_pages;
1753 
1754         /* Compute the maximum number of saveable pages to leave in memory. */
1755         max_size = (count - (size + PAGES_FOR_IO)) / 2
1756                         - 2 * DIV_ROUND_UP(reserved_size, PAGE_SIZE);
1757         /* Compute the desired number of image pages specified by image_size. */
1758         size = DIV_ROUND_UP(image_size, PAGE_SIZE);
1759         if (size > max_size)
1760                 size = max_size;
1761         /*
1762          * If the desired number of image pages is at least as large as the
1763          * current number of saveable pages in memory, allocate page frames for
1764          * the image and we're done.
1765          */
1766         if (size >= saveable) {
1767                 pages = preallocate_image_highmem(save_highmem);
1768                 pages += preallocate_image_memory(saveable - pages, avail_normal);
1769                 goto out;
1770         }
1771 
1772         /* Estimate the minimum size of the image. */
1773         pages = minimum_image_size(saveable);
1774         /*
1775          * To avoid excessive pressure on the normal zone, leave room in it to
1776          * accommodate an image of the minimum size (unless it's already too
1777          * small, in which case don't preallocate pages from it at all).
1778          */
1779         if (avail_normal > pages)
1780                 avail_normal -= pages;
1781         else
1782                 avail_normal = 0;
1783         if (size < pages)
1784                 size = min_t(unsigned long, pages, max_size);
1785 
1786         /*
1787          * Let the memory management subsystem know that we're going to need a
1788          * large number of page frames to allocate and make it free some memory.
1789          * NOTE: If this is not done, performance will be hurt badly in some
1790          * test cases.
1791          */
1792         shrink_all_memory(saveable - size);
1793 
1794         /*
1795          * The number of saveable pages in memory was too high, so apply some
1796          * pressure to decrease it.  First, make room for the largest possible
1797          * image and fail if that doesn't work.  Next, try to decrease the size
1798          * of the image as much as indicated by 'size' using allocations from
1799          * highmem and non-highmem zones separately.
1800          */
1801         pages_highmem = preallocate_image_highmem(highmem / 2);
1802         alloc = count - max_size;
1803         if (alloc > pages_highmem)
1804                 alloc -= pages_highmem;
1805         else
1806                 alloc = 0;
1807         pages = preallocate_image_memory(alloc, avail_normal);
1808         if (pages < alloc) {
1809                 /* We have exhausted non-highmem pages, try highmem. */
1810                 alloc -= pages;
1811                 pages += pages_highmem;
1812                 pages_highmem = preallocate_image_highmem(alloc);
1813                 if (pages_highmem < alloc) {
1814                         pr_err("Image allocation is %lu pages short\n",
1815                                 alloc - pages_highmem);
1816                         goto err_out;
1817                 }
1818                 pages += pages_highmem;
1819                 /*
1820                  * size is the desired number of saveable pages to leave in
1821                  * memory, so try to preallocate (all memory - size) pages.
1822                  */
1823                 alloc = (count - pages) - size;
1824                 pages += preallocate_image_highmem(alloc);
1825         } else {
1826                 /*
1827                  * There are approximately max_size saveable pages at this point
1828                  * and we want to reduce this number down to size.
1829                  */
1830                 alloc = max_size - size;
1831                 size = preallocate_highmem_fraction(alloc, highmem, count);
1832                 pages_highmem += size;
1833                 alloc -= size;
1834                 size = preallocate_image_memory(alloc, avail_normal);
1835                 pages_highmem += preallocate_image_highmem(alloc - size);
1836                 pages += pages_highmem + size;
1837         }
1838 
1839         /*
1840          * We only need as many page frames for the image as there are saveable
1841          * pages in memory, but we have allocated more.  Release the excessive
1842          * ones now.
1843          */
1844         pages -= free_unnecessary_pages();
1845 
1846  out:
1847         stop = ktime_get();
1848         pr_info("Allocated %lu pages for snapshot\n", pages);
1849         swsusp_show_speed(start, stop, pages, "Allocated");
1850 
1851         return 0;

1942 /**
1943  * swsusp_alloc - Allocate memory for hibernation image.
1944  *
1945  * We first try to allocate as many highmem pages as there are
1946  * saveable highmem pages in the system.  If that fails, we allocate
1947  * non-highmem pages for the copies of the remaining highmem ones.
1948  *
1949  * In this approach it is likely that the copies of highmem pages will
1950  * also be located in the high memory, because of the way in which
1951  * copy_data_pages() works.
1952  */
1953 static int swsusp_alloc(struct memory_bitmap *copy_bm,
     /*   */
1954                         unsigned int nr_pages, unsigned int nr_highmem)
1955 {
1956         if (nr_highmem > 0) {
1957                 if (get_highmem_buffer(PG_ANY))
1958                         goto err_out;
1959                 if (nr_highmem > alloc_highmem) {
1960                         nr_highmem -= alloc_highmem;
1961                         nr_pages += alloc_highmem_pages(copy_bm, nr_highmem);
1962                 }
1963         }
1964         if (nr_pages > alloc_normal) {
1965                 nr_pages -= alloc_normal;
1966                 while (nr_pages-- > 0) {
1967                         struct page *page;
1968 
1969                         page = alloc_image_page(GFP_ATOMIC);
1970                         if (!page)
1971                                 goto err_out;
1972                         memory_bm_set_bit(copy_bm, page_to_pfn(page));
1973                 }
1974         }
1975 
1976         return 0;
1977 
1978  err_out:
1979         swsusp_free();
1980         return -ENOMEM;
1981 }

1983 asmlinkage __visible int swsusp_save(void)
     /*   */
1984 {
1985         unsigned int nr_pages, nr_highmem;
1986 
1987         pr_info("Creating image:\n");
1988 
1989         drain_local_pages(NULL);
1990         nr_pages = count_data_pages();
1991         nr_highmem = count_highmem_pages();
1992         pr_info("Need to copy %u pages\n", nr_pages + nr_highmem);
1993 
1994         if (!enough_free_mem(nr_pages, nr_highmem)) {
1995                 pr_err("Not enough free memory\n");
1996                 return -ENOMEM;
1997         }
1998 
1999         if (swsusp_alloc(&copy_bm, nr_pages, nr_highmem)) {
2000                 pr_err("Memory allocation failed\n");
2001                 return -ENOMEM;
2002         }
2003 
2004         /*
2005          * During allocating of suspend pagedir, new cold pages may appear.
2006          * Kill them.
2007          */
2008         drain_local_pages(NULL);
2009         copy_data_pages(&copy_bm, &orig_bm);
2010 
2011         /*
2012          * End of critical section. From now on, we can write to memory,
2013          * but we should not touch disk. This specially means we must _not_
2014          * touch swap space! Except we must write out our image of course.
2015          */
2016 
2017         nr_pages += nr_highmem;
2018         nr_copy_pages = nr_pages;
2019         nr_meta_pages = DIV_ROUND_UP(nr_pages * sizeof(long), PAGE_SIZE);
2020 
2021         pr_info("Image created (%d pages copied)\n", nr_pages);
2022 
2023         return 0;
2024 }

2085 /**
2086  * snapshot_read_next - Get the address to read the next image page from.
2087  * @handle: Snapshot handle to be used for the reading.
2088  *
2089  * On the first call, @handle should point to a zeroed snapshot_handle
2090  * structure.  The structure gets populated then and a pointer to it should be
2091  * passed to this function every next time.
2092  *
2093  * On success, the function returns a positive number.  Then, the caller
2094  * is allowed to read up to the returned number of bytes from the memory
2095  * location computed by the data_of() macro.
2096  *
2097  * The function returns 0 to indicate the end of the data stream condition,
2098  * and negative numbers are returned on errors.  If that happens, the structure
2099  * pointed to by @handle is not updated and should not be used any more.
2100  */
2101 int snapshot_read_next(struct snapshot_handle *handle)
     /*   */
2102 {
2103         if (handle->cur > nr_meta_pages + nr_copy_pages)
2104                 return 0;
2105 
2106         if (!buffer) {
2107                 /* This makes the buffer be freed by swsusp_free() */
2108                 buffer = get_image_page(GFP_ATOMIC, PG_ANY);
2109                 if (!buffer)
2110                         return -ENOMEM;
2111         }
2112         if (!handle->cur) {
2113                 int error;
2114 
2115                 error = init_header((struct swsusp_info *)buffer);
2116                 if (error)
2117                         return error;
2118                 handle->buffer = buffer;
2119                 memory_bm_position_reset(&orig_bm);
2120                 memory_bm_position_reset(&copy_bm);
2121         } else if (handle->cur <= nr_meta_pages) {
2122                 clear_page(buffer);
2123                 pack_pfns(buffer, &orig_bm);
2124         } else {
2125                 struct page *page;
2126 
2127                 page = pfn_to_page(memory_bm_next_pfn(&copy_bm));
2128                 if (PageHighMem(page)) {
2129                         /*
2130                          * Highmem pages are copied to the buffer,
2131                          * because we can't return with a kmapped
2132                          * highmem page (we may not be called again).
2133                          */

2135 
2136                         kaddr = kmap_atomic(page);
2137                         copy_page(buffer, kaddr);
2138                         kunmap_atomic(kaddr);
2139                         handle->buffer = buffer;
2140                 } else {
2141                         handle->buffer = page_address(page);
2142                 }
2143         }
2144         handle->cur++;
2145         return PAGE_SIZE;
2146 }

2483         memory_bm_free(bm, PG_UNSAFE_KEEP);
2484         if (nr_highmem > 0) {
2485                 error = prepare_highmem_image(bm, &nr_highmem);
2486                 if (error)
2487                         goto Free;
2488         }
2489         /*
2490          * Reserve some safe pages for potential later use.
2491          *
2492          * NOTE: This way we make sure there will be enough safe pages for the
2493          * chain_alloc() in get_buffer().  It is a bit wasteful, but
2494          * nr_copy_pages cannot be greater than 50% of the memory anyway.
2495          *
2496          * nr_copy_pages cannot be less than allocated_unsafe_pages too.
2497          */
2498         nr_pages = nr_copy_pages - nr_highmem - allocated_unsafe_pages;
2499         nr_pages = DIV_ROUND_UP(nr_pages, PBES_PER_LINKED_PAGE);
2500         while (nr_pages > 0) {
2501                 lp = get_image_page(GFP_ATOMIC, PG_SAFE);
2502                 if (!lp) {
2503                         error = -ENOMEM;
2504                         goto Free;
2505                 }
2506                 lp->next = safe_pages_list;
2507                 safe_pages_list = lp;
2508                 nr_pages--;
2509         }
2510         /* Preallocate memory for the image */
2511         nr_pages = nr_copy_pages - nr_highmem - allocated_unsafe_pages;
2512         while (nr_pages > 0) {
2513                 lp = (struct linked_page *)get_zeroed_page(GFP_ATOMIC);
2514                 if (!lp) {
2515                         error = -ENOMEM;
2516                         goto Free;
2517                 }
2518                 if (!swsusp_page_is_free(virt_to_page(lp))) {
2519                         /* The page is "safe", add it to the list */
2520                         lp->next = safe_pages_list;
2521                         safe_pages_list = lp;
2522                 }
2523                 /* Mark the page as allocated */
2524                 swsusp_set_page_forbidden(virt_to_page(lp));
2525                 swsusp_set_page_free(virt_to_page(lp));
2526                 nr_pages--;
2527         }
2528         return 0;
2529 
2530  Free:
2531         swsusp_free();
2532         return error;
2533 }

2535 /**
2536  * get_buffer - Get the address to store the next image data page.
2537  *
2538  * Get the address that snapshot_write_next() should return to its caller to
2539  * write to.
2540  */
2541 static void *get_buffer(struct memory_bitmap *bm, struct chain_allocator *ca)
     /*   */
2542 {
2543         struct pbe *pbe;
2544         struct page *page;
2545         unsigned long pfn = memory_bm_next_pfn(bm);
2546 
2547         if (pfn == BM_END_OF_MAP)
2548                 return ERR_PTR(-EFAULT);
2549 
2550         page = pfn_to_page(pfn);
2551         if (PageHighMem(page))
2552                 return get_highmem_page_buffer(page, ca);
2553 
2554         if (swsusp_page_is_forbidden(page) && swsusp_page_is_free(page))
2555                 /*
2556                  * We have allocated the "original" page frame and we can
2557                  * use it directly to store the loaded page.
2558                  */
2559                 return page_address(page);
2560 
2561         /*
2562          * The "original" page frame has not been allocated and we have to
2563          * use a "safe" page frame to store the loaded page.
2564          */
2565         pbe = chain_alloc(ca, sizeof(struct pbe));
2566         if (!pbe) {
2567                 swsusp_free();
2568                 return ERR_PTR(-ENOMEM);
2569         }
2570         pbe->orig_address = page_address(page);
2571         pbe->address = safe_pages_list;
2572         safe_pages_list = safe_pages_list->next;
2573         pbe->next = restore_pblist;
2574         restore_pblist = pbe;
2575         return pbe->address;
2576 }

2578 /**
2579  * snapshot_write_next - Get the address to store the next image page.
2580  * @handle: Snapshot handle structure to guide the writing.
2581  *
2582  * On the first call, @handle should point to a zeroed snapshot_handle
2583  * structure.  The structure gets populated then and a pointer to it should be
2584  * passed to this function every next time.
2585  *
2586  * On success, the function returns a positive number.  Then, the caller
2587  * is allowed to write up to the returned number of bytes to the memory
2588  * location computed by the data_of() macro.
2589  *
2590  * The function returns 0 to indicate the "end of file" condition.  Negative
2591  * numbers are returned on errors, in which cases the structure pointed to by
2592  * @handle is not updated and should not be used any more.
2593  */
2594 int snapshot_write_next(struct snapshot_handle *handle)
     /*   */
2595 {
2596         static struct chain_allocator ca;
2597         int error = 0;
2598 
2599         /* Check if we have already loaded the entire image */
2600         if (handle->cur > 1 && handle->cur > nr_meta_pages + nr_copy_pages)
2601                 return 0;
2602 
2603         handle->sync_read = 1;
2604 
2605         if (!handle->cur) {
2606                 if (!buffer)
2607                         /* This makes the buffer be freed by swsusp_free() */
2608                         buffer = get_image_page(GFP_ATOMIC, PG_ANY);
2609 
2610                 if (!buffer)
2611                         return -ENOMEM;
2612 
2613                 handle->buffer = buffer;
2614         } else if (handle->cur == 1) {
2615                 error = load_header(buffer);
2616                 if (error)
2617                         return error;
2618 
2619                 safe_pages_list = NULL;
2620 
2621                 error = memory_bm_create(&copy_bm, GFP_ATOMIC, PG_ANY);
2622                 if (error)
2623                         return error;
2624 
2625                 hibernate_restore_protection_begin();
2626         } else if (handle->cur <= nr_meta_pages + 1) {
2627                 error = unpack_orig_pfns(buffer, &copy_bm);
2628                 if (error)
2629                         return error;
2630 
2631                 if (handle->cur == nr_meta_pages + 1) {
2632                         error = prepare_image(&orig_bm, &copy_bm);
2633                         if (error)
2634                                 return error;
2635 
2636                         chain_init(&ca, GFP_ATOMIC, PG_SAFE);
2637                         memory_bm_position_reset(&orig_bm);
2638                         restore_pblist = NULL;
2639                         handle->buffer = get_buffer(&orig_bm, &ca);
2640                         handle->sync_read = 0;
2641                         if (IS_ERR(handle->buffer))
2642                                 return PTR_ERR(handle->buffer);
2643                 }
2644         } else {
2645                 copy_last_highmem_page();
2646                 hibernate_restore_protect_page(handle->buffer);
2647                 handle->buffer = get_buffer(&orig_bm, &ca);
2648                 if (IS_ERR(handle->buffer))
2649                         return PTR_ERR(handle->buffer);
2650                 if (handle->buffer != buffer)
2651                         handle->sync_read = 0;
2652         }
2653         handle->cur++;
2654         return PAGE_SIZE;
2655 }

2657 /**
2658  * snapshot_write_finalize - Complete the loading of a hibernation image.
2659  *
2660  * Must be called after the last call to snapshot_write_next() in case the last
2661  * page in the image happens to be a highmem page and its contents should be
2662  * stored in highmem.  Additionally, it recycles bitmap memory that's not
2663  * necessary any more.
2664  */
2665 void snapshot_write_finalize(struct snapshot_handle *handle)
     /*   */
2666 {
2667         copy_last_highmem_page();
2668         hibernate_restore_protect_page(handle->buffer);
2669         /* Do that only if we have loaded the image entirely */
2670         if (handle->cur > 1 && handle->cur > nr_meta_pages + nr_copy_pages) {
2671                 memory_bm_recycle(&orig_bm);
2672                 free_highmem_data();
2673         }
2674 }

2698 /**
2699  * restore_highmem - Put highmem image pages into their original locations.
2700  *
2701  * For each highmem page that was in use before hibernation and is included in
2702  * the image, and also has been allocated by the "restore" kernel, swap its
2703  * current contents with the previous (ie. "before hibernation") ones.
2704  *
2705  * If the restore eventually fails, we can call this function once again and
2706  * restore the highmem state as seen by the restore kernel.
2707  */
2708 int restore_highmem(void)
     /*   */
2709 {
2710         struct highmem_pbe *pbe = highmem_pblist;
2711         void *buf;
2712 
2713         if (!pbe)
2714                 return 0;
2715 
2716         buf = get_image_page(GFP_ATOMIC, PG_SAFE);
2717         if (!buf)
2718                 return -ENOMEM;
2719 
2720         while (pbe) {
2721                 swap_two_pages_data(pbe->copy_page, pbe->orig_page, buf);
2722                 pbe = pbe->next;
2723         }
2724         free_image_page(buf, PG_UNSAFE_CLEAR);
2725         return 0;
