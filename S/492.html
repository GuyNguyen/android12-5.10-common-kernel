 155 /*
 156  * Returns true if current's euid is same as p's uid or euid,
 157  * or has CAP_SYS_NICE to p's user_ns.
 158  *
 159  * Called with rcu_read_lock, creds are safe
 160  */
 161 static bool set_one_prio_perm(struct task_struct *p)
 162 {
 163         const struct cred *cred = current_cred(), *pcred = __task_cred(p);
 164 
 165         if (uid_eq(pcred->uid,  cred->euid) ||
 166             uid_eq(pcred->euid, cred->euid))
 167                 return true;
 168         if (ns_capable(pcred->user_ns, CAP_SYS_NICE))
 169                 return true;
 170         return false;
 171 }

 173 /*
 174  * set the priority of a task
 175  * - the caller must hold the RCU read lock
 176  */
 177 static int set_one_prio(struct task_struct *p, int niceval, int error)
 178 {
 179         int no_nice;
 180 
 181         if (!set_one_prio_perm(p)) {
 182                 error = -EPERM;
 183                 goto out;
 184         }
 185         if (niceval < task_nice(p) && !can_nice(p, niceval)) {
 186                 error = -EACCES;
 187                 goto out;
 188         }
 189         no_nice = security_task_setnice(p, niceval);
 190         if (no_nice) {
 191                 error = no_nice;
 192                 goto out;
 193         }
 194         if (error == -ESRCH)
 195                 error = 0;
 196         set_user_nice(p, niceval);
 197 out:
 198         return error;
 199 }

 201 SYSCALL_DEFINE3(setpriority, int, which, int, who, int, niceval)
 202 {
 203         struct task_struct *g, *p;
 204         struct user_struct *user;
 205         const struct cred *cred = current_cred();
 206         int error = -EINVAL;
 207         struct pid *pgrp;
 208         kuid_t uid;
 209 
 210         if (which > PRIO_USER || which < PRIO_PROCESS)
 211                 goto out;
 212 
 213         /* normalize: avoid signed division (rounding problems) */
 214         error = -ESRCH;
 215         if (niceval < MIN_NICE)
 216                 niceval = MIN_NICE;
 217         if (niceval > MAX_NICE)
 218                 niceval = MAX_NICE;
 219 
 220         rcu_read_lock();
 221         read_lock(&tasklist_lock);
 222         switch (which) {
 223         case PRIO_PROCESS:
 224                 if (who)
 225                         p = find_task_by_vpid(who);
 226                 else
 227                         p = current;
 228                 if (p)
 229                         error = set_one_prio(p, niceval, error);
 230                 break;
 231         case PRIO_PGRP:
 232                 if (who)
 233                         pgrp = find_vpid(who);
 234                 else
 235                         pgrp = task_pgrp(current);
 236                 do_each_pid_thread(pgrp, PIDTYPE_PGID, p) {
 237                         error = set_one_prio(p, niceval, error);
 238                 } while_each_pid_thread(pgrp, PIDTYPE_PGID, p);
 239                 break;
 240         case PRIO_USER:
 241                 uid = make_kuid(cred->user_ns, who);
 242                 user = cred->user;
 243                 if (!who)
 244                         uid = cred->uid;
 245                 else if (!uid_eq(uid, cred->uid)) {
 246                         user = find_user(uid);
 247                         if (!user)
 248                                 goto out_unlock;        /* No processes for this user */
 249                 }
 250                 do_each_thread(g, p) {
 251                         if (uid_eq(task_uid(p), uid) && task_pid_vnr(p))
 252                                 error = set_one_prio(p, niceval, error);
 253                 } while_each_thread(g, p);
 254                 if (!uid_eq(uid, cred->uid))
 255                         free_uid(user);         /* For find_user() */
 256                 break;
 257         }
 258 out_unlock:
 259         read_unlock(&tasklist_lock);
 260         rcu_read_unlock();
 261 out:
 262         return error;
 263 }

 336 /*
 337  * Unprivileged users may change the real gid to the effective gid
 338  * or vice versa.  (BSD-style)
 339  *
 340  * If you set the real gid at all, or set the effective gid to a value not
 341  * equal to the real gid, then the saved gid is set to the new effective gid.
 342  *
 343  * This makes it possible for a setgid program to completely drop its
 344  * privileges, which is often a useful assertion to make when you are doing
 345  * a security audit over a program.
 346  *
 347  * The general idea is that a program which uses just setregid() will be
 348  * 100% compatible with BSD.  A program which uses just setgid() will be
 349  * 100% compatible with POSIX with saved IDs.
 350  *
 351  * SMP: There are not races, the GIDs are checked only by filesystem
 352  *      operations (as far as semantic preservation is concerned).
 353  */
 354 #ifdef CONFIG_MULTIUSER
 355 long __sys_setregid(gid_t rgid, gid_t egid)
 356 {
 357         struct user_namespace *ns = current_user_ns();
 358         const struct cred *old;
 359         struct cred *new;
 360         int retval;
 361         kgid_t krgid, kegid;
 362 
 363         krgid = make_kgid(ns, rgid);
 364         kegid = make_kgid(ns, egid);
 365 
 366         if ((rgid != (gid_t) -1) && !gid_valid(krgid))
 367                 return -EINVAL;
 368         if ((egid != (gid_t) -1) && !gid_valid(kegid))
 369                 return -EINVAL;
 370 
 371         new = prepare_creds();
 372         if (!new)
 373                 return -ENOMEM;
 374         old = current_cred();
 375 
 376         retval = -EPERM;
 377         if (rgid != (gid_t) -1) {
 378                 if (gid_eq(old->gid, krgid) ||
 379                     gid_eq(old->egid, krgid) ||
 380                     ns_capable_setid(old->user_ns, CAP_SETGID))
 381                         new->gid = krgid;
 382                 else
 383                         goto error;
 384         }
 385         if (egid != (gid_t) -1) {
 386                 if (gid_eq(old->gid, kegid) ||
 387                     gid_eq(old->egid, kegid) ||
 388                     gid_eq(old->sgid, kegid) ||
 389                     ns_capable_setid(old->user_ns, CAP_SETGID))
 390                         new->egid = kegid;
 391                 else
 392                         goto error;
 393         }
 394 
 395         if (rgid != (gid_t) -1 ||
 396             (egid != (gid_t) -1 && !gid_eq(kegid, old->gid)))
 397                 new->sgid = new->egid;
 398         new->fsgid = new->egid;
 399 
 400         retval = security_task_fix_setgid(new, old, LSM_SETID_RE);
 401         if (retval < 0)
 402                 goto error;
 403 
 404         return commit_creds(new);
 405 
 406 error:
 407         abort_creds(new);
 408         return retval;
 409 }

 462 /*
 463  * change the user struct in a credentials set to match the new UID
 464  */
 465 static int set_user(struct cred *new)
 466 {
 467         struct user_struct *new_user;
 468 
 469         new_user = alloc_uid(new->uid);
 470         if (!new_user)
 471                 return -EAGAIN;
 472 
 473         /*
 474          * We don't fail in case of NPROC limit excess here because too many
 475          * poorly written programs don't check set*uid() return code, assuming
 476          * it never fails if called by root.  We may still enforce NPROC limit
 477          * for programs doing set*uid()+execve() by harmlessly deferring the
 478          * failure to the execve() stage.
 479          */
 480         if (atomic_read(&new_user->processes) >= rlimit(RLIMIT_NPROC) &&
 481                         new_user != INIT_USER)
 482                 current->flags |= PF_NPROC_EXCEEDED;
 483         else
 484                 current->flags &= ~PF_NPROC_EXCEEDED;
 485 
 486         free_uid(new->user);
 487         new->user = new_user;
 488         return 0;
 489 }

 571 /*
 572  * setuid() is implemented like SysV with SAVED_IDS
 573  *
 574  * Note that SAVED_ID's is deficient in that a setuid root program
 575  * like sendmail, for example, cannot set its uid to be a normal
 576  * user and then switch back, because if you're root, setuid() sets
 577  * the saved uid too.  If you don't like this, blame the bright people
 578  * in the POSIX committee and/or USG.  Note that the BSD-style setreuid()
 579  * will allow a root program to temporarily drop privileges and be able to
 580  * regain them by swapping the real and effective uid.
 581  */
 582 long __sys_setuid(uid_t uid)
 583 {
 584         struct user_namespace *ns = current_user_ns();
 585         const struct cred *old;
 586         struct cred *new;
 587         int retval;
 588         kuid_t kuid;
 589 
 590         kuid = make_kuid(ns, uid);
 591         if (!uid_valid(kuid))
 592                 return -EINVAL;
 593 
 594         new = prepare_creds();
 595         if (!new)
 596                 return -ENOMEM;
 597         old = current_cred();
 598 
 599         retval = -EPERM;

1026         struct task_struct *p;
1027         struct task_struct *group_leader = current->group_leader;
1028         struct pid *pgrp;
1029         int err;
1030 
1031         if (!pid)
1032                 pid = task_pid_vnr(group_leader);
1033         if (!pgid)
1034                 pgid = pid;
1035         if (pgid < 0)
1036                 return -EINVAL;
1037         rcu_read_lock();
1038 
1039         /* From this point forward we keep holding onto the tasklist lock
1040          * so that our parent does not change from under us. -DaveM
1041          */
1042         write_lock_irq(&tasklist_lock);
1043 
1044         err = -ESRCH;
1045         p = find_task_by_vpid(pid);
1046         if (!p)
1047                 goto out;

1051                 goto out;

1054                 err = -EPERM;
1055                 if (task_session(p) != task_session(group_leader))
1056                         goto out;
1057                 err = -EACCES;
1058                 if (!(p->flags & PF_FORKNOEXEC))
1059                         goto out;
1060         } else {
1061                 err = -ESRCH;
1062                 if (p != group_leader)
1063                         goto out;
1064         }
1065 
1066         err = -EPERM;
1067         if (p->signal->leader)
1068                 goto out;

1073

1546 /* make sure you are allowed to change @tsk limits before calling this */
1547 int do_prlimit(struct task_struct *tsk, unsigned int resource,
1548                 struct rlimit *new_rlim, struct rlimit *old_rlim)
1549 {
1550         struct rlimit *rlim;
1551         int retval = 0;
1552 
1553         if (resource >= RLIM_NLIMITS)
1554                 return -EINVAL;
1555         if (new_rlim) {
1556                 if (new_rlim->rlim_cur > new_rlim->rlim_max)
1557                         return -EINVAL;
1558                 if (resource == RLIMIT_NOFILE &&
1559                                 new_rlim->rlim_max > sysctl_nr_open)
1560                         return -EPERM;
1561         }
1562 
1563         /* protect tsk->signal and tsk->sighand from disappearing */
1564         read_lock(&tasklist_lock);
1565         if (!tsk->sighand) {
1566                 retval = -ESRCH;
1567                 goto out;
1568         }
1569 
1570         rlim = tsk->signal->rlim + resource;
1571         task_lock(tsk->group_leader);
1572         if (new_rlim) {
1573                 /* Keep the capable check against init_user_ns until
1574                    cgroups can contain all limits */
1575                 if (new_rlim->rlim_max > rlim->rlim_max &&
1576                                 !capable(CAP_SYS_RESOURCE))
1577                         retval = -EPERM;
1578                 if (!retval)
1579                         retval = security_task_setrlimit(tsk, resource, new_rlim);
1580         }
1581         if (!retval) {
1582                 if (old_rlim)
1583                         *old_rlim = *rlim;
1584                 if (new_rlim)
1585                         *rlim = *new_rlim;
1586         }
1587         task_unlock(tsk->group_leader);
1588 
1589         /*
1590          * RLIMIT_CPU handling. Arm the posix CPU timer if the limit is not
1591          * infite. In case of RLIM_INFINITY the posix CPU timer code
1592          * ignores the rlimit.
1593          */
1594          if (!retval && new_rlim && resource == RLIMIT_CPU &&
1595              new_rlim->rlim_cur != RLIM_INFINITY &&
1596              IS_ENABLED(CONFIG_POSIX_TIMERS))
1597                 update_rlimit_cpu(tsk, new_rlim->rlim_cur);
1598 out:
1599         read_unlock(&tasklist_lock);
1600         return retval;
1601 }

1682 /*
1683  * It would make sense to put struct rusage in the task_struct,
1684  * except that would make the task_struct be *really big*.  After
1685  * task_struct gets moved into malloc'ed memory, it would
1686  * make sense to do this.  It will make moving the rest of the information
1687  * a lot simpler!  (Which we're not doing right now because we're not
1688  * measuring them yet).
1689  *
1690  * When sampling multiple threads for RUSAGE_SELF, under SMP we might have
1691  * races with threads incrementing their own counters.  But since word
1692  * reads are atomic, we either get new values or old values and we don't
1693  * care which for the sums.  We always take the siglock to protect reading
1694  * the c* fields from p->signal from races with exit.c updating those
1695  * fields when reaping, so a sample either gets all the additions of a
1696  * given child after it's reaped, or none so this sample is before reaping.
1697  *
1698  * Locking:
1699  * We need to take the siglock for CHILDEREN, SELF and BOTH
1700  * for  the cases current multithreaded, non-current single threaded
1701  * non-current multithreaded.  Thread traversal is now safe with
1702  * the siglock held.
1703  * Strictly speaking, we donot need to take the siglock if we are current and
1704  * single threaded,  as no one else can take our signal_struct away, no one
1705  * else can  reap the  children to update signal->c* counters, and no one else
1706  * can race with the signal-> fields. If we do not take any lock, the
1707  * signal-> fields could be read out of order while another thread was just
1708  * exiting. So we should  place a read memory barrier when we avoid the lock.
1709  * On the writer side,  write memory barrier is implied in  __exit_signal
1710  * as __exit_signal releases  the siglock spinlock after updating the signal->
1711  * fields. But we don't do this yet to keep things simple.
1712  *
1713  */
1714 
1715 static void accumulate_thread_rusage(struct task_struct *t, struct rusage *r)
1716 {
1717         r->ru_nvcsw += t->nvcsw;
1718         r->ru_nivcsw += t->nivcsw;
1719         r->ru_minflt += t->min_flt;
1720         r->ru_majflt += t->maj_flt;
1721         r->ru_inblock += task_io_get_inblock(t);
1722         r->ru_oublock += task_io_get_oublock(t);
1723 }

1725 void getrusage(struct task_struct *p, int who, struct rusage *r)
1726 {
1727         struct task_struct *t;
1728         unsigned long flags;
1729         u64 tgutime, tgstime, utime, stime;
1730         unsigned long maxrss = 0;
1731 
1732         memset((char *)r, 0, sizeof (*r));
1733         utime = stime = 0;
1734 
1735         if (who == RUSAGE_THREAD) {
1736                 task_cputime_adjusted(current, &utime, &stime);
1737                 accumulate_thread_rusage(p, r);
1738                 maxrss = p->signal->maxrss;
1739                 goto out;
1740         }
1741 
1742         if (!lock_task_sighand(p, &flags))
1743                 return;
1744 
1745         switch (who) {
1746         case RUSAGE_BOTH:
1747         case RUSAGE_CHILDREN:
1748                 utime = p->signal->cutime;
1749                 stime = p->signal->cstime;
1750                 r->ru_nvcsw = p->signal->cnvcsw;
1751                 r->ru_nivcsw = p->signal->cnivcsw;
1752                 r->ru_minflt = p->signal->cmin_flt;
1753                 r->ru_majflt = p->signal->cmaj_flt;
1754                 r->ru_inblock = p->signal->cinblock;
1755                 r->ru_oublock = p->signal->coublock;
1756                 maxrss = p->signal->cmaxrss;
1757 
1758                 if (who == RUSAGE_CHILDREN)
1759                         break;
1760                 fallthrough;
1761 
1762         case RUSAGE_SELF:
1763                 thread_group_cputime_adjusted(p, &tgutime, &tgstime);
1764                 utime += tgutime;
1765                 stime += tgstime;
1766                 r->ru_nvcsw += p->signal->nvcsw;
1767                 r->ru_nivcsw += p->signal->nivcsw;
1768                 r->ru_minflt += p->signal->min_flt;
1769                 r->ru_majflt += p->signal->maj_flt;
1770                 r->ru_inblock += p->signal->inblock;
1771                 r->ru_oublock += p->signal->oublock;
1772                 if (maxrss < p->signal->maxrss)
1773                         maxrss = p->signal->maxrss;
1774                 t = p;
1775                 do {
1776                         accumulate_thread_rusage(t, r);
1777                 } while_each_thread(p, t);
1778                 break;
1779 
1780         default:
1781                 BUG();
1782         }
1783         unlock_task_sighand(p, &flags);
1784 
1785 out:
1786         r->ru_utime = ns_to_kernel_old_timeval(utime);
1787         r->ru_stime = ns_to_kernel_old_timeval(stime);
1788 
1789         if (who != RUSAGE_CHILDREN) {
1790                 struct mm_struct *mm = get_task_mm(p);
1791 
1792                 if (mm) {
1793                         setmax_mm_hiwater_rss(&maxrss, mm);
1794                         mmput(mm);
1795                 }
1796         }
1797         r->ru_maxrss = maxrss * (PAGE_SIZE / 1024); /* convert pages to KBs */
1798 }

1894 /*
1895  * Check arithmetic relations of passed addresses.
1896  *
1897  * WARNING: we don't require any capability here so be very careful
1898  * in what is allowed for modification from userspace.
1899  */
1900 static int validate_prctl_map_addr(struct prctl_mm_map *prctl_map)
1901 {
1902         unsigned long mmap_max_addr = TASK_SIZE;
1903         int error = -EINVAL, i;
1904 
1905         static const unsigned char offsets[] = {
1906                 offsetof(struct prctl_mm_map, start_code),
1907                 offsetof(struct prctl_mm_map, end_code),
1908                 offsetof(struct prctl_mm_map, start_data),
1909                 offsetof(struct prctl_mm_map, end_data),
1910                 offsetof(struct prctl_mm_map, start_brk),
1911                 offsetof(struct prctl_mm_map, brk),
1912                 offsetof(struct prctl_mm_map, start_stack),
1913                 offsetof(struct prctl_mm_map, arg_start),
1914                 offsetof(struct prctl_mm_map, arg_end),
1915                 offsetof(struct prctl_mm_map, env_start),
1916                 offsetof(struct prctl_mm_map, env_end),
1917         };
1918 
1919         /*
1920          * Make sure the members are not somewhere outside
1921          * of allowed address space.
1922          */
1923         for (i = 0; i < ARRAY_SIZE(offsets); i++) {
1924                 u64 val = *(u64 *)((char *)prctl_map + offsets[i]);
1925 
1926                 if ((unsigned long)val >= mmap_max_addr ||
1927                     (unsigned long)val < mmap_min_addr)
1928                         goto out;
1929         }
1930 
1931         /*
1932          * Make sure the pairs are ordered.
1933          */
1934 #define __prctl_check_order(__m1, __op, __m2)                           \
1935         ((unsigned long)prctl_map->__m1 __op                            \
1936          (unsigned long)prctl_map->__m2) ? 0 : -EINVAL
1937         error  = __prctl_check_order(start_code, <, end_code);
1938         error |= __prctl_check_order(start_data,<=, end_data);
1939         error |= __prctl_check_order(start_brk, <=, brk);
1940         error |= __prctl_check_order(arg_start, <=, arg_end);
1941         error |= __prctl_check_order(env_start, <=, env_end);
1942         if (error)
1943                 goto out;
1944 #undef __prctl_check_order
1945 
1946         error = -EINVAL;
1947 
1948         /*
1949          * @brk should be after @end_data in traditional maps.
1950          */
1951         if (prctl_map->start_brk <= prctl_map->end_data ||
1952             prctl_map->brk <= prctl_map->end_data)
1953                 goto out;
1954 
1955         /*
1956          * Neither we should allow to override limits if they set.
1957          */
1958         if (check_data_rlimit(rlimit(RLIMIT_DATA), prctl_map->brk,
1959                               prctl_map->start_brk, prctl_map->end_data,
1960                               prctl_map->start_data))
1961                         goto out;
1962 
1963         error = 0;
1964 out:
1965         return error;
1966 }

2708 /**
2709  * do_sysinfo - fill in sysinfo struct
2710  * @info: pointer to buffer to fill
2711  */
2712 static int do_sysinfo(struct sysinfo *info)
2713 {
2714         unsigned long mem_total, sav_total;
2715         unsigned int mem_unit, bitcount;
2716         struct timespec64 tp;
2717 
2718         memset(info, 0, sizeof(struct sysinfo));
2719 
2720         ktime_get_boottime_ts64(&tp);
2721         timens_add_boottime(&tp);
2722         info->uptime = tp.tv_sec + (tp.tv_nsec ? 1 : 0);
2723 
2724         get_avenrun(info->loads, 0, SI_LOAD_SHIFT - FSHIFT);
2725 
2726         info->procs = nr_threads;
2727 
2728         si_meminfo(info);
2729         si_swapinfo(info);
2730 
2731         /*
2732          * If the sum of all the available memory (i.e. ram + swap)
2733          * is less than can be stored in a 32 bit unsigned long then
2734          * we can be binary compatible with 2.2.x kernels.  If not,
2735          * well, in that case 2.2.x was broken anyways...
2736          *
2737          *  -Erik Andersen <andersee@debian.org>
2738          */
2739 
2740         mem_total = info->totalram + info->totalswap;
2741         if (mem_total < info->totalram || mem_total < info->totalswap)
2742                 goto out;
2743         bitcount = 0;
2744         mem_unit = info->mem_unit;
2745         while (mem_unit > 1) {
2746                 bitcount++;
2747                 mem_unit >>= 1;
2748                 sav_total = mem_total;
2749                 mem_total <<= 1;
2750                 if (mem_total < sav_total)
2751                         goto out;
2752         }
2753 
2754         /*
2755          * If mem_total did not overflow, multiply all memory values by
2756          * info->mem_unit and set it to 1.  This leaves things compatible
2757          * with 2.2.x, and also retains compatibility with earlier 2.4.x
2758          * kernels...
2759          */
2760 
2761         info->mem_unit = 1;
2762         info->totalram <<= bitcount;
2763         info->freeram <<= bitcount;
2764         info->sharedram <<= bitcount;
2765         info->bufferram <<= bitcount;
2766         info->totalswap <<= bitcount;
2767         info->freeswap <<= bitcount;
2768         info->totalhigh <<= bitcount;
2769         info->freehigh <<= bitcount;
2770 
2771 out:
2772         return 0;
2773 }

2775 SYSCALL_DEFINE1(sysinfo, struct sysinfo __user *, info)
2776 {
2777         struct sysinfo val;
2778 
2779         do_sysinfo(&val);
2780 
2781         if (copy_to_user(info, &val, sizeof(struct sysinfo)))
2782                 return -EFAULT;
2783 
2784         return 0;
2785 }
