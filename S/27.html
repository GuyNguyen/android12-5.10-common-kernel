  35 /*
  36  * The least significant 3 bits of the owner value has the following
  37  * meanings when set.
  38  *  - Bit 0: RWSEM_READER_OWNED - The rwsem is owned by readers
  39  *  - Bit 1: RWSEM_RD_NONSPINNABLE - Readers cannot spin on this lock.
  40  *  - Bit 2: RWSEM_WR_NONSPINNABLE - Writers cannot spin on this lock.
  41  *
  42  * When the rwsem is either owned by an anonymous writer, or it is
  43  * reader-owned, but a spinning writer has timed out, both nonspinnable
  44  * bits will be set to disable optimistic spinning by readers and writers.
  45  * In the later case, the last unlocking reader should then check the
  46  * writer nonspinnable bit and clear it only to give writers preference
  47  * to acquire the lock via optimistic spinning, but not readers. Similar
  48  * action is also done in the reader slowpath.
  49 
  50  * When a writer acquires a rwsem, it puts its task_struct pointer
  51  * into the owner field. It is cleared after an unlock.
  52  *
  53  * When a reader acquires a rwsem, it will also puts its task_struct
  54  * pointer into the owner field with the RWSEM_READER_OWNED bit set.
  55  * On unlock, the owner field will largely be left untouched. So
  56  * for a free or reader-owned rwsem, the owner value may contain
  57  * information about the last reader that acquires the rwsem.
  58  *
  59  * That information may be helpful in debugging cases where the system
  60  * seems to hang on a reader owned rwsem especially if only one reader
  61  * is involved. Ideally we would like to track all the readers that own
  62  * a rwsem, but the overhead is simply too big.
  63  *
  64  * Reader optimistic spinning is helpful when the reader critical section
  65  * is short and there aren't that many readers around. It makes readers
  66  * relatively more preferred than writers. When a writer times out spinning
  67  * on a reader-owned lock and set the nospinnable bits, there are two main
  68  * reasons for that.
  69  *
  70  *  1) The reader critical section is long, perhaps the task sleeps after
  71  *     acquiring the read lock.
  72  *  2) There are just too many readers contending the lock causing it to
  73  *     take a while to service all of them.
  74  *
  75  * In the former case, long reader critical section will impede the progress
  76  * of writers which is usually more important for system performance. In
  77  * the later case, reader optimistic spinning tends to make the reader
  78  * groups that contain readers that acquire the lock together smaller
  79  * leading to more of them. That may hurt performance in some cases. In
  80  * other words, the setting of nonspinnable bits indicates that reader
  81  * optimistic spinning may not be helpful for those workloads that cause
  82  * it.
  83  *
  84  * Therefore, any writers that had observed the setting of the writer
  85  * nonspinnable bit for a given rwsem after they fail to acquire the lock
  86  * via optimistic spinning will set the reader nonspinnable bit once they
  87  * acquire the write lock. Similarly, readers that observe the setting
  88  * of reader nonspinnable bit at slowpath entry will set the reader
  89  * nonspinnable bits when they acquire the read lock via the wakeup path.
  90  *
  91  * Once the reader nonspinnable bit is on, it will only be reset when
  92  * a writer is able to acquire the rwsem in the fast path or somehow a
  93  * reader or writer in the slowpath doesn't observe the nonspinable bit.
  94  *
  95  * This is to discourage reader optmistic spinning on that particular
  96  * rwsem and make writers more preferred. This adaptive disabling of reader
  97  * optimistic spinning will alleviate the negative side effect of this
  98  * feature.
  99  */

 120 /*
 121  * On 64-bit architectures, the bit definitions of the count are:
 122  *
 123  * Bit  0    - writer locked bit
 124  * Bit  1    - waiters present bit
 125  * Bit  2    - lock handoff bit
 126  * Bits 3-7  - reserved
 127  * Bits 8-62 - 55-bit reader count
 128  * Bit  63   - read fail bit
 129  *
 130  * On 32-bit architectures, the bit definitions of the count are:
 131  *
 132  * Bit  0    - writer locked bit
 133  * Bit  1    - waiters present bit
 134  * Bit  2    - lock handoff bit
 135  * Bits 3-7  - reserved
 136  * Bits 8-30 - 23-bit reader count
 137  * Bit  31   - read fail bit
 138  *
 139  * It is not likely that the most significant bit (read fail bit) will ever
 140  * be set. This guard bit is still checked anyway in the down_read() fastpath
 141  * just in case we need to use up more of the reader bits for other purpose
 142  * in the future.
 143  *
 144  * atomic_long_fetch_add() is used to obtain reader lock, whereas
 145  * atomic_long_cmpxchg() will be used to obtain writer lock.
 146  *
 147  * There are three places where the lock handoff bit may be set or cleared.
 148  * 1) rwsem_mark_wake() for readers.
 149  * 2) rwsem_try_write_lock() for writers.
 150  * 3) Error path of rwsem_down_write_slowpath().
 151  *
 152  * For all the above cases, wait_lock will be held. A writer must also
 153  * be the first one in the wait_list to be eligible for setting the handoff
 154  * bit. So concurrent setting/clearing of handoff bit is not possible.
 155  */
 156 #define RWSEM_WRITER_LOCKED     (1UL << 0)
 157 #define RWSEM_FLAG_WAITERS      (1UL << 1)
 158 #define RWSEM_FLAG_HANDOFF      (1UL << 2)
 159 #define RWSEM_FLAG_READFAIL     (1UL << (BITS_PER_LONG - 1))

 259  * Set the RWSEM_NONSPINNABLE bits if the RWSEM_READER_OWNED flag
 260  * remains set. Otherwise, the operation will be aborted.
 261  */
 262 static inline void rwsem_set_nonspinnable(struct rw_semaphore *sem)
 263 {
 264         unsigned long owner = atomic_long_read(&sem->owner);
 265 
 266         do {
 267                 if (!(owner & RWSEM_READER_OWNED))
 268                         break;
 269                 if (owner & RWSEM_NONSPINNABLE)
 270                         break;
 271         } while (!atomic_long_try_cmpxchg(&sem->owner, &owner,
 272                                           owner | RWSEM_NONSPINNABLE));

 305 /*
 306  * Guide to the rw_semaphore's count field.
 307  *
 308  * When the RWSEM_WRITER_LOCKED bit in count is set, the lock is owned
 309  * by a writer.
 310  *
 311  * The lock is owned by readers when
 312  * (1) the RWSEM_WRITER_LOCKED isn't set in count,
 313  * (2) some of the reader bits are set in count, and
 314  * (3) the owner field has RWSEM_READ_OWNED bit set.
 315  *
 316  * Having some reader bits set is not enough to guarantee a readers owned
 317  * lock as the readers may be in the process of backing out from the count
 318  * and a writer has just released the lock. So another writer may steal
 319  * the lock immediately after that.
 320  */

 379 /*
 380  * handle the lock release when processes blocked on it that can now run
 381  * - if we come here from up_xxxx(), then the RWSEM_FLAG_WAITERS bit must
 382  *   have been set.
 383  * - there must be someone on the queue
 384  * - the wait_lock must be held by the caller
 385  * - tasks are marked for wakeup, the caller must later invoke wake_up_q()
 386  *   to actually wakeup the blocked task(s) and drop the reference count,
 387  *   preferably when the wait_lock is released
 388  * - woken process blocks are discarded from the list after having task zeroed
 389  * - writers are only marked woken if downgrading is false
 390  */
 391 static void rwsem_mark_wake(struct rw_semaphore *sem,
 392                             enum rwsem_wake_type wake_type,
 393                             struct wake_q_head *wake_q)
 394 {
 395         struct rwsem_waiter *waiter, *tmp;
 396         long oldcount, woken = 0, adjustment = 0;
 397         struct list_head wlist;
 398 
 399         lockdep_assert_held(&sem->wait_lock);
 400 
 401         /*
 402          * Take a peek at the queue head waiter such that we can determine
 403          * the wakeup(s) to perform.
 404          */
 405         waiter = rwsem_first_waiter(sem);
 406 
 407         if (waiter->type == RWSEM_WAITING_FOR_WRITE) {
 408                 if (wake_type == RWSEM_WAKE_ANY) {
 409                         /*
 410                          * Mark writer at the front of the queue for wakeup.
 411                          * Until the task is actually later awoken later by
 412                          * the caller, other writers are able to steal it.
 413                          * Readers, on the other hand, will block as they
 414                          * will notice the queued writer.
 415                          */
 416                         wake_q_add(wake_q, waiter->task);
 417                         lockevent_inc(rwsem_wake_writer);
 418                 }
 419 
 420                 return;
 421         }
 422 
 423         /*
 424          * No reader wakeup if there are too many of them already.
 425          */
 426         if (unlikely(atomic_long_read(&sem->count) < 0))
 427                 return;
 428 
 429         /*
 430          * Writers might steal the lock before we grant it to the next reader.
 431          * We prefer to do the first reader grant before counting readers
 432          * so we can bail out early if a writer stole the lock.
 433          */
 434         if (wake_type != RWSEM_WAKE_READ_OWNED) {
 435                 struct task_struct *owner;
 436 
 437                 adjustment = RWSEM_READER_BIAS;
 438                 oldcount = atomic_long_fetch_add(adjustment, &sem->count);
 439                 if (unlikely(oldcount & RWSEM_WRITER_MASK)) {
 440                         /*
 441                          * When we've been waiting "too" long (for writers
 442                          * to give up the lock), request a HANDOFF to
 443                          * force the issue.
 444                          */
 445                         if (!(oldcount & RWSEM_FLAG_HANDOFF) &&
 446                             time_after(jiffies, waiter->timeout)) {
 447                                 adjustment -= RWSEM_FLAG_HANDOFF;
 448                                 lockevent_inc(rwsem_rlock_handoff);
 449                         }
 450 
 451                         atomic_long_add(-adjustment, &sem->count);
 452                         return;
 453                 }
 454                 /*
 455                  * Set it to reader-owned to give spinners an early
 456                  * indication that readers now have the lock.
 457                  * The reader nonspinnable bit seen at slowpath entry of
 458                  * the reader is copied over.
 459                  */
 460                 owner = waiter->task;
 461                 if (waiter->last_rowner & RWSEM_RD_NONSPINNABLE) {
 462                         owner = (void *)((unsigned long)owner | RWSEM_RD_NONSPINNABLE);
 463                         lockevent_inc(rwsem_opt_norspin);
 464                 }
 465                 __rwsem_set_reader_owned(sem, owner);
 466         }
 467 
 468         /*
 469          * Grant up to MAX_READERS_WAKEUP read locks to all the readers in the
 470          * queue. We know that the woken will be at least 1 as we accounted
 471          * for above. Note we increment the 'active part' of the count by the
 472          * number of readers before waking any processes up.
 473          *
 474          * This is an adaptation of the phase-fair R/W locks where at the
 475          * reader phase (first waiter is a reader), all readers are eligible
 476          * to acquire the lock at the same time irrespective of their order
 477          * in the queue. The writers acquire the lock according to their
 478          * order in the queue.
 479          *
 480          * We have to do wakeup in 2 passes to prevent the possibility that
 481          * the reader count may be decremented before it is incremented. It
 482          * is because the to-be-woken waiter may not have slept yet. So it
 483          * may see waiter->task got cleared, finish its critical section and
 484          * do an unlock before the reader count increment.
 485          *
 486          * 1) Collect the read-waiters in a separate list, count them and
 487          *    fully increment the reader count in rwsem.
 488          * 2) For each waiters in the new list, clear waiter->task and
 489          *    put them into wake_q to be woken up later.
 490          */
 491         INIT_LIST_HEAD(&wlist);
 492         list_for_each_entry_safe(waiter, tmp, &sem->wait_list, list) {
 493                 if (waiter->type == RWSEM_WAITING_FOR_WRITE)
 494                         continue;
 495 
 496                 woken++;
 497                 list_move_tail(&waiter->list, &wlist);
 498 
 499                 /*
 500                  * Limit # of readers that can be woken up per wakeup call.
 501                  */
 502                 if (woken >= MAX_READERS_WAKEUP)
 503                         break;
 504         }
 505 
 506         adjustment = woken * RWSEM_READER_BIAS - adjustment;
 507         lockevent_cond_inc(rwsem_wake_reader, woken);
 508         if (list_empty(&sem->wait_list)) {
 509                 /* hit end of list above */
 510                 adjustment -= RWSEM_FLAG_WAITERS;
 511         }
 512 
 513         /*
 514          * When we've woken a reader, we no longer need to force writers
 515          * to give up the lock and we can clear HANDOFF.
 516          */
 517         if (woken && (atomic_long_read(&sem->count) & RWSEM_FLAG_HANDOFF))
 518                 adjustment -= RWSEM_FLAG_HANDOFF;
 519 
 520         if (adjustment)
 521                 atomic_long_add(adjustment, &sem->count);
 522 
 523         /* 2nd pass */
 524         list_for_each_entry_safe(waiter, tmp, &wlist, list) {
 525                 struct task_struct *tsk;
 526 
 527                 tsk = waiter->task;
 528                 get_task_struct(tsk);
 529 
 530                 /*
 531                  * Ensure calling get_task_struct() before setting the reader
 532                  * waiter to nil such that rwsem_down_read_slowpath() cannot
 533                  * race with do_exit() by always holding a reference count
 534                  * to the task to wakeup.
 535                  */
 536                 smp_store_release(&waiter->task, NULL);
 537                 /*
 538                  * Ensure issuing the wakeup (either by us or someone else)
 539                  * after setting the reader waiter to nil.
 540                  */
 541                 wake_q_add_safe(wake_q, tsk);
 542         }
 543 }

 545 /*
 546  * This function must be called with the sem->wait_lock held to prevent
 547  * race conditions between checking the rwsem wait list and setting the
 548  * sem->count accordingly.
 549  *
 550  * If wstate is WRITER_HANDOFF, it will make sure that either the handoff
 551  * bit is set or the lock is acquired with handoff bit cleared.
 552  */
 553 static inline bool rwsem_try_write_lock(struct rw_semaphore *sem,
 554                                         enum writer_wait_state wstate)
 555 {
 556         long count, new;
 557 
 558         lockdep_assert_held(&sem->wait_lock);
 559 
 560         count = atomic_long_read(&sem->count);
 561         do {
 562                 bool has_handoff = !!(count & RWSEM_FLAG_HANDOFF);
 563 
 564                 if (has_handoff && wstate == WRITER_NOT_FIRST)
 565                         return false;
 566 
 567                 new = count;
 568 
 569                 if (count & RWSEM_LOCK_MASK) {
 570                         if (has_handoff || (wstate != WRITER_HANDOFF))
 571                                 return false;
 572 
 573                         new |= RWSEM_FLAG_HANDOFF;
 574                 } else {
 575                         new |= RWSEM_WRITER_LOCKED;
 576                         new &= ~RWSEM_FLAG_HANDOFF;
 577 
 578                         if (list_is_singular(&sem->wait_list))
 579                                 new &= ~RWSEM_FLAG_WAITERS;
 580                 }
 581         } while (!atomic_long_try_cmpxchg_acquire(&sem->count, &count, new));
 582 
 583         /*
 584          * We have either acquired the lock with handoff bit cleared or
 585          * set the handoff bit.
 586          */
 587         if (new & RWSEM_FLAG_HANDOFF)
 588                 return false;
 589 
 590         rwsem_set_owner(sem);
 591         return true;
 592 }

 595 /*
 596  * Try to acquire read lock before the reader is put on wait queue.
 597  * Lock acquisition isn't allowed if the rwsem is locked or a writer handoff
 598  * is ongoing.
 599  */
 600 static inline bool rwsem_try_read_lock_unqueued(struct rw_semaphore *sem)
 601 {
 602         long count = atomic_long_read(&sem->count);
 603 
 604         if (count & (RWSEM_WRITER_MASK | RWSEM_FLAG_HANDOFF))
 605                 return false;
 606 
 607         count = atomic_long_fetch_add_acquire(RWSEM_READER_BIAS, &sem->count);
 608         if (!(count & (RWSEM_WRITER_MASK | RWSEM_FLAG_HANDOFF))) {
 609                 rwsem_set_reader_owned(sem);
 610                 lockevent_inc(rwsem_opt_rlock);
 611                 return true;
 612         }
 613 
 614         /* Back out the change */
 615         atomic_long_add(-RWSEM_READER_BIAS, &sem->count);
 616         return false;
 617 }

 646 static inline bool rwsem_can_spin_on_owner(struct rw_semaphore *sem,
 647                                            unsigned long nonspinnable)
 648 {
 649         struct task_struct *owner;
 650         unsigned long flags;
 651         bool ret = true;
 652 
 653         if (need_resched()) {
 654                 lockevent_inc(rwsem_opt_fail);
 655                 return false;
 656         }
 657 
 658         preempt_disable();
 659         rcu_read_lock();

 665             (owner && !(flags & RWSEM_READER_OWNED) && !owner_on_cpu(owner)))
 666                 ret = false;
 667         rcu_read_unlock();
 668         preempt_enable();
 669

 671         return ret;

 674 /*
 675  * The rwsem_spin_on_owner() function returns the folowing 4 values
 676  * depending on the lock owner state.
 677  *   OWNER_NULL  : owner is currently NULL
 678  *   OWNER_WRITER: when owner changes and is a writer
 679  *   OWNER_READER: when owner changes and the new owner may be a reader.
 680  *   OWNER_NONSPINNABLE:
 681  *                 when optimistic spinning has to stop because either the
 682  *                 owner stops running, is unknown, or its timeslice has
 683  *                 been used up.
 684  */
 685 enum owner_state {
 686         OWNER_NULL              = 1 << 0,
 687         OWNER_WRITER            = 1 << 1,
 688         OWNER_READER            = 1 << 2,
 689         OWNER_NONSPINNABLE      = 1 << 3,
 690 };
 691 #define OWNER_SPINNABLE         (OWNER_NULL | OWNER_WRITER | OWNER_READER)
 692 
 693 static inline enum owner_state
 694 rwsem_owner_state(struct task_struct *owner, unsigned long flags, unsigned long nonspinnable)
 695 {
 696         if (flags & nonspinnable)
 697                 return OWNER_NONSPINNABLE;
 698 
 699         if (flags & RWSEM_READER_OWNED)
 700                 return OWNER_READER;
 701 
 702         return owner ? OWNER_WRITER : OWNER_NULL;
 703 }
 704 
 705 static noinline enum owner_state
 706 rwsem_spin_on_owner(struct rw_semaphore *sem, unsigned long nonspinnable)
 707 {
 708         struct task_struct *new, *owner;
 709         unsigned long flags, new_flags;
 710         enum owner_state state;
 711 
 712         owner = rwsem_owner_flags(sem, &flags);
 713         state = rwsem_owner_state(owner, flags, nonspinnable);
 714         if (state != OWNER_WRITER)
 715                 return state;
 716 
 717         rcu_read_lock();
 718         for (;;) {
 719                 /*
 720                  * When a waiting writer set the handoff flag, it may spin
 721                  * on the owner as well. Once that writer acquires the lock,
 722                  * we can spin on it. So we don't need to quit even when the
 723                  * handoff bit is set.
 724                  */
 725                 new = rwsem_owner_flags(sem, &new_flags);
 726                 if ((new != owner) || (new_flags != flags)) {
 727                         state = rwsem_owner_state(new, new_flags, nonspinnable);
 728                         break;
 729                 }
 730 
 731                 /*
 732                  * Ensure we emit the owner->on_cpu, dereference _after_
 733                  * checking sem->owner still matches owner, if that fails,
 734                  * owner might point to free()d memory, if it still matches,
 735                  * the rcu_read_lock() ensures the memory stays valid.
 736                  */
 737                 barrier();
 738 
 739                 if (need_resched() || !owner_on_cpu(owner)) {
 740                         state = OWNER_NONSPINNABLE;
 741                         break;
 742                 }
 743 
 744                 cpu_relax();
 745         }
 746         rcu_read_unlock();
 747 
 748         return state;
 749 }

 751 /*
 752  * Calculate reader-owned rwsem spinning threshold for writer
 753  *
 754  * The more readers own the rwsem, the longer it will take for them to
 755  * wind down and free the rwsem. So the empirical formula used to
 756  * determine the actual spinning time limit here is:
 757  *
 758  *   Spinning threshold = (10 + nr_readers/2)us
 759  *
 760  * The limit is capped to a maximum of 25us (30 readers). This is just
 761  * a heuristic and is subjected to change in the future.
 762  */
 763 static inline u64 rwsem_rspin_threshold(struct rw_semaphore *sem)
 764 {
 765         long count = atomic_long_read(&sem->count);
 766         int readers = count >> RWSEM_READER_SHIFT;
 767         u64 delta;
 768 
 769         if (readers > 30)
 770                 readers = 30;
 771         delta = (20 + readers) * NSEC_PER_USEC / 2;
 772 
 773         return sched_clock() + delta;
 774 }

 776 static bool rwsem_optimistic_spin(struct rw_semaphore *sem, bool wlock)
 777 {
 778         bool taken = false;
 779         int prev_owner_state = OWNER_NULL;
 780         int loop = 0;
 781         u64 rspin_threshold = 0;
 782         unsigned long nonspinnable = wlock ? RWSEM_WR_NONSPINNABLE
 783                                            : RWSEM_RD_NONSPINNABLE;
 784 
 785         preempt_disable();
 786 
 787         /* sem->wait_lock should not be held when doing optimistic spinning */
 788         if (!osq_lock(&sem->osq))
 789                 goto done;
 790 
 791         /*
 792          * Optimistically spin on the owner field and attempt to acquire the
 793          * lock whenever the owner changes. Spinning will be stopped when:
 794          *  1) the owning writer isn't running; or
 795          *  2) readers own the lock and spinning time has exceeded limit.
 796          */
 797         for (;;) {
 798                 enum owner_state owner_state;
 799 
 800                 owner_state = rwsem_spin_on_owner(sem, nonspinnable);
 801                 if (!(owner_state & OWNER_SPINNABLE))
 802                         break;
 803 
 804                 /*
 805                  * Try to acquire the lock
 806                  */
 807                 taken = wlock ? rwsem_try_write_lock_unqueued(sem)
 808                               : rwsem_try_read_lock_unqueued(sem);
 809 
 810                 if (taken)
 811                         break;
 812 
 813                 /*
 814                  * Time-based reader-owned rwsem optimistic spinning
 815                  */
 816                 if (wlock && (owner_state == OWNER_READER)) {
 817                         /*
 818                          * Re-initialize rspin_threshold every time when
 819                          * the owner state changes from non-reader to reader.
 820                          * This allows a writer to steal the lock in between
 821                          * 2 reader phases and have the threshold reset at
 822                          * the beginning of the 2nd reader phase.
 823                          */
 824                         if (prev_owner_state != OWNER_READER) {
 825                                 if (rwsem_test_oflags(sem, nonspinnable))
 826                                         break;
 827                                 rspin_threshold = rwsem_rspin_threshold(sem);
 828                                 loop = 0;
 829                         }
 830 
 831                         /*
 832                          * Check time threshold once every 16 iterations to
 833                          * avoid calling sched_clock() too frequently so
 834                          * as to reduce the average latency between the times
 835                          * when the lock becomes free and when the spinner
 836                          * is ready to do a trylock.
 837                          */
 838                         else if (!(++loop & 0xf) && (sched_clock() > rspin_threshold)) {
 839                                 rwsem_set_nonspinnable(sem);
 840                                 lockevent_inc(rwsem_opt_nospin);
 841                                 break;
 842                         }
 843                 }
 844 
 845                 /*
 846                  * An RT task cannot do optimistic spinning if it cannot
 847                  * be sure the lock holder is running or live-lock may
 848                  * happen if the current task and the lock holder happen
 849                  * to run in the same CPU. However, aborting optimistic
 850                  * spinning while a NULL owner is detected may miss some
 851                  * opportunity where spinning can continue without causing
 852                  * problem.
 853                  *
 854                  * There are 2 possible cases where an RT task may be able
 855                  * to continue spinning.
 856                  *
 857                  * 1) The lock owner is in the process of releasing the
 858                  *    lock, sem->owner is cleared but the lock has not
 859                  *    been released yet.
 860                  * 2) The lock was free and owner cleared, but another
 861                  *    task just comes in and acquire the lock before
 862                  *    we try to get it. The new owner may be a spinnable
 863                  *    writer.
 864                  *
 865                  * To take advantage of two scenarios listed agove, the RT
 866                  * task is made to retry one more time to see if it can
 867                  * acquire the lock or continue spinning on the new owning
 868                  * writer. Of course, if the time lag is long enough or the
 869                  * new owner is not a writer or spinnable, the RT task will
 870                  * quit spinning.
 871                  *
 872                  * If the owner is a writer, the need_resched() check is
 873                  * done inside rwsem_spin_on_owner(). If the owner is not
 874                  * a writer, need_resched() check needs to be done here.
 875                  */
 876                 if (owner_state != OWNER_WRITER) {
 877                         if (need_resched())
 878                                 break;
 879                         if (rt_task(current) &&
 880                            (prev_owner_state != OWNER_WRITER))
 881                                 break;
 882                 }
 883                 prev_owner_state = owner_state;
 884 
 885                 /*
 886                  * The cpu_relax() call is a compiler barrier which forces
 887                  * everything in this loop to be re-loaded. We don't need
 888                  * memory barriers as we'll eventually observe the right
 889                  * values at the cost of a few extra spins.
 890                  */
 891                 cpu_relax();
 892         }
 893         osq_unlock(&sem->osq);
 894 done:
 895         preempt_enable();
 896         lockevent_cond_inc(rwsem_opt_fail, !taken);
 897         return taken;
 898 }

 900 /*
 901  * Clear the owner's RWSEM_WR_NONSPINNABLE bit if it is set. This should
 902  * only be called when the reader count reaches 0.
 903  *
 904  * This give writers better chance to acquire the rwsem first before
 905  * readers when the rwsem was being held by readers for a relatively long
 906  * period of time. Race can happen that an optimistic spinner may have
 907  * just stolen the rwsem and set the owner, but just clearing the
 908  * RWSEM_WR_NONSPINNABLE bit will do no harm anyway.
 909  */
 910 static inline void clear_wr_nonspinnable(struct rw_semaphore *sem)
 911 {
 912         if (rwsem_test_oflags(sem, RWSEM_WR_NONSPINNABLE))
 913                 atomic_long_andnot(RWSEM_WR_NONSPINNABLE, &sem->owner);
 914 }

 916 /*
 917  * This function is called when the reader fails to acquire the lock via
 918  * optimistic spinning. In this case we will still attempt to do a trylock
 919  * when comparing the rwsem state right now with the state when entering
 920  * the slowpath indicates that the reader is still in a valid reader phase.
 921  * This happens when the following conditions are true:
 922  *
 923  * 1) The lock is currently reader owned, and
 924  * 2) The lock is previously not reader-owned or the last read owner changes.
 925  *
 926  * In the former case, we have transitioned from a writer phase to a
 927  * reader-phase while spinning. In the latter case, it means the reader
 928  * phase hasn't ended when we entered the optimistic spinning loop. In
 929  * both cases, the reader is eligible to acquire the lock. This is the
 930  * secondary path where a read lock is acquired optimistically.
 931  *
 932  * The reader non-spinnable bit wasn't set at time of entry or it will
 933  * not be here at all.
 934  */
 935 static inline bool rwsem_reader_phase_trylock(struct rw_semaphore *sem,
 936                                               unsigned long last_rowner)
 937 {
 938         unsigned long owner = atomic_long_read(&sem->owner);
 939 
 940         if (!(owner & RWSEM_READER_OWNED))
 941                 return false;
 942 
 943         if (((owner ^ last_rowner) & ~RWSEM_OWNER_FLAGS_MASK) &&
 944             rwsem_try_read_lock_unqueued(sem)) {
 945                 lockevent_inc(rwsem_opt_rlock2);
 946                 lockevent_add(rwsem_opt_fail, -1);
 947                 return true;
 948         }
 949         return false;
 950 }

 979 /*
 980  * Wait for the read lock to be granted
 981  */
 982 static struct rw_semaphore __sched *
 983 rwsem_down_read_slowpath(struct rw_semaphore *sem, int state)
 984 {
 985         long count, adjustment = -RWSEM_READER_BIAS;
 986         struct rwsem_waiter waiter;
 987         DEFINE_WAKE_Q(wake_q);
 988         bool wake = false;
 989         bool already_on_list = false;
 990 
 991         /*
 992          * Save the current read-owner of rwsem, if available, and the
 993          * reader nonspinnable bit.
 994          */
 995         waiter.last_rowner = atomic_long_read(&sem->owner);
 996         if (!(waiter.last_rowner & RWSEM_READER_OWNED))
 997                 waiter.last_rowner &= RWSEM_RD_NONSPINNABLE;
 998 
 999         if (!rwsem_can_spin_on_owner(sem, RWSEM_RD_NONSPINNABLE))
1000                 goto queue;
1001 
1002         /*
1003          * Undo read bias from down_read() and do optimistic spinning.
1004          */
1005         atomic_long_add(-RWSEM_READER_BIAS, &sem->count);
1006         adjustment = 0;
1007         if (rwsem_optimistic_spin(sem, false)) {
1008                 /* rwsem_optimistic_spin() implies ACQUIRE on success */
1009                 /*
1010                  * Wake up other readers in the wait list if the front
1011                  * waiter is a reader.
1012                  */
1013                 if ((atomic_long_read(&sem->count) & RWSEM_FLAG_WAITERS)) {
1014                         raw_spin_lock_irq(&sem->wait_lock);
1015                         if (!list_empty(&sem->wait_list))
1016                                 rwsem_mark_wake(sem, RWSEM_WAKE_READ_OWNED,
1017                                                 &wake_q);
1018                         raw_spin_unlock_irq(&sem->wait_lock);
1019                         wake_up_q(&wake_q);
1020                 }
1021                 return sem;
1022         } else if (rwsem_reader_phase_trylock(sem, waiter.last_rowner)) {
1023                 /* rwsem_reader_phase_trylock() implies ACQUIRE on success */
1024                 return sem;
1025         }
1026 
1027 queue:
1028         waiter.task = current;
1029         waiter.type = RWSEM_WAITING_FOR_READ;
1030         waiter.timeout = jiffies + RWSEM_WAIT_TIMEOUT;
1031 
1032         raw_spin_lock_irq(&sem->wait_lock);
1033         if (list_empty(&sem->wait_list)) {
1034                 /*
1035                  * In case the wait queue is empty and the lock isn't owned
1036                  * by a writer or has the handoff bit set, this reader can
1037                  * exit the slowpath and return immediately as its
1038                  * RWSEM_READER_BIAS has already been set in the count.
1039                  */
1040                 if (adjustment && !(atomic_long_read(&sem->count) &
1041                      (RWSEM_WRITER_MASK | RWSEM_FLAG_HANDOFF))) {
1042                         /* Provide lock ACQUIRE */
1043                         smp_acquire__after_ctrl_dep();
1044                         raw_spin_unlock_irq(&sem->wait_lock);
1045                         rwsem_set_reader_owned(sem);
1046                         lockevent_inc(rwsem_rlock_fast);
1047                         return sem;
1048                 }
1049                 adjustment += RWSEM_FLAG_WAITERS;
1050         }
1051         trace_android_vh_alter_rwsem_list_add(
1052                                         &waiter,
1053                                         sem, &already_on_list);
1054         if (!already_on_list)
1055                 list_add_tail(&waiter.list, &sem->wait_list);
1056 
1057         /* we're now waiting on the lock, but no longer actively locking */
1058         if (adjustment)
1059                 count = atomic_long_add_return(adjustment, &sem->count);
1060         else
1061                 count = atomic_long_read(&sem->count);
1062 
1063         /*
1064          * If there are no active locks, wake the front queued process(es).
1065          *
1066          * If there are no writers and we are first in the queue,
1067          * wake our own waiter to join the existing active readers !
1068          */
1069         if (!(count & RWSEM_LOCK_MASK)) {
1070                 clear_wr_nonspinnable(sem);
1071                 wake = true;
1072         }
1073         if (wake || (!(count & RWSEM_WRITER_MASK) &&
1074                     (adjustment & RWSEM_FLAG_WAITERS)))
1075                 rwsem_mark_wake(sem, RWSEM_WAKE_ANY, &wake_q);
1076 
1077         trace_android_vh_rwsem_wake(sem);
1078         raw_spin_unlock_irq(&sem->wait_lock);
1079         wake_up_q(&wake_q);
1080 
1081         /* wait to be given the lock */
1082         trace_android_vh_rwsem_read_wait_start(sem);
1083         for (;;) {
1084                 set_current_state(state);
1085                 if (!smp_load_acquire(&waiter.task)) {
1086                         /* Matches rwsem_mark_wake()'s smp_store_release(). */
1087                         break;
1088                 }
1089                 if (signal_pending_state(state, current)) {
1090                         raw_spin_lock_irq(&sem->wait_lock);
1091                         if (waiter.task)
1092                                 goto out_nolock;
1093                         raw_spin_unlock_irq(&sem->wait_lock);
1094                         /* Ordered by sem->wait_lock against rwsem_mark_wake(). */
1095                         break;
1096                 }
1097                 schedule();
1098                 lockevent_inc(rwsem_sleep_reader);
1099         }
1100 
1101         __set_current_state(TASK_RUNNING);
1102         trace_android_vh_rwsem_read_wait_finish(sem);
1103         lockevent_inc(rwsem_rlock);
1104         return sem;
1105 
1106 out_nolock:
1107         list_del(&waiter.list);
1108         if (list_empty(&sem->wait_list)) {
1109                 atomic_long_andnot(RWSEM_FLAG_WAITERS|RWSEM_FLAG_HANDOFF,
1110                                    &sem->count);
1111         }
1112         raw_spin_unlock_irq(&sem->wait_lock);
1113         __set_current_state(TASK_RUNNING);
1114         trace_android_vh_rwsem_read_wait_finish(sem);
1115         lockevent_inc(rwsem_rlock_fail);
1116         return ERR_PTR(-EINTR);
1117 }

1132 /*
1133  * Wait until we successfully acquire the write lock
1134  */
1135 static struct rw_semaphore *
1136 rwsem_down_write_slowpath(struct rw_semaphore *sem, int state)
1137 {
1138         long count;
1139         bool disable_rspin;
1140         enum writer_wait_state wstate;
1141         struct rwsem_waiter waiter;
1142         struct rw_semaphore *ret = sem;
1143         DEFINE_WAKE_Q(wake_q);
1144         bool already_on_list = false;
1145 
1146         /* do optimistic spinning and steal lock if possible */
1147         if (rwsem_can_spin_on_owner(sem, RWSEM_WR_NONSPINNABLE) &&
1148             rwsem_optimistic_spin(sem, true)) {
1149                 /* rwsem_optimistic_spin() implies ACQUIRE on success */
1150                 return sem;
1151         }
1152 
1153         /*
1154          * Disable reader optimistic spinning for this rwsem after
1155          * acquiring the write lock when the setting of the nonspinnable
1156          * bits are observed.
1157          */
1158         disable_rspin = atomic_long_read(&sem->owner) & RWSEM_NONSPINNABLE;
1159 
1160         /*
1161          * Optimistic spinning failed, proceed to the slowpath
1162          * and block until we can acquire the sem.
1163          */
1164         waiter.task = current;
1165         waiter.type = RWSEM_WAITING_FOR_WRITE;
1166         waiter.timeout = jiffies + RWSEM_WAIT_TIMEOUT;
1167 
1168         raw_spin_lock_irq(&sem->wait_lock);
1169 
1170         /* account for this before adding a new element to the list */
1171         wstate = list_empty(&sem->wait_list) ? WRITER_FIRST : WRITER_NOT_FIRST;
1172 
1173         trace_android_vh_alter_rwsem_list_add(
1174                                         &waiter,
1175                                         sem, &already_on_list);
1176         if (!already_on_list)
1177                 list_add_tail(&waiter.list, &sem->wait_list);
1178 
1179         /* we're now waiting on the lock */
1180         if (wstate == WRITER_NOT_FIRST) {
1181                 count = atomic_long_read(&sem->count);
1182 
1183                 /*
1184                  * If there were already threads queued before us and:
1185                  *  1) there are no no active locks, wake the front
1186                  *     queued process(es) as the handoff bit might be set.
1187                  *  2) there are no active writers and some readers, the lock
1188                  *     must be read owned; so we try to wake any read lock
1189                  *     waiters that were queued ahead of us.
1190                  */
1191                 if (count & RWSEM_WRITER_MASK)
1192                         goto wait;
1193 
1194                 rwsem_mark_wake(sem, (count & RWSEM_READER_MASK)
1195                                         ? RWSEM_WAKE_READERS
1196                                         : RWSEM_WAKE_ANY, &wake_q);
1197 
1198                 if (!wake_q_empty(&wake_q)) {
1199                         /*
1200                          * We want to minimize wait_lock hold time especially
1201                          * when a large number of readers are to be woken up.
1202                          */
1203                         raw_spin_unlock_irq(&sem->wait_lock);
1204                         wake_up_q(&wake_q);
1205                         wake_q_init(&wake_q);   /* Used again, reinit */
1206                         raw_spin_lock_irq(&sem->wait_lock);
1207                 }
1208         } else {
1209                 atomic_long_or(RWSEM_FLAG_WAITERS, &sem->count);
1210         }
1211 
1212 wait:
1213         trace_android_vh_rwsem_wake(sem);
1214         /* wait until we successfully acquire the lock */
1215         trace_android_vh_rwsem_write_wait_start(sem);
1216         set_current_state(state);
1217         for (;;) {
1218                 if (rwsem_try_write_lock(sem, wstate)) {
1219                         /* rwsem_try_write_lock() implies ACQUIRE on success */
1220                         break;
1221                 }
1222 
1223                 raw_spin_unlock_irq(&sem->wait_lock);
1224 
1225                 /*
1226                  * After setting the handoff bit and failing to acquire
1227                  * the lock, attempt to spin on owner to accelerate lock
1228                  * transfer. If the previous owner is a on-cpu writer and it
1229                  * has just released the lock, OWNER_NULL will be returned.
1230                  * In this case, we attempt to acquire the lock again
1231                  * without sleeping.
1232                  */
1233                 if (wstate == WRITER_HANDOFF &&
1234                     rwsem_spin_on_owner(sem, RWSEM_NONSPINNABLE) == OWNER_NULL)
1235                         goto trylock_again;
1236 
1237                 /* Block until there are no active lockers. */
1238                 for (;;) {
1239                         if (signal_pending_state(state, current))
1240                                 goto out_nolock;
1241 
1242                         schedule();
1243                         lockevent_inc(rwsem_sleep_writer);
1244                         set_current_state(state);
1245                         /*
1246                          * If HANDOFF bit is set, unconditionally do
1247                          * a trylock.
1248                          */
1249                         if (wstate == WRITER_HANDOFF)
1250                                 break;
1251 
1252                         if ((wstate == WRITER_NOT_FIRST) &&
1253                             (rwsem_first_waiter(sem) == &waiter))
1254                                 wstate = WRITER_FIRST;
1255 
1256                         count = atomic_long_read(&sem->count);
1257                         if (!(count & RWSEM_LOCK_MASK))
1258                                 break;
1259 
1260                         /*
1261                          * The setting of the handoff bit is deferred
1262                          * until rwsem_try_write_lock() is called.
1263                          */
1264                         if ((wstate == WRITER_FIRST) && (rt_task(current) ||
1265                             time_after(jiffies, waiter.timeout))) {
1266                                 wstate = WRITER_HANDOFF;
1267                                 lockevent_inc(rwsem_wlock_handoff);
1268                                 break;
1269                         }
1270                 }
1271 trylock_again:
1272                 raw_spin_lock_irq(&sem->wait_lock);
1273         }
1274         __set_current_state(TASK_RUNNING);
1275         trace_android_vh_rwsem_write_wait_finish(sem);
1276         list_del(&waiter.list);
1277         rwsem_disable_reader_optspin(sem, disable_rspin);
1278         raw_spin_unlock_irq(&sem->wait_lock);
1279         lockevent_inc(rwsem_wlock);
1280 
1281         return ret;
1282 
1283 out_nolock:
1284         __set_current_state(TASK_RUNNING);
1285         trace_android_vh_rwsem_write_wait_finish(sem);
1286         raw_spin_lock_irq(&sem->wait_lock);
1287         list_del(&waiter.list);
1288 
1289         if (unlikely(wstate == WRITER_HANDOFF))
1290                 atomic_long_add(-RWSEM_FLAG_HANDOFF,  &sem->count);
1291 
1292         if (list_empty(&sem->wait_list))
1293                 atomic_long_andnot(RWSEM_FLAG_WAITERS, &sem->count);
1294         else
1295                 rwsem_mark_wake(sem, RWSEM_WAKE_ANY, &wake_q);
1296         raw_spin_unlock_irq(&sem->wait_lock);
1297         wake_up_q(&wake_q);
1298         lockevent_inc(rwsem_wlock_fail);
1299 
1300         return ERR_PTR(-EINTR);
1301 }

1303 /*
1304  * handle waking up a waiter on the semaphore
1305  * - up_read/up_write has decremented the active part of count if we come here
1306  */
1307 static struct rw_semaphore *rwsem_wake(struct rw_semaphore *sem, long count)
1308 {
1309         unsigned long flags;
1310         DEFINE_WAKE_Q(wake_q);
1311 
1312         raw_spin_lock_irqsave(&sem->wait_lock, flags);
1313 
1314         if (!list_empty(&sem->wait_list))
1315                 rwsem_mark_wake(sem, RWSEM_WAKE_ANY, &wake_q);
1316         trace_android_vh_rwsem_wake_finish(sem);
1317 
1318         raw_spin_unlock_irqrestore(&sem->wait_lock, flags);
1319         wake_up_q(&wake_q);
1320 
1321         return sem;
1322 }

1325  * downgrade a write lock into a read lock
1326  * - caller incremented waiting part of count and discovered it still negative
1327  * - just wake up any readers at the front of the queue
1328  */

1337                 rwsem_mark_wake(sem, RWSEM_WAKE_READ_OWNED, &wake_q);
1338 
1339         raw_spin_unlock_irqrestore(&sem->wait_lock, flags);
1340         wake_up_q(&wake_q);

1382 static inline int __down_read_trylock(struct rw_semaphore *sem)
1383 {
1384         long tmp;
1385 
1386         DEBUG_RWSEMS_WARN_ON(sem->magic != sem, sem);
1387 
1388         /*
1389          * Optimize for the case when the rwsem is not locked at all.
1390          */
1391         tmp = RWSEM_UNLOCKED_VALUE;
1392         do {
1393                 if (atomic_long_try_cmpxchg_acquire(&sem->count, &tmp,
1394                                         tmp + RWSEM_READER_BIAS)) {
1395                         rwsem_set_reader_owned(sem);
1396                         return 1;
1397                 }
1398         } while (!(tmp & RWSEM_READ_FAILED_MASK));
1399         return 0;
1400 }

1446  * unlock after reading
1447  */
1448 static inline void __up_read(struct rw_semaphore *sem)
1449 {
1450         long tmp;
1451 
1452         DEBUG_RWSEMS_WARN_ON(sem->magic != sem, sem);
1453         DEBUG_RWSEMS_WARN_ON(!is_rwsem_reader_owned(sem), sem);
1454 
1455         rwsem_clear_reader_owned(sem);

1457         DEBUG_RWSEMS_WARN_ON(tmp < 0, sem);

1459                       RWSEM_FLAG_WAITERS)) {

1465 /*
1466  * unlock after writing
1467  */
1468 static inline void __up_write(struct rw_semaphore *sem)
1469 {
1470         long tmp;
1471 
1472         DEBUG_RWSEMS_WARN_ON(sem->magic != sem, sem);
1473         /*
1474          * sem->owner may differ from current if the ownership is transferred
1475          * to an anonymous writer by setting the RWSEM_NONSPINNABLE bits.
1476          */
1477         DEBUG_RWSEMS_WARN_ON((rwsem_owner(sem) != current) &&
1478                             !rwsem_test_oflags(sem, RWSEM_NONSPINNABLE), sem);
1479 
1480         rwsem_clear_owner(sem);
1481         tmp = atomic_long_fetch_add_release(-RWSEM_WRITER_LOCKED, &sem->count);
1482         if (unlikely(tmp & RWSEM_FLAG_WAITERS))
1483                 rwsem_wake(sem, tmp);
1484 }

1486 /*
1487  * downgrade write lock to read lock
1488  */
1489 static inline void __downgrade_write(struct rw_semaphore *sem)
1490 {
1491         long tmp;
1492 
1493         /*
1494          * When downgrading from exclusive to shared ownership,
1495          * anything inside the write-locked region cannot leak
1496          * into the read side. In contrast, anything in the
1497          * read-locked region is ok to be re-ordered into the
1498          * write side. As such, rely on RELEASE semantics.
1499          */
1500         DEBUG_RWSEMS_WARN_ON(rwsem_owner(sem) != current, sem);
1501         tmp = atomic_long_fetch_add_release(
1502                 -RWSEM_WRITER_LOCKED+RWSEM_READER_BIAS, &sem->count);
1503         rwsem_set_reader_owned(sem);
1504         if (tmp & RWSEM_FLAG_WAITERS)
1505                 rwsem_downgrade_wake(sem);
1506 }
