  68 /*
  69  * The timer wheel has LVL_DEPTH array levels. Each level provides an array of
  70  * LVL_SIZE buckets. Each level is driven by its own clock and therefor each
  71  * level has a different granularity.
  72  *
  73  * The level granularity is:            LVL_CLK_DIV ^ lvl
  74  * The level clock frequency is:        HZ / (LVL_CLK_DIV ^ level)
  75  *
  76  * The array level of a newly armed timer depends on the relative expiry
  77  * time. The farther the expiry time is away the higher the array level and
  78  * therefor the granularity becomes.
  79  *
  80  * Contrary to the original timer wheel implementation, which aims for 'exact'
  81  * expiry of the timers, this implementation removes the need for recascading
  82  * the timers into the lower array levels. The previous 'classic' timer wheel
  83  * implementation of the kernel already violated the 'exact' expiry by adding
  84  * slack to the expiry time to provide batched expiration. The granularity
  85  * levels provide implicit batching.
  86  *
  87  * This is an optimization of the original timer wheel implementation for the
  88  * majority of the timer wheel use cases: timeouts. The vast majority of
  89  * timeout timers (networking, disk I/O ...) are canceled before expiry. If
  90  * the timeout expires it indicates that normal operation is disturbed, so it
  91  * does not matter much whether the timeout comes with a slight delay.
  92  *
  93  * The only exception to this are networking timers with a small expiry
  94  * time. They rely on the granularity. Those fit into the first wheel level,
  95  * which has HZ granularity.
  96  *
  97  * We don't have cascading anymore. timers with a expiry time above the
  98  * capacity of the last wheel level are force expired at the maximum timeout
  99  * value of the last wheel level. From data sampling we know that the maximum
 100  * value observed is 5 days (network connection tracking), so this should not
 101  * be an issue.
 102  *
 103  * The currently chosen array constants values are a good compromise between
 104  * array size and granularity.
 105  *
 106  * This results in the following granularity and range levels:
 107  *
 108  * HZ 1000 steps
 109  * Level Offset  Granularity            Range
 110  *  0      0         1 ms                0 ms -         63 ms
 111  *  1     64         8 ms               64 ms -        511 ms
 112  *  2    128        64 ms              512 ms -       4095 ms (512ms - ~4s)
 113  *  3    192       512 ms             4096 ms -      32767 ms (~4s - ~32s)

 115  *  5    320     32768 ms (~32s)    262144 ms -    2097151 ms (~4m - ~34m)
 116  *  6    384    262144 ms (~4m)    2097152 ms -   16777215 ms (~34m - ~4h)
 117  *  7    448   2097152 ms (~34m)  16777216 ms -  134217727 ms (~4h - ~1d)
 118  *  8    512  16777216 ms (~4h)  134217728 ms - 1073741822 ms (~1d - ~12d)
 119  *
 120  * HZ  300
 121  * Level Offset  Granularity            Range
 122  *  0      0         3 ms                0 ms -        210 ms
 123  *  1     64        26 ms              213 ms -       1703 ms (213ms - ~1s)
 124  *  2    128       213 ms             1706 ms -      13650 ms (~1s - ~13s)
 125  *  3    192      1706 ms (~1s)      13653 ms -     109223 ms (~13s - ~1m)

 279 static unsigned long round_jiffies_common(unsigned long j, int cpu,
     /*   */
 280                 bool force_up)
 281 {
 282         int rem;
 283         unsigned long original = j;
 284 
 285         /*
 286          * We don't want all cpus firing their timers at once hitting the
 287          * same lock or cachelines, so we skew each extra cpu with an extra
 288          * 3 jiffies. This 3 jiffies came originally from the mm/ code which
 289          * already did this.
 290          * The skew is done by adding 3*cpunr, then round, then subtract this
 291          * extra offset again.
 292          */
 293         j += cpu * 3;
 294 
 295         rem = j % HZ;
 296 
 297         /*
 298          * If the target jiffie is just after a whole second (which can happen
 299          * due to delays of the timer irq, long irq off times etc etc) then
 300          * we should round down to the whole second, not up. Use 1/4th second
 301          * as cutoff for this rounding as an extreme upper bound for this.
 302          * But never round down if @force_up is set.
 303          */
 304         if (rem < HZ/4 && !force_up) /* round down */
 305                 j = j - rem;
 306         else /* round up */
 307                 j = j - rem + HZ;
 308 
 309         /* now that we have rounded, subtract the extra skew again */
 310         j -= cpu * 3;
 311 
 312         /*
 313          * Make sure j is still in the future. Otherwise return the
 314          * unmodified value.
 315          */
 316         return time_is_after_jiffies(j) ? j : original;
 317 }

 319 /**
 320  * __round_jiffies - function to round jiffies to a full second
 321  * @j: the time in (absolute) jiffies that should be rounded
 322  * @cpu: the processor number on which the timeout will happen
 323  *
 324  * __round_jiffies() rounds an absolute time in the future (in jiffies)
 325  * up or down to (approximately) full seconds. This is useful for timers
 326  * for which the exact time they fire does not matter too much, as long as
 327  * they fire approximately every X seconds.
 328  *
 329  * By rounding these timers to whole seconds, all such timers will fire
 330  * at the same time, rather than at various times spread out. The goal
 331  * of this is to have the CPU wake up less, which saves power.
 332  *
 333  * The exact rounding is skewed for each processor to avoid all
 334  * processors firing at the exact same time, which could lead
 335  * to lock contention or spurious cache line bouncing.
 336  *
 337  * The return value is the rounded version of the @j parameter.
 338  */
 339 unsigned long __round_jiffies(unsigned long j, int cpu)
     /*   */
 340 {
 341         return round_jiffies_common(j, cpu, false);
 342 }

 493 /*
 494  * Helper function to calculate the array index for a given expiry
 495  * time.
 496  */
 497 static inline unsigned calc_index(unsigned long expires, unsigned lvl,
     /*   */
 498                                   unsigned long *bucket_expiry)
 499 {
 500 
 501         /*
 502          * The timer wheel has to guarantee that a timer does not fire
 503          * early. Early expiry can happen due to:
 504          * - Timer is armed at the edge of a tick
 505          * - Truncation of the expiry time in the outer wheel levels
 506          *
 507          * Round up with level granularity to prevent this.
 508          */
 509         trace_android_vh_timer_calc_index(lvl, &expires);
 510         expires = (expires + LVL_GRAN(lvl)) >> LVL_SHIFT(lvl);
 511         *bucket_expiry = expires << LVL_SHIFT(lvl);
 512         return LVL_OFFS(lvl) + (expires & LVL_MASK);
 513 }
 514 
 515 static int calc_wheel_index(unsigned long expires, unsigned long clk,
     /*   */
 516                             unsigned long *bucket_expiry)
 517 {
 518         unsigned long delta = expires - clk;
 519         unsigned int idx;
 520 
 521         if (delta < LVL_START(1)) {
 522                 idx = calc_index(expires, 0, bucket_expiry);
 523         } else if (delta < LVL_START(2)) {
 524                 idx = calc_index(expires, 1, bucket_expiry);
 525         } else if (delta < LVL_START(3)) {
 526                 idx = calc_index(expires, 2, bucket_expiry);
 527         } else if (delta < LVL_START(4)) {
 528                 idx = calc_index(expires, 3, bucket_expiry);
 529         } else if (delta < LVL_START(5)) {
 530                 idx = calc_index(expires, 4, bucket_expiry);
 531         } else if (delta < LVL_START(6)) {
 532                 idx = calc_index(expires, 5, bucket_expiry);
 533         } else if (delta < LVL_START(7)) {
 534                 idx = calc_index(expires, 6, bucket_expiry);
 535         } else if (LVL_DEPTH > 8 && delta < LVL_START(8)) {
 536                 idx = calc_index(expires, 7, bucket_expiry);
 537         } else if ((long) delta < 0) {
 538                 idx = clk & LVL_MASK;
 539                 *bucket_expiry = clk;
 540         } else {
 541                 /*
 542                  * Force expire obscene large timeouts to expire at the
 543                  * capacity limit of the wheel.
 544                  */
 545                 if (delta >= WHEEL_TIMEOUT_CUTOFF)
 546                         expires = clk + WHEEL_TIMEOUT_MAX;
 547 
 548                 idx = calc_index(expires, LVL_DEPTH - 1, bucket_expiry);
 549         }
 550         return idx;
 551 }

 553 static void
 554 trigger_dyntick_cpu(struct timer_base *base, struct timer_list *timer)
     /*   */
 555 {
 556         if (!is_timers_nohz_active())
 557                 return;
 558 
 559         /*
 560          * TODO: This wants some optimizing similar to the code below, but we
 561          * will do that when we switch from push to pull for deferrable timers.
 562          */
 563         if (timer->flags & TIMER_DEFERRABLE) {
 564                 if (tick_nohz_full_cpu(base->cpu))
 565                         wake_up_nohz_cpu(base->cpu);
 566                 return;
 567         }
 568 
 569         /*
 570          * We might have to IPI the remote CPU if the base is idle and the
 571          * timer is not deferrable. If the other CPU is on the way to idle
 572          * then it can't set base->is_idle as we hold the base lock:
 573          */
 574         if (base->is_idle)
 575                 wake_up_nohz_cpu(base->cpu);
 576 }

 578 /*
 579  * Enqueue the timer into the hash bucket, mark it pending in
 580  * the bitmap, store the index in the timer flags then wake up
 581  * the target CPU if needed.
 582  */
 583 static void enqueue_timer(struct timer_base *base, struct timer_list *timer,
     /*   */
 584                           unsigned int idx, unsigned long bucket_expiry)
 585 {
 586 
 587         hlist_add_head(&timer->entry, base->vectors + idx);
 588         __set_bit(idx, base->pending_map);
 589         timer_set_idx(timer, idx);
 590 
 591         trace_timer_start(timer, timer->expires, timer->flags);
 592 
 593         /*
 594          * Check whether this is the new first expiring timer. The
 595          * effective expiry time of the timer is required here
 596          * (bucket_expiry) instead of timer->expires.
 597          */
 598         if (time_before(bucket_expiry, base->next_expiry)) {
 599                 /*
 600                  * Set the next expiry time and kick the CPU so it
 601                  * can reevaluate the wheel:
 602                  */
 603                 base->next_expiry = bucket_expiry;
 604                 base->next_expiry_recalc = false;
 605                 trigger_dyntick_cpu(base, timer);
 606         }
 607 }

 609 static void internal_add_timer(struct timer_base *base, struct timer_list *timer)
     /*   */
 610 {
 611         unsigned long bucket_expiry;
 612         unsigned int idx;
 613 
 614         idx = calc_wheel_index(timer->expires, base->clk, &bucket_expiry);
 615         enqueue_timer(base, timer, idx, bucket_expiry);
 616 }

 896 static inline void forward_timer_base(struct timer_base *base)
     /*   */
 897 {
 898         unsigned long jnow = READ_ONCE(jiffies);
 899 
 900         /*
 901          * No need to forward if we are close enough below jiffies.
 902          * Also while executing timers, base->clk is 1 offset ahead
 903          * of jiffies to avoid endless requeuing to current jffies.
 904          */
 905         if ((long)(jnow - base->clk) < 1)
 906                 return;
 907 
 908         /*
 909          * If the next expiry value is > jiffies, then we fast forward to
 910          * jiffies otherwise we forward to the next expiry value.
 911          */
 912         if (time_after(base->next_expiry, jnow)) {
 913                 base->clk = jnow;
 914         } else {
 915                 if (WARN_ON_ONCE(time_before(base->next_expiry, base->clk)))
 916                         return;
 917                 base->clk = base->next_expiry;
 918         }
 919 }

 922 /*
 923  * We are using hashed locking: Holding per_cpu(timer_bases[x]).lock means
 924  * that all timers which are tied to this base are locked, and the base itself
 925  * is locked too.
 926  *
 927  * So __run_timers/migrate_timers can safely modify all timers which could
 928  * be found in the base->vectors array.
 929  *
 930  * When a timer is migrating then the TIMER_MIGRATING flag is set and we need
 931  * to wait until the migration is done.
 932  */
 933 static struct timer_base *lock_timer_base(struct timer_list *timer,
     /*   */
 934                                           unsigned long *flags)
 935         __acquires(timer->base->lock)
 936 {
 937         for (;;) {
 938                 struct timer_base *base;
 939                 u32 tf;
 940 
 941                 /*
 942                  * We need to use READ_ONCE() here, otherwise the compiler
 943                  * might re-read @tf between the check for TIMER_MIGRATING
 944                  * and spin_lock().
 945                  */
 946                 tf = READ_ONCE(timer->flags);
 947 
 948                 if (!(tf & TIMER_MIGRATING)) {
 949                         base = get_timer_base(tf);
 950                         raw_spin_lock_irqsave(&base->lock, *flags);
 951                         if (timer->flags == tf)
 952                                 return base;
 953                         raw_spin_unlock_irqrestore(&base->lock, *flags);
 954                 }
 955                 cpu_relax();
 956         }
 957 }

 959 #define MOD_TIMER_PENDING_ONLY          0x01
 960 #define MOD_TIMER_REDUCE                0x02
 961 #define MOD_TIMER_NOTPENDING            0x04
 962 
 963 static inline int
 964 __mod_timer(struct timer_list *timer, unsigned long expires, unsigned int options)
     /*   */
 965 {
 966         unsigned long clk = 0, flags, bucket_expiry;
 967         struct timer_base *base, *new_base;
 968         unsigned int idx = UINT_MAX;
 969         int ret = 0;
 970 
 971         BUG_ON(!timer->function);
 972 
 973         /*
 974          * This is a common optimization triggered by the networking code - if
 975          * the timer is re-modified to have the same timeout or ends up in the
 976          * same array bucket then just return:
 977          */
 978         if (!(options & MOD_TIMER_NOTPENDING) && timer_pending(timer)) {
 979                 /*
 980                  * The downside of this optimization is that it can result in
 981                  * larger granularity than you would get from adding a new
 982                  * timer with this expiry.
 983                  */
 984                 long diff = timer->expires - expires;
 985 
 986                 if (!diff)
 987                         return 1;
 988                 if (options & MOD_TIMER_REDUCE && diff <= 0)
 989                         return 1;
 990 
 991                 /*
 992                  * We lock timer base and calculate the bucket index right
 993                  * here. If the timer ends up in the same bucket, then we
 994                  * just update the expiry time and avoid the whole
 995                  * dequeue/enqueue dance.
 996                  */
 997                 base = lock_timer_base(timer, &flags);
 998                 forward_timer_base(base);
 999 
1000                 if (timer_pending(timer) && (options & MOD_TIMER_REDUCE) &&
1001                     time_before_eq(timer->expires, expires)) {
1002                         ret = 1;
1003                         goto out_unlock;
1004                 }
1005 
1006                 clk = base->clk;
1007                 idx = calc_wheel_index(expires, clk, &bucket_expiry);
1008 
1009                 /*
1010                  * Retrieve and compare the array index of the pending
1011                  * timer. If it matches set the expiry to the new value so a
1012                  * subsequent call will exit in the expires check above.
1013                  */
1014                 if (idx == timer_get_idx(timer)) {
1015                         if (!(options & MOD_TIMER_REDUCE))
1016                                 timer->expires = expires;
1017                         else if (time_after(timer->expires, expires))
1018                                 timer->expires = expires;
1019                         ret = 1;
1020                         goto out_unlock;
1021                 }
1022         } else {
1023                 base = lock_timer_base(timer, &flags);
1024                 forward_timer_base(base);
1025         }
1026 
1027         ret = detach_if_pending(timer, base, false);
1028         if (!ret && (options & MOD_TIMER_PENDING_ONLY))
1029                 goto out_unlock;
1030 
1031         new_base = get_target_base(base, timer->flags);
1032 
1033         if (base != new_base) {
1034                 /*
1035                  * We are trying to schedule the timer on the new base.
1036                  * However we can't change timer's base while it is running,
1037                  * otherwise del_timer_sync() can't detect that the timer's
1038                  * handler yet has not finished. This also guarantees that the
1039                  * timer is serialized wrt itself.
1040                  */
1041                 if (likely(base->running_timer != timer)) {
1042                         /* See the comment in lock_timer_base() */
1043                         timer->flags |= TIMER_MIGRATING;
1044 
1045                         raw_spin_unlock(&base->lock);
1046                         base = new_base;
1047                         raw_spin_lock(&base->lock);
1048                         WRITE_ONCE(timer->flags,
1049                                    (timer->flags & ~TIMER_BASEMASK) | base->cpu);
1050                         forward_timer_base(base);
1051                 }
1052         }
1053 
1054         debug_timer_activate(timer);
1055 
1056         timer->expires = expires;
1057         /*
1058          * If 'idx' was calculated above and the base time did not advance
1059          * between calculating 'idx' and possibly switching the base, only
1060          * enqueue_timer() is required. Otherwise we need to (re)calculate
1061          * the wheel index via internal_add_timer().
1062          */
1063         if (idx != UINT_MAX && clk == base->clk)
1064                 enqueue_timer(base, timer, idx, bucket_expiry);
1065         else
1066                 internal_add_timer(base, timer);
1067 
1068 out_unlock:
1069         raw_spin_unlock_irqrestore(&base->lock, flags);
1070 
1071         return ret;
1072 }

1320  * @timer: the timer to be deactivated

1324  * CPUs.

1327  * otherwise this function is meaningless. It must not be called from
1328  * interrupt contexts unless the timer is an irqsafe one. The caller must
1329  * not hold locks which would prevent completion of the timer's
1330  * handler. The timer's handler must not call add_timer_on(). Upon exit the
1331  * timer is not queued and the handler is not running on any CPU.

1334  *   interrupt context while calling this function. Even if the lock has

1337  *    CPU0                             CPU1
1338  *    ----                             ----
1339  *                                     <SOFTIRQ>
1340  *                                       call_timer_fn();
1341  *                                       base->running_timer = mytimer;
1342  *    spin_lock_irq(somelock);
1343  *                                     <IRQ>
1344  *                                        spin_lock(somelock);

1346  *    while (base->running_timer == mytimer);
1347  *

1357 
1358 #ifdef CONFIG_LOCKDEP

1360

1362          * If lockdep gives a backtrace here, please reference
1363          * the synchronization rules above.
1364          */
1365         local_irq_save(flags);
1366         lock_map_acquire(&timer->lockdep_map);
1367         lock_map_release(&timer->lockdep_map);
1368         local_irq_restore(flags);
1369 #endif
1370         /*
1371          * don't use it in hardirq context, because it
1372          * could lead to deadlock.
1373          */
1374         WARN_ON(in_irq() && !(timer->flags & TIMER_IRQSAFE));
1375 
1376         do {
1377                 ret = try_to_del_timer_sync(timer);
1378 
1379                 if (unlikely(ret < 0)) {
1380                         del_timer_wait_running(timer);
1381                         cpu_relax();
1382                 }
1383         } while (ret < 0);
1384

1390 static void call_timer_fn(struct timer_list *timer,
     /*   */
1391                           void (*fn)(struct timer_list *),
1392                           unsigned long baseclk)
1393 {
1394         int count = preempt_count();
1395 
1396 #ifdef CONFIG_LOCKDEP
1397         /*
1398          * It is permissible to free the timer from inside the
1399          * function that is called from it, this we need to take into
1400          * account for lockdep too. To avoid bogus "held lock freed"
1401          * warnings as well as problems when looking into
1402          * timer->lockdep_map, make a copy and use that here.
1403          */

1405 
1406         lockdep_copy_map(&lockdep_map, &timer->lockdep_map);
1407 #endif
1408         /*
1409          * Couple the lock chain with the lock chain at
1410          * del_timer_sync() by acquiring the lock_map around the fn()
1411          * call here and in del_timer_sync().
1412          */
1413         lock_map_acquire(&lockdep_map);
1414 
1415         trace_timer_expire_entry(timer, baseclk);
1416         fn(timer);
1417         trace_timer_expire_exit(timer);
1418 
1419         lock_map_release(&lockdep_map);
1420 
1421         if (count != preempt_count()) {
1422                 WARN_ONCE(1, "timer: %pS preempt leak: %08x -> %08x\n",

1425                  * Restore the preempt count. That gives us a decent
1426                  * chance to survive and extract information. If the
1427                  * callback kept a lock held, bad luck, but not worse
1428                  * than the BUG() we had.
1429                  */
1430                 preempt_count_set(count);
1431         }

1469 static int collect_expired_timers(struct timer_base *base,
     /*   */
1470                                   struct hlist_head *heads)
1471 {
1472         unsigned long clk = base->clk = base->next_expiry;
1473         struct hlist_head *vec;
1474         int i, levels = 0;
1475         unsigned int idx;
1476 
1477         for (i = 0; i < LVL_DEPTH; i++) {
1478                 idx = (clk & LVL_MASK) + i * LVL_SIZE;
1479 
1480                 if (__test_and_clear_bit(idx, base->pending_map)) {
1481                         vec = base->vectors + idx;
1482                         hlist_move_list(vec, heads++);
1483                         levels++;
1484                 }
1485                 /* Is it time to look at the next level? */
1486                 if (clk & LVL_CLK_MASK)
1487                         break;
1488                 /* Shift clock for the next level granularity */
1489                 clk >>= LVL_CLK_SHIFT;
1490         }
1491         return levels;
1492 }

1494 /*
1495  * Find the next pending bucket of a level. Search from level start (@offset)
1496  * + @clk upwards and if nothing there, search from start of the level
1497  * (@offset) up to @offset + clk.
1498  */
1499 static int next_pending_bucket(struct timer_base *base, unsigned offset,
     /*   */
1500                                unsigned clk)
1501 {
1502         unsigned pos, start = offset + clk;
1503         unsigned end = offset + LVL_SIZE;
1504 
1505         pos = find_next_bit(base->pending_map, end, start);
1506         if (pos < end)
1507                 return pos - start;
1508 
1509         pos = find_next_bit(base->pending_map, start, offset);
1510         return pos < start ? pos + LVL_SIZE - start : -1;
1511 }

1513 /*
1514  * Search the first expiring timer in the various clock levels. Caller must
1515  * hold base->lock.
1516  */
1517 static unsigned long __next_timer_interrupt(struct timer_base *base)
     /*   */
1518 {
1519         unsigned long clk, next, adj;
1520         unsigned lvl, offset = 0;
1521 
1522         next = base->clk + NEXT_TIMER_MAX_DELTA;
1523         clk = base->clk;
1524         for (lvl = 0; lvl < LVL_DEPTH; lvl++, offset += LVL_SIZE) {
1525                 int pos = next_pending_bucket(base, offset, clk & LVL_MASK);
1526                 unsigned long lvl_clk = clk & LVL_CLK_MASK;
1527 
1528                 if (pos >= 0) {
1529                         unsigned long tmp = clk + (unsigned long) pos;
1530 
1531                         tmp <<= LVL_SHIFT(lvl);
1532                         if (time_before(tmp, next))
1533                                 next = tmp;
1534 
1535                         /*
1536                          * If the next expiration happens before we reach
1537                          * the next level, no need to check further.
1538                          */
1539                         if (pos <= ((LVL_CLK_DIV - lvl_clk) & LVL_CLK_MASK))
1540                                 break;
1541                 }
1542                 /*
1543                  * Clock for the next level. If the current level clock lower
1544                  * bits are zero, we look at the next level as is. If not we
1545                  * need to advance it by one because that's going to be the
1546                  * next expiring bucket in that level. base->clk is the next
1547                  * expiring jiffie. So in case of:
1548                  *
1549                  * LVL5 LVL4 LVL3 LVL2 LVL1 LVL0
1550                  *  0    0    0    0    0    0
1551                  *
1552                  * we have to look at all levels @index 0. With
1553                  *
1554                  * LVL5 LVL4 LVL3 LVL2 LVL1 LVL0
1555                  *  0    0    0    0    0    2
1556                  *
1557                  * LVL0 has the next expiring bucket @index 2. The upper
1558                  * levels have the next expiring bucket @index 1.
1559                  *
1560                  * In case that the propagation wraps the next level the same
1561                  * rules apply:
1562                  *
1563                  * LVL5 LVL4 LVL3 LVL2 LVL1 LVL0
1564                  *  0    0    0    0    F    2
1565                  *
1566                  * So after looking at LVL0 we get:
1567                  *
1568                  * LVL5 LVL4 LVL3 LVL2 LVL1
1569                  *  0    0    0    1    0
1570                  *
1571                  * So no propagation from LVL1 to LVL2 because that happened
1572                  * with the add already, but then we need to propagate further
1573                  * from LVL2 to LVL3.
1574                  *
1575                  * So the simple check whether the lower bits of the current
1576                  * level are 0 or not is sufficient for all cases.
1577                  */
1578                 adj = lvl_clk ? 1 : 0;
1579                 clk >>= LVL_CLK_SHIFT;
1580                 clk += adj;
1581         }
1582 
1583         base->next_expiry_recalc = false;
1584 
1585         return next;
1586 }

1622 /**
1623  * get_next_timer_interrupt - return the time (clock mono) of the next timer
1624  * @basej:      base time jiffies
1625  * @basem:      base time clock monotonic
1626  *
1627  * Returns the tick aligned clock monotonic time of the next pending
1628  * timer or KTIME_MAX if no timer is pending.
1629  */
1630 u64 get_next_timer_interrupt(unsigned long basej, u64 basem)
     /*   */
1631 {
1632         struct timer_base *base = this_cpu_ptr(&timer_bases[BASE_STD]);
1633         u64 expires = KTIME_MAX;
1634         unsigned long nextevt;
1635         bool is_max_delta;
1636 
1637         /*
1638          * Pretend that there is no timer pending if the cpu is offline.
1639          * Possible pending timers will be migrated later to an active cpu.
1640          */
1641         if (cpu_is_offline(smp_processor_id()))
1642                 return expires;
1643 
1644         raw_spin_lock(&base->lock);
1645         if (base->next_expiry_recalc)
1646                 base->next_expiry = __next_timer_interrupt(base);
1647         nextevt = base->next_expiry;
1648         is_max_delta = (nextevt == base->clk + NEXT_TIMER_MAX_DELTA);
1649 
1650         /*
1651          * We have a fresh next event. Check whether we can forward the
1652          * base. We can only do that when @basej is past base->clk
1653          * otherwise we might rewind base->clk.
1654          */
1655         if (time_after(basej, base->clk)) {
1656                 if (time_after(nextevt, basej))
1657                         base->clk = basej;
1658                 else if (time_after(nextevt, base->clk))
1659                         base->clk = nextevt;
1660         }
1661 
1662         if (time_before_eq(nextevt, basej)) {
1663                 expires = basem;
1664                 base->is_idle = false;
1665         } else {
1666                 if (!is_max_delta)
1667                         expires = basem + (u64)(nextevt - basej) * TICK_NSEC;
1668                 /*
1669                  * If we expect to sleep more than a tick, mark the base idle.
1670                  * Also the tick is stopped so any added timer must forward
1671                  * the base clk itself to keep granularity small. This idle
1672                  * logic is only maintained for the BASE_STD base, deferrable
1673                  * timers may still see large granularity skew (by design).
1674                  */
1675                 if ((expires - basem) > TICK_NSEC)
1676                         base->is_idle = true;
1677         }
1678         raw_spin_unlock(&base->lock);
1679 
1680         return cmp_next_hrtimer_event(basem, expires);
1681 }

1725 /**
1726  * __run_timers - run all expired timers (if any) on this CPU.
1727  * @base: the timer vector to be processed.
1728  */
1729 static inline void __run_timers(struct timer_base *base)
     /*   */
1730 {
1731         struct hlist_head heads[LVL_DEPTH];
1732         int levels;
1733 
1734         if (time_before(jiffies, base->next_expiry))
1735                 return;
1736 
1737         timer_base_lock_expiry(base);
1738         raw_spin_lock_irq(&base->lock);
1739 
1740         while (time_after_eq(jiffies, base->clk) &&
1741                time_after_eq(jiffies, base->next_expiry)) {
1742                 levels = collect_expired_timers(base, heads);
1743                 /*
1744                  * The only possible reason for not finding any expired
1745                  * timer at this clk is that all matching timers have been
1746                  * dequeued.
1747                  */
1748                 WARN_ON_ONCE(!levels && !base->next_expiry_recalc);
1749                 base->clk++;
1750                 base->next_expiry = __next_timer_interrupt(base);
1751 
1752                 while (levels--)
1753                         expire_timers(base, heads + levels);
1754         }
1755         raw_spin_unlock_irq(&base->lock);
1756         timer_base_unlock_expiry(base);
1757 }

1807 /**
1808  * schedule_timeout - sleep until timeout
1809  * @timeout: timeout value in jiffies
1810  *
1811  * Make the current task sleep until @timeout jiffies have elapsed.
1812  * The function behavior depends on the current task state
1813  * (see also set_current_state() description):
1814  *
1815  * %TASK_RUNNING - the scheduler is called, but the task does not sleep
1816  * at all. That happens because sched_submit_work() does nothing for
1817  * tasks in %TASK_RUNNING state.
1818  *
1819  * %TASK_UNINTERRUPTIBLE - at least @timeout jiffies are guaranteed to
1820  * pass before the routine returns unless the current task is explicitly
1821  * woken up, (e.g. by wake_up_process()).
1822  *
1823  * %TASK_INTERRUPTIBLE - the routine may return early if a signal is
1824  * delivered to the current task or the current task is explicitly woken
1825  * up.
1826  *
1827  * The current task state is guaranteed to be %TASK_RUNNING when this
1828  * routine returns.
1829  *
1830  * Specifying a @timeout value of %MAX_SCHEDULE_TIMEOUT will schedule
1831  * the CPU away without a bound on the timeout. In this case the return
1832  * value will be %MAX_SCHEDULE_TIMEOUT.
1833  *
1834  * Returns 0 when the timer has expired otherwise the remaining time in
1835  * jiffies will be returned. In all cases the return value is guaranteed
1836  * to be non-negative.
1837  */
1838 signed long __sched schedule_timeout(signed long timeout)
     /*   */
1839 {
1840         struct process_timer timer;
1841         unsigned long expire;
1842 
1843         switch (timeout)
1844         {
1845         case MAX_SCHEDULE_TIMEOUT:
1846                 /*
1847                  * These two special cases are useful to be comfortable
1848                  * in the caller. Nothing more. We could take
1849                  * MAX_SCHEDULE_TIMEOUT from one of the negative value
1850                  * but I' d like to return a valid offset (>=0) to allow
1851                  * the caller to do everything it want with the retval.
1852                  */
1853                 schedule();
1854                 goto out;
1855         default:
1856                 /*
1857                  * Another bit of PARANOID. Note that the retval will be
1858                  * 0 since no piece of kernel is supposed to do a check
1859                  * for a negative retval of schedule_timeout() (since it
1860                  * should never happens anyway). You just have the printk()
1861                  * that will tell you if something is gone wrong and where.
1862                  */
1863                 if (timeout < 0) {
1864                         printk(KERN_ERR "schedule_timeout: wrong timeout "
1865                                 "value %lx\n", timeout);
1866                         dump_stack();
1867                         current->state = TASK_RUNNING;
1868                         goto out;
1869                 }
1870         }
1871 
1872         expire = timeout + jiffies;
1873 
1874         timer.task = current;
1875         timer_setup_on_stack(&timer.timer, process_timeout, 0);
1876         __mod_timer(&timer.timer, expire, MOD_TIMER_NOTPENDING);
1877         schedule();
1878         del_singleshot_timer_sync(&timer.timer);
1879 
1880         /* Remove the timer from the object tracker */
1881         destroy_timer_on_stack(&timer.timer);
1882 
1883         timeout = expire - jiffies;
1884 
1885  out:
1886         return timeout < 0 ? 0 : timeout;
1887 }
1888 EXPORT_SYMBOL(schedule_timeout);

     /*   */
1955 {
1956         struct timer_base *old_base;
1957         struct timer_base *new_base;

1984                 raw_spin_unlock_irq(&new_base->lock);

2051 /**
2052  * usleep_range - Sleep for an approximate time
2053  * @min: Minimum time in usecs to sleep
2054  * @max: Maximum time in usecs to sleep
2055  *
2056  * In non-atomic context where the exact wakeup time is flexible, use
2057  * usleep_range() instead of udelay().  The sleep improves responsiveness
2058  * by avoiding the CPU-hogging busy-wait of udelay(), and the range reduces
2059  * power usage by allowing hrtimers to take advantage of an already-
2060  * scheduled interrupt instead of scheduling a new one just for this sleep.
2061  */
2062 void __sched usleep_range(unsigned long min, unsigned long max)
     /*   */
2063 {
2064         ktime_t exp = ktime_add_us(ktime_get(), min);
2065         u64 delta = (u64)(max - min) * NSEC_PER_USEC;
2066 
2067         for (;;) {
2068                 __set_current_state(TASK_UNINTERRUPTIBLE);
2069                 /* Do not return before the requested sleep time has elapsed */
2070                 if (!schedule_hrtimeout_range(&exp, delta, HRTIMER_MODE_ABS))
2071                         break;
2072         }
2073 }
2074 EXPORT_SYMBOL(usleep_range);
