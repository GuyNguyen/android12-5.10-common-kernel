  90 /* scan for the sequence $<data>#<checksum> */
  91 static void get_packet(char *buffer)
  92 {
  93         unsigned char checksum;
  94         unsigned char xmitcsum;
  95         int count;
  96         char ch;
  97 
  98         do {
  99                 /*
 100                  * Spin and wait around for the start character, ignore all
 101                  * other characters:
 102                  */
 103                 while ((ch = (gdbstub_read_wait())) != '$')
 104                         /* nothing */;
 105 
 106                 kgdb_connected = 1;
 107                 checksum = 0;
 108                 xmitcsum = -1;
 109 
 110                 count = 0;
 111 
 112                 /*
 113                  * now, read until a # or end of buffer is found:
 114                  */
 115                 while (count < (BUFMAX - 1)) {
 116                         ch = gdbstub_read_wait();
 117                         if (ch == '#')
 118                                 break;
 119                         checksum = checksum + ch;
 120                         buffer[count] = ch;
 121                         count = count + 1;
 122                 }
 123 
 124                 if (ch == '#') {
 125                         xmitcsum = hex_to_bin(gdbstub_read_wait()) << 4;
 126                         xmitcsum += hex_to_bin(gdbstub_read_wait());
 127 
 128                         if (checksum != xmitcsum)
 129                                 /* failed checksum */
 130                                 dbg_io_ops->write_char('-');
 131                         else
 132                                 /* successful transfer */
 133                                 dbg_io_ops->write_char('+');
 134                         if (dbg_io_ops->flush)
 135                                 dbg_io_ops->flush();
 136                 }
 137                 buffer[count] = 0;
 138         } while (checksum != xmitcsum);
 139 }

 141 /*
 142  * Send the packet in buffer.
 143  * Check for gdb connection if asked for.
 144  */
 145 static void put_packet(char *buffer)
 146 {
 147         unsigned char checksum;
 148         int count;
 149         char ch;
 150 
 151         /*
 152          * $<packet info>#<checksum>.
 153          */
 154         while (1) {
 155                 dbg_io_ops->write_char('$');
 156                 checksum = 0;
 157                 count = 0;
 158 
 159                 while ((ch = buffer[count])) {
 160                         dbg_io_ops->write_char(ch);
 161                         checksum += ch;
 162                         count++;
 163                 }
 164 
 165                 dbg_io_ops->write_char('#');
 166                 dbg_io_ops->write_char(hex_asc_hi(checksum));
 167                 dbg_io_ops->write_char(hex_asc_lo(checksum));
 168                 if (dbg_io_ops->flush)
 169                         dbg_io_ops->flush();
 170 
 171                 /* Now see what we get in reply. */
 172                 ch = gdbstub_read_wait();
 173 
 174                 if (ch == 3)
 175                         ch = gdbstub_read_wait();
 176 
 177                 /* If we get an ACK, we are done. */
 178                 if (ch == '+')
 179                         return;
 180 
 181                 /*
 182                  * If we get the start of another packet, this means
 183                  * that GDB is attempting to reconnect.  We will NAK
 184                  * the packet being sent, and stop trying to send this
 185                  * packet.
 186                  */
 187                 if (ch == '$') {
 188                         dbg_io_ops->write_char('-');
 189                         if (dbg_io_ops->flush)
 190                                 dbg_io_ops->flush();
 191                         return;
 192                 }
 193         }
 194 }

 196 static char gdbmsgbuf[BUFMAX + 1];
 197 
 198 void gdbstub_msg_write(const char *s, int len)
 199 {
 200         char *bufptr;
 201         int wcount;
 202         int i;
 203 
 204         if (len == 0)
 205                 len = strlen(s);
 206 
 207         /* 'O'utput */
 208         gdbmsgbuf[0] = 'O';
 209 
 210         /* Fill and send buffers... */
 211         while (len > 0) {
 212                 bufptr = gdbmsgbuf + 1;
 213 
 214                 /* Calculate how many this time */
 215                 if ((len << 1) > (BUFMAX - 2))
 216                         wcount = (BUFMAX - 2) >> 1;
 217                 else
 218                         wcount = len;
 219 
 220                 /* Pack in hex chars */
 221                 for (i = 0; i < wcount; i++)
 222                         bufptr = hex_byte_pack(bufptr, s[i]);
 223                 *bufptr = '\0';
 224 
 225                 /* Move up */
 226                 s += wcount;
 227                 len -= wcount;
 228 
 229                 /* Write packet */
 230                 put_packet(gdbmsgbuf);
 231         }
 232 }

 234 /*
 235  * Convert the memory pointed to by mem into hex, placing result in
 236  * buf.  Return a pointer to the last char put in buf (null). May
 237  * return an error.
 238  */
 239 char *kgdb_mem2hex(char *mem, char *buf, int count)
 240 {
 241         char *tmp;
 242         int err;
 243 
 244         /*
 245          * We use the upper half of buf as an intermediate buffer for the
 246          * raw memory copy.  Hex conversion will work against this one.
 247          */
 248         tmp = buf + count;
 249 
 250         err = copy_from_kernel_nofault(tmp, mem, count);
 251         if (err)
 252                 return NULL;
 253         while (count > 0) {
 254                 buf = hex_byte_pack(buf, *tmp);
 255                 tmp++;
 256                 count--;
 257         }
 258         *buf = 0;
 259 
 260         return buf;
 261 }

 263 /*
 264  * Convert the hex array pointed to by buf into binary to be placed in
 265  * mem.  Return a pointer to the character AFTER the last byte
 266  * written.  May return an error.
 267  */
 268 int kgdb_hex2mem(char *buf, char *mem, int count)
 269 {
 270         char *tmp_raw;
 271         char *tmp_hex;
 272 
 273         /*
 274          * We use the upper half of buf as an intermediate buffer for the
 275          * raw memory that is converted from hex.
 276          */
 277         tmp_raw = buf + count * 2;
 278 
 279         tmp_hex = tmp_raw - 1;
 280         while (tmp_hex >= buf) {
 281                 tmp_raw--;
 282                 *tmp_raw = hex_to_bin(*tmp_hex--);
 283                 *tmp_raw |= hex_to_bin(*tmp_hex--) << 4;
 284         }
 285 
 286         return copy_to_kernel_nofault(mem, tmp_raw, count);
 287 }

 289 /*
 290  * While we find nice hex chars, build a long_val.
 291  * Return number of chars processed.
 292  */
 293 int kgdb_hex2long(char **ptr, unsigned long *long_val)
 294 {
 295         int hex_val;
 296         int num = 0;
 297         int negate = 0;
 298 
 299         *long_val = 0;
 300 
 301         if (**ptr == '-') {
 302                 negate = 1;
 303                 (*ptr)++;
 304         }
 305         while (**ptr) {
 306                 hex_val = hex_to_bin(**ptr);
 307                 if (hex_val < 0)
 308                         break;
 309 
 310                 *long_val = (*long_val << 4) | hex_val;
 311                 num++;
 312                 (*ptr)++;
 313         }
 314 
 315         if (negate)
 316                 *long_val = -*long_val;
 317 
 318         return num;
 319 }

 321 /*
 322  * Copy the binary array pointed to by buf into mem.  Fix $, #, and
 323  * 0x7d escaped with 0x7d. Return -EFAULT on failure or 0 on success.
 324  * The input buf is overwitten with the result to write to mem.
 325  */
 326 static int kgdb_ebin2mem(char *buf, char *mem, int count)
 327 {
 328         int size = 0;
 329         char *c = buf;
 330 
 331         while (count-- > 0) {
 332                 c[size] = *buf++;
 333                 if (c[size] == 0x7d)
 334                         c[size] = *buf++ ^ 0x20;
 335                 size++;
 336         }
 337 
 338         return copy_to_kernel_nofault(mem, c, size);
 339 }

 367 /* Write memory due to an 'M' or 'X' packet. */
 368 static int write_mem_msg(int binary)
 369 {
 370         char *ptr = &remcom_in_buffer[1];
 371         unsigned long addr;
 372         unsigned long length;
 373         int err;
 374 
 375         if (kgdb_hex2long(&ptr, &addr) > 0 && *(ptr++) == ',' &&
 376             kgdb_hex2long(&ptr, &length) > 0 && *(ptr++) == ':') {
 377                 if (binary)
 378                         err = kgdb_ebin2mem(ptr, (char *)addr, length);
 379                 else
 380                         err = kgdb_hex2mem(ptr, (char *)addr, length);
 381                 if (err)
 382                         return err;
 383                 if (CACHE_FLUSH_IS_SAFE)
 384                         flush_icache_range(addr, addr + length);
 385                 return 0;
 386         }
 387 
 388         return -EINVAL;
 389 }

 391 static void error_packet(char *pkt, int error)
 392 {
 393         error = -error;
 394         pkt[0] = 'E';
 395         pkt[1] = hex_asc[(error / 10)];
 396         pkt[2] = hex_asc[(error % 10)];
 397         pkt[3] = '\0';
 398 }

 400 /*
 401  * Thread ID accessors. We represent a flat TID space to GDB, where
 402  * the per CPU idle threads (which under Linux all have PID 0) are
 403  * remapped to negative TIDs.
 404  */
 405 
 406 #define BUF_THREAD_ID_SIZE      8
 407 
 408 static char *pack_threadid(char *pkt, unsigned char *id)
 409 {
 410         unsigned char *limit;
 411         int lzero = 1;
 412 
 413         limit = id + (BUF_THREAD_ID_SIZE / 2);
 414         while (id < limit) {
 415                 if (!lzero || *id != 0) {
 416                         pkt = hex_byte_pack(pkt, *id);
 417                         lzero = 0;
 418                 }
 419                 id++;
 420         }
 421 
 422         if (lzero)
 423                 pkt = hex_byte_pack(pkt, 0);
 424 
 425         return pkt;
 426 }

 433 static struct task_struct *getthread(struct pt_regs *regs, int tid)
 434 {
 435         /*
 436          * Non-positive TIDs are remapped to the cpu shadow information
 437          */
 438         if (tid == 0 || tid == -1)
 439                 tid = -atomic_read(&kgdb_active) - 2;
 440         if (tid < -1 && tid > -NR_CPUS - 2) {
 441                 if (kgdb_info[-tid - 2].task)
 442                         return kgdb_info[-tid - 2].task;
 443                 else
 444                         return idle_task(-tid - 2);
 445         }
 446         if (tid <= 0) {
 447                 printk(KERN_ERR "KGDB: Internal thread select error\n");
 448                 dump_stack();
 449                 return NULL;
 450         }
 451 
 452         /*
 453          * find_task_by_pid_ns() does not take the tasklist lock anymore
 454          * but is nicely RCU locked - hence is a pretty resilient
 455          * thing to use:
 456          */
 457         return find_task_by_pid_ns(tid, &init_pid_ns);
 458 }

 494 static void gdb_get_regs_helper(struct kgdb_state *ks)
 495 {
 496         struct task_struct *thread;
 497         void *local_debuggerinfo;
 498         int i;
 499 
 500         thread = kgdb_usethread;
 501         if (!thread) {
 502                 thread = kgdb_info[ks->cpu].task;
 503                 local_debuggerinfo = kgdb_info[ks->cpu].debuggerinfo;
 504         } else {
 505                 local_debuggerinfo = NULL;
 506                 for_each_online_cpu(i) {
 507                         /*
 508                          * Try to find the task on some other
 509                          * or possibly this node if we do not
 510                          * find the matching task then we try
 511                          * to approximate the results.
 512                          */
 513                         if (thread == kgdb_info[i].task)
 514                                 local_debuggerinfo = kgdb_info[i].debuggerinfo;
 515                 }
 516         }
 517 
 518         /*
 519          * All threads that don't have debuggerinfo should be
 520          * in schedule() sleeping, since all other CPUs
 521          * are in kgdb_wait, and thus have debuggerinfo.
 522          */
 523         if (local_debuggerinfo) {
 524                 pt_regs_to_gdb_regs(gdb_regs, local_debuggerinfo);
 525         } else {
 526                 /*
 527                  * Pull stuff saved during switch_to; nothing
 528                  * else is accessible (or even particularly
 529                  * relevant).
 530                  *
 531                  * This should be enough for a stack trace.
 532                  */
 533                 sleeping_thread_to_gdb_regs(gdb_regs, thread);
 534         }
 535 }

 613 /* Handle the 'P' individual regster set */
 614 static void gdb_cmd_reg_set(struct kgdb_state *ks)
 615 {
 616         unsigned long regnum;
 617         char *ptr = &remcom_in_buffer[1];
 618         int i = 0;
 619 
 620         kgdb_hex2long(&ptr, &regnum);
 621         if (*ptr++ != '=' ||
 622             !(!kgdb_usethread || kgdb_usethread == current) ||
 623             !dbg_get_reg(regnum, gdb_regs, ks->linux_regs)) {
 624                 error_packet(remcom_out_buffer, -EINVAL);
 625                 return;
 626         }
 627         memset(gdb_regs, 0, sizeof(gdb_regs));
 628         while (i < sizeof(gdb_regs) * 2)
 629                 if (hex_to_bin(ptr[i]) >= 0)
 630                         i++;
 631                 else
 632                         break;
 633         i = i / 2;
 634         kgdb_hex2mem(ptr, (char *)gdb_regs, i);
 635         dbg_set_reg(regnum, gdb_regs, ks->linux_regs);
 636         strcpy(remcom_out_buffer, "OK");
 637 }

 651 /* Handle the 'D' or 'k', detach or kill packets */
 652 static void gdb_cmd_detachkill(struct kgdb_state *ks)
 653 {
 654         int error;
 655 
 656         /* The detach case */
 657         if (remcom_in_buffer[0] == 'D') {
 658                 error = dbg_remove_all_break();
 659                 if (error < 0) {
 660                         error_packet(remcom_out_buffer, error);
 661                 } else {
 662                         strcpy(remcom_out_buffer, "OK");
 663                         kgdb_connected = 0;
 664                 }
 665                 put_packet(remcom_out_buffer);
 666         } else {
 667                 /*
 668                  * Assume the kill case, with no exit code checking,
 669                  * trying to force detach the debugger:
 670                  */
 671                 dbg_remove_all_break();
 672                 kgdb_connected = 0;
 673         }
 674 }

 690                 kgdb_connected = 0;
 691 
 692                 return 1;
 693         }
 694         return 0;
 695 }

 697 /* Handle the 'q' query packets */
 698 static void gdb_cmd_query(struct kgdb_state *ks)
 699 {
 700         struct task_struct *g;
 701         struct task_struct *p;
 702         unsigned char thref[BUF_THREAD_ID_SIZE];
 703         char *ptr;
 704         int i;
 705         int cpu;
 706         int finished = 0;
 707 
 708         switch (remcom_in_buffer[1]) {
 709         case 's':
 710         case 'f':
 711                 if (memcmp(remcom_in_buffer + 2, "ThreadInfo", 10))
 712                         break;
 713 
 714                 i = 0;
 715                 remcom_out_buffer[0] = 'm';
 716                 ptr = remcom_out_buffer + 1;
 717                 if (remcom_in_buffer[1] == 'f') {
 718                         /* Each cpu is a shadow thread */
 719                         for_each_online_cpu(cpu) {
 720                                 ks->thr_query = 0;
 721                                 int_to_threadref(thref, -cpu - 2);
 722                                 ptr = pack_threadid(ptr, thref);
 723                                 *(ptr++) = ',';
 724                                 i++;
 725                         }
 726                 }
 727 
 728                 for_each_process_thread(g, p) {
 729                         if (i >= ks->thr_query && !finished) {
 730                                 int_to_threadref(thref, p->pid);
 731                                 ptr = pack_threadid(ptr, thref);
 732                                 *(ptr++) = ',';
 733                                 ks->thr_query++;
 734                                 if (ks->thr_query % KGDB_MAX_THREAD_QUERY == 0)
 735                                         finished = 1;
 736                         }
 737                         i++;
 738                 }
 739 
 740                 *(--ptr) = '\0';
 741                 break;
 742 
 743         case 'C':
 744                 /* Current thread id */
 745                 strcpy(remcom_out_buffer, "QC");
 746                 ks->threadid = shadow_pid(current->pid);
 747                 int_to_threadref(thref, ks->threadid);
 748                 pack_threadid(remcom_out_buffer + 2, thref);
 749                 break;
 750         case 'T':
 751                 if (memcmp(remcom_in_buffer + 1, "ThreadExtraInfo,", 16))
 752                         break;
 753 
 754                 ks->threadid = 0;
 755                 ptr = remcom_in_buffer + 17;
 756                 kgdb_hex2long(&ptr, &ks->threadid);
 757                 if (!getthread(ks->linux_regs, ks->threadid)) {
 758                         error_packet(remcom_out_buffer, -EINVAL);
 759                         break;
 760                 }
 761                 if ((int)ks->threadid > 0) {
 762                         kgdb_mem2hex(getthread(ks->linux_regs,
 763                                         ks->threadid)->comm,
 764                                         remcom_out_buffer, 16);
 765                 } else {
 766                         static char tmpstr[23 + BUF_THREAD_ID_SIZE];
 767 
 768                         sprintf(tmpstr, "shadowCPU%d",
 769                                         (int)(-ks->threadid - 2));
 770                         kgdb_mem2hex(tmpstr, remcom_out_buffer, strlen(tmpstr));
 771                 }
 772                 break;
 773 #ifdef CONFIG_KGDB_KDB
 774         case 'R':
 775                 if (strncmp(remcom_in_buffer, "qRcmd,", 6) == 0) {
 776                         int len = strlen(remcom_in_buffer + 6);
 777 
 778                         if ((len % 2) != 0) {
 779                                 strcpy(remcom_out_buffer, "E01");
 780                                 break;
 781                         }
 782                         kgdb_hex2mem(remcom_in_buffer + 6,
 783                                      remcom_out_buffer, len);
 784                         len = len / 2;
 785                         remcom_out_buffer[len++] = 0;
 786 
 787                         kdb_common_init_state(ks);
 788                         kdb_parse(remcom_out_buffer);
 789                         kdb_common_deinit_state();
 790 
 791                         strcpy(remcom_out_buffer, "OK");
 792                 }
 793                 break;
 794 #endif
 795 #ifdef CONFIG_HAVE_ARCH_KGDB_QXFER_PKT
 796         case 'S':
 797                 if (!strncmp(remcom_in_buffer, "qSupported:", 11))
 798                         strcpy(remcom_out_buffer, kgdb_arch_gdb_stub_feature);
 799                 break;
 800         case 'X':
 801                 if (!strncmp(remcom_in_buffer, "qXfer:", 6))
 802                         kgdb_arch_handle_qxfer_pkt(remcom_in_buffer,
 803                                                    remcom_out_buffer);
 804                 break;
 805 #endif
 806         default:
 807                 break;
 808         }
 809 }

 811 /* Handle the 'H' task query packets */
 812 static void gdb_cmd_task(struct kgdb_state *ks)
 813 {
 814         struct task_struct *thread;
 815         char *ptr;
 816 
 817         switch (remcom_in_buffer[1]) {
 818         case 'g':
 819                 ptr = &remcom_in_buffer[2];
 820                 kgdb_hex2long(&ptr, &ks->threadid);
 821                 thread = getthread(ks->linux_regs, ks->threadid);
 822                 if (!thread && ks->threadid > 0) {
 823                         error_packet(remcom_out_buffer, -EINVAL);
 824                         break;
 825                 }
 826                 kgdb_usethread = thread;
 827                 ks->kgdb_usethreadid = ks->threadid;
 828                 strcpy(remcom_out_buffer, "OK");
 829                 break;
 830         case 'c':
 831                 ptr = &remcom_in_buffer[2];
 832                 kgdb_hex2long(&ptr, &ks->threadid);
 833                 if (!ks->threadid) {
 834                         kgdb_contthread = NULL;
 835                 } else {
 836                         thread = getthread(ks->linux_regs, ks->threadid);
 837                         if (!thread && ks->threadid > 0) {
 838                                 error_packet(remcom_out_buffer, -EINVAL);
 839                                 break;
 840                         }
 841                         kgdb_contthread = thread;
 842                 }
 843                 strcpy(remcom_out_buffer, "OK");
 844                 break;
 845         }
 846 }

 862 /* Handle the 'z' or 'Z' breakpoint remove or set packets */
 863 static void gdb_cmd_break(struct kgdb_state *ks)
 864 {
 865         /*
 866          * Since GDB-5.3, it's been drafted that '0' is a software
 867          * breakpoint, '1' is a hardware breakpoint, so let's do that.
 868          */
 869         char *bpt_type = &remcom_in_buffer[1];
 870         char *ptr = &remcom_in_buffer[2];
 871         unsigned long addr;
 872         unsigned long length;
 873         int error = 0;
 874 
 875         if (arch_kgdb_ops.set_hw_breakpoint && *bpt_type >= '1') {
 876                 /* Unsupported */
 877                 if (*bpt_type > '4')
 878                         return;
 879         } else {
 880                 if (*bpt_type != '0' && *bpt_type != '1')
 881                         /* Unsupported. */
 882                         return;
 883         }
 884 
 885         /*
 886          * Test if this is a hardware breakpoint, and
 887          * if we support it:
 888          */
 889         if (*bpt_type == '1' && !(arch_kgdb_ops.flags & KGDB_HW_BREAKPOINT))
 890                 /* Unsupported. */
 891                 return;
 892 
 893         if (*(ptr++) != ',') {
 894                 error_packet(remcom_out_buffer, -EINVAL);
 895                 return;
 896         }
 897         if (!kgdb_hex2long(&ptr, &addr)) {
 898                 error_packet(remcom_out_buffer, -EINVAL);
 899                 return;
 900         }
 901         if (*(ptr++) != ',' ||
 902                 !kgdb_hex2long(&ptr, &length)) {
 903                 error_packet(remcom_out_buffer, -EINVAL);
 904                 return;
 905         }
 906 
 907         if (remcom_in_buffer[0] == 'Z' && *bpt_type == '0')
 908                 error = dbg_set_sw_break(addr);
 909         else if (remcom_in_buffer[0] == 'z' && *bpt_type == '0')
 910                 error = dbg_remove_sw_break(addr);
 911         else if (remcom_in_buffer[0] == 'Z')
 912                 error = arch_kgdb_ops.set_hw_breakpoint(addr,
 913                         (int)length, *bpt_type - '0');
 914         else if (remcom_in_buffer[0] == 'z')
 915                 error = arch_kgdb_ops.remove_hw_breakpoint(addr,
 916                         (int) length, *bpt_type - '0');
 917 
 918         if (error == 0)
 919                 strcpy(remcom_out_buffer, "OK");
 920         else
 921                 error_packet(remcom_out_buffer, error);
 922 }

 924 /* Handle the 'C' signal / exception passing packets */
 925 static int gdb_cmd_exception_pass(struct kgdb_state *ks)
 926 {
 927         /* C09 == pass exception
 928          * C15 == detach kgdb, pass exception
 929          */
 930         if (remcom_in_buffer[1] == '0' && remcom_in_buffer[2] == '9') {
 931 
 932                 ks->pass_exception = 1;
 933                 remcom_in_buffer[0] = 'c';
 934 
 935         } else if (remcom_in_buffer[1] == '1' && remcom_in_buffer[2] == '5') {
 936 
 937                 ks->pass_exception = 1;
 938                 remcom_in_buffer[0] = 'D';
 939                 dbg_remove_all_break();
 940                 kgdb_connected = 0;
 941                 return 1;
 942 
 943         } else {
 944                 gdbstub_msg_write("KGDB only knows signal 9 (pass)"
 945                         " and 15 (pass and disconnect)\n"
 946                         "Executing a continue without signal passing\n", 0);
 947                 remcom_in_buffer[0] = 'c';
 948         }
 949 
 950         /* Indicate fall through */
 951         return -1;
 952 }

 954 /*
 955  * This function performs all gdbserial command procesing
 956  */
 957 int gdb_serial_stub(struct kgdb_state *ks)
 958 {
 959         int error = 0;
 960         int tmp;
 961 
 962         /* Initialize comm buffer and globals. */
 963         memset(remcom_out_buffer, 0, sizeof(remcom_out_buffer));
 964         kgdb_usethread = kgdb_info[ks->cpu].task;
 965         ks->kgdb_usethreadid = shadow_pid(kgdb_info[ks->cpu].task->pid);
 966         ks->pass_exception = 0;
 967 
 968         if (kgdb_connected) {
 969                 unsigned char thref[BUF_THREAD_ID_SIZE];
 970                 char *ptr;
 971 
 972                 /* Reply to host that an exception has occurred */
 973                 ptr = remcom_out_buffer;
 974                 *ptr++ = 'T';
 975                 ptr = hex_byte_pack(ptr, ks->signo);
 976                 ptr += strlen(strcpy(ptr, "thread:"));
 977                 int_to_threadref(thref, shadow_pid(current->pid));
 978                 ptr = pack_threadid(ptr, thref);
 979                 *ptr++ = ';';
 980                 put_packet(remcom_out_buffer);
 981         }
 982 
 983         while (1) {
 984                 error = 0;
 985 
 986                 /* Clear the out buffer. */
 987                 memset(remcom_out_buffer, 0, sizeof(remcom_out_buffer));
 988 
 989                 get_packet(remcom_in_buffer);
 990 
 991                 switch (remcom_in_buffer[0]) {
 992                 case '?': /* gdbserial status */
 993                         gdb_cmd_status(ks);
 994                         break;
 995                 case 'g': /* return the value of the CPU registers */
 996                         gdb_cmd_getregs(ks);
 997                         break;
 998                 case 'G': /* set the value of the CPU registers - return OK */
 999                         gdb_cmd_setregs(ks);
1000                         break;
1001                 case 'm': /* mAA..AA,LLLL  Read LLLL bytes at address AA..AA */
1002                         gdb_cmd_memread(ks);
1003                         break;
1004                 case 'M': /* MAA..AA,LLLL: Write LLLL bytes at address AA..AA */
1005                         gdb_cmd_memwrite(ks);
1006                         break;
1007 #if DBG_MAX_REG_NUM > 0
1008                 case 'p': /* pXX Return gdb register XX (in hex) */
1009                         gdb_cmd_reg_get(ks);
1010                         break;
1011                 case 'P': /* PXX=aaaa Set gdb register XX to aaaa (in hex) */
1012                         gdb_cmd_reg_set(ks);
1013                         break;
1014 #endif /* DBG_MAX_REG_NUM > 0 */
1015                 case 'X': /* XAA..AA,LLLL: Write LLLL bytes at address AA..AA */
1016                         gdb_cmd_binwrite(ks);
1017                         break;
1018                         /* kill or detach. KGDB should treat this like a
1019                          * continue.
1020                          */
1021                 case 'D': /* Debugger detach */
1022                 case 'k': /* Debugger detach via kill */
1023                         gdb_cmd_detachkill(ks);
1024                         goto default_handle;
1025                 case 'R': /* Reboot */
1026                         if (gdb_cmd_reboot(ks))
1027                                 goto default_handle;
1028                         break;
1029                 case 'q': /* query command */
1030                         gdb_cmd_query(ks);
1031                         break;
1032                 case 'H': /* task related */
1033                         gdb_cmd_task(ks);
1034                         break;
1035                 case 'T': /* Query thread status */
1036                         gdb_cmd_thread(ks);
1037                         break;
1038                 case 'z': /* Break point remove */
1039                 case 'Z': /* Break point set */
1040                         gdb_cmd_break(ks);
1041                         break;
1042 #ifdef CONFIG_KGDB_KDB
1043                 case '3': /* Escape into back into kdb */
1044                         if (remcom_in_buffer[1] == '\0') {
1045                                 gdb_cmd_detachkill(ks);
1046                                 return DBG_PASS_EVENT;
1047                         }
1048 #endif
1049                         fallthrough;
1050                 case 'C': /* Exception passing */
1051                         tmp = gdb_cmd_exception_pass(ks);
1052                         if (tmp > 0)
1053                                 goto default_handle;
1054                         if (tmp == 0)
1055                                 break;
1056                         fallthrough;    /* on tmp < 0 */
1057                 case 'c': /* Continue packet */
1058                 case 's': /* Single step packet */
1059                         if (kgdb_contthread && kgdb_contthread != current) {
1060                                 /* Can't switch threads in kgdb */
1061                                 error_packet(remcom_out_buffer, -EINVAL);
1062                                 break;
1063                         }
1064                         fallthrough;    /* to default processing */
1065                 default:
1066 default_handle:
1067                         error = kgdb_arch_handle_exception(ks->ex_vector,
1068                                                 ks->signo,
1069                                                 ks->err_code,
1070                                                 remcom_in_buffer,
1071                                                 remcom_out_buffer,
1072                                                 ks->linux_regs);
1073                         /*
1074                          * Leave cmd processing on error, detach,
1075                          * kill, continue, or single step.
1076                          */
1077                         if (error >= 0 || remcom_in_buffer[0] == 'D' ||
1078                             remcom_in_buffer[0] == 'k') {
1079                                 error = 0;
1080                                 goto kgdb_exit;
1081                         }
1082 
1083                 }
1084 
1085                 /* reply to the request */
1086                 put_packet(remcom_out_buffer);
1087         }
1088 
1089 kgdb_exit:
1090         if (ks->pass_exception)
1091                 error = 1;
1092         return error;
1093 }

1123 /**
1124  * gdbstub_exit - Send an exit message to GDB
1125  * @status: The exit code to report.
1126  */
1127 void gdbstub_exit(int status)
1128 {
1129         unsigned char checksum, ch, buffer[3];
1130         int loop;
1131 
1132         if (!kgdb_connected)
1133                 return;
1134         kgdb_connected = 0;
1135 
1136         if (!dbg_io_ops || dbg_kdb_mode)
1137                 return;
1138 
1139         buffer[0] = 'W';
1140         buffer[1] = hex_asc_hi(status);
1141         buffer[2] = hex_asc_lo(status);
1142 
1143         dbg_io_ops->write_char('$');
1144         checksum = 0;
1145 
1146         for (loop = 0; loop < 3; loop++) {
1147                 ch = buffer[loop];
1148                 checksum += ch;
1149                 dbg_io_ops->write_char(ch);
1150         }
1151 
1152         dbg_io_ops->write_char('#');
1153         dbg_io_ops->write_char(hex_asc_hi(checksum));
1154         dbg_io_ops->write_char(hex_asc_lo(checksum));
1155 
1156         /* make sure the output is flushed, lest the bootloader clobber it */
1157         if (dbg_io_ops->flush)
1158                 dbg_io_ops->flush();
1159 }
