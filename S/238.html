 304 /*
 305  * There are two global locks guarding cpuset structures - cpuset_mutex and
 306  * callback_lock. We also require taking task_lock() when dereferencing a
 307  * task's cpuset pointer. See "The task_lock() exception", at the end of this
 308  * comment.
 309  *
 310  * A task must hold both locks to modify cpusets.  If a task holds
 311  * cpuset_mutex, then it blocks others wanting that mutex, ensuring that it
 312  * is the only task able to also acquire callback_lock and be able to
 313  * modify cpusets.  It can perform various checks on the cpuset structure
 314  * first, knowing nothing will change.  It can also allocate memory while
 315  * just holding cpuset_mutex.  While it is performing these checks, various
 316  * callback routines can briefly acquire callback_lock to query cpusets.
 317  * Once it is ready to make the changes, it takes callback_lock, blocking
 318  * everyone else.
 319  *
 320  * Calls to the kernel memory allocator can not be made while holding
 321  * callback_lock, as that would risk double tripping on callback_lock
 322  * from one of the callbacks into the cpuset code from within
 323  * __alloc_pages().
 324  *
 325  * If a task is only holding callback_lock, then it has read-only
 326  * access to cpusets.
 327  *
 328  * Now, the task_struct fields mems_allowed and mempolicy may be changed
 329  * by other task, we use alloc_lock in the task_struct fields to protect
 330  * them.
 331  *
 332  * The cpuset_common_file_read() handlers only hold callback_lock across
 333  * small pieces of code, such as when reading out possibly multi-word
 334  * cpumasks and nodemasks.
 335  *
 336  * Accessing a task's cpuset should be done in accordance with the
 337  * guidelines for accessing subsystem state in kernel/cgroup.c
 338  */

 367 /*
 368  * Return in pmask the portion of a task's cpusets's cpus_allowed that
 369  * are online and are capable of running the task.  If none are found,
 370  * walk up the cpuset hierarchy until we find one that does have some
 371  * appropriate cpus.
 372  *
 373  * One way or another, we guarantee to return some non-empty subset
 374  * of cpu_active_mask.
 375  *
 376  * Call with callback_lock or cpuset_mutex held.
 377  */
 378 static void guarantee_online_cpus(struct task_struct *tsk,
     /*   */
 379                                   struct cpumask *pmask)
 380 {
 381         const struct cpumask *possible_mask = task_cpu_possible_mask(tsk);
 382         struct cpuset *cs;
 383 
 384         if (WARN_ON(!cpumask_and(pmask, possible_mask, cpu_active_mask)))
 385                 cpumask_copy(pmask, cpu_active_mask);
 386 
 387         rcu_read_lock();
 388         cs = task_cs(tsk);
 389 
 390         while (!cpumask_intersects(cs->effective_cpus, pmask)) {
 391                 cs = parent_cs(cs);
 392                 if (unlikely(!cs)) {
 393                         /*
 394                          * The top cpuset doesn't have any online cpu as a
 395                          * consequence of a race between cpuset_hotplug_work
 396                          * and cpu hotplug notifier.  But we know the top
 397                          * cpuset's effective_cpus is on its way to be
 398                          * identical to cpu_online_mask.
 399                          */
 400                         goto out_unlock;
 401                 }
 402         }
 403         cpumask_and(pmask, pmask, cs->effective_cpus);
 404 
 405 out_unlock:
 406         rcu_read_unlock();
 407 }

 409 /*
 410  * Return in *pmask the portion of a cpusets's mems_allowed that
 411  * are online, with memory.  If none are online with memory, walk
 412  * up the cpuset hierarchy until we find one that does have some
 413  * online mems.  The top cpuset always has some mems online.
 414  *
 415  * One way or another, we guarantee to return some non-empty subset
 416  * of node_states[N_MEMORY].
 417  *
 418  * Call with callback_lock or cpuset_mutex held.
 419  */
 420 static void guarantee_online_mems(struct cpuset *cs, nodemask_t *pmask)
     /*   */
 421 {
 422         while (!nodes_intersects(cs->effective_mems, node_states[N_MEMORY]))
 423                 cs = parent_cs(cs);
 424         nodes_and(*pmask, cs->effective_mems, node_states[N_MEMORY]);
 425 }

 462 /**
 463  * alloc_cpumasks - allocate three cpumasks for cpuset
 464  * @cs:  the cpuset that have cpumasks to be allocated.
 465  * @tmp: the tmpmasks structure pointer
 466  * Return: 0 if successful, -ENOMEM otherwise.
 467  *
 468  * Only one of the two input arguments should be non-NULL.
 469  */
 470 static inline int alloc_cpumasks(struct cpuset *cs, struct tmpmasks *tmp)
     /*   */
 471 {
 472         cpumask_var_t *pmask1, *pmask2, *pmask3;
 473 
 474         if (cs) {
 475                 pmask1 = &cs->cpus_allowed;
 476                 pmask2 = &cs->effective_cpus;
 477                 pmask3 = &cs->subparts_cpus;
 478         } else {
 479                 pmask1 = &tmp->new_cpus;
 480                 pmask2 = &tmp->addmask;
 481                 pmask3 = &tmp->delmask;
 482         }
 483 
 484         if (!zalloc_cpumask_var(pmask1, GFP_KERNEL))
 485                 return -ENOMEM;
 486 
 487         if (!zalloc_cpumask_var(pmask2, GFP_KERNEL))
 488                 goto free_one;
 489 
 490         if (!zalloc_cpumask_var(pmask3, GFP_KERNEL))
 491                 goto free_two;
 492 
 493         if (cs && !zalloc_cpumask_var(&cs->cpus_requested, GFP_KERNEL))
 494                 goto free_three;
 495 
 496         return 0;
 497 
 498 free_three:
 499         free_cpumask_var(*pmask3);
 500 free_two:
 501         free_cpumask_var(*pmask2);
 502 free_one:
 503         free_cpumask_var(*pmask1);
 504         return -ENOMEM;
 505 }

 560 /*
 561  * validate_change() - Used to validate that any proposed cpuset change
 562  *                     follows the structural rules for cpusets.
 563  *
 564  * If we replaced the flag and mask values of the current cpuset
 565  * (cur) with those values in the trial cpuset (trial), would
 566  * our various subset and exclusive rules still be valid?  Presumes
 567  * cpuset_mutex held.
 568  *
 569  * 'cur' is the address of an actual, in-use cpuset.  Operations
 570  * such as list traversal that depend on the actual address of the
 571  * cpuset in the list must use cur below, not trial.
 572  *
 573  * 'trial' is the address of bulk structure copy of cur, with
 574  * perhaps one or more of the fields cpus_allowed, mems_allowed,
 575  * or flags changed to new, trial values.
 576  *
 577  * Return 0 if valid, -errno if not.
 578  */
 579 
 580 static int validate_change(struct cpuset *cur, struct cpuset *trial)
     /*   */
 581 {
 582         struct cgroup_subsys_state *css;
 583         struct cpuset *c, *par;
 584         int ret;
 585 
 586         rcu_read_lock();
 587 
 588         /* Each of our child cpusets must be a subset of us */
 589         ret = -EBUSY;
 590         cpuset_for_each_child(c, css, cur)
 591                 if (!is_cpuset_subset(c, trial))
 592                         goto out;
 593 
 594         /* Remaining checks don't apply to root cpuset */
 595         ret = 0;
 596         if (cur == &top_cpuset)
 597                 goto out;
 598 
 599         par = parent_cs(cur);
 600 
 601         /* On legacy hiearchy, we must be a subset of our parent cpuset. */
 602         ret = -EACCES;
 603         if (!is_in_v2_mode() && !is_cpuset_subset(trial, par))
 604                 goto out;
 605 
 606         /*
 607          * If either I or some sibling (!= me) is exclusive, we can't
 608          * overlap
 609          */
 610         ret = -EINVAL;
 611         cpuset_for_each_child(c, css, par) {
 612                 if ((is_cpu_exclusive(trial) || is_cpu_exclusive(c)) &&
 613                     c != cur &&
 614                     cpumask_intersects(trial->cpus_requested, c->cpus_requested))
 615                         goto out;
 616                 if ((is_mem_exclusive(trial) || is_mem_exclusive(c)) &&
 617                     c != cur &&
 618                     nodes_intersects(trial->mems_allowed, c->mems_allowed))
 619                         goto out;
 620         }
 621 
 622         /*
 623          * Cpusets with tasks - existing or newly being attached - can't
 624          * be changed to have empty cpus_allowed or mems_allowed.
 625          */
 626         ret = -ENOSPC;
 627         if ((cgroup_is_populated(cur->css.cgroup) || cur->attach_in_progress)) {
 628                 if (!cpumask_empty(cur->cpus_allowed) &&
 629                     cpumask_empty(trial->cpus_allowed))
 630                         goto out;
 631                 if (!nodes_empty(cur->mems_allowed) &&
 632                     nodes_empty(trial->mems_allowed))
 633                         goto out;
 634         }
 635 
 636         /*
 637          * We can't shrink if we won't have enough room for SCHED_DEADLINE
 638          * tasks.
 639          */
 640         ret = -EBUSY;
 641         if (is_cpu_exclusive(cur) &&
 642             !cpuset_cpumask_can_shrink(cur->cpus_allowed,
 643                                        trial->cpus_allowed))
 644                 goto out;
 645 
 646         ret = 0;
 647 out:
 648         rcu_read_unlock();
 649         return ret;
 650 }

 697 /*
 698  * generate_sched_domains()
 699  *
 700  * This function builds a partial partition of the systems CPUs
 701  * A 'partial partition' is a set of non-overlapping subsets whose
 702  * union is a subset of that set.
 703  * The output of this function needs to be passed to kernel/sched/core.c
 704  * partition_sched_domains() routine, which will rebuild the scheduler's
 705  * load balancing domains (sched domains) as specified by that partial
 706  * partition.
 707  *
 708  * See "What is sched_load_balance" in Documentation/admin-guide/cgroup-v1/cpusets.rst
 709  * for a background explanation of this.
 710  *
 711  * Does not return errors, on the theory that the callers of this
 712  * routine would rather not worry about failures to rebuild sched
 713  * domains when operating in the severe memory shortage situations
 714  * that could cause allocation failures below.
 715  *
 716  * Must be called with cpuset_mutex held.
 717  *
 718  * The three key local variables below are:
 719  *    cp - cpuset pointer, used (together with pos_css) to perform a
 720  *         top-down scan of all cpusets. For our purposes, rebuilding
 721  *         the schedulers sched domains, we can ignore !is_sched_load_
 722  *         balance cpusets.
 723  *  csa  - (for CpuSet Array) Array of pointers to all the cpusets
 724  *         that need to be load balanced, for convenient iterative
 725  *         access by the subsequent code that finds the best partition,
 726  *         i.e the set of domains (subsets) of CPUs such that the
 727  *         cpus_allowed of every cpuset marked is_sched_load_balance
 728  *         is a subset of one of these domains, while there are as
 729  *         many such domains as possible, each as small as possible.
 730  * doms  - Conversion of 'csa' to an array of cpumasks, for passing to
 731  *         the kernel/sched/core.c routine partition_sched_domains() in a
 732  *         convenient format, that can be easily compared to the prior
 733  *         value to determine what partition elements (sched domains)
 734  *         were changed (added or removed.)
 735  *
 736  * Finding the best partition (set of domains):
 737  *      The triple nested loops below over i, j, k scan over the
 738  *      load balanced cpusets (using the array of cpuset pointers in
 739  *      csa[]) looking for pairs of cpusets that have overlapping
 740  *      cpus_allowed, but which don't have the same 'pn' partition
 741  *      number and gives them in the same partition number.  It keeps
 742  *      looping on the 'restart' label until it can no longer find
 743  *      any such pairs.
 744  *
 745  *      The union of the cpus_allowed masks from the set of
 746  *      all cpusets having the same 'pn' value then form the one
 747  *      element of the partition (one sched domain) to be passed to
 748  *      partition_sched_domains().
 749  */
 750 static int generate_sched_domains(cpumask_var_t **domains,
     /*   */
 751                         struct sched_domain_attr **attributes)
 752 {
 753         struct cpuset *cp;      /* top-down scan of cpusets */
 754         struct cpuset **csa;    /* array of all cpuset ptrs */
 755         int csn;                /* how many cpuset ptrs in csa so far */
 756         int i, j, k;            /* indices for partition finding loops */
 757         cpumask_var_t *doms;    /* resulting partition; i.e. sched domains */
 758         struct sched_domain_attr *dattr;  /* attributes for custom domains */
 759         int ndoms = 0;          /* number of sched domains in result */
 760         int nslot;              /* next empty doms[] struct cpumask slot */
 761         struct cgroup_subsys_state *pos_css;
 762         bool root_load_balance = is_sched_load_balance(&top_cpuset);
 763 
 764         doms = NULL;
 765         dattr = NULL;
 766         csa = NULL;
 767 
 768         /* Special case for the 99% of systems with one, full, sched domain */
 769         if (root_load_balance && !top_cpuset.nr_subparts_cpus) {
 770                 ndoms = 1;
 771                 doms = alloc_sched_domains(ndoms);
 772                 if (!doms)
 773                         goto done;
 774 
 775                 dattr = kmalloc(sizeof(struct sched_domain_attr), GFP_KERNEL);
 776                 if (dattr) {
 777                         *dattr = SD_ATTR_INIT;
 778                         update_domain_attr_tree(dattr, &top_cpuset);
 779                 }
 780                 cpumask_and(doms[0], top_cpuset.effective_cpus,
 781                             housekeeping_cpumask(HK_FLAG_DOMAIN));
 782 
 783                 goto done;
 784         }
 785 
 786         csa = kmalloc_array(nr_cpusets(), sizeof(cp), GFP_KERNEL);
 787         if (!csa)
 788                 goto done;
 789         csn = 0;
 790 
 791         rcu_read_lock();
 792         if (root_load_balance)
 793                 csa[csn++] = &top_cpuset;
 794         cpuset_for_each_descendant_pre(cp, pos_css, &top_cpuset) {
 795                 if (cp == &top_cpuset)
 796                         continue;
 797                 /*
 798                  * Continue traversing beyond @cp iff @cp has some CPUs and
 799                  * isn't load balancing.  The former is obvious.  The
 800                  * latter: All child cpusets contain a subset of the
 801                  * parent's cpus, so just skip them, and then we call
 802                  * update_domain_attr_tree() to calc relax_domain_level of
 803                  * the corresponding sched domain.
 804                  *
 805                  * If root is load-balancing, we can skip @cp if it
 806                  * is a subset of the root's effective_cpus.
 807                  */
 808                 if (!cpumask_empty(cp->cpus_allowed) &&
 809                     !(is_sched_load_balance(cp) &&
 810                       cpumask_intersects(cp->cpus_allowed,
 811                                          housekeeping_cpumask(HK_FLAG_DOMAIN))))
 812                         continue;
 813 
 814                 if (root_load_balance &&
 815                     cpumask_subset(cp->cpus_allowed, top_cpuset.effective_cpus))
 816                         continue;
 817 
 818                 if (is_sched_load_balance(cp) &&
 819                     !cpumask_empty(cp->effective_cpus))
 820                         csa[csn++] = cp;
 821 
 822                 /* skip @cp's subtree if not a partition root */
 823                 if (!is_partition_root(cp))
 824                         pos_css = css_rightmost_descendant(pos_css);
 825         }
 826         rcu_read_unlock();
 827 
 828         for (i = 0; i < csn; i++)
 829                 csa[i]->pn = i;
 830         ndoms = csn;
 831 
 832 restart:
 833         /* Find the best partition (set of sched domains) */
 834         for (i = 0; i < csn; i++) {
 835                 struct cpuset *a = csa[i];
 836                 int apn = a->pn;
 837 
 838                 for (j = 0; j < csn; j++) {
 839                         struct cpuset *b = csa[j];
 840                         int bpn = b->pn;
 841 
 842                         if (apn != bpn && cpusets_overlap(a, b)) {
 843                                 for (k = 0; k < csn; k++) {
 844                                         struct cpuset *c = csa[k];
 845 
 846                                         if (c->pn == bpn)
 847                                                 c->pn = apn;
 848                                 }
 849                                 ndoms--;        /* one less element */
 850                                 goto restart;
 851                         }
 852                 }
 853         }
 854 
 855         /*
 856          * Now we know how many domains to create.
 857          * Convert <csn, csa> to <ndoms, doms> and populate cpu masks.
 858          */
 859         doms = alloc_sched_domains(ndoms);
 860         if (!doms)
 861                 goto done;
 862 
 863         /*
 864          * The rest of the code, including the scheduler, can deal with
 865          * dattr==NULL case. No need to abort if alloc fails.
 866          */
 867         dattr = kmalloc_array(ndoms, sizeof(struct sched_domain_attr),
 868                               GFP_KERNEL);
 869 
 870         for (nslot = 0, i = 0; i < csn; i++) {
 871                 struct cpuset *a = csa[i];
 872                 struct cpumask *dp;
 873                 int apn = a->pn;
 874 
 875                 if (apn < 0) {
 876                         /* Skip completed partitions */
 877                         continue;
 878                 }
 879 
 880                 dp = doms[nslot];
 881 
 882                 if (nslot == ndoms) {
 883                         static int warnings = 10;
 884                         if (warnings) {
 885                                 pr_warn("rebuild_sched_domains confused: nslot %d, ndoms %d, csn %d, i %d, apn %d\n",
 886                                         nslot, ndoms, csn, i, apn);
 887                                 warnings--;
 888                         }
 889                         continue;
 890                 }
 891 
 892                 cpumask_clear(dp);
 893                 if (dattr)
 894                         *(dattr + nslot) = SD_ATTR_INIT;
 895                 for (j = i; j < csn; j++) {
 896                         struct cpuset *b = csa[j];
 897 
 898                         if (apn == b->pn) {
 899                                 cpumask_or(dp, dp, b->effective_cpus);
 900                                 cpumask_and(dp, dp, housekeeping_cpumask(HK_FLAG_DOMAIN));
 901                                 if (dattr)
 902                                         update_domain_attr_tree(dattr + nslot, b);
 903 
 904                                 /* Done with this partition */
 905                                 b->pn = -1;
 906                         }
 907                 }
 908                 nslot++;
 909         }
 910         BUG_ON(nslot != ndoms);
 911 
 912 done:
 913         kfree(csa);
 914 
 915         /*
 916          * Fallback to the default domain if kmalloc() failed.
 917          * See comments in partition_sched_domains().
 918          */
 919         if (doms == NULL)
 920                 ndoms = 1;
 921 
 922         *domains    = doms;
 923         *attributes = dattr;
 924         return ndoms;
 925 }

 986 /*
 987  * Rebuild scheduler domains.
 988  *
 989  * If the flag 'sched_load_balance' of any cpuset with non-empty
 990  * 'cpus' changes, or if the 'cpus' allowed changes in any cpuset
 991  * which has that flag enabled, or if any cpuset with a non-empty
 992  * 'cpus' is removed, then call this routine to rebuild the
 993  * scheduler's dynamic sched domains.
 994  *
 995  * Call with cpuset_mutex held.  Takes get_online_cpus().
 996  */
 997 static void rebuild_sched_domains_locked(void)
     /*   */
 998 {
 999         struct cgroup_subsys_state *pos_css;
1000         struct sched_domain_attr *attr;
1001         cpumask_var_t *doms;
1002         struct cpuset *cs;
1003         int ndoms;
1004 
1005         lockdep_assert_held(&cpuset_mutex);
1006 
1007         /*
1008          * If we have raced with CPU hotplug, return early to avoid
1009          * passing doms with offlined cpu to partition_sched_domains().
1010          * Anyways, cpuset_hotplug_workfn() will rebuild sched domains.
1011          *
1012          * With no CPUs in any subpartitions, top_cpuset's effective CPUs
1013          * should be the same as the active CPUs, so checking only top_cpuset
1014          * is enough to detect racing CPU offlines.
1015          */
1016         if (!top_cpuset.nr_subparts_cpus &&
1017             !cpumask_equal(top_cpuset.effective_cpus, cpu_active_mask))
1018                 return;
1019 
1020         /*
1021          * With subpartition CPUs, however, the effective CPUs of a partition
1022          * root should be only a subset of the active CPUs.  Since a CPU in any
1023          * partition root could be offlined, all must be checked.
1024          */
1025         if (top_cpuset.nr_subparts_cpus) {
1026                 rcu_read_lock();
1027                 cpuset_for_each_descendant_pre(cs, pos_css, &top_cpuset) {
1028                         if (!is_partition_root(cs)) {
1029                                 pos_css = css_rightmost_descendant(pos_css);
1030                                 continue;
1031                         }
1032                         if (!cpumask_subset(cs->effective_cpus,
1033                                             cpu_active_mask)) {
1034                                 rcu_read_unlock();
1035                                 return;
1036                         }
1037                 }
1038                 rcu_read_unlock();
1039         }
1040 
1041         /* Generate domain masks and attrs */
1042         ndoms = generate_sched_domains(&doms, &attr);
1043 
1044         /* Have scheduler rebuild the domains */
1045         partition_and_rebuild_sched_domains(ndoms, doms, attr);
1046 }

1093 /**
1094  * compute_effective_cpumask - Compute the effective cpumask of the cpuset
1095  * @new_cpus: the temp variable for the new effective_cpus mask
1096  * @cs: the cpuset the need to recompute the new effective_cpus mask
1097  * @parent: the parent cpuset
1098  *
1099  * If the parent has subpartition CPUs, include them in the list of
1100  * allowable CPUs in computing the new effective_cpus mask. Since offlined
1101  * CPUs are not removed from subparts_cpus, we have to use cpu_active_mask
1102  * to mask those out.
1103  */
1104 static void compute_effective_cpumask(struct cpumask *new_cpus,
     /*   */
1105                                       struct cpuset *cs, struct cpuset *parent)
1106 {

1108                 cpumask_or(new_cpus, parent->effective_cpus,
1109                            parent->subparts_cpus);
1110                 cpumask_and(new_cpus, new_cpus, cs->cpus_requested);
1111                 cpumask_and(new_cpus, new_cpus, cpu_active_mask);
1112         } else {
1113                 cpumask_and(new_cpus, cs->cpus_requested, parent_cs(cs)->effective_cpus);
1114         }
1115 }

1139  *
1140  * For partcmd_disable, the cpuset is being transofrmed from a partition
1141  * root back to a non-partition root. any CPUs in cpus_allowed that are in
1142  * parent's subparts_cpus will be taken away from that cpumask and put back
1143  * into parent's effective_cpus. 0 should always be returned.
1144  *
1145  * For partcmd_update, if the optional newmask is specified, the cpu
1146  * list is to be changed from cpus_allowed to newmask. Otherwise,
1147  * cpus_allowed is assumed to remain the same. The cpuset should either
1148  * be a partition root or an invalid partition root. The partition root
1149  * state may change if newmask is NULL and none of the requested CPUs can
1150  * be granted by the parent. The function will return 1 if changes to
1151  * parent's subparts_cpus and effective_cpus happen or 0 otherwise.
1152  * Error code should only be returned when newmask is non-NULL.
1153  *
1154  * The partcmd_enable and partcmd_disable commands are used by
1155  * update_prstate(). The partcmd_update command is used by
1156  * update_cpumasks_hier() with newmask NULL and update_cpumask() with
1157  * newmask set.
1158  *
1159  * The checking is more strict when enabling partition root than the
1160  * other two commands.
1161  *
1162  * Because of the implicit cpu exclusive nature of a partition root,
1163  * cpumask changes that violates the cpu exclusivity rule will not be
1164  * permitted when checked by validate_change(). The validate_change()
1165  * function will also prevent any changes to the cpu list if it is not
1166  * a superset of children's cpu lists.
1167  */
1168 static int update_parent_subparts_cpumask(struct cpuset *cpuset, int cmd,
     /*   */
1169                                           struct cpumask *newmask,
1170                                           struct tmpmasks *tmp)
1171 {
1172         struct cpuset *parent = parent_cs(cpuset);
1173         int adding;     /* Moving cpus from effective_cpus to subparts_cpus */
1174         int deleting;   /* Moving cpus from subparts_cpus to effective_cpus */
1175         bool part_error = false;        /* Partition error? */
1176 
1177         lockdep_assert_held(&cpuset_mutex);
1178 
1179         /*
1180          * The parent must be a partition root.
1181          * The new cpumask, if present, or the current cpus_allowed must
1182          * not be empty.
1183          */
1184         if (!is_partition_root(parent) ||
1185            (newmask && cpumask_empty(newmask)) ||
1186            (!newmask && cpumask_empty(cpuset->cpus_allowed)))
1187                 return -EINVAL;
1188 
1189         /*
1190          * Enabling/disabling partition root is not allowed if there are
1191          * online children.
1192          */
1193         if ((cmd != partcmd_update) && css_has_online_children(&cpuset->css))
1194                 return -EBUSY;
1195 
1196         /*
1197          * Enabling partition root is not allowed if not all the CPUs
1198          * can be granted from parent's effective_cpus or at least one
1199          * CPU will be left after that.
1200          */
1201         if ((cmd == partcmd_enable) &&
1202            (!cpumask_subset(cpuset->cpus_allowed, parent->effective_cpus) ||
1203              cpumask_equal(cpuset->cpus_allowed, parent->effective_cpus)))
1204                 return -EINVAL;
1205 
1206         /*
1207          * A cpumask update cannot make parent's effective_cpus become empty.
1208          */
1209         adding = deleting = false;
1210         if (cmd == partcmd_enable) {
1211                 cpumask_copy(tmp->addmask, cpuset->cpus_allowed);
1212                 adding = true;
1213         } else if (cmd == partcmd_disable) {
1214                 deleting = cpumask_and(tmp->delmask, cpuset->cpus_allowed,
1215                                        parent->subparts_cpus);
1216         } else if (newmask) {
1217                 /*
1218                  * partcmd_update with newmask:
1219                  *
1220                  * delmask = cpus_allowed & ~newmask & parent->subparts_cpus
1221                  * addmask = newmask & parent->effective_cpus
1222                  *                   & ~parent->subparts_cpus
1223                  */
1224                 cpumask_andnot(tmp->delmask, cpuset->cpus_allowed, newmask);
1225                 deleting = cpumask_and(tmp->delmask, tmp->delmask,
1226                                        parent->subparts_cpus);
1227 
1228                 cpumask_and(tmp->addmask, newmask, parent->effective_cpus);
1229                 adding = cpumask_andnot(tmp->addmask, tmp->addmask,
1230                                         parent->subparts_cpus);
1231                 /*
1232                  * Return error if the new effective_cpus could become empty.
1233                  */
1234                 if (adding &&
1235                     cpumask_equal(parent->effective_cpus, tmp->addmask)) {
1236                         if (!deleting)
1237                                 return -EINVAL;
1238                         /*
1239                          * As some of the CPUs in subparts_cpus might have
1240                          * been offlined, we need to compute the real delmask
1241                          * to confirm that.
1242                          */
1243                         if (!cpumask_and(tmp->addmask, tmp->delmask,
1244                                          cpu_active_mask))
1245                                 return -EINVAL;
1246                         cpumask_copy(tmp->addmask, parent->effective_cpus);
1247                 }
1248         } else {
1249                 /*
1250                  * partcmd_update w/o newmask:
1251                  *
1252                  * addmask = cpus_allowed & parent->effectiveb_cpus
1253                  *
1254                  * Note that parent's subparts_cpus may have been
1255                  * pre-shrunk in case there is a change in the cpu list.
1256                  * So no deletion is needed.
1257                  */
1258                 adding = cpumask_and(tmp->addmask, cpuset->cpus_allowed,
1259                                      parent->effective_cpus);
1260                 part_error = cpumask_equal(tmp->addmask,
1261                                            parent->effective_cpus);
1262         }
1263 
1264         if (cmd == partcmd_update) {
1265                 int prev_prs = cpuset->partition_root_state;
1266 
1267                 /*
1268                  * Check for possible transition between PRS_ENABLED
1269                  * and PRS_ERROR.
1270                  */
1271                 switch (cpuset->partition_root_state) {
1272                 case PRS_ENABLED:
1273                         if (part_error)
1274                                 cpuset->partition_root_state = PRS_ERROR;
1275                         break;
1276                 case PRS_ERROR:
1277                         if (!part_error)
1278                                 cpuset->partition_root_state = PRS_ENABLED;
1279                         break;
1280                 }
1281                 /*
1282                  * Set part_error if previously in invalid state.
1283                  */
1284                 part_error = (prev_prs == PRS_ERROR);
1285         }
1286 
1287         if (!part_error && (cpuset->partition_root_state == PRS_ERROR))
1288                 return 0;       /* Nothing need to be done */
1289 
1290         if (cpuset->partition_root_state == PRS_ERROR) {
1291                 /*
1292                  * Remove all its cpus from parent's subparts_cpus.
1293                  */
1294                 adding = false;
1295                 deleting = cpumask_and(tmp->delmask, cpuset->cpus_allowed,
1296                                        parent->subparts_cpus);
1297         }
1298 
1299         if (!adding && !deleting)
1300                 return 0;
1301 
1302         /*
1303          * Change the parent's subparts_cpus.
1304          * Newly added CPUs will be removed from effective_cpus and
1305          * newly deleted ones will be added back to effective_cpus.
1306          */
1307         spin_lock_irq(&callback_lock);
1308         if (adding) {
1309                 cpumask_or(parent->subparts_cpus,
1310                            parent->subparts_cpus, tmp->addmask);
1311                 cpumask_andnot(parent->effective_cpus,
1312                                parent->effective_cpus, tmp->addmask);
1313         }
1314         if (deleting) {
1315                 cpumask_andnot(parent->subparts_cpus,
1316                                parent->subparts_cpus, tmp->delmask);
1317                 /*
1318                  * Some of the CPUs in subparts_cpus might have been offlined.
1319                  */
1320                 cpumask_and(tmp->delmask, tmp->delmask, cpu_active_mask);
1321                 cpumask_or(parent->effective_cpus,
1322                            parent->effective_cpus, tmp->delmask);
1323         }
1324 
1325         parent->nr_subparts_cpus = cpumask_weight(parent->subparts_cpus);
1326         spin_unlock_irq(&callback_lock);
1327 
1328         return cmd == partcmd_update;
1329 }

1331 /*
1332  * update_cpumasks_hier - Update effective cpumasks and tasks in the subtree
1333  * @cs:  the cpuset to consider
1334  * @tmp: temp variables for calculating effective_cpus & partition setup
1335  *
1336  * When congifured cpumask is changed, the effective cpumasks of this cpuset
1337  * and all its descendants need to be updated.
1338  *
1339  * On legacy hierachy, effective_cpus will be the same with cpu_allowed.
1340  *
1341  * Called with cpuset_mutex held
1342  */
1343 static void update_cpumasks_hier(struct cpuset *cs, struct tmpmasks *tmp)
     /*   */
1344 {
1345         struct cpuset *cp;
1346         struct cgroup_subsys_state *pos_css;
1347         bool need_rebuild_sched_domains = false;
1348 
1349         rcu_read_lock();
1350         cpuset_for_each_descendant_pre(cp, pos_css, cs) {
1351                 struct cpuset *parent = parent_cs(cp);
1352 
1353                 compute_effective_cpumask(tmp->new_cpus, cp, parent);
1354 
1355                 /*
1356                  * If it becomes empty, inherit the effective mask of the
1357                  * parent, which is guaranteed to have some CPUs.
1358                  */
1359                 if (is_in_v2_mode() && cpumask_empty(tmp->new_cpus)) {
1360                         cpumask_copy(tmp->new_cpus, parent->effective_cpus);
1361                         if (!cp->use_parent_ecpus) {
1362                                 cp->use_parent_ecpus = true;
1363                                 parent->child_ecpus_count++;
1364                         }
1365                 } else if (cp->use_parent_ecpus) {
1366                         cp->use_parent_ecpus = false;
1367                         WARN_ON_ONCE(!parent->child_ecpus_count);
1368                         parent->child_ecpus_count--;
1369                 }
1370 
1371                 /*
1372                  * Skip the whole subtree if the cpumask remains the same
1373                  * and has no partition root state.
1374                  */
1375                 if (!cp->partition_root_state &&
1376                     cpumask_equal(tmp->new_cpus, cp->effective_cpus)) {
1377                         pos_css = css_rightmost_descendant(pos_css);
1378                         continue;
1379                 }
1380 
1381                 /*
1382                  * update_parent_subparts_cpumask() should have been called
1383                  * for cs already in update_cpumask(). We should also call
1384                  * update_tasks_cpumask() again for tasks in the parent
1385                  * cpuset if the parent's subparts_cpus changes.
1386                  */
1387                 if ((cp != cs) && cp->partition_root_state) {
1388                         switch (parent->partition_root_state) {
1389                         case PRS_DISABLED:
1390                                 /*
1391                                  * If parent is not a partition root or an
1392                                  * invalid partition root, clear the state
1393                                  * state and the CS_CPU_EXCLUSIVE flag.
1394                                  */
1395                                 WARN_ON_ONCE(cp->partition_root_state
1396                                              != PRS_ERROR);
1397                                 cp->partition_root_state = 0;
1398 
1399                                 /*
1400                                  * clear_bit() is an atomic operation and
1401                                  * readers aren't interested in the state
1402                                  * of CS_CPU_EXCLUSIVE anyway. So we can
1403                                  * just update the flag without holding
1404                                  * the callback_lock.
1405                                  */
1406                                 clear_bit(CS_CPU_EXCLUSIVE, &cp->flags);
1407                                 break;
1408 
1409                         case PRS_ENABLED:
1410                                 if (update_parent_subparts_cpumask(cp, partcmd_update, NULL, tmp))
1411                                         update_tasks_cpumask(parent);
1412                                 break;
1413 
1414                         case PRS_ERROR:
1415                                 /*
1416                                  * When parent is invalid, it has to be too.
1417                                  */
1418                                 cp->partition_root_state = PRS_ERROR;
1419                                 if (cp->nr_subparts_cpus) {
1420                                         cp->nr_subparts_cpus = 0;
1421                                         cpumask_clear(cp->subparts_cpus);
1422                                 }
1423                                 break;
1424                         }
1425                 }
1426 
1427                 if (!css_tryget_online(&cp->css))
1428                         continue;
1429                 rcu_read_unlock();
1430 
1431                 spin_lock_irq(&callback_lock);
1432 
1433                 cpumask_copy(cp->effective_cpus, tmp->new_cpus);
1434                 if (cp->nr_subparts_cpus &&
1435                    (cp->partition_root_state != PRS_ENABLED)) {
1436                         cp->nr_subparts_cpus = 0;
1437                         cpumask_clear(cp->subparts_cpus);
1438                 } else if (cp->nr_subparts_cpus) {
1439                         /*
1440                          * Make sure that effective_cpus & subparts_cpus
1441                          * are mutually exclusive.
1442                          *
1443                          * In the unlikely event that effective_cpus
1444                          * becomes empty. we clear cp->nr_subparts_cpus and
1445                          * let its child partition roots to compete for
1446                          * CPUs again.
1447                          */
1448                         cpumask_andnot(cp->effective_cpus, cp->effective_cpus,
1449                                        cp->subparts_cpus);
1450                         if (cpumask_empty(cp->effective_cpus)) {
1451                                 cpumask_copy(cp->effective_cpus, tmp->new_cpus);
1452                                 cpumask_clear(cp->subparts_cpus);
1453                                 cp->nr_subparts_cpus = 0;
1454                         } else if (!cpumask_subset(cp->subparts_cpus,
1455                                                    tmp->new_cpus)) {
1456                                 cpumask_andnot(cp->subparts_cpus,
1457                                         cp->subparts_cpus, tmp->new_cpus);
1458                                 cp->nr_subparts_cpus
1459                                         = cpumask_weight(cp->subparts_cpus);
1460                         }
1461                 }
1462                 spin_unlock_irq(&callback_lock);
1463 
1464                 WARN_ON(!is_in_v2_mode() &&
1465                         !cpumask_equal(cp->cpus_allowed, cp->effective_cpus));
1466 
1467                 update_tasks_cpumask(cp);
1468 
1469                 /*
1470                  * On legacy hierarchy, if the effective cpumask of any non-
1471                  * empty cpuset is changed, we need to rebuild sched domains.
1472                  * On default hierarchy, the cpuset needs to be a partition
1473                  * root as well.
1474                  */
1475                 if (!cpumask_empty(cp->cpus_allowed) &&
1476                     is_sched_load_balance(cp) &&
1477                    (!cgroup_subsys_on_dfl(cpuset_cgrp_subsys) ||
1478                     is_partition_root(cp)))
1479                         need_rebuild_sched_domains = true;
1480 
1481                 rcu_read_lock();
1482                 css_put(&cp->css);
1483         }
1484         rcu_read_unlock();
1485 
1486         if (need_rebuild_sched_domains)
1487                 rebuild_sched_domains_locked();
1488 }

1490 /**
1491  * update_sibling_cpumasks - Update siblings cpumasks
1492  * @parent:  Parent cpuset
1493  * @cs:      Current cpuset
1494  * @tmp:     Temp variables
1495  */
1496 static void update_sibling_cpumasks(struct cpuset *parent, struct cpuset *cs,
     /*   */
1497                                     struct tmpmasks *tmp)
1498 {
1499         struct cpuset *sibling;
1500         struct cgroup_subsys_state *pos_css;
1501 
1502         /*
1503          * Check all its siblings and call update_cpumasks_hier()
1504          * if their use_parent_ecpus flag is set in order for them
1505          * to use the right effective_cpus value.
1506          */
1507         rcu_read_lock();
1508         cpuset_for_each_child(sibling, pos_css, parent) {
1509                 if (sibling == cs)
1510                         continue;
1511                 if (!sibling->use_parent_ecpus)
1512                         continue;
1513 
1514                 update_cpumasks_hier(sibling, tmp);
1515         }
1516         rcu_read_unlock();
1517 }

1519 /**
1520  * update_cpumask - update the cpus_allowed mask of a cpuset and all tasks in it
1521  * @cs: the cpuset to consider
1522  * @trialcs: trial cpuset
1523  * @buf: buffer of cpu numbers written to this cpuset
1524  */
1525 static int update_cpumask(struct cpuset *cs, struct cpuset *trialcs,
     /*   */
1526                           const char *buf)
1527 {
1528         int retval;
1529         struct tmpmasks tmp;
1530 
1531         /* top_cpuset.cpus_allowed tracks cpu_online_mask; it's read-only */
1532         if (cs == &top_cpuset)
1533                 return -EACCES;
1534 
1535         /*
1536          * An empty cpus_requested is ok only if the cpuset has no tasks.
1537          * Since cpulist_parse() fails on an empty mask, we special case
1538          * that parsing.  The validate_change() call ensures that cpusets
1539          * with tasks have cpus.
1540          */
1541         if (!*buf) {
1542                 cpumask_clear(trialcs->cpus_requested);
1543         } else {
1544                 retval = cpulist_parse(buf, trialcs->cpus_requested);
1545                 if (retval < 0)
1546                         return retval;
1547         }
1548 
1549         if (!cpumask_subset(trialcs->cpus_requested, cpu_present_mask))
1550                 return -EINVAL;
1551 
1552         cpumask_and(trialcs->cpus_allowed, trialcs->cpus_requested, cpu_active_mask);
1553 
1554         /* Nothing to do if the cpus didn't change */
1555         if (cpumask_equal(cs->cpus_requested, trialcs->cpus_requested))
1556                 return 0;
1557 
1558         retval = validate_change(cs, trialcs);
1559         if (retval < 0)
1560                 return retval;
1561 
1562 #ifdef CONFIG_CPUMASK_OFFSTACK
1563         /*
1564          * Use the cpumasks in trialcs for tmpmasks when they are pointers
1565          * to allocated cpumasks.
1566          */
1567         tmp.addmask  = trialcs->subparts_cpus;
1568         tmp.delmask  = trialcs->effective_cpus;
1569         tmp.new_cpus = trialcs->cpus_allowed;
1570 #endif
1571 
1572         if (cs->partition_root_state) {
1573                 /* Cpumask of a partition root cannot be empty */
1574                 if (cpumask_empty(trialcs->cpus_allowed))
1575                         return -EINVAL;
1576                 if (update_parent_subparts_cpumask(cs, partcmd_update,
1577                                         trialcs->cpus_allowed, &tmp) < 0)
1578                         return -EINVAL;
1579         }
1580 
1581         spin_lock_irq(&callback_lock);
1582         cpumask_copy(cs->cpus_allowed, trialcs->cpus_allowed);
1583         cpumask_copy(cs->cpus_requested, trialcs->cpus_requested);
1584 
1585         /*
1586          * Make sure that subparts_cpus is a subset of cpus_allowed.
1587          */
1588         if (cs->nr_subparts_cpus) {
1589                 cpumask_andnot(cs->subparts_cpus, cs->subparts_cpus,
1590                                cs->cpus_allowed);
1591                 cs->nr_subparts_cpus = cpumask_weight(cs->subparts_cpus);
1592         }
1593         spin_unlock_irq(&callback_lock);
1594 
1595         update_cpumasks_hier(cs, &tmp);
1596 
1597         if (cs->partition_root_state) {
1598                 struct cpuset *parent = parent_cs(cs);
1599 
1600                 /*
1601                  * For partition root, update the cpumasks of sibling
1602                  * cpusets if they use parent's effective_cpus.
1603                  */
1604                 if (parent->child_ecpus_count)
1605                         update_sibling_cpumasks(parent, cs, &tmp);
1606         }
1607         return 0;
1608 }

1611  * Migrate memory region from one set of nodes to another.  This is
1612  * performed asynchronously as it can be called from process migration path
1613  * holding locks involved in process management.  All mm migrations are
1614  * performed in the queued order and can be waited for by flushing

     /*   */

1658 /*
1659  * cpuset_change_task_nodemask - change task's mems_allowed and mempolicy
1660  * @tsk: the task to change
1661  * @newmems: new nodes that the task will be set
1662  *
1663  * We use the mems_allowed_seq seqlock to safely update both tsk->mems_allowed
1664  * and rebind an eventual tasks' mempolicy. If the task is allocating in
1665  * parallel, it might temporarily see an empty intersection, which results in
1666  * a seqlock check and retry before OOM or allocation failure.
1667  */
1668 static void cpuset_change_task_nodemask(struct task_struct *tsk,
     /*   */
1669                                         nodemask_t *newmems)
1670 {
1671         task_lock(tsk);
1672 
1673         local_irq_disable();
1674         write_seqcount_begin(&tsk->mems_allowed_seq);
1675 
1676         nodes_or(tsk->mems_allowed, tsk->mems_allowed, *newmems);
1677         mpol_rebind_task(tsk, newmems);
1678         tsk->mems_allowed = *newmems;
1679 
1680         write_seqcount_end(&tsk->mems_allowed_seq);
1681         local_irq_enable();
1682 
1683         task_unlock(tsk);
1684 }

1686 static void *cpuset_being_rebound;
1687 
1688 /**
1689  * update_tasks_nodemask - Update the nodemasks of tasks in the cpuset.
1690  * @cs: the cpuset in which each task's mems_allowed mask needs to be changed
1691  *
1692  * Iterate through each task of @cs updating its mems_allowed to the
1693  * effective cpuset's.  As this function is called with cpuset_mutex held,
1694  * cpuset membership stays stable.
1695  */
1696 static void update_tasks_nodemask(struct cpuset *cs)
     /*   */
1697 {
1698         static nodemask_t newmems;      /* protected by cpuset_mutex */
1699         struct css_task_iter it;
1700         struct task_struct *task;
1701 
1702         cpuset_being_rebound = cs;              /* causes mpol_dup() rebind */
1703 
1704         guarantee_online_mems(cs, &newmems);
1705 
1706         /*
1707          * The mpol_rebind_mm() call takes mmap_lock, which we couldn't
1708          * take while holding tasklist_lock.  Forks can happen - the
1709          * mpol_dup() cpuset_being_rebound check will catch such forks,
1710          * and rebind their vma mempolicies too.  Because we still hold
1711          * the global cpuset_mutex, we know that no other rebind effort
1712          * will be contending for the global variable cpuset_being_rebound.
1713          * It's ok if we rebind the same mm twice; mpol_rebind_mm()
1714          * is idempotent.  Also migrate pages in each mm to new nodes.
1715          */
1716         css_task_iter_start(&cs->css, 0, &it);
1717         while ((task = css_task_iter_next(&it))) {
1718                 struct mm_struct *mm;
1719                 bool migrate;
1720

2052 /*
2053  * Frequency meter - How fast is some event occurring?
2054  *
2055  * These routines manage a digitally filtered, constant time based,
2056  * event frequency meter.  There are four routines:
2057  *   fmeter_init() - initialize a frequency meter.
2058  *   fmeter_markevent() - called each time the event happens.
2059  *   fmeter_getrate() - returns the recent rate of such events.
2060  *   fmeter_update() - internal routine used to update fmeter.
2061  *
2062  * A common data structure is passed to each of these routines,
2063  * which is used to keep track of the state required to manage the
2064  * frequency meter and its digital filter.
2065  *
2066  * The filter works on the number of events marked per unit time.
2067  * The filter is single-pole low-pass recursive (IIR).  The time unit
2068  * is 1 second.  Arithmetic is done using 32-bit integers scaled to
2069  * simulate 3 decimal digits of precision (multiplied by 1000).
2070  *
2071  * With an FM_COEF of 933, and a time base of 1 second, the filter
2072  * has a half-life of 10 seconds, meaning that if the events quit
2073  * happening, then the rate returned from the fmeter_getrate()
2074  * will be cut in half each 10 seconds, until it converges to zero.
2075  *
2076  * It is not worth doing a real infinitely recursive filter.  If more
2077  * than FM_MAXTICKS ticks have elapsed since the last filter event,
2078  * just compute FM_MAXTICKS ticks worth, by which point the level
2079  * will be stable.
2080  *
2081  * Limit the count of unprocessed events to FM_MAXCNT, so as to avoid
2082  * arithmetic overflow in the fmeter_update() routine.
2083  *
2084  * Given the simple 32 bit integer arithmetic used, this meter works
2085  * best for reporting rates between one per millisecond (msec) and
2086  * one per 32 (approx) seconds.  At constant rates faster than one
2087  * per msec it maxes out at values just under 1,000,000.  At constant
2088  * rates between one per msec, and one per second it will stabilize
2089  * to a value N*1000, where N is the rate of events per second.
2090  * At constant rates between one per second and one per 32 seconds,
2091  * it will be choppy, moving up on the seconds that have an event,
2092  * and then decaying until the next event.  At rates slower than
2093  * about one in 32 seconds, it decays all the way back to zero between
2094  * each event.
2095  */
2096 
2097 #define FM_COEF 933             /* coefficient for half-life of 10 secs */
2098 #define FM_MAXTICKS ((u32)99)   /* useless computing more ticks than this */
2099 #define FM_MAXCNT 1000000       /* limit cnt to avoid overflow */
2100 #define FM_SCALE 1000           /* faux fixed point scale */
2101 
2102 /* Initialize a frequency meter */
2103 static void fmeter_init(struct fmeter *fmp)
     /*   */
2104 {
2105         fmp->cnt = 0;
2106         fmp->val = 0;
2107         fmp->time = 0;
2108         spin_lock_init(&fmp->lock);
2109 }

2111 /* Internal meter update - process cnt events and update value */
2112 static void fmeter_update(struct fmeter *fmp)
     /*   */
2113 {
2114         time64_t now;
2115         u32 ticks;
2116 
2117         now = ktime_get_seconds();
2118         ticks = now - fmp->time;
2119 
2120         if (ticks == 0)
2121                 return;
2122 
2123         ticks = min(FM_MAXTICKS, ticks);
2124         while (ticks-- > 0)
2125                 fmp->val = (FM_COEF * fmp->val) / FM_SCALE;
2126         fmp->time = now;
2127 
2128         fmp->val += ((FM_SCALE - FM_COEF) * fmp->cnt) / FM_SCALE;
2129         fmp->cnt = 0;
2130 }
2131 
2132 /* Process any previous ticks, then bump cnt by one (times scale). */
2133 static void fmeter_markevent(struct fmeter *fmp)
     /*   */
2134 {
2135         spin_lock(&fmp->lock);
2136         fmeter_update(fmp);
2137         fmp->cnt = min(FM_MAXCNT, fmp->cnt + FM_SCALE);
2138         spin_unlock(&fmp->lock);
2139 }

2141 /* Process any previous ticks, then return current value. */
2142 static int fmeter_getrate(struct fmeter *fmp)
     /*   */
2143 {
2144         int val;
2145 
2146         spin_lock(&fmp->lock);
2147         fmeter_update(fmp);
2148         val = fmp->val;
2149         spin_unlock(&fmp->lock);
2150         return val;
2151 }

2356         int retval = -ENODEV;
2357 
2358         get_online_cpus();
2359         mutex_lock(&cpuset_mutex);
2360         if (!is_cpuset_online(cs))
2361                 goto out_unlock;
2362 
2363         switch (type) {
2364         case FILE_SCHED_RELAX_DOMAIN_LEVEL:
2365                 retval = update_relax_domain_level(cs, val);
2366                 break;

2377 /*
2378  * Common handling for a write to a "cpus" or "mems" file.
2379  */
2380 static ssize_t cpuset_write_resmask(struct kernfs_open_file *of,
     /*   */
2381                                     char *buf, size_t nbytes, loff_t off)
2382 {
2383         struct cpuset *cs = css_cs(of_css(of));
2384         struct cpuset *trialcs;
2385         int retval = -ENODEV;
2386 
2387         buf = strstrip(buf);
2388 
2389         /*
2390          * CPU or memory hotunplug may leave @cs w/o any execution
2391          * resources, in which case the hotplug code asynchronously updates
2392          * configuration and transfers all tasks to the nearest ancestor
2393          * which can execute.
2394          *
2395          * As writes to "cpus" or "mems" may restore @cs's execution
2396          * resources, wait for the previously scheduled operations before
2397          * proceeding, so that we don't end up keep removing tasks added
2398          * after execution capability is restored.
2399          *
2400          * cpuset_hotplug_work calls back into cgroup core via
2401          * cgroup_transfer_tasks() and waiting for it from a cgroupfs
2402          * operation like this one can lead to a deadlock through kernfs
2403          * active_ref protection.  Let's break the protection.  Losing the
2404          * protection is okay as we check whether @cs is online after
2405          * grabbing cpuset_mutex anyway.  This only happens on the legacy
2406          * hierarchies.
2407          */
2408         css_get(&cs->css);
2409         kernfs_break_active_protection(of->kn);
2410         flush_work(&cpuset_hotplug_work);
2411 
2412         get_online_cpus();
2413         mutex_lock(&cpuset_mutex);
2414         if (!is_cpuset_online(cs))
2415                 goto out_unlock;
2416 
2417         trialcs = alloc_trial_cpuset(cs);
2418         if (!trialcs) {
2419                 retval = -ENOMEM;
2420                 goto out_unlock;
2421         }
2422 
2423         switch (of_cft(of)->private) {
2424         case FILE_CPULIST:
2425                 retval = update_cpumask(cs, trialcs, buf);
2426                 break;
2427         case FILE_MEMLIST:
2428                 retval = update_nodemask(cs, trialcs, buf);
2429                 break;
2430         default:
2431                 retval = -EINVAL;
2432                 break;
2433         }
2434 
2435         free_cpuset(trialcs);
2436 out_unlock:
2437         mutex_unlock(&cpuset_mutex);
2438         put_online_cpus();
2439         kernfs_unbreak_active_protection(of->kn);
2440         css_put(&cs->css);
2441         flush_workqueue(cpuset_migrate_mm_wq);
2442         return retval ?: nbytes;
2443 }

2773 static int cpuset_css_online(struct cgroup_subsys_state *css)
     /*   */
2774 {
2775         struct cpuset *cs = css_cs(css);
2776         struct cpuset *parent = parent_cs(cs);
2777         struct cpuset *tmp_cs;
2778         struct cgroup_subsys_state *pos_css;
2779 
2780         if (!parent)
2781                 return 0;
2782 
2783         get_online_cpus();
2784         mutex_lock(&cpuset_mutex);
2785 
2786         set_bit(CS_ONLINE, &cs->flags);
2787         if (is_spread_page(parent))
2788                 set_bit(CS_SPREAD_PAGE, &cs->flags);
2789         if (is_spread_slab(parent))
2790                 set_bit(CS_SPREAD_SLAB, &cs->flags);
2791 
2792         cpuset_inc();
2793 
2794         spin_lock_irq(&callback_lock);
2795         if (is_in_v2_mode()) {
2796                 cpumask_copy(cs->effective_cpus, parent->effective_cpus);
2797                 cs->effective_mems = parent->effective_mems;
2798                 cs->use_parent_ecpus = true;
2799                 parent->child_ecpus_count++;
2800         }
2801         spin_unlock_irq(&callback_lock);
2802 
2803         if (!test_bit(CGRP_CPUSET_CLONE_CHILDREN, &css->cgroup->flags))
2804                 goto out_unlock;
2805 
2806         /*
2807          * Clone @parent's configuration if CGRP_CPUSET_CLONE_CHILDREN is
2808          * set.  This flag handling is implemented in cgroup core for
2809          * histrical reasons - the flag may be specified during mount.
2810          *
2811          * Currently, if any sibling cpusets have exclusive cpus or mem, we
2812          * refuse to clone the configuration - thereby refusing the task to
2813          * be entered, and as a result refusing the sys_unshare() or
2814          * clone() which initiated it.  If this becomes a problem for some
2815          * users who wish to allow that scenario, then this could be
2816          * changed to grant parent->cpus_allowed-sibling_cpus_exclusive
2817          * (and likewise for mems) to the new cgroup.
2818          */
2819         rcu_read_lock();
2820         cpuset_for_each_child(tmp_cs, pos_css, parent) {
2821                 if (is_mem_exclusive(tmp_cs) || is_cpu_exclusive(tmp_cs)) {
2822                         rcu_read_unlock();
2823                         goto out_unlock;
2824                 }
2825         }
2826         rcu_read_unlock();
2827 
2828         spin_lock_irq(&callback_lock);
2829         cs->mems_allowed = parent->mems_allowed;
2830         cs->effective_mems = parent->mems_allowed;
2831         cpumask_copy(cs->cpus_allowed, parent->cpus_allowed);
2832         cpumask_copy(cs->cpus_requested, parent->cpus_requested);
2833         cpumask_copy(cs->effective_cpus, parent->cpus_allowed);
2834         spin_unlock_irq(&callback_lock);
2835 out_unlock:
2836         mutex_unlock(&cpuset_mutex);
2837         put_online_cpus();
2838         return 0;
2839 }

2844  * will call rebuild_sched_domains_locked(). That is not needed
2845  * in the default hierarchy where only changes in partition
2846  * will cause repartitioning.
2847  *

2855

2967 /*

2969  * or memory nodes, we need to walk over the cpuset hierarchy,

     /*   */

3158 /**
3159  * cpuset_hotplug_workfn - handle CPU/memory hotunplug for a cpuset
3160  *
3161  * This function is called after either CPU or memory configuration has
3162  * changed and updates cpuset accordingly.  The top_cpuset is always
3163  * synchronized to cpu_active_mask and N_MEMORY, which is necessary in
3164  * order to make cpusets transparent (of no affect) on systems that are
3165  * actively using CPU hotplug but making no active use of cpusets.
3166  *
3167  * Non-root cpusets are only affected by offlining.  If any CPUs or memory
3168  * nodes have been taken down, cpuset_hotplug_update_tasks() is invoked on
3169  * all descendants.
3170  *
3171  * Note that CPU offlining during suspend is ignored.  We don't modify
3172  * cpusets across suspend/resume cycles at all.
3173  */
3174 void cpuset_hotplug_workfn(struct work_struct *work)
     /*   */
3175 {
3176         static cpumask_t new_cpus;
3177         static nodemask_t new_mems;
3178         bool cpus_updated, mems_updated;
3179         bool on_dfl = is_in_v2_mode();
3180         struct tmpmasks tmp, *ptmp = NULL;
3181 
3182         if (on_dfl && !alloc_cpumasks(NULL, &tmp))
3183                 ptmp = &tmp;
3184 
3185         mutex_lock(&cpuset_mutex);
3186 
3187         /* fetch the available cpus/mems and find out which changed how */
3188         cpumask_copy(&new_cpus, cpu_active_mask);
3189         new_mems = node_states[N_MEMORY];
3190 
3191         /*
3192          * If subparts_cpus is populated, it is likely that the check below
3193          * will produce a false positive on cpus_updated when the cpu list
3194          * isn't changed. It is extra work, but it is better to be safe.
3195          */

3197         mems_updated = !nodes_equal(top_cpuset.effective_mems, new_mems);

3206                  * do not show up in effective_cpus. If no CPU is left,
3207                  * we clear the subparts_cpus & let the child partitions
3208                  * fight for the CPUs again.
3209                  */

3211                         if (cpumask_subset(&new_cpus,
3212                                            top_cpuset.subparts_cpus)) {
3213                                 top_cpuset.nr_subparts_cpus = 0;
3214                                 cpumask_clear(top_cpuset.subparts_cpus);
3215                         } else {
3216                                 cpumask_andnot(&new_cpus, &new_cpus,
3217                                                top_cpuset.subparts_cpus);
3218                         }
3219                 }
3220                 cpumask_copy(top_cpuset.effective_cpus, &new_cpus);
3221                 spin_unlock_irq(&callback_lock);
3222                 /* we don't mess with cpumasks of tasks in top_cpuset */
3223         }
3224

3227                 spin_lock_irq(&callback_lock);
3228                 if (!on_dfl)
3229                         top_cpuset.mems_allowed = new_mems;
3230                 top_cpuset.effective_mems = new_mems;
3231                 spin_unlock_irq(&callback_lock);
3232                 update_tasks_nodemask(&top_cpuset);
3233         }
3234 
3235         mutex_unlock(&cpuset_mutex);
3236 
3237         /* if cpus or mems changed, we need to propagate to descendants */
3238         if (cpus_updated || mems_updated) {
3239                 struct cpuset *cs;
3240                 struct cgroup_subsys_state *pos_css;
3241 
3242                 rcu_read_lock();

3244                         if (cs == &top_cpuset || !css_tryget_online(&cs->css))
3245                                 continue;

3248                         cpuset_hotplug_update_tasks(cs, ptmp);
3249 
3250                         rcu_read_lock();
3251                         css_put(&cs->css);
3252                 }
3253                 rcu_read_unlock();

3255

3257         if (cpus_updated || force_rebuild) {
3258                 force_rebuild = false;
3259                 rebuild_sched_domains();
3260         }
3261 
3262         free_cpumasks(NULL, ptmp);
3263 }

3439 /**
3440  * cpuset_node_allowed - Can we allocate on a memory node?
3441  * @node: is this an allowed node?
3442  * @gfp_mask: memory allocation flags
3443  *
3444  * If we're in interrupt, yes, we can always allocate.  If @node is set in
3445  * current's mems_allowed, yes.  If it's not a __GFP_HARDWALL request and this
3446  * node is set in the nearest hardwalled cpuset ancestor to current's cpuset,
3447  * yes.  If current has access to memory reserves as an oom victim, yes.
3448  * Otherwise, no.
3449  *
3450  * GFP_USER allocations are marked with the __GFP_HARDWALL bit,
3451  * and do not allow allocations outside the current tasks cpuset
3452  * unless the task has been OOM killed.
3453  * GFP_KERNEL allocations are not so marked, so can escape to the
3454  * nearest enclosing hardwalled ancestor cpuset.
3455  *
3456  * Scanning up parent cpusets requires callback_lock.  The
3457  * __alloc_pages() routine only calls here with __GFP_HARDWALL bit
3458  * _not_ set if it's a GFP_KERNEL allocation, and all nodes in the
3459  * current tasks mems_allowed came up empty on the first pass over
3460  * the zonelist.  So only GFP_KERNEL allocations, if all nodes in the
3461  * cpuset are short of memory, might require taking the callback_lock.
3462  *
3463  * The first call here from mm/page_alloc:get_page_from_freelist()
3464  * has __GFP_HARDWALL set in gfp_mask, enforcing hardwall cpusets,
3465  * so no allocation on a node outside the cpuset is allowed (unless
3466  * in interrupt, of course).
3467  *
3468  * The second pass through get_page_from_freelist() doesn't even call
3469  * here for GFP_ATOMIC calls.  For those calls, the __alloc_pages()
3470  * variable 'wait' is not set, and the bit ALLOC_CPUSET is not set
3471  * in alloc_flags.  That logic and the checks below have the combined
3472  * affect that:
3473  *      in_interrupt - any node ok (current task context irrelevant)
3474  *      GFP_ATOMIC   - any node ok
3475  *      tsk_is_oom_victim   - any node ok
3476  *      GFP_KERNEL   - any node in enclosing hardwalled cpuset ok
3477  *      GFP_USER     - only nodes in current tasks mems allowed ok.
3478  */
3479 bool __cpuset_node_allowed(int node, gfp_t gfp_mask)
     /*   */
3480 {
3481         struct cpuset *cs;              /* current cpuset ancestors */
3482         int allowed;                    /* is allocation in zone z allowed? */
3483         unsigned long flags;
3484 
3485         if (in_interrupt())
3486                 return true;
3487         if (node_isset(node, current->mems_allowed))
3488                 return true;
3489         /*
3490          * Allow tasks that have access to memory reserves because they have
3491          * been OOM killed to get memory anywhere.
3492          */
3493         if (unlikely(tsk_is_oom_victim(current)))
3494                 return true;
3495         if (gfp_mask & __GFP_HARDWALL)  /* If hardwall request, stop here */
3496                 return false;
3497 
3498         if (current->flags & PF_EXITING) /* Let dying task have memory */
3499                 return true;
3500 
3501         /* Not hardwall and node outside mems_allowed: scan up cpusets */
3502         spin_lock_irqsave(&callback_lock, flags);
3503 
3504         rcu_read_lock();
3505         cs = nearest_hardwall_ancestor(task_cs(current));
3506         allowed = node_isset(node, cs->mems_allowed);
3507         rcu_read_unlock();
3508 
3509         spin_unlock_irqrestore(&callback_lock, flags);
3510         return allowed;
3511 }

3513 /**

3515  * cpuset_slab_spread_node() - On which node to begin search for a slab page
3516  *
3517  * If a task is marked PF_SPREAD_PAGE or PF_SPREAD_SLAB (as for
3518  * tasks in a cpuset with is_spread_page or is_spread_slab set),
3519  * and if the memory allocation used cpuset_mem_spread_node()
3520  * to determine on which node to start looking, as it will for
3521  * certain page cache or slab cache pages such as used for file
3522  * system buffers and inode caches, then instead of starting on the
3523  * local node to look for a free page, rather spread the starting
3524  * node around the tasks mems_allowed nodes.
3525  *
3526  * We don't have to worry about the returned node being offline
3527  * because "it can't happen", and even if it did, it would be ok.
3528  *
3529  * The routines calling guarantee_online_mems() are careful to
3530  * only set nodes in task->mems_allowed that are online.  So it
3531  * should not be possible for the following code to return an
