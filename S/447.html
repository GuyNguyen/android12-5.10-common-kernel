 110 /*
 111  * Increase resolution of nice-level calculations for 64-bit architectures.
 112  * The extra resolution improves shares distribution and load balancing of
 113  * low-weight task groups (eg. nice +19 on an autogroup), deeper taskgroup
 114  * hierarchies, especially on larger systems. This is not a user-visible change
 115  * and does not change the user-interface for setting shares/weights.
 116  *
 117  * We increase resolution only if we have enough bits to allow this increased
 118  * resolution (i.e. 64-bit). The costs for increasing resolution when 32-bit
 119  * are pretty high and the returns do not justify the increased costs.
 120  *
 121  * Really only required when CONFIG_FAIR_GROUP_SCHED=y is also set, but to
 122  * increase coverage and consistency always enable it on 64-bit platforms.
 123  */
 124 #ifdef CONFIG_64BIT
 125 # define NICE_0_LOAD_SHIFT      (SCHED_FIXEDPOINT_SHIFT + SCHED_FIXEDPOINT_SHIFT)
 126 # define scale_load(w)          ((w) << SCHED_FIXEDPOINT_SHIFT)
 127 # define scale_load_down(w) \
 128 ({ \
 129         unsigned long __w = (w); \
 130         if (__w) \
 131                 __w = max(2UL, __w >> SCHED_FIXEDPOINT_SHIFT); \
 132         __w; \
 133 })
 134 #else
 135 # define NICE_0_LOAD_SHIFT      (SCHED_FIXEDPOINT_SHIFT)
 136 # define scale_load(w)          (w)
 137 # define scale_load_down(w)     (w)
 138 #endif

 280 /*
 281  * To keep the bandwidth of -deadline tasks under control
 282  * we need some place where:
 283  *  - store the maximum -deadline bandwidth of each cpu;
 284  *  - cache the fraction of bandwidth that is currently allocated in
 285  *    each root domain;
 286  *
 287  * This is all done in the data structure below. It is similar to the
 288  * one used for RT-throttling (rt_bandwidth), with the main difference
 289  * that, since here we are only interested in admission control, we
 290  * do not decrease any runtime while the group "executes", neither we
 291  * need a timer to replenish it.
 292  *
 293  * With respect to SMP, bandwidth is given on a per root domain basis,
 294  * meaning that:
 295  *  - bw (< 100%) is the deadline bandwidth of each CPU;
 296  *  - total_bw is the currently allocated bandwidth in each root domain;
 297  */
 298 struct dl_bw {
 299         raw_spinlock_t          lock;
 300         u64                     bw;
 301         u64                     total_bw;
 302 };

2120 /*
2121  * fair double_lock_balance: Safely acquires both rq->locks in a fair
2122  * way at the expense of forcing extra atomic operations in all
2123  * invocations.  This assures that the double_lock is acquired using the
2124  * same underlying policy as the spinlock_t on this architecture, which
2125  * reduces latency compared to the unfair variant below.  However, it
2126  * also adds more overhead and therefore may reduce throughput.
2127  */
2128 static inline int _double_lock_balance(struct rq *this_rq, struct rq *busiest)
     /*   */
2129         __releases(this_rq->lock)
2130         __acquires(busiest->lock)
2131         __acquires(this_rq->lock)
2132 {
2133         raw_spin_unlock(&this_rq->lock);
2134         double_rq_lock(this_rq, busiest);
2135 
2136         return 1;
2137 }

2140 /*
2141  * Unfair double_lock_balance: Optimizes throughput at the expense of
2142  * latency by eliminating extra atomic operations when the locks are
2143  * already in proper order on entry.  This favors lower CPU-ids and will
2144  * grant the double lock to lower CPUs over higher ids under contention,
2145  * regardless of entry order into the function.
2146  */

     /*   */

2441 /**
2442  * uclamp_rq_util_with - clamp @util with @rq and @p effective uclamp values.
2443  * @rq:         The rq to clamp against. Must not be NULL.
2444  * @util:       The util value to clamp.
2445  * @p:          The task to clamp against. Can be NULL if you want to clamp
2446  *              against @rq only.
2447  *
2448  * Clamps the passed @util to the max(@rq, @p) effective uclamp values.
2449  *
2450  * If sched_uclamp_used static key is disabled, then just return the util
2451  * without any clamping since uclamp aggregation at the rq level in the fast
2452  * path is disabled, rendering this operation a NOP.
2453  *
2454  * Use uclamp_eff_value() if you don't care about uclamp values at rq level. It
2455  * will return the correct effective uclamp value of the task even if the
2456  * static key is disabled.
2457  */
2458 static __always_inline
2459 unsigned long uclamp_rq_util_with(struct rq *rq, unsigned long util,
     /*   */
2460                                   struct task_struct *p)
2461 {
2462         unsigned long min_util;
2463         unsigned long max_util;
2464 
2465         if (!static_branch_likely(&sched_uclamp_used))
2466                 return util;
2467 
2468         min_util = READ_ONCE(rq->uclamp[UCLAMP_MIN].value);
2469         max_util = READ_ONCE(rq->uclamp[UCLAMP_MAX].value);
2470 
2471         if (p) {
2472                 min_util = max(min_util, uclamp_eff_value(p, UCLAMP_MIN));
2473                 max_util = max(max_util, uclamp_eff_value(p, UCLAMP_MAX));
2474         }
2475 
2476         /*
2477          * Since CPU's {min,max}_util clamps are MAX aggregated considering
2478          * RUNNABLE tasks with _different_ clamps, we can end up with an
2479          * inversion. Fix it now when the clamps are applied.
2480          */
2481         if (unlikely(min_util >= max_util))
2482                 return min_util;
2483 
2484         return clamp(util, min_util, max_util);
2485 }
