   1 /*
   2  * Pressure stall information for CPU, memory and IO
   3  *
   4  * Copyright (c) 2018 Facebook, Inc.
   5  * Author: Johannes Weiner <hannes@cmpxchg.org>
   6  *
   7  * Polling support by Suren Baghdasaryan <surenb@google.com>
   8  * Copyright (c) 2018 Google, Inc.
   9  *
  10  * When CPU, memory and IO are contended, tasks experience delays that
  11  * reduce throughput and introduce latencies into the workload. Memory
  12  * and IO contention, in addition, can cause a full loss of forward
  13  * progress in which the CPU goes idle.
  14  *
  15  * This code aggregates individual task delays into resource pressure
  16  * metrics that indicate problems with both workload health and
  17  * resource utilization.
  18  *
  19  *                      Model
  20  *
  21  * The time in which a task can execute on a CPU is our baseline for
  22  * productivity. Pressure expresses the amount of time in which this
  23  * potential cannot be realized due to resource contention.
  24  *
  25  * This concept of productivity has two components: the workload and
  26  * the CPU. To measure the impact of pressure on both, we define two
  27  * contention states for a resource: SOME and FULL.
  28  *
  29  * In the SOME state of a given resource, one or more tasks are
  30  * delayed on that resource. This affects the workload's ability to
  31  * perform work, but the CPU may still be executing other tasks.
  32  *
  33  * In the FULL state of a given resource, all non-idle tasks are
  34  * delayed on that resource such that nobody is advancing and the CPU
  35  * goes idle. This leaves both workload and CPU unproductive.
  36  *
  37  * (Naturally, the FULL state doesn't exist for the CPU resource.)
  38  *
  39  *      SOME = nr_delayed_tasks != 0
  40  *      FULL = nr_delayed_tasks != 0 && nr_running_tasks == 0
  41  *
  42  * The percentage of wallclock time spent in those compound stall
  43  * states gives pressure numbers between 0 and 100 for each resource,
  44  * where the SOME percentage indicates workload slowdowns and the FULL
  45  * percentage indicates reduced CPU utilization:
  46  *
  47  *      %SOME = time(SOME) / period
  48  *      %FULL = time(FULL) / period
  49  *
  50  *                      Multiple CPUs
  51  *
  52  * The more tasks and available CPUs there are, the more work can be
  53  * performed concurrently. This means that the potential that can go
  54  * unrealized due to resource contention *also* scales with non-idle
  55  * tasks and CPUs.
  56  *
  57  * Consider a scenario where 257 number crunching tasks are trying to
  58  * run concurrently on 256 CPUs. If we simply aggregated the task
  59  * states, we would have to conclude a CPU SOME pressure number of
  60  * 100%, since *somebody* is waiting on a runqueue at all
  61  * times. However, that is clearly not the amount of contention the
  62  * workload is experiencing: only one out of 256 possible exceution
  63  * threads will be contended at any given time, or about 0.4%.
  64  *
  65  * Conversely, consider a scenario of 4 tasks and 4 CPUs where at any
  66  * given time *one* of the tasks is delayed due to a lack of memory.
  67  * Again, looking purely at the task state would yield a memory FULL
  68  * pressure number of 0%, since *somebody* is always making forward
  69  * progress. But again this wouldn't capture the amount of execution
  70  * potential lost, which is 1 out of 4 CPUs, or 25%.
  71  *
  72  * To calculate wasted potential (pressure) with multiple processors,
  73  * we have to base our calculation on the number of non-idle tasks in
  74  * conjunction with the number of available CPUs, which is the number
  75  * of potential execution threads. SOME becomes then the proportion of
  76  * delayed tasks to possibe threads, and FULL is the share of possible
  77  * threads that are unproductive due to delays:
  78  *
  79  *      threads = min(nr_nonidle_tasks, nr_cpus)
  80  *         SOME = min(nr_delayed_tasks / threads, 1)
  81  *         FULL = (threads - min(nr_running_tasks, threads)) / threads
  82  *
  83  * For the 257 number crunchers on 256 CPUs, this yields:
  84  *
  85  *      threads = min(257, 256)
  86  *         SOME = min(1 / 256, 1)             = 0.4%
  87  *         FULL = (256 - min(257, 256)) / 256 = 0%
  88  *
  89  * For the 1 out of 4 memory-delayed tasks, this yields:
  90  *
  91  *      threads = min(4, 4)
  92  *         SOME = min(1 / 4, 1)               = 25%
  93  *         FULL = (4 - min(3, 4)) / 4         = 25%
  94  *
  95  * [ Substitute nr_cpus with 1, and you can see that it's a natural
  96  *   extension of the single-CPU model. ]
  97  *
  98  *                      Implementation
  99  *
 100  * To assess the precise time spent in each such state, we would have
 101  * to freeze the system on task changes and start/stop the state
 102  * clocks accordingly. Obviously that doesn't scale in practice.
 103  *
 104  * Because the scheduler aims to distribute the compute load evenly
 105  * among the available CPUs, we can track task state locally to each
 106  * CPU and, at much lower frequency, extrapolate the global state for
 107  * the cumulative stall times and the running averages.
 108  *
 109  * For each runqueue, we track:
 110  *
 111  *         tSOME[cpu] = time(nr_delayed_tasks[cpu] != 0)
 112  *         tFULL[cpu] = time(nr_delayed_tasks[cpu] && !nr_running_tasks[cpu])
 113  *      tNONIDLE[cpu] = time(nr_nonidle_tasks[cpu] != 0)
 114  *
 115  * and then periodically aggregate:
 116  *
 117  *      tNONIDLE = sum(tNONIDLE[i])
 118  *
 119  *         tSOME = sum(tSOME[i] * tNONIDLE[i]) / tNONIDLE
 120  *         tFULL = sum(tFULL[i] * tNONIDLE[i]) / tNONIDLE
 121  *
 122  *         %SOME = tSOME / period
 123  *         %FULL = tFULL / period
 124  *
 125  * This gives us an approximation of pressure that is practical
 126  * cost-wise, yet way more sensitive and accurate than periodic
 127  * sampling of the aggregate task states would be.
 128  */

 185 {
 186         int cpu;
 187 
 188         for_each_possible_cpu(cpu)
 189                 seqcount_init(&per_cpu_ptr(group->pcpu, cpu)->seq);
 190         group->avg_last_update = sched_clock();
 191         group->avg_next_update = group->avg_last_update + psi_period;
 192         INIT_DELAYED_WORK(&group->avgs_work, psi_avgs_work);

 197         memset(group->nr_triggers, 0, sizeof(group->nr_triggers));
 198         group->poll_states = 0;
 199         group->poll_min_period = U32_MAX;
 200         memset(group->polling_total, 0, sizeof(group->polling_total));
 201         group->polling_next_update = ULLONG_MAX;
 202         group->polling_until = 0;
 203         rcu_assign_pointer(group->poll_task, NULL);

 217 static bool test_state(unsigned int *tasks, enum psi_states state)
 218 {
 219         switch (state) {
 220         case PSI_IO_SOME:
 221                 return tasks[NR_IOWAIT];
 222         case PSI_IO_FULL:
 223                 return tasks[NR_IOWAIT] && !tasks[NR_RUNNING];
 224         case PSI_MEM_SOME:
 225                 return tasks[NR_MEMSTALL];
 226         case PSI_MEM_FULL:
 227                 return tasks[NR_MEMSTALL] && !tasks[NR_RUNNING];
 228         case PSI_CPU_SOME:
 229                 return tasks[NR_RUNNING] > tasks[NR_ONCPU];
 230         case PSI_NONIDLE:
 231                 return tasks[NR_IOWAIT] || tasks[NR_MEMSTALL] ||
 232                         tasks[NR_RUNNING];
 233         default:
 234                 return false;
 235         }
 236 }

 238 static void get_recent_times(struct psi_group *group, int cpu,
 239                              enum psi_aggregators aggregator, u32 *times,
 240                              u32 *pchanged_states)
 241 {
 242         struct psi_group_cpu *groupc = per_cpu_ptr(group->pcpu, cpu);
 243         u64 now, state_start;
 244         enum psi_states s;
 245         unsigned int seq;
 246         u32 state_mask;
 247 
 248         *pchanged_states = 0;
 249 
 250         /* Snapshot a coherent view of the CPU state */
 251         do {
 252                 seq = read_seqcount_begin(&groupc->seq);
 253                 now = cpu_clock(cpu);
 254                 memcpy(times, groupc->times, sizeof(groupc->times));
 255                 state_mask = groupc->state_mask;
 256                 state_start = groupc->state_start;
 257         } while (read_seqcount_retry(&groupc->seq, seq));
 258 
 259         /* Calculate state time deltas against the previous snapshot */
 260         for (s = 0; s < NR_PSI_STATES; s++) {
 261                 u32 delta;
 262                 /*
 263                  * In addition to already concluded states, we also
 264                  * incorporate currently active states on the CPU,
 265                  * since states may last for many sampling periods.
 266                  *
 267                  * This way we keep our delta sampling buckets small
 268                  * (u32) and our reported pressure close to what's
 269                  * actually happening.
 270                  */
 271                 if (state_mask & (1 << s))
 272                         times[s] += now - state_start;
 273 
 274                 delta = times[s] - groupc->times_prev[aggregator][s];
 275                 groupc->times_prev[aggregator][s] = times[s];
 276 
 277                 times[s] = delta;
 278                 if (delta)
 279                         *pchanged_states |= (1 << s);
 280         }
 281 }

 283 static void calc_avgs(unsigned long avg[3], int missed_periods,
 284                       u64 time, u64 period)
 285 {
 286         unsigned long pct;
 287 
 288         /* Fill in zeroes for periods of no activity */
 289         if (missed_periods) {
 290                 avg[0] = calc_load_n(avg[0], EXP_10s, 0, missed_periods);
 291                 avg[1] = calc_load_n(avg[1], EXP_60s, 0, missed_periods);
 292                 avg[2] = calc_load_n(avg[2], EXP_300s, 0, missed_periods);
 293         }
 294 
 295         /* Sample the most recent active period */
 296         pct = div_u64(time * 100, period);
 297         pct *= FIXED_1;
 298         avg[0] = calc_load(avg[0], EXP_10s, pct);
 299         avg[1] = calc_load(avg[1], EXP_60s, pct);
 300         avg[2] = calc_load(avg[2], EXP_300s, pct);
 301 }

 303 static void collect_percpu_times(struct psi_group *group,
 304                                  enum psi_aggregators aggregator,
 305                                  u32 *pchanged_states)
 306 {
 307         u64 deltas[NR_PSI_STATES - 1] = { 0, };
 308         unsigned long nonidle_total = 0;
 309         u32 changed_states = 0;
 310         int cpu;
 311         int s;
 312 
 313         /*
 314          * Collect the per-cpu time buckets and average them into a
 315          * single time sample that is normalized to wallclock time.
 316          *
 317          * For averaging, each CPU is weighted by its non-idle time in
 318          * the sampling period. This eliminates artifacts from uneven
 319          * loading, or even entirely idle CPUs.
 320          */
 321         for_each_possible_cpu(cpu) {
 322                 u32 times[NR_PSI_STATES];
 323                 u32 nonidle;
 324                 u32 cpu_changed_states;
 325 
 326                 get_recent_times(group, cpu, aggregator, times,
 327                                 &cpu_changed_states);
 328                 changed_states |= cpu_changed_states;
 329 
 330                 nonidle = nsecs_to_jiffies(times[PSI_NONIDLE]);
 331                 nonidle_total += nonidle;
 332 
 333                 for (s = 0; s < PSI_NONIDLE; s++)
 334                         deltas[s] += (u64)times[s] * nonidle;
 335         }
 336 
 337         /*
 338          * Integrate the sample into the running statistics that are
 339          * reported to userspace: the cumulative stall times and the
 340          * decaying averages.
 341          *
 342          * Pressure percentages are sampled at PSI_FREQ. We might be
 343          * called more often when the user polls more frequently than
 344          * that; we might be called less often when there is no task
 345          * activity, thus no data, and clock ticks are sporadic. The
 346          * below handles both.
 347          */
 348 
 349         /* total= */
 350         for (s = 0; s < NR_PSI_STATES - 1; s++)
 351                 group->total[aggregator][s] +=
 352                                 div_u64(deltas[s], max(nonidle_total, 1UL));
 353 
 354         if (pchanged_states)
 355                 *pchanged_states = changed_states;
 356 }

 358 static u64 update_averages(struct psi_group *group, u64 now)
 359 {
 360         unsigned long missed_periods = 0;
 361         u64 expires, period;
 362         u64 avg_next_update;
 363         int s;
 364 
 365         /* avgX= */
 366         expires = group->avg_next_update;
 367         if (now - expires >= psi_period)
 368                 missed_periods = div_u64(now - expires, psi_period);
 369 
 370         /*
 371          * The periodic clock tick can get delayed for various
 372          * reasons, especially on loaded systems. To avoid clock
 373          * drift, we schedule the clock in fixed psi_period intervals.
 374          * But the deltas we sample out of the per-cpu buckets above
 375          * are based on the actual time elapsing between clock ticks.
 376          */
 377         avg_next_update = expires + ((1 + missed_periods) * psi_period);
 378         period = now - (group->avg_last_update + (missed_periods * psi_period));
 379         group->avg_last_update = now;
 380 
 381         for (s = 0; s < NR_PSI_STATES - 1; s++) {
 382                 u32 sample;
 383 
 384                 sample = group->total[PSI_AVGS][s] - group->avg_total[s];
 385                 /*
 386                  * Due to the lockless sampling of the time buckets,
 387                  * recorded time deltas can slip into the next period,
 388                  * which under full pressure can result in samples in
 389                  * excess of the period length.
 390                  *
 391                  * We don't want to report non-sensical pressures in
 392                  * excess of 100%, nor do we want to drop such events
 393                  * on the floor. Instead we punt any overage into the
 394                  * future until pressure subsides. By doing this we
 395                  * don't underreport the occurring pressure curve, we
 396                  * just report it delayed by one period length.
 397                  *
 398                  * The error isn't cumulative. As soon as another
 399                  * delta slips from a period P to P+1, by definition
 400                  * it frees up its time T in P.
 401                  */
 402                 if (sample > period)
 403                         sample = period;
 404                 group->avg_total[s] += sample;
 405                 calc_avgs(group->avg[s], missed_periods, sample, period);
 406         }
 407 
 408         return avg_next_update;
 409 }

 411 static void psi_avgs_work(struct work_struct *work)
 412 {
 413         struct delayed_work *dwork;
 414         struct psi_group *group;
 415         u32 changed_states;
 416         bool nonidle;
 417         u64 now;
 418 
 419         dwork = to_delayed_work(work);
 420         group = container_of(dwork, struct psi_group, avgs_work);
 421 
 422         mutex_lock(&group->avgs_lock);
 423 
 424         now = sched_clock();
 425 
 426         collect_percpu_times(group, PSI_AVGS, &changed_states);
 427         nonidle = changed_states & (1 << PSI_NONIDLE);
 428         /*
 429          * If there is task activity, periodically fold the per-cpu
 430          * times and feed samples into the running averages. If things
 431          * are idle and there is no data to process, stop the clock.
 432          * Once restarted, we'll catch up the running averages in one
 433          * go - see calc_avgs() and missed_periods.
 434          */
 435         if (now >= group->avg_next_update)
 436                 group->avg_next_update = update_averages(group, now);
 437 
 438         if (nonidle) {
 439                 schedule_delayed_work(dwork, nsecs_to_jiffies(
 440                                 group->avg_next_update - now) + 1);
 441         }
 442 
 443         mutex_unlock(&group->avgs_lock);
 444 }

 446 /* Trigger tracking window manupulations */
 447 static void window_reset(struct psi_window *win, u64 now, u64 value,
 448                          u64 prev_growth)
 449 {
 450         win->start_time = now;
 451         win->start_value = value;
 452         win->prev_growth = prev_growth;
 453 }
 454 
 455 /*
 456  * PSI growth tracking window update and growth calculation routine.
 457  *
 458  * This approximates a sliding tracking window by interpolating
 459  * partially elapsed windows using historical growth data from the
 460  * previous intervals. This minimizes memory requirements (by not storing
 461  * all the intermediate values in the previous window) and simplifies
 462  * the calculations. It works well because PSI signal changes only in
 463  * positive direction and over relatively small window sizes the growth
 464  * is close to linear.
 465  */
 466 static u64 window_update(struct psi_window *win, u64 now, u64 value)
 467 {
 468         u64 elapsed;
 469         u64 growth;
 470 
 471         elapsed = now - win->start_time;
 472         growth = value - win->start_value;
 473         /*
 474          * After each tracking window passes win->start_value and
 475          * win->start_time get reset and win->prev_growth stores
 476          * the average per-window growth of the previous window.
 477          * win->prev_growth is then used to interpolate additional
 478          * growth from the previous window assuming it was linear.
 479          */
 480         if (elapsed > win->size)
 481                 window_reset(win, now, value, growth);
 482         else {
 483                 u32 remaining;
 484 
 485                 remaining = win->size - elapsed;
 486                 growth += div64_u64(win->prev_growth * remaining, win->size);
 487         }
 488 
 489         return growth;
 490 }

 492 static void init_triggers(struct psi_group *group, u64 now)
 493 {
 494         struct psi_trigger *t;
 495 
 496         list_for_each_entry(t, &group->triggers, node)
 497                 window_reset(&t->win, now,
 498                                 group->total[PSI_POLL][t->state], 0);
 499         memcpy(group->polling_total, group->total[PSI_POLL],
 500                    sizeof(group->polling_total));
 501         group->polling_next_update = now + group->poll_min_period;
 502 }

 504 static u64 update_triggers(struct psi_group *group, u64 now)
 505 {
 506         struct psi_trigger *t;
 507         bool new_stall = false;
 508         u64 *total = group->total[PSI_POLL];
 509 
 510         /*
 511          * On subsequent updates, calculate growth deltas and let
 512          * watchers know when their specified thresholds are exceeded.
 513          */
 514         list_for_each_entry(t, &group->triggers, node) {
 515                 u64 growth;
 516 
 517                 /* Check for stall activity */
 518                 if (group->polling_total[t->state] == total[t->state])
 519                         continue;
 520 
 521                 /*
 522                  * Multiple triggers might be looking at the same state,
 523                  * remember to update group->polling_total[] once we've
 524                  * been through all of them. Also remember to extend the
 525                  * polling time if we see new stall activity.
 526                  */
 527                 new_stall = true;
 528 
 529                 /* Calculate growth since last update */
 530                 growth = window_update(&t->win, now, total[t->state]);
 531                 if (growth < t->threshold)
 532                         continue;
 533 
 534                 /* Limit event signaling to once per window */
 535                 if (now < t->last_event_time + t->win.size)
 536                         continue;
 537 
 538                 trace_android_vh_psi_event(t);
 539 
 540                 /* Generate an event */
 541                 if (cmpxchg(&t->event, 0, 1) == 0)
 542                         wake_up_interruptible(&t->event_wait);
 543                 t->last_event_time = now;
 544         }
 545 
 546         trace_android_vh_psi_group(group);
 547 
 548         if (new_stall)
 549                 memcpy(group->polling_total, total,
 550                                 sizeof(group->polling_total));
 551 
 552         return now + group->poll_min_period;
 553 }

 555 /* Schedule polling if it's not already scheduled. */
 556 static void psi_schedule_poll_work(struct psi_group *group, unsigned long delay)
 557 {
 558         struct task_struct *task;
 559 
 560         /*
 561          * Do not reschedule if already scheduled.
 562          * Possible race with a timer scheduled after this check but before
 563          * mod_timer below can be tolerated because group->polling_next_update
 564          * will keep updates on schedule.
 565          */
 566         if (timer_pending(&group->poll_timer))
 567                 return;
 568 
 569         rcu_read_lock();
 570 
 571         task = rcu_dereference(group->poll_task);
 572         /*
 573          * kworker might be NULL in case psi_trigger_destroy races with
 574          * psi_task_change (hotpath) which can't use locks
 575          */
 576         if (likely(task))
 577                 mod_timer(&group->poll_timer, jiffies + delay);
 578 
 579         rcu_read_unlock();
 580 }

 582 static void psi_poll_work(struct psi_group *group)
 583 {
 584         u32 changed_states;
 585         u64 now;
 586 
 587         mutex_lock(&group->trigger_lock);
 588 
 589         now = sched_clock();
 590 
 591         collect_percpu_times(group, PSI_POLL, &changed_states);
 592 
 593         if (changed_states & group->poll_states) {
 594                 /* Initialize trigger windows when entering polling mode */
 595                 if (now > group->polling_until)
 596                         init_triggers(group, now);
 597 
 598                 /*
 599                  * Keep the monitor active for at least the duration of the
 600                  * minimum tracking window as long as monitor states are
 601                  * changing.
 602                  */
 603                 group->polling_until = now +
 604                         group->poll_min_period * UPDATES_PER_WINDOW;
 605         }
 606 
 607         if (now > group->polling_until) {
 608                 group->polling_next_update = ULLONG_MAX;
 609                 goto out;
 610         }
 611 
 612         if (now >= group->polling_next_update)
 613                 group->polling_next_update = update_triggers(group, now);
 614 
 615         psi_schedule_poll_work(group,
 616                 nsecs_to_jiffies(group->polling_next_update - now) + 1);
 617 
 618 out:
 619         mutex_unlock(&group->trigger_lock);
 620 }
 621 
 622 static int psi_poll_worker(void *data)
 623 {
 624         struct psi_group *group = (struct psi_group *)data;
 625 
 626         sched_set_fifo_low(current);
 627 
 628         while (true) {
 629                 wait_event_interruptible(group->poll_wait,
 630                                 atomic_cmpxchg(&group->poll_wakeup, 1, 0) ||
 631                                 kthread_should_stop());
 632                 if (kthread_should_stop())
 633                         break;
 634 
 635                 psi_poll_work(group);
 636         }
 637         return 0;
 638 }

 640 static void poll_timer_fn(struct timer_list *t)
 641 {
 642         struct psi_group *group = from_timer(group, t, poll_timer);
 643 
 644         atomic_set(&group->poll_wakeup, 1);
 645         wake_up_interruptible(&group->poll_wait);
 646 }

 648 static void record_times(struct psi_group_cpu *groupc, int cpu,
 649                          bool memstall_tick)
 650 {
 651         u32 delta;
 652         u64 now;
 653 
 654         now = cpu_clock(cpu);
 655         delta = now - groupc->state_start;
 656         groupc->state_start = now;
 657 
 658         if (groupc->state_mask & (1 << PSI_IO_SOME)) {
 659                 groupc->times[PSI_IO_SOME] += delta;
 660                 if (groupc->state_mask & (1 << PSI_IO_FULL))
 661                         groupc->times[PSI_IO_FULL] += delta;
 662         }
 663 
 664         if (groupc->state_mask & (1 << PSI_MEM_SOME)) {
 665                 groupc->times[PSI_MEM_SOME] += delta;
 666                 if (groupc->state_mask & (1 << PSI_MEM_FULL))
 667                         groupc->times[PSI_MEM_FULL] += delta;
 668                 else if (memstall_tick) {
 669                         u32 sample;
 670                         /*
 671                          * Since we care about lost potential, a
 672                          * memstall is FULL when there are no other
 673                          * working tasks, but also when the CPU is
 674                          * actively reclaiming and nothing productive
 675                          * could run even if it were runnable.
 676                          *
 677                          * When the timer tick sees a reclaiming CPU,
 678                          * regardless of runnable tasks, sample a FULL
 679                          * tick (or less if it hasn't been a full tick
 680                          * since the last state change).
 681                          */
 682                         sample = min(delta, (u32)jiffies_to_nsecs(1));
 683                         groupc->times[PSI_MEM_FULL] += sample;
 684                 }
 685         }
 686 
 687         if (groupc->state_mask & (1 << PSI_CPU_SOME))
 688                 groupc->times[PSI_CPU_SOME] += delta;
 689 
 690         if (groupc->state_mask & (1 << PSI_NONIDLE))
 691                 groupc->times[PSI_NONIDLE] += delta;
 692 }

 694 static void psi_group_change(struct psi_group *group, int cpu,
 695                              unsigned int clear, unsigned int set,
 696                              bool wake_clock)
 697 {
 698         struct psi_group_cpu *groupc;
 699         u32 state_mask = 0;
 700         unsigned int t, m;
 701         enum psi_states s;
 702 
 703         groupc = per_cpu_ptr(group->pcpu, cpu);
 704 
 705         /*
 706          * First we assess the aggregate resource states this CPU's
 707          * tasks have been in since the last change, and account any
 708          * SOME and FULL time these may have resulted in.
 709          *
 710          * Then we update the task counts according to the state
 711          * change requested through the @clear and @set bits.
 712          */
 713         write_seqcount_begin(&groupc->seq);
 714 
 715         record_times(groupc, cpu, false);
 716 
 717         for (t = 0, m = clear; m; m &= ~(1 << t), t++) {
 718                 if (!(m & (1 << t)))
 719                         continue;
 720                 if (groupc->tasks[t]) {
 721                         groupc->tasks[t]--;
 722                 } else if (!psi_bug) {
 723                         printk_deferred(KERN_ERR "psi: task underflow! cpu=%d t=%d tasks=[%u %u %u %u] clear=%x set=%x\n",
 724                                         cpu, t, groupc->tasks[0],
 725                                         groupc->tasks[1], groupc->tasks[2],
 726                                         groupc->tasks[3], clear, set);
 727                         psi_bug = 1;
 728                 }
 729         }
 730 
 731         for (t = 0; set; set &= ~(1 << t), t++)
 732                 if (set & (1 << t))
 733                         groupc->tasks[t]++;
 734 
 735         /* Calculate state mask representing active states */
 736         for (s = 0; s < NR_PSI_STATES; s++) {
 737                 if (test_state(groupc->tasks, s))
 738                         state_mask |= (1 << s);
 739         }
 740         groupc->state_mask = state_mask;
 741 
 742         write_seqcount_end(&groupc->seq);
 743 
 744         if (state_mask & group->poll_states)
 745                 psi_schedule_poll_work(group, 1);
 746 
 747         if (wake_clock && !delayed_work_pending(&group->avgs_work))
 748                 schedule_delayed_work(&group->avgs_work, PSI_FREQ);
 749 }

 752 {
 753 #if defined CONFIG_CGROUPS && defined CONFIG_PSI_PER_CGROUP_ACCT
 754         struct cgroup *cgroup = NULL;

 757                 cgroup = task->cgroups->dfl_cgrp;

 775 static void psi_flags_change(struct task_struct *task, int clear, int set)
 776 {
 777         if (((task->psi_flags & set) ||
 778              (task->psi_flags & clear) != clear) &&
 779             !psi_bug) {
 780                 printk_deferred(KERN_ERR "psi: inconsistent task state! task=%d:%s cpu=%d psi_flags=%x clear=%x set=%x\n",
 781                                 task->pid, task->comm, task_cpu(task),
 782                                 task->psi_flags, clear, set);
 783                 psi_bug = 1;
 784         }
 785 
 786         task->psi_flags &= ~clear;
 787         task->psi_flags |= set;
 788 }

 790 void psi_task_change(struct task_struct *task, int clear, int set)
 791 {
 792         int cpu = task_cpu(task);
 793         struct psi_group *group;
 794         bool wake_clock = true;
 795         void *iter = NULL;
 796 
 797         if (!task->pid)
 798                 return;
 799 
 800         psi_flags_change(task, clear, set);
 801 
 802         /*
 803          * Periodic aggregation shuts off if there is a period of no
 804          * task changes, so we wake it back up if necessary. However,
 805          * don't do this if the task change is the aggregation worker
 806          * itself going to sleep, or we'll ping-pong forever.
 807          */
 808         if (unlikely((clear & TSK_RUNNING) &&
 809                      (task->flags & PF_WQ_WORKER) &&
 810                      wq_worker_last_func(task) == psi_avgs_work))
 811                 wake_clock = false;
 812 
 813         while ((group = iterate_groups(task, &iter)))
 814                 psi_group_change(group, cpu, clear, set, wake_clock);
 815 }

 817 void psi_task_switch(struct task_struct *prev, struct task_struct *next,
 818                      bool sleep)
 819 {
 820         struct psi_group *group, *common = NULL;
 821         int cpu = task_cpu(prev);
 822         void *iter;
 823 
 824         if (next->pid) {
 825                 psi_flags_change(next, 0, TSK_ONCPU);
 826                 /*
 827                  * When moving state between tasks, the group that
 828                  * contains them both does not change: we can stop
 829                  * updating the tree once we reach the first common
 830                  * ancestor. Iterate @next's ancestors until we
 831                  * encounter @prev's state.
 832                  */
 833                 iter = NULL;
 834                 while ((group = iterate_groups(next, &iter))) {
 835                         if (per_cpu_ptr(group->pcpu, cpu)->tasks[NR_ONCPU]) {
 836                                 common = group;
 837                                 break;
 838                         }
 839 
 840                         psi_group_change(group, cpu, 0, TSK_ONCPU, true);
 841                 }
 842         }
 843 
 844         /*
 845          * If this is a voluntary sleep, dequeue will have taken care
 846          * of the outgoing TSK_ONCPU alongside TSK_RUNNING already. We
 847          * only need to deal with it during preemption.
 848          */
 849         if (sleep)
 850                 return;
 851 
 852         if (prev->pid) {
 853                 psi_flags_change(prev, TSK_ONCPU, 0);
 854 
 855                 iter = NULL;
 856                 while ((group = iterate_groups(prev, &iter)) && group != common)
 857                         psi_group_change(group, cpu, TSK_ONCPU, 0, true);
 858         }
 859 }

 861 void psi_memstall_tick(struct task_struct *task, int cpu)
 862 {
 863         struct psi_group *group;
 864         void *iter = NULL;
 865 
 866         while ((group = iterate_groups(task, &iter))) {
 867                 struct psi_group_cpu *groupc;
 868 
 869                 groupc = per_cpu_ptr(group->pcpu, cpu);
 870                 write_seqcount_begin(&groupc->seq);
 871                 record_times(groupc, cpu, true);
 872                 write_seqcount_end(&groupc->seq);
 873         }
 874 }

 876 /**
 877  * psi_memstall_enter - mark the beginning of a memory stall section
 878  * @flags: flags to handle nested sections
 879  *
 880  * Marks the calling task as being stalled due to a lack of memory,
 881  * such as waiting for a refault or performing reclaim.
 882  */
 883 void psi_memstall_enter(unsigned long *flags)
 884 {
 885         struct rq_flags rf;
 886         struct rq *rq;
 887 
 888         if (static_branch_likely(&psi_disabled))
 889                 return;
 890 
 891         *flags = current->in_memstall;
 892         if (*flags)
 893                 return;
 894         /*
 895          * in_memstall setting & accounting needs to be atomic wrt
 896          * changes to the task's scheduling state, otherwise we can
 897          * race with CPU migration.
 898          */
 899         rq = this_rq_lock_irq(&rf);
 900 
 901         current->in_memstall = 1;
 902         psi_task_change(current, 0, TSK_MEMSTALL);
 903 
 904         rq_unlock_irq(rq, &rf);
 905 }
 906 
 907 /**
 908  * psi_memstall_leave - mark the end of an memory stall section
 909  * @flags: flags to handle nested memdelay sections
 910  *
 911  * Marks the calling task as no longer stalled due to lack of memory.
 912  */
 913 void psi_memstall_leave(unsigned long *flags)
 914 {
 915         struct rq_flags rf;
 916         struct rq *rq;
 917 
 918         if (static_branch_likely(&psi_disabled))
 919                 return;
 920 
 921         if (*flags)
 922                 return;
 923         /*
 924          * in_memstall clearing & accounting needs to be atomic wrt
 925          * changes to the task's scheduling state, otherwise we could
 926          * race with CPU migration.
 927          */
 928         rq = this_rq_lock_irq(&rf);
 929 
 930         current->in_memstall = 0;
 931         psi_task_change(current, TSK_MEMSTALL, 0);
 932 
 933         rq_unlock_irq(rq, &rf);
 934 }

 960 /**
 961  * cgroup_move_task - move task to a different cgroup
 962  * @task: the task
 963  * @to: the target css_set
 964  *
 965  * Move task to a new cgroup and safely migrate its associated stall
 966  * state between the different groups.
 967  *
 968  * This function acquires the task's rq lock to lock out concurrent
 969  * changes to the task's scheduling state and - in case the task is
 970  * running - concurrent changes to its stall state.
 971  */
 972 void cgroup_move_task(struct task_struct *task, struct css_set *to)
 973 {
 974         unsigned int task_flags = 0;
 975         struct rq_flags rf;
 976         struct rq *rq;
 977 
 978         if (static_branch_likely(&psi_disabled)) {
 979                 /*
 980                  * Lame to do this here, but the scheduler cannot be locked
 981                  * from the outside, so we move cgroups from inside sched/.
 982                  */
 983                 rcu_assign_pointer(task->cgroups, to);
 984                 return;
 985         }
 986 
 987         rq = task_rq_lock(task, &rf);
 988 
 989         if (task_on_rq_queued(task)) {
 990                 task_flags = TSK_RUNNING;
 991                 if (task_current(rq, task))
 992                         task_flags |= TSK_ONCPU;
 993         } else if (task->in_iowait)
 994                 task_flags = TSK_IOWAIT;
 995 
 996         if (task->in_memstall)
 997                 task_flags |= TSK_MEMSTALL;
 998 
 999         if (task_flags)
1000                 psi_task_change(task, task_flags, 0);
1001 
1002         /* See comment above */
1003         rcu_assign_pointer(task->cgroups, to);
1004 
1005         if (task_flags)
1006                 psi_task_change(task, 0, task_flags);
1007 
1008         task_rq_unlock(rq, task, &rf);
1009 }

1012 int psi_show(struct seq_file *m, struct psi_group *group, enum psi_res res)
1013 {
1014         int full;
1015         u64 now;
1016 
1017         if (static_branch_likely(&psi_disabled))
1018                 return -EOPNOTSUPP;
1019 
1020         /* Update averages before reporting them */
1021         mutex_lock(&group->avgs_lock);
1022         now = sched_clock();
1023         collect_percpu_times(group, PSI_AVGS, NULL);
1024         if (now >= group->avg_next_update)
1025                 group->avg_next_update = update_averages(group, now);
1026         mutex_unlock(&group->avgs_lock);
1027 
1028         for (full = 0; full < 2 - (res == PSI_CPU); full++) {
1029                 unsigned long avg[3];
1030                 u64 total;
1031                 int w;
1032 
1033                 for (w = 0; w < 3; w++)
1034                         avg[w] = group->avg[res * 2 + full][w];
1035                 total = div_u64(group->total[PSI_AVGS][res * 2 + full],
1036                                 NSEC_PER_USEC);

1038                 seq_printf(m, "%s avg10=%lu.%02lu avg60=%lu.%02lu avg300=%lu.%02lu total=%llu\n",
1039                            full ? "full" : "some",
1040                            LOAD_INT(avg[0]), LOAD_FRAC(avg[0]),
1041                            LOAD_INT(avg[1]), LOAD_FRAC(avg[1]),
1042                            LOAD_INT(avg[2]), LOAD_FRAC(avg[2]),
1043                            total);
1044         }
1045

1079 struct psi_trigger *psi_trigger_create(struct psi_group *group,
1080                         char *buf, size_t nbytes, enum psi_res res)
1081 {
1082         struct psi_trigger *t;
1083         enum psi_states state;
1084         u32 threshold_us;
1085         u32 window_us;
1086 
1087         if (static_branch_likely(&psi_disabled))
1088                 return ERR_PTR(-EOPNOTSUPP);
1089 
1090         if (sscanf(buf, "some %u %u", &threshold_us, &window_us) == 2)
1091                 state = PSI_IO_SOME + res * 2;
1092         else if (sscanf(buf, "full %u %u", &threshold_us, &window_us) == 2)
1093                 state = PSI_IO_FULL + res * 2;
1094         else
1095                 return ERR_PTR(-EINVAL);
1096 
1097         if (state >= PSI_NONIDLE)
1098                 return ERR_PTR(-EINVAL);
1099 
1100         if (window_us < WINDOW_MIN_US ||
1101                 window_us > WINDOW_MAX_US)
1102                 return ERR_PTR(-EINVAL);
1103 
1104         /* Check threshold */
1105         if (threshold_us == 0 || threshold_us > window_us)
1106                 return ERR_PTR(-EINVAL);
1107 
1108         t = kmalloc(sizeof(*t), GFP_KERNEL);
1109         if (!t)
1110                 return ERR_PTR(-ENOMEM);
1111 
1112         t->group = group;
1113         t->state = state;
1114         t->threshold = threshold_us * NSEC_PER_USEC;
1115         t->win.size = window_us * NSEC_PER_USEC;
1116         window_reset(&t->win, 0, 0, 0);
1117 
1118         t->event = 0;
1119         t->last_event_time = 0;
1120         init_waitqueue_head(&t->event_wait);
1121         kref_init(&t->refcount);
1122 
1123         mutex_lock(&group->trigger_lock);
1124 
1125         if (!rcu_access_pointer(group->poll_task)) {
1126                 struct task_struct *task;
1127 
1128                 task = kthread_create(psi_poll_worker, group, "psimon");
1129                 if (IS_ERR(task)) {
1130                         kfree(t);
1131                         mutex_unlock(&group->trigger_lock);
1132                         return ERR_CAST(task);
1133                 }
1134                 atomic_set(&group->poll_wakeup, 0);
1135                 init_waitqueue_head(&group->poll_wait);
1136                 wake_up_process(task);
1137                 timer_setup(&group->poll_timer, poll_timer_fn, 0);
1138                 rcu_assign_pointer(group->poll_task, task);
1139         }
1140 
1141         list_add(&t->node, &group->triggers);
1142         group->poll_min_period = min(group->poll_min_period,
1143                 div_u64(t->win.size, UPDATES_PER_WINDOW));
1144         group->nr_triggers[t->state]++;
1145         group->poll_states |= (1 << t->state);
1146 
1147         mutex_unlock(&group->trigger_lock);
1148 
1149         return t;
1150 }

1152 static void psi_trigger_destroy(struct kref *ref)
1153 {
1154         struct psi_trigger *t = container_of(ref, struct psi_trigger, refcount);
1155         struct psi_group *group = t->group;
1156         struct task_struct *task_to_destroy = NULL;
1157 
1158         if (static_branch_likely(&psi_disabled))
1159                 return;
1160 
1161         /*
1162          * Wakeup waiters to stop polling. Can happen if cgroup is deleted
1163          * from under a polling process.
1164          */
1165         wake_up_interruptible(&t->event_wait);
1166 
1167         mutex_lock(&group->trigger_lock);
1168 
1169         if (!list_empty(&t->node)) {
1170                 struct psi_trigger *tmp;
1171                 u64 period = ULLONG_MAX;
1172 
1173                 list_del(&t->node);
1174                 group->nr_triggers[t->state]--;
1175                 if (!group->nr_triggers[t->state])
1176                         group->poll_states &= ~(1 << t->state);
1177                 /* reset min update period for the remaining triggers */
1178                 list_for_each_entry(tmp, &group->triggers, node)
1179                         period = min(period, div_u64(tmp->win.size,
1180                                         UPDATES_PER_WINDOW));
1181                 group->poll_min_period = period;
1182                 /* Destroy poll_task when the last trigger is destroyed */
1183                 if (group->poll_states == 0) {
1184                         group->polling_until = 0;
1185                         task_to_destroy = rcu_dereference_protected(
1186                                         group->poll_task,
1187                                         lockdep_is_held(&group->trigger_lock));
1188                         rcu_assign_pointer(group->poll_task, NULL);
1189                 }
1190         }
1191 
1192         mutex_unlock(&group->trigger_lock);
1193 
1194         /*
1195          * Wait for both *trigger_ptr from psi_trigger_replace and
1196          * poll_task RCUs to complete their read-side critical sections
1197          * before destroying the trigger and optionally the poll_task
1198          */
1199         synchronize_rcu();
1200         /*
1201          * Destroy the kworker after releasing trigger_lock to prevent a
1202          * deadlock while waiting for psi_poll_work to acquire trigger_lock
1203          */
1204         if (task_to_destroy) {
1205                 /*
1206                  * After the RCU grace period has expired, the worker
1207                  * can no longer be found through group->poll_task.
1208                  * But it might have been already scheduled before
1209                  * that - deschedule it cleanly before destroying it.
1210                  */
1211                 del_timer_sync(&group->poll_timer);
1212                 kthread_stop(task_to_destroy);
1213         }
1214         kfree(t);
1215 }

1217 void psi_trigger_replace(void **trigger_ptr, struct psi_trigger *new)
1218 {
1219         struct psi_trigger *old = *trigger_ptr;
1220 
1221         if (static_branch_likely(&psi_disabled))
1222                 return;
1223 
1224         rcu_assign_pointer(*trigger_ptr, new);
1225         if (old)
1226                 kref_put(&old->refcount, psi_trigger_destroy);
1227 }

1229 __poll_t psi_trigger_poll(void **trigger_ptr,
1230                                 struct file *file, poll_table *wait)
1231 {
1232         __poll_t ret = DEFAULT_POLLMASK;
1233         struct psi_trigger *t;
1234 
1235         if (static_branch_likely(&psi_disabled))
1236                 return DEFAULT_POLLMASK | EPOLLERR | EPOLLPRI;
1237 
1238         rcu_read_lock();
1239 
1240         t = rcu_dereference(*(void __rcu __force **)trigger_ptr);
1241         if (!t) {
1242                 rcu_read_unlock();
1243                 return DEFAULT_POLLMASK | EPOLLERR | EPOLLPRI;
1244         }
1245         kref_get(&t->refcount);
1246 
1247         rcu_read_unlock();
1248 
1249         poll_wait(file, &t->event_wait, wait);
1250 
1251         if (cmpxchg(&t->event, 1, 0) == 1)
1252                 ret |= EPOLLPRI;
1253 
1254         kref_put(&t->refcount, psi_trigger_destroy);
1255 
1256         return ret;
1257 }

1259 static ssize_t psi_write(struct file *file, const char __user *user_buf,
1260                          size_t nbytes, enum psi_res res)
1261 {
1262         char buf[32];
1263         size_t buf_size;
1264         struct seq_file *seq;
1265         struct psi_trigger *new;
1266 
1267         if (static_branch_likely(&psi_disabled))
1268                 return -EOPNOTSUPP;
1269 
1270         if (!nbytes)
1271                 return -EINVAL;
1272 
1273         buf_size = min(nbytes, sizeof(buf));
1274         if (copy_from_user(buf, user_buf, buf_size))
1275                 return -EFAULT;
1276 
1277         buf[buf_size - 1] = '\0';
1278 
1279         new = psi_trigger_create(&psi_system, buf, nbytes, res);
1280         if (IS_ERR(new))
1281                 return PTR_ERR(new);
1282 
1283         seq = file->private_data;
1284         /* Take seq->lock to protect seq->private from concurrent writes */
1285         mutex_lock(&seq->lock);
1286         psi_trigger_replace(&seq->private, new);
1287         mutex_unlock(&seq->lock);
1288 
1289         return nbytes;
1290 }
