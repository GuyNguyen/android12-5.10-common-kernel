  53  * Must be called with interrupts disabled !
  54  */
  55 static void tick_do_update_jiffies64(ktime_t now)
     /*   */
  56 {
  57         unsigned long ticks = 0;
  58         ktime_t delta;
  59 
  60         /*
  61          * Do a quick check without holding jiffies_lock:
  62          * The READ_ONCE() pairs with two updates done later in this function.
  63          */
  64         delta = ktime_sub(now, READ_ONCE(last_jiffies_update));
  65         if (delta < tick_period)
  66                 return;
  67 
  68         /* Reevaluate with jiffies_lock held */
  69         raw_spin_lock(&jiffies_lock);
  70         write_seqcount_begin(&jiffies_seq);
  71 
  72         delta = ktime_sub(now, last_jiffies_update);
  73         if (delta >= tick_period) {
  74 
  75                 delta = ktime_sub(delta, tick_period);
  76                 /* Pairs with the lockless read in this function. */
  77                 WRITE_ONCE(last_jiffies_update,
  78                            ktime_add(last_jiffies_update, tick_period));
  79 
  80                 /* Slow path for long timeouts */
  81                 if (unlikely(delta >= tick_period)) {
  82                         s64 incr = ktime_to_ns(tick_period);
  83 
  84                         ticks = ktime_divns(delta, incr);
  85 
  86                         /* Pairs with the lockless read in this function. */
  87                         WRITE_ONCE(last_jiffies_update,
  88                                    ktime_add_ns(last_jiffies_update,
  89                                                 incr * ticks));
  90                 }
  91                 do_timer(++ticks);
  92 
  93                 /* Keep the tick_next_period variable up to date */
  94                 tick_next_period = ktime_add(last_jiffies_update, tick_period);
  95         } else {
  96                 write_seqcount_end(&jiffies_seq);
  97                 raw_spin_unlock(&jiffies_lock);
  98                 return;
  99         }
 100         write_seqcount_end(&jiffies_seq);
 101         raw_spin_unlock(&jiffies_lock);
 102         update_wall_time();
 103 }

 108 static ktime_t tick_init_jiffy_update(void)
     /*   */
 109 {
 110         ktime_t period;
 111 
 112         raw_spin_lock(&jiffies_lock);
 113         write_seqcount_begin(&jiffies_seq);
 114         /* Did we start the jiffies update yet ? */
 115         if (last_jiffies_update == 0)
 116                 last_jiffies_update = tick_next_period;
 117         period = last_jiffies_update;
 118         write_seqcount_end(&jiffies_seq);
 119         raw_spin_unlock(&jiffies_lock);
 120         return period;
 121 }

 123 static void tick_sched_do_timer(struct tick_sched *ts, ktime_t now)
     /*   */
 124 {
 125         int cpu = smp_processor_id();
 126 
 127 #ifdef CONFIG_NO_HZ_COMMON
 128         /*
 129          * Check if the do_timer duty was dropped. We don't care about
 130          * concurrency: This happens only when the CPU in charge went
 131          * into a long sleep. If two CPUs happen to assign themselves to
 132          * this duty, then the jiffies update is still serialized by
 133          * jiffies_lock.
 134          *
 135          * If nohz_full is enabled, this should not happen because the
 136          * tick_do_timer_cpu never relinquishes.
 137          */
 138         if (unlikely(tick_do_timer_cpu == TICK_DO_TIMER_NONE)) {
 139 #ifdef CONFIG_NO_HZ_FULL
 140                 WARN_ON(tick_nohz_full_running);
 141 #endif
 142                 tick_do_timer_cpu = cpu;
 143         }
 144 #endif
 145 
 146         /* Check, if the jiffies need an update */
 147         if (tick_do_timer_cpu == cpu) {
 148                 tick_do_update_jiffies64(now);
 149                 trace_android_vh_jiffies_update(NULL);
 150         }
 151 
 152         if (ts->inidle)
 153                 ts->got_idle_tick = 1;
 154 }

 156 static void tick_sched_handle(struct tick_sched *ts, struct pt_regs *regs)
     /*   */
 157 {
 158 #ifdef CONFIG_NO_HZ_COMMON
 159         /*
 160          * When we are idle and the tick is stopped, we have to touch
 161          * the watchdog as we might not schedule for a really long
 162          * time. This happens on complete idle SMP systems while
 163          * waiting on the login prompt. We also increment the "start of
 164          * idle" jiffy stamp so the idle accounting adjustment we do
 165          * when we go busy again does not account too much ticks.
 166          */
 167         if (ts->tick_stopped) {
 168                 touch_softlockup_watchdog_sched();
 169                 if (is_idle_task(current))
 170                         ts->idle_jiffies++;
 171                 /*
 172                  * In case the current tick fired too early past its expected
 173                  * expiration, make sure we don't bypass the next clock reprogramming
 174                  * to the same deadline.
 175                  */
 176                 ts->next_tick = 0;
 177         }
 178 #endif
 179         update_process_times(user_mode(regs));
 180         profile_tick(CPU_PROFILING);
 181 }
 182 #endif

 190 static bool check_tick_dependency(atomic_t *dep)
     /*   */
 191 {
 192         int val = atomic_read(dep);
 193 
 194         if (val & TICK_DEP_MASK_POSIX_TIMER) {
 195                 trace_tick_stop(0, TICK_DEP_MASK_POSIX_TIMER);
 196                 return true;
 197         }
 198 
 199         if (val & TICK_DEP_MASK_PERF_EVENTS) {
 200                 trace_tick_stop(0, TICK_DEP_MASK_PERF_EVENTS);
 201                 return true;
 202         }
 203 
 204         if (val & TICK_DEP_MASK_SCHED) {
 205                 trace_tick_stop(0, TICK_DEP_MASK_SCHED);
 206                 return true;
 207         }
 208 
 209         if (val & TICK_DEP_MASK_CLOCK_UNSTABLE) {
 210                 trace_tick_stop(0, TICK_DEP_MASK_CLOCK_UNSTABLE);
 211                 return true;
 212         }
 213 
 214         if (val & TICK_DEP_MASK_RCU) {
 215                 trace_tick_stop(0, TICK_DEP_MASK_RCU);
 216                 return true;
 217         }

 254 /*
 255  * Kick this CPU if it's full dynticks in order to force it to
 256  * re-evaluate its dependency on the tick and restart it if necessary.
 257  * This kick, unlike tick_nohz_full_kick_cpu() and tick_nohz_full_kick_all(),
 258  * is NMI safe.
 259  */
 260 static void tick_nohz_full_kick(void)
     /*   */
 261 {
 262         if (!tick_nohz_full_cpu(smp_processor_id()))
 263                 return;
 264 
 265         irq_work_queue(this_cpu_ptr(&nohz_full_kick_work));
 266 }

 280 /*
 281  * Kick all full dynticks CPUs in order to force these to re-evaluate
 282  * their dependency on the tick and restart it if necessary.

 284 static void tick_nohz_full_kick_all(void)
     /*   */
 285 {
 286         int cpu;
 287 
 288         if (!tick_nohz_full_running)
 289                 return;
 290 
 291         preempt_disable();
 292         for_each_cpu_and(cpu, tick_nohz_full_mask, cpu_online_mask)
 293                 tick_nohz_full_kick_cpu(cpu);

 321 /*
 322  * Set per-CPU tick dependency. Used by scheduler and perf events in order to
 323  * manage events throttling.
 324  */
 325 void tick_nohz_dep_set_cpu(int cpu, enum tick_dep_bits bit)
     /*   */
 326 {
 327         int prev;
 328         struct tick_sched *ts;
 329 
 330         ts = per_cpu_ptr(&tick_cpu_sched, cpu);
 331 
 332         prev = atomic_fetch_or(BIT(bit), &ts->tick_dep_mask);
 333         if (!prev) {
 334                 preempt_disable();
 335                 /* Perf needs local kick that is NMI safe */
 336                 if (cpu == smp_processor_id()) {
 337                         tick_nohz_full_kick();
 338                 } else {
 339                         /* Remote irq work not NMI-safe */
 340                         if (!WARN_ON_ONCE(in_nmi()))
 341                                 tick_nohz_full_kick_cpu(cpu);
 342                 }
 343                 preempt_enable();
 344         }
 345 }

 398 /*
 399  * Re-evaluate the need for the tick as we switch the current task.
 400  * It might need the tick due to per task/process properties:
 401  * perf events, posix CPU timers, ...
 402  */
 403 void __tick_nohz_task_switch(void)
     /*   */
 404 {
 405         unsigned long flags;
 406         struct tick_sched *ts;
 407 
 408         local_irq_save(flags);
 409 
 410         if (!tick_nohz_full_cpu(smp_processor_id()))
 411                 goto out;
 412 
 413         ts = this_cpu_ptr(&tick_cpu_sched);
 414 
 415         if (ts->tick_stopped) {
 416                 if (atomic_read(&current->tick_dep_mask) ||
 417                     atomic_read(&current->signal->tick_dep_mask))
 418                         tick_nohz_full_kick();
 419         }
 420 out:
 421         local_irq_restore(flags);
 422 }

 433 static int tick_nohz_cpu_down(unsigned int cpu)
     /*   */
 434 {
 435         /*
 436          * The tick_do_timer_cpu CPU handles housekeeping duty (unbound
 437          * timers, workqueues, timekeeping, ...) on behalf of full dynticks
 438          * CPUs. It must remain online when nohz full is enabled.
 439          */
 440         if (tick_nohz_full_running && tick_do_timer_cpu == cpu)
 441                 return -EBUSY;
 442         return 0;

 445 void __init tick_nohz_init(void)
     /*   */
 446 {
 447         int cpu, ret;
 448 
 449         if (!tick_nohz_full_running)
 450                 return;
 451 
 452         /*
 453          * Full dynticks uses irq work to drive the tick rescheduling on safe
 454          * locking contexts. But then we need irq work to raise its own
 455          * interrupts to avoid circular dependency on the tick
 456          */
 457         if (!arch_irq_work_has_interrupt()) {
 458                 pr_warn("NO_HZ: Can't run full dynticks because arch doesn't support irq work self-IPIs\n");
 459                 cpumask_clear(tick_nohz_full_mask);
 460                 tick_nohz_full_running = false;
 461                 return;
 462         }
 463 
 464         if (IS_ENABLED(CONFIG_PM_SLEEP_SMP) &&
 465                         !IS_ENABLED(CONFIG_PM_SLEEP_SMP_NONZERO_CPU)) {
 466                 cpu = smp_processor_id();
 467 
 468                 if (cpumask_test_cpu(cpu, tick_nohz_full_mask)) {
 469                         pr_warn("NO_HZ: Clearing %d from nohz_full range "
 470                                 "for timekeeping\n", cpu);
 471                         cpumask_clear_cpu(cpu, tick_nohz_full_mask);
 472                 }
 473         }
 474 
 475         for_each_cpu(cpu, tick_nohz_full_mask)
 476                 context_tracking_cpu_set(cpu);
 477 
 478         ret = cpuhp_setup_state_nocalls(CPUHP_AP_ONLINE_DYN,
 479                                         "kernel/nohz:predown", NULL,
 480                                         tick_nohz_cpu_down);
 481         WARN_ON(ret < 0);
 482         pr_info("NO_HZ: Full dynticks CPUs: %*pbl.\n",
 483                 cpumask_pr_args(tick_nohz_full_mask));
 484 }

 520 /**
 521  * tick_nohz_update_jiffies - update jiffies when idle was interrupted
 522  *
 523  * Called from interrupt entry when the CPU was idle
 524  *
 525  * In case the sched_tick was stopped on this CPU, we have to check if jiffies
 526  * must be updated. Otherwise an interrupt handler could use a stale jiffy
 527  * value. We do this unconditionally on any CPU, as we don't know whether the
 528  * CPU, which has the update task assigned is in a long sleep.
 529  */
 530 static void tick_nohz_update_jiffies(ktime_t now)
     /*   */
 531 {
 532         unsigned long flags;
 533 
 534         __this_cpu_write(tick_cpu_sched.idle_waketime, now);
 535 
 536         local_irq_save(flags);
 537         tick_do_update_jiffies64(now);
 538         local_irq_restore(flags);
 539 
 540         touch_softlockup_watchdog_sched();
 541 }

 581  * get_cpu_idle_time_us - get the total idle time of a CPU
 582  * @cpu: CPU number to query
 583  * @last_update_time: variable to store update time in. Do not update
 584  * counters if NULL.
 585  *
 586  * Return the cumulative idle time (since boot) for a given
 587  * CPU, in microseconds.
 588  *
 589  * This time is measured via accounting rather than sampling,
 590  * and is as accurate as ktime_get() is.
 591  *
 592  * This function returns -1 if NOHZ is not enabled.
 593  */
 594 u64 get_cpu_idle_time_us(int cpu, u64 *last_update_time)
     /*   */
 595 {
 596         struct tick_sched *ts = &per_cpu(tick_cpu_sched, cpu);
 597         ktime_t now, idle;
 598 
 599         if (!tick_nohz_active)
 600                 return -1;
 601 
 602         now = ktime_get();
 603         if (last_update_time) {
 604                 update_ts_time_stats(cpu, ts, now, last_update_time);
 605                 idle = ts->idle_sleeptime;
 606         } else {
 607                 if (ts->idle_active && !nr_iowait_cpu(cpu)) {
 608                         ktime_t delta = ktime_sub(now, ts->idle_entrytime);
 609 
 610                         idle = ktime_add(ts->idle_sleeptime, delta);
 611                 } else {
 612                         idle = ts->idle_sleeptime;
 613                 }
 614         }
 615 
 616         return ktime_to_us(idle);
 617 
 618 }
 619 EXPORT_SYMBOL_GPL(get_cpu_idle_time_us);

 661 static void tick_nohz_restart(struct tick_sched *ts, ktime_t now)
     /*   */
 662 {
 663         hrtimer_cancel(&ts->sched_timer);
 664         hrtimer_set_expires(&ts->sched_timer, ts->last_tick);
 665 
 666         /* Forward the time to expire in the future */
 667         hrtimer_forward(&ts->sched_timer, now, tick_period);
 668 
 669         if (ts->nohz_mode == NOHZ_MODE_HIGHRES) {
 670                 hrtimer_start_expires(&ts->sched_timer,
 671                                       HRTIMER_MODE_ABS_PINNED_HARD);
 672         } else {
 673                 tick_program_event(hrtimer_get_expires(&ts->sched_timer), 1);
 674         }
 675 
 676         /*
 677          * Reset to make sure next tick stop doesn't get fooled by past
 678          * cached clock deadline.
 679          */
 680         ts->next_tick = 0;
 681 }

 688 static ktime_t tick_nohz_next_event(struct tick_sched *ts, int cpu)
     /*   */
 689 {
 690         u64 basemono, next_tick, next_tmr, next_rcu, delta, expires;
 691         unsigned long basejiff;
 692         unsigned int seq;
 693 
 694         /* Read jiffies and the time when jiffies were updated last */
 695         do {
 696                 seq = read_seqcount_begin(&jiffies_seq);
 697                 basemono = last_jiffies_update;
 698                 basejiff = jiffies;
 699         } while (read_seqcount_retry(&jiffies_seq, seq));
 700         ts->last_jiffies = basejiff;
 701         ts->timer_expires_base = basemono;
 702 
 703         /*
 704          * Keep the periodic tick, when RCU, architecture or irq_work
 705          * requests it.
 706          * Aside of that check whether the local timer softirq is
 707          * pending. If so its a bad idea to call get_next_timer_interrupt()
 708          * because there is an already expired timer, so it will request
 709          * immeditate expiry, which rearms the hardware timer with a
 710          * minimal delta which brings us back to this place
 711          * immediately. Lather, rinse and repeat...
 712          */
 713         if (rcu_needs_cpu(basemono, &next_rcu) || arch_needs_cpu() ||
 714             irq_work_needs_cpu() || local_timer_softirq_pending()) {
 715                 next_tick = basemono + TICK_NSEC;
 716         } else {
 717                 /*
 718                  * Get the next pending timer. If high resolution
 719                  * timers are enabled this only takes the timer wheel
 720                  * timers into account. If high resolution timers are
 721                  * disabled this also looks at the next expiring
 722                  * hrtimer.
 723                  */
 724                 next_tmr = get_next_timer_interrupt(basejiff, basemono);
 725                 ts->next_timer = next_tmr;
 726                 /* Take the next rcu event into account */
 727                 next_tick = next_rcu < next_tmr ? next_rcu : next_tmr;
 728         }
 729 
 730         /*
 731          * If the tick is due in the next period, keep it ticking or
 732          * force prod the timer.
 733          */
 734         delta = next_tick - basemono;
 735         if (delta <= (u64)TICK_NSEC) {
 736                 /*
 737                  * Tell the timer code that the base is not idle, i.e. undo
 738                  * the effect of get_next_timer_interrupt():
 739                  */
 740                 timer_clear_idle();
 741                 /*
 742                  * We've not stopped the tick yet, and there's a timer in the
 743                  * next period, so no point in stopping it either, bail.
 744                  */
 745                 if (!ts->tick_stopped) {
 746                         ts->timer_expires = 0;
 747                         goto out;
 748                 }
 749         }
 750 
 751         /*
 752          * If this CPU is the one which had the do_timer() duty last, we limit
 753          * the sleep time to the timekeeping max_deferment value.
 754          * Otherwise we can sleep as long as we want.
 755          */
 756         delta = timekeeping_max_deferment();
 757         if (cpu != tick_do_timer_cpu &&
 758             (tick_do_timer_cpu != TICK_DO_TIMER_NONE || !ts->do_timer_last))
 759                 delta = KTIME_MAX;
 760 
 761         /* Calculate the next expiry time */
 762         if (delta < (KTIME_MAX - basemono))
 763                 expires = basemono + delta;
 764         else
 765                 expires = KTIME_MAX;
 766 
 767         ts->timer_expires = min_t(u64, expires, next_tick);
 768 
 769 out:
 770         return ts->timer_expires;
 771 }

 773 static void tick_nohz_stop_tick(struct tick_sched *ts, int cpu)
     /*   */
 774 {
 775         struct clock_event_device *dev = __this_cpu_read(tick_cpu_device.evtdev);
 776         u64 basemono = ts->timer_expires_base;
 777         u64 expires = ts->timer_expires;
 778         ktime_t tick = expires;
 779 
 780         /* Make sure we won't be trying to stop it twice in a row. */
 781         ts->timer_expires_base = 0;
 782 
 783         /*
 784          * If this CPU is the one which updates jiffies, then give up
 785          * the assignment and let it be taken by the CPU which runs
 786          * the tick timer next, which might be this CPU as well. If we
 787          * don't drop this here the jiffies might be stale and
 788          * do_timer() never invoked. Keep track of the fact that it
 789          * was the one which had the do_timer() duty last.
 790          */
 791         if (cpu == tick_do_timer_cpu) {
 792                 tick_do_timer_cpu = TICK_DO_TIMER_NONE;
 793                 ts->do_timer_last = 1;
 794         } else if (tick_do_timer_cpu != TICK_DO_TIMER_NONE) {
 795                 ts->do_timer_last = 0;
 796         }
 797 
 798         /* Skip reprogram of event if its not changed */
 799         if (ts->tick_stopped && (expires == ts->next_tick)) {
 800                 /* Sanity check: make sure clockevent is actually programmed */
 801                 if (tick == KTIME_MAX || ts->next_tick == hrtimer_get_expires(&ts->sched_timer))
 802                         return;
 803 
 804                 WARN_ON_ONCE(1);
 805                 printk_once("basemono: %llu ts->next_tick: %llu dev->next_event: %llu timer->active: %d timer->expires: %llu\n",
 806                             basemono, ts->next_tick, dev->next_event,
 807                             hrtimer_active(&ts->sched_timer), hrtimer_get_expires(&ts->sched_timer));
 808         }
 809 
 810         /*
 811          * nohz_stop_sched_tick can be called several times before
 812          * the nohz_restart_sched_tick is called. This happens when
 813          * interrupts arrive which do not cause a reschedule. In the
 814          * first call we save the current tick time, so we can restart
 815          * the scheduler tick in nohz_restart_sched_tick.
 816          */
 817         if (!ts->tick_stopped) {
 818                 calc_load_nohz_start();
 819                 quiet_vmstat();
 820 
 821                 ts->last_tick = hrtimer_get_expires(&ts->sched_timer);
 822                 ts->tick_stopped = 1;
 823                 trace_tick_stop(1, TICK_DEP_MASK_NONE);
 824         }
 825 
 826         ts->next_tick = tick;
 827 
 828         /*
 829          * If the expiration time == KTIME_MAX, then we simply stop
 830          * the tick timer.
 831          */
 832         if (unlikely(expires == KTIME_MAX)) {
 833                 if (ts->nohz_mode == NOHZ_MODE_HIGHRES)
 834                         hrtimer_cancel(&ts->sched_timer);
 835                 return;
 836         }
 837 
 838         if (ts->nohz_mode == NOHZ_MODE_HIGHRES) {
 839                 hrtimer_start(&ts->sched_timer, tick,
 840                               HRTIMER_MODE_ABS_PINNED_HARD);
 841         } else {
 842                 hrtimer_set_expires(&ts->sched_timer, tick);
 843                 tick_program_event(tick, 1);
 844         }
 845 }

 902 static bool can_stop_idle_tick(int cpu, struct tick_sched *ts)
     /*   */
 903 {
 904         /*
 905          * If this CPU is offline and it is the one which updates
 906          * jiffies, then give up the assignment and let it be taken by
 907          * the CPU which runs the tick timer next. If we don't drop
 908          * this here the jiffies might be stale and do_timer() never
 909          * invoked.
 910          */
 911         if (unlikely(!cpu_online(cpu))) {
 912                 if (cpu == tick_do_timer_cpu)
 913                         tick_do_timer_cpu = TICK_DO_TIMER_NONE;
 914                 /*
 915                  * Make sure the CPU doesn't get fooled by obsolete tick
 916                  * deadline if it comes back online later.
 917                  */
 918                 ts->next_tick = 0;
 919                 return false;
 920         }
 921 
 922         if (unlikely(ts->nohz_mode == NOHZ_MODE_INACTIVE))
 923                 return false;
 924 
 925         if (need_resched())
 926                 return false;
 927 
 928         if (unlikely(local_softirq_pending())) {
 929                 static int ratelimit;
 930 
 931                 if (ratelimit < 10 &&
 932                     (local_softirq_pending() & SOFTIRQ_STOP_IDLE_MASK)) {
 933                         pr_warn("NOHZ tick-stop error: Non-RCU local softirq work is pending, handler #%02x!!!\n",
 934                                 (unsigned int) local_softirq_pending());
 935                         ratelimit++;
 936                 }
 937                 return false;
 938         }
 939 
 940         if (tick_nohz_full_enabled()) {
 941                 /*
 942                  * Keep the tick alive to guarantee timekeeping progression
 943                  * if there are full dynticks CPUs around
 944                  */
 945                 if (tick_do_timer_cpu == cpu)
 946                         return false;
 947 
 948                 /* Should not happen for nohz-full */
 949                 if (WARN_ON_ONCE(tick_do_timer_cpu == TICK_DO_TIMER_NONE))
 950                         return false;
 951         }
 952 
 953         return true;
 954 }

 956 static void __tick_nohz_idle_stop_tick(struct tick_sched *ts)
     /*   */
 957 {
 958         ktime_t expires;
 959         int cpu = smp_processor_id();
 960 
 961         /*
 962          * If tick_nohz_get_sleep_length() ran tick_nohz_next_event(), the
 963          * tick timer expiration time is known already.
 964          */
 965         if (ts->timer_expires_base)
 966                 expires = ts->timer_expires;
 967         else if (can_stop_idle_tick(cpu, ts))
 968                 expires = tick_nohz_next_event(ts, cpu);
 969         else
 970                 return;
 971 
 972         ts->idle_calls++;
 973 
 974         if (expires > 0LL) {
 975                 int was_stopped = ts->tick_stopped;
 976 
 977                 tick_nohz_stop_tick(ts, cpu);
 978 
 979                 ts->idle_sleeps++;
 980                 ts->idle_expires = expires;
 981 
 982                 if (!was_stopped && ts->tick_stopped) {
 983                         ts->idle_jiffies = ts->last_jiffies;
 984                         nohz_balance_enter_idle(cpu);
 985                 }
 986         } else {
 987                 tick_nohz_retain_tick(ts);
 988         }
 989 }

1078 /**
1079  * tick_nohz_get_sleep_length - return the expected length of the current sleep
1080  * @delta_next: duration until the next event if the tick cannot be stopped
1081  *
1082  * Called from power state control code with interrupts disabled
1083  */
1084 ktime_t tick_nohz_get_sleep_length(ktime_t *delta_next)
     /*   */
1085 {
1086         struct clock_event_device *dev = __this_cpu_read(tick_cpu_device.evtdev);
1087         struct tick_sched *ts = this_cpu_ptr(&tick_cpu_sched);
1088         int cpu = smp_processor_id();
1089         /*
1090          * The idle entry time is expected to be a sufficient approximation of
1091          * the current time at this point.
1092          */
1093         ktime_t now = ts->idle_entrytime;
1094         ktime_t next_event;
1095 
1096         WARN_ON_ONCE(!ts->inidle);
1097 
1098         *delta_next = ktime_sub(dev->next_event, now);
1099 
1100         if (!can_stop_idle_tick(cpu, ts))
1101                 return *delta_next;
1102 
1103         next_event = tick_nohz_next_event(ts, cpu);
1104         if (!next_event)
1105                 return *delta_next;
1106 
1107         /*
1108          * If the next highres timer to expire is earlier than next_event, the
1109          * idle governor needs to know that.
1110          */
1111         next_event = min_t(u64, next_event,
1112                            hrtimer_next_event_without(&ts->sched_timer));
1113 
1114         return ktime_sub(next_event, now);
1115 }

1179 /**

1181  *

1183  * This also exit the RCU extended quiescent state. The CPU
1184  * can use RCU again after this function is called.

     /*   */
1187 {

1189         bool idle_active, tick_stopped;
1190         ktime_t now;
1191

1196 
1197         ts->inidle = 0;
1198         idle_active = ts->idle_active;
1199         tick_stopped = ts->tick_stopped;
1200 
1201         if (idle_active || tick_stopped)
1202                 now = ktime_get();
1203 
1204         if (idle_active)
1205                 tick_nohz_stop_idle(ts, now);
1206 
1207         if (tick_stopped)
1208                 __tick_nohz_idle_restart_tick(ts, now);
1209 
1210         local_irq_enable();
1211 }

1213 /*
1214  * The nohz low res interrupt handler
1215  */
1216 static void tick_nohz_handler(struct clock_event_device *dev)
     /*   */
1217 {
1218         struct tick_sched *ts = this_cpu_ptr(&tick_cpu_sched);
1219         struct pt_regs *regs = get_irq_regs();
1220         ktime_t now = ktime_get();
1221 
1222         dev->next_event = KTIME_MAX;
1223 
1224         tick_sched_do_timer(ts, now);
1225         tick_sched_handle(ts, regs);
1226 
1227         /* No need to reprogram if we are running tickless  */
1228         if (unlikely(ts->tick_stopped))
1229                 return;
1230 
1231         hrtimer_forward(&ts->sched_timer, now, tick_period);
1232         tick_program_event(hrtimer_get_expires(&ts->sched_timer), 1);
1233 }
1234 
1235 static inline void tick_nohz_activate(struct tick_sched *ts, int mode)
     /*   */
1236 {
1237         if (!tick_nohz_enabled)
1238                 return;
1239         ts->nohz_mode = mode;
1240         /* One update is enough */
1241         if (!test_and_set_bit(0, &tick_nohz_active))
1242                 timers_update_nohz();
1243 }

1245 /**
1246  * tick_nohz_switch_to_nohz - switch to nohz mode
1247  */
1248 static void tick_nohz_switch_to_nohz(void)
     /*   */
1249 {
1250         struct tick_sched *ts = this_cpu_ptr(&tick_cpu_sched);
1251         ktime_t next;
1252 
1253         if (!tick_nohz_enabled)
1254                 return;
1255 
1256         if (tick_switch_to_oneshot(tick_nohz_handler))
1257                 return;
1258 
1259         /*
1260          * Recycle the hrtimer in ts, so we can share the
1261          * hrtimer_forward with the highres code.
1262          */
1263         hrtimer_init(&ts->sched_timer, CLOCK_MONOTONIC, HRTIMER_MODE_ABS_HARD);
1264         /* Get the next period */
1265         next = tick_init_jiffy_update();
1266 
1267         hrtimer_set_expires(&ts->sched_timer, next);
1268         hrtimer_forward_now(&ts->sched_timer, tick_period);
1269         tick_program_event(hrtimer_get_expires(&ts->sched_timer), 1);
1270         tick_nohz_activate(ts, NOHZ_MODE_LOWRES);
1271 }

1273 static inline void tick_nohz_irq_enter(void)
     /*   */
1274 {
1275         struct tick_sched *ts = this_cpu_ptr(&tick_cpu_sched);
1276         ktime_t now;
1277 
1278         if (!ts->idle_active && !ts->tick_stopped)
1279                 return;
1280         now = ktime_get();
1281         if (ts->idle_active)
1282                 tick_nohz_stop_idle(ts, now);
1283         if (ts->tick_stopped)
1284                 tick_nohz_update_jiffies(now);
1285 }

1305  * High resolution timer specific code
1306  */
1307 #ifdef CONFIG_HIGH_RES_TIMERS
1308 /*
1309  * We rearm the timer until we get disabled by the idle code.
1310  * Called with interrupts disabled.
1311  */
1312 static enum hrtimer_restart tick_sched_timer(struct hrtimer *timer)
     /*   */
1313 {
1314         struct tick_sched *ts =
1315                 container_of(timer, struct tick_sched, sched_timer);
1316         struct pt_regs *regs = get_irq_regs();
1317         ktime_t now = ktime_get();
1318 
1319         tick_sched_do_timer(ts, now);
1320 
1321         /*
1322          * Do not call, when we are not in irq context and have
1323          * no valid regs pointer
1324          */
1325         if (regs)
1326                 tick_sched_handle(ts, regs);
1327         else
1328                 ts->next_tick = 0;
1329 
1330         /* No need to reprogram if we are in idle or full dynticks mode */
1331         if (unlikely(ts->tick_stopped))
1332                 return HRTIMER_NORESTART;
1333 
1334         hrtimer_forward(timer, now, tick_period);
1335 
1336         return HRTIMER_RESTART;
1337 }

1349 /**
1350  * tick_setup_sched_timer - setup the tick emulation timer
1351  */
1352 void tick_setup_sched_timer(void)
     /*   */
1353 {
1354         struct tick_sched *ts = this_cpu_ptr(&tick_cpu_sched);
1355         ktime_t now = ktime_get();
1356 
1357         /*
1358          * Emulate tick processing via per-CPU hrtimers:
1359          */
1360         hrtimer_init(&ts->sched_timer, CLOCK_MONOTONIC, HRTIMER_MODE_ABS_HARD);
1361         ts->sched_timer.function = tick_sched_timer;
1362 
1363         /* Get the next period (per-CPU) */
1364         hrtimer_set_expires(&ts->sched_timer, tick_init_jiffy_update());
1365 
1366         /* Offset the tick to avert jiffies_lock contention. */
1367         if (sched_skew_tick) {
1368                 u64 offset = ktime_to_ns(tick_period) >> 1;
1369                 do_div(offset, num_possible_cpus());
1370                 offset *= smp_processor_id();
1371                 hrtimer_add_expires_ns(&ts->sched_timer, offset);
1372         }
1373 
1374         hrtimer_forward(&ts->sched_timer, now, tick_period);
1375         hrtimer_start_expires(&ts->sched_timer, HRTIMER_MODE_ABS_PINNED_HARD);
1376         tick_nohz_activate(ts, NOHZ_MODE_HIGHRES);
1377 }
1378 #endif /* HIGH_RES_TIMERS */

1415 /**
1416  * Check, if a change happened, which makes oneshot possible.
1417  *
1418  * Called cyclic from the hrtimer softirq (driven by the timer
1419  * softirq) allow_nohz signals, that we can switch into low-res nohz
1420  * mode, because high resolution timers are disabled (either compile
1421  * or runtime). Called with interrupts disabled.
1422  */
1423 int tick_check_oneshot_change(int allow_nohz)
     /*   */
1424 {
1425         struct tick_sched *ts = this_cpu_ptr(&tick_cpu_sched);
1426 
1427         if (!test_and_clear_bit(0, &ts->check_clocks))
1428                 return 0;
1429 
1430         if (ts->nohz_mode != NOHZ_MODE_INACTIVE)
1431                 return 0;
1432 
1433         if (!timekeeping_valid_for_hres() || !tick_is_oneshot_available())
1434                 return 0;
1435 
1436         if (!allow_nohz)
1437                 return 1;
1438 
1439         tick_nohz_switch_to_nohz();
1440         return 0;
1441 }
