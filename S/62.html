 157  * Weak aliases for breakpoint management,
 158  * can be overridden by architectures when needed:
 159  */
 160 int __weak kgdb_arch_set_breakpoint(struct kgdb_bkpt *bpt)
 161 {
 162         int err;
 163 
 164         err = copy_from_kernel_nofault(bpt->saved_instr, (char *)bpt->bpt_addr,
 165                                 BREAK_INSTR_SIZE);
 166         if (err)
 167                 return err;
 168         err = copy_to_kernel_nofault((char *)bpt->bpt_addr,
 169                                  arch_kgdb_ops.gdb_bpt_instr, BREAK_INSTR_SIZE);
 170         return err;
 171 }

 181 int __weak kgdb_validate_break_address(unsigned long addr)
 182 {
 183         struct kgdb_bkpt tmp;
 184         int err;
 185 
 186         if (kgdb_within_blocklist(addr))
 187                 return -EINVAL;
 188 
 189         /* Validate setting the breakpoint and then removing it.  If the
 190          * remove fails, the kernel needs to emit a bad message because we
 191          * are deep trouble not being able to put things back the way we
 192          * found them.
 193          */
 194         tmp.bpt_addr = addr;
 195         err = kgdb_arch_set_breakpoint(&tmp);
 196         if (err)
 197                 return err;
 198         err = kgdb_arch_remove_breakpoint(&tmp);
 199         if (err)
 200                 pr_err("Critical breakpoint error, kernel memory destroyed at: %lx\n",
 201                        addr);
 202         return err;
 203 }

 244 void __weak kgdb_roundup_cpus(void)
 245 {
 246         call_single_data_t *csd;
 247         int this_cpu = raw_smp_processor_id();
 248         int cpu;
 249         int ret;
 250 
 251         for_each_online_cpu(cpu) {
 252                 /* No need to roundup ourselves */
 253                 if (cpu == this_cpu)
 254                         continue;
 255 
 256                 csd = &per_cpu(kgdb_roundup_csd, cpu);
 257 
 258                 /*
 259                  * If it didn't round up last time, don't try again
 260                  * since smp_call_function_single_async() will block.
 261                  *
 262                  * If rounding_up is false then we know that the
 263                  * previous call must have at least started and that
 264                  * means smp_call_function_single_async() won't block.
 265                  */
 266                 if (kgdb_info[cpu].rounding_up)
 267                         continue;
 268                 kgdb_info[cpu].rounding_up = true;
 269 
 270                 csd->func = kgdb_call_nmi_hook;
 271                 ret = smp_call_function_single_async(cpu, csd);
 272                 if (ret)
 273                         kgdb_info[cpu].rounding_up = false;
 274         }
 275 }
 276 NOKPROBE_SYMBOL(kgdb_roundup_cpus);

 284 static void kgdb_flush_swbreak_addr(unsigned long addr)
 285 {
 286         if (!CACHE_FLUSH_IS_SAFE)
 287                 return;
 288 
 289         if (current->mm) {
 290                 int i;
 291 
 292                 for (i = 0; i < VMACACHE_SIZE; i++) {
 293                         if (!current->vmacache.vmas[i])
 294                                 continue;
 295                         flush_cache_range(current->vmacache.vmas[i],
 296                                           addr, addr + BREAK_INSTR_SIZE);
 297                 }
 298         }
 299 
 300         /* Force flush instruction cache if it was outside the mm */
 301         flush_icache_range(addr, addr + BREAK_INSTR_SIZE);
 302 }

 308 int dbg_activate_sw_breakpoints(void)
 309 {
 310         int error;
 311         int ret = 0;
 312         int i;
 313 
 314         for (i = 0; i < KGDB_MAX_BREAKPOINTS; i++) {
 315                 if (kgdb_break[i].state != BP_SET)
 316                         continue;
 317 
 318                 error = kgdb_arch_set_breakpoint(&kgdb_break[i]);
 319                 if (error) {
 320                         ret = error;
 321                         pr_info("BP install failed: %lx\n",
 322                                 kgdb_break[i].bpt_addr);
 323                         continue;
 324                 }
 325 
 326                 kgdb_flush_swbreak_addr(kgdb_break[i].bpt_addr);
 327                 kgdb_break[i].state = BP_ACTIVE;
 328         }
 329         return ret;
 330 }
 331 NOKPROBE_SYMBOL(dbg_activate_sw_breakpoints);

 333 int dbg_set_sw_break(unsigned long addr)
 334 {
 335         int err = kgdb_validate_break_address(addr);
 336         int breakno = -1;
 337         int i;
 338 
 339         if (err)
 340                 return err;
 341 
 342         for (i = 0; i < KGDB_MAX_BREAKPOINTS; i++) {
 343                 if ((kgdb_break[i].state == BP_SET) &&
 344                                         (kgdb_break[i].bpt_addr == addr))
 345                         return -EEXIST;
 346         }
 347         for (i = 0; i < KGDB_MAX_BREAKPOINTS; i++) {
 348                 if (kgdb_break[i].state == BP_REMOVED &&
 349                                         kgdb_break[i].bpt_addr == addr) {
 350                         breakno = i;
 351                         break;
 352                 }
 353         }
 354 
 355         if (breakno == -1) {
 356                 for (i = 0; i < KGDB_MAX_BREAKPOINTS; i++) {
 357                         if (kgdb_break[i].state == BP_UNDEFINED) {
 358                                 breakno = i;
 359                                 break;
 360                         }
 361                 }
 362         }
 363 
 364         if (breakno == -1)
 365                 return -E2BIG;
 366 
 367         kgdb_break[breakno].state = BP_SET;
 368         kgdb_break[breakno].type = BP_BREAKPOINT;
 369         kgdb_break[breakno].bpt_addr = addr;
 370 
 371         return 0;
 372 }

 435 int dbg_remove_all_break(void)
 436 {
 437         int error;
 438         int i;
 439 
 440         /* Clear memory breakpoints. */
 441         for (i = 0; i < KGDB_MAX_BREAKPOINTS; i++) {
 442                 if (kgdb_break[i].state != BP_ACTIVE)
 443                         goto setundefined;
 444                 error = kgdb_arch_remove_breakpoint(&kgdb_break[i]);
 445                 if (error)
 446                         pr_err("breakpoint remove failed: %lx\n",
 447                                kgdb_break[i].bpt_addr);
 448 setundefined:
 449                 kgdb_break[i].state = BP_UNDEFINED;
 450         }
 451 
 452         /* Clear hardware breakpoints. */
 453         if (arch_kgdb_ops.remove_all_hw_break)
 454                 arch_kgdb_ops.remove_all_hw_break();
 455 
 456         return 0;
 457 }

 470 #ifdef CONFIG_KGDB_KDB
 471 void kdb_dump_stack_on_cpu(int cpu)
 472 {
 473         if (cpu == raw_smp_processor_id() || !IS_ENABLED(CONFIG_SMP)) {
 474                 dump_stack();
 475                 return;
 476         }
 477 
 478         if (!(kgdb_info[cpu].exception_state & DCPU_IS_SLAVE)) {
 479                 kdb_printf("ERROR: Task on cpu %d didn't stop in the debugger\n",
 480                            cpu);
 481                 return;
 482         }
 483 
 484         /*
 485          * In general, architectures don't support dumping the stack of a
 486          * "running" process that's not the current one.  From the point of
 487          * view of the Linux, kernel processes that are looping in the kgdb
 488          * slave loop are still "running".  There's also no API (that actually
 489          * works across all architectures) that can do a stack crawl based
 490          * on registers passed as a parameter.
 491          *
 492          * Solve this conundrum by asking slave CPUs to do the backtrace
 493          * themselves.
 494          */
 495         kgdb_info[cpu].exception_state |= DCPU_WANT_BT;
 496         while (kgdb_info[cpu].exception_state & DCPU_WANT_BT)
 497                 cpu_relax();
 498 }

 530 static int kgdb_reenter_check(struct kgdb_state *ks)
 531 {
 532         unsigned long addr;
 533 
 534         if (atomic_read(&kgdb_active) != raw_smp_processor_id())
 535                 return 0;
 536 
 537         /* Panic on recursive debugger calls: */
 538         exception_level++;
 539         addr = kgdb_arch_pc(ks->ex_vector, ks->linux_regs);
 540         dbg_deactivate_sw_breakpoints();
 541 
 542         /*
 543          * If the break point removed ok at the place exception
 544          * occurred, try to recover and print a warning to the end
 545          * user because the user planted a breakpoint in a place that
 546          * KGDB needs in order to function.
 547          */
 548         if (dbg_remove_sw_break(addr) == 0) {
 549                 exception_level = 0;
 550                 kgdb_skipexception(ks->ex_vector, ks->linux_regs);
 551                 dbg_activate_sw_breakpoints();
 552                 pr_crit("re-enter error: breakpoint removed %lx\n", addr);
 553                 WARN_ON_ONCE(1);
 554 
 555                 return 1;
 556         }
 557         dbg_remove_all_break();
 558         kgdb_skipexception(ks->ex_vector, ks->linux_regs);
 559 
 560         if (exception_level > 1) {
 561                 dump_stack();
 562                 kgdb_io_module_registered = false;
 563                 panic("Recursive entry to debugger");
 564         }
 565 
 566         pr_crit("re-enter exception: ALL breakpoints killed\n");
 567 #ifdef CONFIG_KGDB_KDB
 568         /* Allow kdb to debug itself one level */
 569         return 0;
 570 #endif
 571         dump_stack();
 572         panic("Recursive entry to debugger");
 573 
 574         return 1;
 575 }
 576 NOKPROBE_SYMBOL(kgdb_reenter_check);

 586 static int kgdb_cpu_enter(struct kgdb_state *ks, struct pt_regs *regs,
 587                 int exception_state)
 588 {
 589         unsigned long flags;
 590         int sstep_tries = 100;
 591         int error;
 592         int cpu;
 593         int trace_on = 0;
 594         int online_cpus = num_online_cpus();
 595         u64 time_left;
 596 
 597         kgdb_info[ks->cpu].enter_kgdb++;
 598         kgdb_info[ks->cpu].exception_state |= exception_state;
 599 
 600         if (exception_state == DCPU_WANT_MASTER)
 601                 atomic_inc(&masters_in_kgdb);
 602         else
 603                 atomic_inc(&slaves_in_kgdb);
 604 
 605         if (arch_kgdb_ops.disable_hw_break)
 606                 arch_kgdb_ops.disable_hw_break(regs);
 607 
 608 acquirelock:
 609         rcu_read_lock();
 610         /*
 611          * Interrupts will be restored by the 'trap return' code, except when
 612          * single stepping.
 613          */
 614         local_irq_save(flags);
 615 
 616         cpu = ks->cpu;
 617         kgdb_info[cpu].debuggerinfo = regs;
 618         kgdb_info[cpu].task = current;
 619         kgdb_info[cpu].ret_state = 0;
 620         kgdb_info[cpu].irq_depth = hardirq_count() >> HARDIRQ_SHIFT;
 621 
 622         /* Make sure the above info reaches the primary CPU */
 623         smp_mb();
 624 
 625         if (exception_level == 1) {
 626                 if (raw_spin_trylock(&dbg_master_lock))
 627                         atomic_xchg(&kgdb_active, cpu);
 628                 goto cpu_master_loop;
 629         }
 630 
 631         /*
 632          * CPU will loop if it is a slave or request to become a kgdb
 633          * master cpu and acquire the kgdb_active lock:
 634          */
 635         while (1) {
 636 cpu_loop:
 637                 if (kgdb_info[cpu].exception_state & DCPU_NEXT_MASTER) {
 638                         kgdb_info[cpu].exception_state &= ~DCPU_NEXT_MASTER;
 639                         goto cpu_master_loop;
 640                 } else if (kgdb_info[cpu].exception_state & DCPU_WANT_MASTER) {
 641                         if (raw_spin_trylock(&dbg_master_lock)) {
 642                                 atomic_xchg(&kgdb_active, cpu);
 643                                 break;
 644                         }
 645                 } else if (kgdb_info[cpu].exception_state & DCPU_WANT_BT) {
 646                         dump_stack();
 647                         kgdb_info[cpu].exception_state &= ~DCPU_WANT_BT;
 648                 } else if (kgdb_info[cpu].exception_state & DCPU_IS_SLAVE) {
 649                         if (!raw_spin_is_locked(&dbg_slave_lock))
 650                                 goto return_normal;
 651                 } else {
 652 return_normal:
 653                         /* Return to normal operation by executing any
 654                          * hw breakpoint fixup.
 655                          */
 656                         if (arch_kgdb_ops.correct_hw_break)
 657                                 arch_kgdb_ops.correct_hw_break();
 658                         if (trace_on)
 659                                 tracing_on();
 660                         kgdb_info[cpu].debuggerinfo = NULL;
 661                         kgdb_info[cpu].task = NULL;
 662                         kgdb_info[cpu].exception_state &=
 663                                 ~(DCPU_WANT_MASTER | DCPU_IS_SLAVE);
 664                         kgdb_info[cpu].enter_kgdb--;
 665                         smp_mb__before_atomic();
 666                         atomic_dec(&slaves_in_kgdb);
 667                         dbg_touch_watchdogs();
 668                         local_irq_restore(flags);
 669                         rcu_read_unlock();
 670                         return 0;
 671                 }
 672                 cpu_relax();
 673         }
 674 
 675         /*
 676          * For single stepping, try to only enter on the processor
 677          * that was single stepping.  To guard against a deadlock, the
 678          * kernel will only try for the value of sstep_tries before
 679          * giving up and continuing on.
 680          */
 681         if (atomic_read(&kgdb_cpu_doing_single_step) != -1 &&
 682             (kgdb_info[cpu].task &&
 683              kgdb_info[cpu].task->pid != kgdb_sstep_pid) && --sstep_tries) {
 684                 atomic_set(&kgdb_active, -1);
 685                 raw_spin_unlock(&dbg_master_lock);
 686                 dbg_touch_watchdogs();
 687                 local_irq_restore(flags);
 688                 rcu_read_unlock();
 689 
 690                 goto acquirelock;
 691         }
 692 
 693         if (!kgdb_io_ready(1)) {
 694                 kgdb_info[cpu].ret_state = 1;
 695                 goto kgdb_restore; /* No I/O connection, resume the system */
 696         }
 697 
 698         /*
 699          * Don't enter if we have hit a removed breakpoint.
 700          */
 701         if (kgdb_skipexception(ks->ex_vector, ks->linux_regs))
 702                 goto kgdb_restore;
 703 
 704         atomic_inc(&ignore_console_lock_warning);
 705 
 706         /* Call the I/O driver's pre_exception routine */
 707         if (dbg_io_ops->pre_exception)
 708                 dbg_io_ops->pre_exception();
 709 
 710         /*
 711          * Get the passive CPU lock which will hold all the non-primary
 712          * CPU in a spin state while the debugger is active
 713          */
 714         if (!kgdb_single_step)
 715                 raw_spin_lock(&dbg_slave_lock);
 716 
 717 #ifdef CONFIG_SMP
 718         /* If send_ready set, slaves are already waiting */
 719         if (ks->send_ready)
 720                 atomic_set(ks->send_ready, 1);
 721 
 722         /* Signal the other CPUs to enter kgdb_wait() */
 723         else if ((!kgdb_single_step) && kgdb_do_roundup)
 724                 kgdb_roundup_cpus();
 725 #endif
 726 
 727         /*
 728          * Wait for the other CPUs to be notified and be waiting for us:
 729          */
 730         time_left = MSEC_PER_SEC;
 731         while (kgdb_do_roundup && --time_left &&
 732                (atomic_read(&masters_in_kgdb) + atomic_read(&slaves_in_kgdb)) !=
 733                    online_cpus)
 734                 udelay(1000);
 735         if (!time_left)
 736                 pr_crit("Timed out waiting for secondary CPUs.\n");
 737 
 738         /*
 739          * At this point the primary processor is completely
 740          * in the debugger and all secondary CPUs are quiescent
 741          */
 742         dbg_deactivate_sw_breakpoints();
 743         kgdb_single_step = 0;
 744         kgdb_contthread = current;
 745         exception_level = 0;
 746         trace_on = tracing_is_on();
 747         if (trace_on)
 748                 tracing_off();
 749 
 750         while (1) {
 751 cpu_master_loop:
 752                 if (dbg_kdb_mode) {
 753                         kgdb_connected = 1;
 754                         error = kdb_stub(ks);
 755                         if (error == -1)
 756                                 continue;
 757                         kgdb_connected = 0;
 758                 } else {
 759                         error = gdb_serial_stub(ks);
 760                 }
 761 
 762                 if (error == DBG_PASS_EVENT) {
 763                         dbg_kdb_mode = !dbg_kdb_mode;
 764                 } else if (error == DBG_SWITCH_CPU_EVENT) {
 765                         kgdb_info[dbg_switch_cpu].exception_state |=
 766                                 DCPU_NEXT_MASTER;
 767                         goto cpu_loop;
 768                 } else {
 769                         kgdb_info[cpu].ret_state = error;
 770                         break;
 771                 }
 772         }
 773 
 774         dbg_activate_sw_breakpoints();
 775 
 776         /* Call the I/O driver's post_exception routine */
 777         if (dbg_io_ops->post_exception)
 778                 dbg_io_ops->post_exception();
 779 
 780         atomic_dec(&ignore_console_lock_warning);
 781 
 782         if (!kgdb_single_step) {
 783                 raw_spin_unlock(&dbg_slave_lock);
 784                 /* Wait till all the CPUs have quit from the debugger. */
 785                 while (kgdb_do_roundup && atomic_read(&slaves_in_kgdb))
 786                         cpu_relax();
 787         }
 788 
 789 kgdb_restore:
 790         if (atomic_read(&kgdb_cpu_doing_single_step) != -1) {
 791                 int sstep_cpu = atomic_read(&kgdb_cpu_doing_single_step);
 792                 if (kgdb_info[sstep_cpu].task)
 793                         kgdb_sstep_pid = kgdb_info[sstep_cpu].task->pid;
 794                 else
 795                         kgdb_sstep_pid = 0;
 796         }
 797         if (arch_kgdb_ops.correct_hw_break)
 798                 arch_kgdb_ops.correct_hw_break();
 799         if (trace_on)
 800                 tracing_on();
 801 
 802         kgdb_info[cpu].debuggerinfo = NULL;
 803         kgdb_info[cpu].task = NULL;
 804         kgdb_info[cpu].exception_state &=
 805                 ~(DCPU_WANT_MASTER | DCPU_IS_SLAVE);
 806         kgdb_info[cpu].enter_kgdb--;
 807         smp_mb__before_atomic();
 808         atomic_dec(&masters_in_kgdb);
 809         /* Free kgdb_active */
 810         atomic_set(&kgdb_active, -1);
 811         raw_spin_unlock(&dbg_master_lock);
 812         dbg_touch_watchdogs();
 813         local_irq_restore(flags);
 814         rcu_read_unlock();
 815 
 816         return kgdb_info[cpu].ret_state;
 817 }
 818 NOKPROBE_SYMBOL(kgdb_cpu_enter);

 820 /*
 821  * kgdb_handle_exception() - main entry point from a kernel exception
 822  *
 823  * Locking hierarchy:
 824  *      interface locks, if any (begin_session)
 825  *      kgdb lock (kgdb_active)
 826  */
 827 int
 828 kgdb_handle_exception(int evector, int signo, int ecode, struct pt_regs *regs)
 829 {
 830         struct kgdb_state kgdb_var;
 831         struct kgdb_state *ks = &kgdb_var;
 832         int ret = 0;
 833 
 834         if (arch_kgdb_ops.enable_nmi)
 835                 arch_kgdb_ops.enable_nmi(0);
 836         /*
 837          * Avoid entering the debugger if we were triggered due to an oops
 838          * but panic_timeout indicates the system should automatically
 839          * reboot on panic. We don't want to get stuck waiting for input
 840          * on such systems, especially if its "just" an oops.
 841          */
 842         if (signo != SIGTRAP && panic_timeout)
 843                 return 1;
 844 
 845         memset(ks, 0, sizeof(struct kgdb_state));
 846         ks->cpu                 = raw_smp_processor_id();
 847         ks->ex_vector           = evector;
 848         ks->signo               = signo;
 849         ks->err_code            = ecode;
 850         ks->linux_regs          = regs;
 851 
 852         if (kgdb_reenter_check(ks))
 853                 goto out; /* Ouch, double exception ! */
 854         if (kgdb_info[ks->cpu].enter_kgdb != 0)
 855                 goto out;
 856 
 857         ret = kgdb_cpu_enter(ks, regs, DCPU_WANT_MASTER);
 858 out:
 859         if (arch_kgdb_ops.enable_nmi)
 860                 arch_kgdb_ops.enable_nmi(1);
 861         return ret;
 862 }
 863 NOKPROBE_SYMBOL(kgdb_handle_exception);

1123 /**
1124  *      kgdb_register_io_module - register KGDB IO module
1125  *      @new_dbg_io_ops: the io ops vector
1126  *
1127  *      Register it with the KGDB core.
1128  */
1129 int kgdb_register_io_module(struct kgdb_io *new_dbg_io_ops)
1130 {
1131         struct kgdb_io *old_dbg_io_ops;
1132         int err;
1133 
1134         spin_lock(&kgdb_registration_lock);
1135 
1136         old_dbg_io_ops = dbg_io_ops;
1137         if (old_dbg_io_ops) {
1138                 if (!old_dbg_io_ops->deinit) {
1139                         spin_unlock(&kgdb_registration_lock);
1140 
1141                         pr_err("KGDB I/O driver %s can't replace %s.\n",
1142                                 new_dbg_io_ops->name, old_dbg_io_ops->name);
1143                         return -EBUSY;
1144                 }
1145                 pr_info("Replacing I/O driver %s with %s\n",
1146                         old_dbg_io_ops->name, new_dbg_io_ops->name);
1147         }
1148 
1149         if (new_dbg_io_ops->init) {
1150                 err = new_dbg_io_ops->init();
1151                 if (err) {
1152                         spin_unlock(&kgdb_registration_lock);
1153                         return err;
1154                 }
1155         }
1156 
1157         dbg_io_ops = new_dbg_io_ops;
1158 
1159         spin_unlock(&kgdb_registration_lock);
1160 
1161         if (old_dbg_io_ops) {
1162                 old_dbg_io_ops->deinit();
1163                 return 0;
1164         }
1165 
1166         pr_info("Registered I/O driver %s\n", new_dbg_io_ops->name);
1167 
1168         /* Arm KGDB now. */
1169         kgdb_register_callbacks();
1170 
1171         if (kgdb_break_asap &&
1172             (!dbg_is_early || IS_ENABLED(CONFIG_ARCH_HAS_EARLY_DEBUG)))
1173                 kgdb_initial_breakpoint();
1174 
1175         return 0;
1176 }
1177 EXPORT_SYMBOL_GPL(kgdb_register_io_module);

1180  *      kkgdb_unregister_io_module - unregister KGDB IO module
1181  *      @old_dbg_io_ops: the io ops vector
1182  *
1183  *      Unregister it with the KGDB core.
1184  */
1185 void kgdb_unregister_io_module(struct kgdb_io *old_dbg_io_ops)
