     /*   */
 238 {
 239         arch_spinlock_t *l = (void *)lock;
 240         union {
 241                 __u32 val;

 409 static int __bpf_strtoull(const char *buf, size_t buf_len, u64 flags,
     /*   */
 410                           unsigned long long *res, bool *is_negative)
 411 {
 412         unsigned int base = flags & BPF_STRTOX_BASE_MASK;
 413         const char *cur_buf = buf;
 414         size_t cur_len = buf_len;
 415         unsigned int consumed;
 416         size_t val_len;
 417         char str[64];
 418 
 419         if (!buf || !buf_len || !res || !is_negative)
 420                 return -EINVAL;
 421 
 422         if (base != 0 && base != 8 && base != 10 && base != 16)
 423                 return -EINVAL;
 424 
 425         if (flags & ~BPF_STRTOX_BASE_MASK)
 426                 return -EINVAL;
 427 
 428         while (cur_buf < buf + buf_len && isspace(*cur_buf))
 429                 ++cur_buf;
 430 
 431         *is_negative = (cur_buf < buf + buf_len && *cur_buf == '-');
 432         if (*is_negative)
 433                 ++cur_buf;
 434 
 435         consumed = cur_buf - buf;
 436         cur_len -= consumed;
 437         if (!cur_len)
 438                 return -EINVAL;
 439 
 440         cur_len = min(cur_len, sizeof(str) - 1);
 441         memcpy(str, cur_buf, cur_len);
 442         str[cur_len] = '\0';
 443         cur_buf = str;
 444 
 445         cur_buf = _parse_integer_fixup_radix(cur_buf, &base);
 446         val_len = _parse_integer(cur_buf, base, res);
 447 
 448         if (val_len & KSTRTOX_OVERFLOW)
 449                 return -ERANGE;
 450 
 451         if (val_len == 0)
 452                 return -EINVAL;
 453 
 454         cur_buf += val_len;
 455         consumed += cur_buf - str;
 456 
 457         return consumed;
 458 }

 460 static int __bpf_strtoll(const char *buf, size_t buf_len, u64 flags,
     /*   */
 461                          long long *res)
 462 {
 463         unsigned long long _res;
 464         bool is_negative;
 465         int err;
 466 
 467         err = __bpf_strtoull(buf, buf_len, flags, &_res, &is_negative);
 468         if (err < 0)
 469                 return err;
 470         if (is_negative) {
 471                 if ((long long)-_res > 0)
 472                         return -ERANGE;
 473                 *res = -_res;
 474         } else {
 475                 if ((long long)_res < 0)
 476                         return -ERANGE;
 477                 *res = _res;
 478         }
 479         return err;
 480 }

 536 BPF_CALL_4(bpf_get_ns_current_pid_tgid, u64, dev, u64, ino,
     /*   */
 537            struct bpf_pidns_info *, nsdata, u32, size)
 538 {
 539         struct task_struct *task = current;
 540         struct pid_namespace *pidns;
 541         int err = -EINVAL;
 542 
 543         if (unlikely(size != sizeof(struct bpf_pidns_info)))
 544                 goto clear;
 545 
 546         if (unlikely((u64)(dev_t)dev != dev))
 547                 goto clear;
 548 
 549         if (unlikely(!task))
 550                 goto clear;
 551 
 552         pidns = task_active_pid_ns(task);
 553         if (unlikely(!pidns)) {
 554                 err = -ENOENT;
 555                 goto clear;
 556         }
 557 
 558         if (!ns_match(&pidns->ns, (dev_t)dev, ino))
 559                 goto clear;
 560 
 561         nsdata->pid = task_pid_nr_ns(task, pidns);
 562         nsdata->tgid = task_tgid_nr_ns(task, pidns);
 563         return 0;
 564 clear:
 565         memset((void *)nsdata, 0, (size_t) size);
 566         return err;
 567 }

 610         if (unlikely(ret)) {
 611                 memset(dst, 0, size);
 612                 ret = -EFAULT;
 613         }
 614 
 615         return ret;
 616 }
