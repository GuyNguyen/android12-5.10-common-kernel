  35 static int __init set_pelt(char *str)
  36 {
  37         int rc, num;
  38 
  39         rc = kstrtoint(str, 0, &num);
  40         if (rc) {
  41                 pr_err("%s: kstrtoint failed. rc=%d\n", __func__, rc);
  42                 return 0;
  43         }
  44 
  45         switch (num) {
  46         case PELT8_LOAD_AVG_PERIOD:
  47                 pelt_load_avg_period = PELT8_LOAD_AVG_PERIOD;
  48                 pelt_load_avg_max = PELT8_LOAD_AVG_MAX;
  49                 pelt_runnable_avg_yN_inv = pelt8_runnable_avg_yN_inv;
  50                 pr_info("PELT half life is set to %dms\n", num);
  51                 break;
  52         case PELT32_LOAD_AVG_PERIOD:
  53                 pelt_load_avg_period = PELT32_LOAD_AVG_PERIOD;
  54                 pelt_load_avg_max = PELT32_LOAD_AVG_MAX;
  55                 pelt_runnable_avg_yN_inv = pelt32_runnable_avg_yN_inv;
  56                 pr_info("PELT half life is set to %dms\n", num);
  57                 break;
  58         default:
  59                 pr_err("Default PELT half life is 32ms\n");
  60         }
  61 
  62         return 0;
  63 }

  67 /*
  68  * Approximate:
  69  *   val * y^n,    where y^32 ~= 0.5 (~1 scheduling period)
  70  */
  71 static u64 decay_load(u64 val, u64 n)
  72 {
  73         unsigned int local_n;
  74 
  75         if (unlikely(n > LOAD_AVG_PERIOD * 63))
  76                 return 0;
  77 
  78         /* after bounds checking we can collapse to 32-bit */
  79         local_n = n;
  80 
  81         /*
  82          * As y^PERIOD = 1/2, we can combine
  83          *    y^n = 1/2^(n/PERIOD) * y^(n%PERIOD)
  84          * With a look-up table which covers y^n (n<PERIOD)
  85          *
  86          * To achieve constant time decay_load.
  87          */
  88         if (unlikely(local_n >= LOAD_AVG_PERIOD)) {
  89                 val >>= local_n / LOAD_AVG_PERIOD;
  90                 local_n %= LOAD_AVG_PERIOD;
  91         }
  92 
  93         val = mul_u64_u32_shr(val, pelt_runnable_avg_yN_inv[local_n], 32);
  94         return val;
  95 }

  97 static u32 __accumulate_pelt_segments(u64 periods, u32 d1, u32 d3)
  98 {
  99         u32 c1, c2, c3 = d3; /* y^0 == 1 */
 100 
 101         /*
 102          * c1 = d1 y^p
 103          */
 104         c1 = decay_load((u64)d1, periods);
 105 
 106         /*
 107          *            p-1
 108          * c2 = 1024 \Sum y^n
 109          *            n=1
 110          *
 111          *              inf        inf
 112          *    = 1024 ( \Sum y^n - \Sum y^n - y^0 )
 113          *              n=0        n=p
 114          */
 115         c2 = LOAD_AVG_MAX - decay_load(LOAD_AVG_MAX, periods) - 1024;
 116 
 117         return c1 + c2 + c3;
 118 }

 120 /*
 121  * Accumulate the three separate parts of the sum; d1 the remainder
 122  * of the last (incomplete) period, d2 the span of full periods and d3
 123  * the remainder of the (incomplete) current period.
 124  *
 125  *           d1          d2           d3
 126  *           ^           ^            ^
 127  *           |           |            |
 128  *         |<->|<----------------->|<--->|
 129  * ... |---x---|------| ... |------|-----x (now)
 130  *
 131  *                           p-1
 132  * u' = (u + d1) y^p + 1024 \Sum y^n + d3 y^0
 133  *                           n=1
 134  *
 135  *    = u y^p +                                 (Step 1)
 136  *
 137  *                     p-1
 138  *      d1 y^p + 1024 \Sum y^n + d3 y^0         (Step 2)
 139  *                     n=1
 140  */
 141 static __always_inline u32
 142 accumulate_sum(u64 delta, struct sched_avg *sa,
 143                unsigned long load, unsigned long runnable, int running)
 144 {
 145         u32 contrib = (u32)delta; /* p == 0 -> delta < 1024 */
 146         u64 periods;
 147 
 148         delta += sa->period_contrib;
 149         periods = delta / 1024; /* A period is 1024us (~1ms) */
 150 
 151         /*
 152          * Step 1: decay old *_sum if we crossed period boundaries.
 153          */
 154         if (periods) {
 155                 sa->load_sum = decay_load(sa->load_sum, periods);
 156                 sa->runnable_sum =
 157                         decay_load(sa->runnable_sum, periods);
 158                 sa->util_sum = decay_load((u64)(sa->util_sum), periods);
 159 
 160                 /*
 161                  * Step 2
 162                  */
 163                 delta %= 1024;
 164                 if (load) {
 165                         /*
 166                          * This relies on the:
 167                          *
 168                          * if (!load)
 169                          *      runnable = running = 0;
 170                          *
 171                          * clause from ___update_load_sum(); this results in
 172                          * the below usage of @contrib to dissapear entirely,
 173                          * so no point in calculating it.
 174                          */
 175                         contrib = __accumulate_pelt_segments(periods,
 176                                         1024 - sa->period_contrib, delta);
 177                 }
 178         }
 179         sa->period_contrib = delta;
 180 
 181         if (load)
 182                 sa->load_sum += load * contrib;
 183         if (runnable)
 184                 sa->runnable_sum += runnable * contrib << SCHED_CAPACITY_SHIFT;
 185         if (running)
 186                 sa->util_sum += contrib << SCHED_CAPACITY_SHIFT;
 187 
 188         return periods;
 189 }

 191 /*
 192  * We can represent the historical contribution to runnable average as the
 193  * coefficients of a geometric series.  To do this we sub-divide our runnable
 194  * history into segments of approximately 1ms (1024us); label the segment that
 195  * occurred N-ms ago p_N, with p_0 corresponding to the current period, e.g.
 196  *
 197  * [<- 1024us ->|<- 1024us ->|<- 1024us ->| ...
 198  *      p0            p1           p2
 199  *     (now)       (~1ms ago)  (~2ms ago)
 200  *
 201  * Let u_i denote the fraction of p_i that the entity was runnable.
 202  *
 203  * We then designate the fractions u_i as our co-efficients, yielding the
 204  * following representation of historical load:
 205  *   u_0 + u_1*y + u_2*y^2 + u_3*y^3 + ...
 206  *
 207  * We choose y based on the with of a reasonably scheduling period, fixing:
 208  *   y^32 = 0.5
 209  *
 210  * This means that the contribution to load ~32ms ago (u_32) will be weighted
 211  * approximately half as much as the contribution to load within the last ms
 212  * (u_0).
 213  *
 214  * When a period "rolls over" and we have new u_0`, multiplying the previous
 215  * sum again by y is sufficient to update:
 216  *   load_avg = u_0` + y*(u_0 + u_1*y + u_2*y^2 + ... )
 217  *            = u_0 + u_1*y + u_2*y^2 + ... [re-labeling u_i --> u_{i+1}]
 218  */
 219 static __always_inline int
 220 ___update_load_sum(u64 now, struct sched_avg *sa,
 221                   unsigned long load, unsigned long runnable, int running)
 222 {
 223         u64 delta;
 224 
 225         delta = now - sa->last_update_time;
 226         /*
 227          * This should only happen when time goes backwards, which it
 228          * unfortunately does during sched clock init when we swap over to TSC.
 229          */
 230         if ((s64)delta < 0) {
 231                 sa->last_update_time = now;
 232                 return 0;
 233         }
 234 
 235         /*
 236          * Use 1024ns as the unit of measurement since it's a reasonable
 237          * approximation of 1us and fast to compute.
 238          */
 239         delta >>= 10;
 240         if (!delta)
 241                 return 0;
 242 
 243         sa->last_update_time += delta << 10;
 244 
 245         /*
 246          * running is a subset of runnable (weight) so running can't be set if
 247          * runnable is clear. But there are some corner cases where the current
 248          * se has been already dequeued but cfs_rq->curr still points to it.
 249          * This means that weight will be 0 but not running for a sched_entity
 250          * but also for a cfs_rq if the latter becomes idle. As an example,
 251          * this happens during idle_balance() which calls
 252          * update_blocked_averages().
 253          *
 254          * Also see the comment in accumulate_sum().
 255          */
 256         if (!load)
 257                 runnable = running = 0;
 258 
 259         /*
 260          * Now we know we crossed measurement unit boundaries. The *_avg
 261          * accrues by two steps:
 262          *
 263          * Step 1: accumulate *_sum since last_update_time. If we haven't
 264          * crossed period boundaries, finish.
 265          */
 266         if (!accumulate_sum(delta, sa, load, runnable, running))
 267                 return 0;
 268 
 269         return 1;
 270 }

 272 /*
 273  * When syncing *_avg with *_sum, we must take into account the current
 274  * position in the PELT segment otherwise the remaining part of the segment
 275  * will be considered as idle time whereas it's not yet elapsed and this will
 276  * generate unwanted oscillation in the range [1002..1024[.
 277  *
 278  * The max value of *_sum varies with the position in the time segment and is
 279  * equals to :
 280  *
 281  *   LOAD_AVG_MAX*y + sa->period_contrib
 282  *
 283  * which can be simplified into:
 284  *
 285  *   LOAD_AVG_MAX - 1024 + sa->period_contrib
 286  *
 287  * because LOAD_AVG_MAX*y == LOAD_AVG_MAX-1024
 288  *
 289  * The same care must be taken when a sched entity is added, updated or
 290  * removed from a cfs_rq and we need to update sched_avg. Scheduler entities
 291  * and the cfs rq, to which they are attached, have the same position in the
 292  * time segment because they use the same clock. This means that we can use
 293  * the period_contrib of cfs_rq when updating the sched_avg of a sched_entity
 294  * if it's more convenient.
 295  */
 296 static __always_inline void
 297 ___update_load_avg(struct sched_avg *sa, unsigned long load)
 298 {
 299         u32 divider = get_pelt_divider(sa);
 300 
 301         /*
 302          * Step 2: update *_avg.
 303          */
 304         sa->load_avg = div_u64(load * sa->load_sum, divider);
 305         sa->runnable_avg = div_u64(sa->runnable_sum, divider);
 306         WRITE_ONCE(sa->util_avg, sa->util_sum / divider);
 307 }

 309 /*
 310  * sched_entity:
 311  *
 312  *   task:
 313  *     se_weight()   = se->load.weight
 314  *     se_runnable() = !!on_rq
 315  *
 316  *   group: [ see update_cfs_group() ]
 317  *     se_weight()   = tg->weight * grq->load_avg / tg->load_avg
 318  *     se_runnable() = grq->h_nr_running
 319  *
 320  *   runnable_sum = se_runnable() * runnable = grq->runnable_sum
 321  *   runnable_avg = runnable_sum
 322  *
 323  *   load_sum := runnable
 324  *   load_avg = se_weight(se) * load_sum
 325  *
 326  * cfq_rq:
 327  *
 328  *   runnable_sum = \Sum se->avg.runnable_sum
 329  *   runnable_avg = \Sum se->avg.runnable_avg
 330  *
 331  *   load_sum = \Sum se_weight(se) * se->avg.load_sum
 332  *   load_avg = \Sum se->avg.load_avg
 333  */

 335 int __update_load_avg_blocked_se(u64 now, struct sched_entity *se)
 336 {
 337         if (___update_load_sum(now, &se->avg, 0, 0, 0)) {
 338                 ___update_load_avg(&se->avg, se_weight(se));
 339                 trace_pelt_se_tp(se);
 340                 return 1;
 341         }
 342 
 343         return 0;
 344 }

 348 {

 430  * thermal:
 431  *
 432  *   load_sum = \Sum se->avg.load_sum but se->avg.load_sum is not tracked

 434  *   util_avg and runnable_load_avg are not supported and meaningless.

 459 #ifdef CONFIG_HAVE_SCHED_AVG_IRQ
 460 /*
 461  * irq:
 462  *
 463  *   util_sum = \Sum se->avg.util_sum but se->avg.util_sum is not tracked
 464  *   util_sum = cpu_scale * load_sum
 465  *   runnable_sum = util_sum
 466  *
 467  *   load_avg and runnable_avg are not supported and meaningless.
 468  *
 469  */
 470 
 471 int update_irq_load_avg(struct rq *rq, u64 running)
 472 {
 473         int ret = 0;
 474 
 475         /*
 476          * We can't use clock_pelt because irq time is not accounted in
 477          * clock_task. Instead we directly scale the running time to
 478          * reflect the real amount of computation
 479          */
 480         running = cap_scale(running, arch_scale_freq_capacity(cpu_of(rq)));
 481         running = cap_scale(running, arch_scale_cpu_capacity(cpu_of(rq)));
 482 
 483         /*
 484          * We know the time that has been used by interrupt since last update
 485          * but we don't when. Let be pessimistic and assume that interrupt has
 486          * happened just before the update. This is not so far from reality
 487          * because interrupt will most probably wake up task and trig an update
 488          * of rq clock during which the metric is updated.
 489          * We start to decay with normal context time and then we add the
 490          * interrupt context time.
 491          * We can safely remove running from rq->clock because
 492          * rq->clock += delta with delta >= running
 493          */
 494         ret = ___update_load_sum(rq->clock - running, &rq->avg_irq,
 495                                 0,
 496                                 0,
 497                                 0);
 498         ret += ___update_load_sum(rq->clock, &rq->avg_irq,
 499                                 1,
 500                                 1,
 501                                 1);
 502 
 503         if (ret) {
 504                 ___update_load_avg(&rq->avg_irq, 1);
 505                 trace_pelt_irq_tp(rq);
 506         }
 507 
 508         return ret;
 509 }
 510 #endif
