  49 /*
  50  * READ this before attempting to hack on futexes!
  51  *
  52  * Basic futex operation and ordering guarantees
  53  * =============================================
  54  *
  55  * The waiter reads the futex value in user space and calls
  56  * futex_wait(). This function computes the hash bucket and acquires
  57  * the hash bucket lock. After that it reads the futex user space value
  58  * again and verifies that the data has not changed. If it has not changed
  59  * it enqueues itself into the hash bucket, releases the hash bucket lock
  60  * and schedules.
  61  *
  62  * The waker side modifies the user space value of the futex and calls
  63  * futex_wake(). This function computes the hash bucket and acquires the
  64  * hash bucket lock. Then it looks for waiters on that futex in the hash
  65  * bucket and wakes them.
  66  *
  67  * In futex wake up scenarios where no tasks are blocked on a futex, taking
  68  * the hb spinlock can be avoided and simply return. In order for this
  69  * optimization to work, ordering guarantees must exist so that the waiter
  70  * being added to the list is acknowledged when the list is concurrently being
  71  * checked by the waker, avoiding scenarios like the following:
  72  *
  73  * CPU 0                               CPU 1
  74  * val = *futex;
  75  * sys_futex(WAIT, futex, val);
  76  *   futex_wait(futex, val);
  77  *   uval = *futex;
  78  *                                     *futex = newval;
  79  *                                     sys_futex(WAKE, futex);
  80  *                                       futex_wake(futex);
  81  *                                       if (queue_empty())
  82  *                                         return;
  83  *   if (uval == val)
  84  *      lock(hash_bucket(futex));
  85  *      queue();
  86  *     unlock(hash_bucket(futex));
  87  *     schedule();
  88  *
  89  * This would cause the waiter on CPU 0 to wait forever because it
  90  * missed the transition of the user space value from val to newval
  91  * and the waker did not find the waiter in the hash bucket queue.
  92  *
  93  * The correct serialization ensures that a waiter either observes
  94  * the changed user space value before blocking or is woken by a
  95  * concurrent waker:
  96  *
  97  * CPU 0                                 CPU 1
  98  * val = *futex;
  99  * sys_futex(WAIT, futex, val);
 100  *   futex_wait(futex, val);
 101  *
 102  *   waiters++; (a)
 103  *   smp_mb(); (A) <-- paired with -.
 104  *                                  |
 105  *   lock(hash_bucket(futex));      |
 106  *                                  |
 107  *   uval = *futex;                 |
 108  *                                  |        *futex = newval;
 109  *                                  |        sys_futex(WAKE, futex);
 110  *                                  |          futex_wake(futex);
 111  *                                  |
 112  *                                  `--------> smp_mb(); (B)
 113  *   if (uval == val)
 114  *     queue();
 115  *     unlock(hash_bucket(futex));
 116  *     schedule();                         if (waiters)
 117  *                                           lock(hash_bucket(futex));
 118  *   else                                    wake_waiters(futex);
 119  *     waiters--; (b)                        unlock(hash_bucket(futex));
 120  *
 121  * Where (A) orders the waiters increment and the futex value read through
 122  * atomic operations (see hb_waiters_inc) and where (B) orders the write
 123  * to futex and the waiters read (see hb_waiters_pending()).
 124  *
 125  * This yields the following case (where X:=waiters, Y:=futex):
 126  *
 127  *      X = Y = 0
 128  *
 129  *      w[X]=1          w[Y]=1
 130  *      MB              MB
 131  *      r[Y]=y          r[X]=x
 132  *
 133  * Which guarantees that x==0 && y==0 is impossible; which translates back into
 134  * the guarantee that we cannot both miss the futex variable change and the
 135  * enqueue.
 136  *
 137  * Note that a new waiter is accounted for in (a) even when it is possible that
 138  * the wait call can return error, in which case we backtrack from it in (b).
 139  * Refer to the comment in queue_lock().
 140  *
 141  * Similarly, in order to account for waiters being requeued on another
 142  * address we always increment the waiters for the destination bucket before
 143  * acquiring the lock. It then decrements them again  after releasing it -
 144  * the code that actually moves the futex(es) between hash buckets (requeue_futex)
 145  * will do the additional required waiter count housekeeping. This is done for
 146  * double_lock_hb() and double_unlock_hb(), respectively.
 147  */

 421 /*
 422  * Generate a machine wide unique identifier for this inode.
 423  *
 424  * This relies on u64 not wrapping in the life-time of the machine; which with
 425  * 1ns resolution means almost 585 years.
 426  *
 427  * This further relies on the fact that a well formed program will not unmap
 428  * the file while it has a (shared) futex waiting on it. This mapping will have
 429  * a file reference which pins the mount and inode.
 430  *
 431  * If for some reason an inode gets evicted and read back in again, it will get
 432  * a new sequence number and will _NOT_ match, even though it is the exact same
 433  * file.
 434  *
 435  * It is important that match_futex() will never have a false-positive, esp.
 436  * for PI futexes that can mess up the state. The above argues that false-negatives
 437  * are only possible for malformed programs.
 438  */
 439 static u64 get_inode_sequence_number(struct inode *inode)
     /*   */
 440 {
 441         static atomic64_t i_seq;
 442         u64 old;
 443 
 444         /* Does the inode already have a sequence number? */
 445         old = atomic64_read(&inode->i_sequence);
 446         if (likely(old))
 447                 return old;
 448 
 449         for (;;) {
 450                 u64 new = atomic64_add_return(1, &i_seq);
 451                 if (WARN_ON_ONCE(!new))
 452                         continue;
 453 
 454                 old = atomic64_cmpxchg_relaxed(&inode->i_sequence, 0, new);
 455                 if (old)
 456                         return old;
 457                 return new;
 458         }
 459 }

 461 /**
 462  * get_futex_key() - Get parameters which are the keys for a futex
 463  * @uaddr:      virtual address of the futex
 464  * @fshared:    false for a PROCESS_PRIVATE futex, true for PROCESS_SHARED
 465  * @key:        address where result is stored.
 466  * @rw:         mapping needs to be read/write (values: FUTEX_READ,
 467  *              FUTEX_WRITE)
 468  *
 469  * Return: a negative error code or 0
 470  *
 471  * The key words are stored in @key on success.
 472  *
 473  * For shared mappings (when @fshared), the key is:
 474  *
 475  *   ( inode->i_sequence, page->index, offset_within_page )
 476  *
 477  * [ also see get_inode_sequence_number() ]
 478  *
 479  * For private mappings (or when !@fshared), the key is:
 480  *
 481  *   ( current->mm, address, 0 )
 482  *
 483  * This allows (cross process, where applicable) identification of the futex
 484  * without keeping the page pinned for the duration of the FUTEX_WAIT.
 485  *
 486  * lock_page() might sleep, the caller should not hold a spinlock.
 487  */
 488 static int get_futex_key(u32 __user *uaddr, bool fshared, union futex_key *key,
     /*   */
 489                          enum futex_access rw)
 490 {
 491         unsigned long address = (unsigned long)uaddr;
 492         struct mm_struct *mm = current->mm;
 493         struct page *page, *tail;
 494         struct address_space *mapping;
 495         int err, ro = 0;
 496 
 497         /*
 498          * The futex address must be "naturally" aligned.
 499          */
 500         key->both.offset = address % PAGE_SIZE;
 501         if (unlikely((address % sizeof(u32)) != 0))
 502                 return -EINVAL;
 503         address -= key->both.offset;
 504 
 505         if (unlikely(!access_ok(uaddr, sizeof(u32))))
 506                 return -EFAULT;
 507 
 508         if (unlikely(should_fail_futex(fshared)))
 509                 return -EFAULT;
 510 
 511         /*
 512          * PROCESS_PRIVATE futexes are fast.
 513          * As the mm cannot disappear under us and the 'key' only needs
 514          * virtual address, we dont even have to find the underlying vma.
 515          * Note : We do have to check 'uaddr' is a valid user address,
 516          *        but access_ok() should be faster than find_vma()
 517          */
 518         if (!fshared) {
 519                 key->private.mm = mm;
 520                 key->private.address = address;
 521                 return 0;
 522         }
 523 
 524 again:
 525         /* Ignore any VERIFY_READ mapping (futex common case) */
 526         if (unlikely(should_fail_futex(true)))
 527                 return -EFAULT;
 528 
 529         err = get_user_pages_fast(address, 1, FOLL_WRITE, &page);
 530         /*
 531          * If write access is not required (eg. FUTEX_WAIT), try
 532          * and get read-only access.
 533          */
 534         if (err == -EFAULT && rw == FUTEX_READ) {
 535                 err = get_user_pages_fast(address, 1, 0, &page);
 536                 ro = 1;
 537         }
 538         if (err < 0)
 539                 return err;
 540         else
 541                 err = 0;
 542 
 543         /*
 544          * The treatment of mapping from this point on is critical. The page
 545          * lock protects many things but in this context the page lock
 546          * stabilizes mapping, prevents inode freeing in the shared
 547          * file-backed region case and guards against movement to swap cache.
 548          *
 549          * Strictly speaking the page lock is not needed in all cases being
 550          * considered here and page lock forces unnecessarily serialization
 551          * From this point on, mapping will be re-verified if necessary and
 552          * page lock will be acquired only if it is unavoidable
 553          *
 554          * Mapping checks require the head page for any compound page so the
 555          * head page and mapping is looked up now. For anonymous pages, it
 556          * does not matter if the page splits in the future as the key is
 557          * based on the address. For filesystem-backed pages, the tail is
 558          * required as the index of the page determines the key. For
 559          * base pages, there is no tail page and tail == page.
 560          */
 561         tail = page;
 562         page = compound_head(page);
 563         mapping = READ_ONCE(page->mapping);
 564 
 565         /*
 566          * If page->mapping is NULL, then it cannot be a PageAnon
 567          * page; but it might be the ZERO_PAGE or in the gate area or
 568          * in a special mapping (all cases which we are happy to fail);
 569          * or it may have been a good file page when get_user_pages_fast
 570          * found it, but truncated or holepunched or subjected to
 571          * invalidate_complete_page2 before we got the page lock (also
 572          * cases which we are happy to fail).  And we hold a reference,
 573          * so refcount care in invalidate_complete_page's remove_mapping
 574          * prevents drop_caches from setting mapping to NULL beneath us.
 575          *
 576          * The case we do have to guard against is when memory pressure made
 577          * shmem_writepage move it from filecache to swapcache beneath us:
 578          * an unlikely race, but we do need to retry for page->mapping.
 579          */
 580         if (unlikely(!mapping)) {
 581                 int shmem_swizzled;
 582 
 583                 /*
 584                  * Page lock is required to identify which special case above
 585                  * applies. If this is really a shmem page then the page lock
 586                  * will prevent unexpected transitions.
 587                  */
 588                 lock_page(page);
 589                 shmem_swizzled = PageSwapCache(page) || page->mapping;
 590                 unlock_page(page);
 591                 put_user_page(page);
 592 
 593                 if (shmem_swizzled)
 594                         goto again;
 595 
 596                 return -EFAULT;
 597         }
 598 
 599         /*
 600          * Private mappings are handled in a simple way.
 601          *
 602          * If the futex key is stored on an anonymous page, then the associated
 603          * object is the mm which is implicitly pinned by the calling process.
 604          *
 605          * NOTE: When userspace waits on a MAP_SHARED mapping, even if
 606          * it's a read-only handle, it's expected that futexes attach to
 607          * the object not the particular process.
 608          */
 609         if (PageAnon(page)) {
 610                 /*
 611                  * A RO anonymous page will never change and thus doesn't make
 612                  * sense for futex operations.
 613                  */
 614                 if (unlikely(should_fail_futex(true)) || ro) {
 615                         err = -EFAULT;
 616                         goto out;
 617                 }
 618 
 619                 key->both.offset |= FUT_OFF_MMSHARED; /* ref taken on mm */
 620                 key->private.mm = mm;
 621                 key->private.address = address;
 622 
 623         } else {
 624                 struct inode *inode;
 625 
 626                 /*
 627                  * The associated futex object in this case is the inode and
 628                  * the page->mapping must be traversed. Ordinarily this should
 629                  * be stabilised under page lock but it's not strictly
 630                  * necessary in this case as we just want to pin the inode, not
 631                  * update the radix tree or anything like that.
 632                  *
 633                  * The RCU read lock is taken as the inode is finally freed
 634                  * under RCU. If the mapping still matches expectations then the
 635                  * mapping->host can be safely accessed as being a valid inode.
 636                  */
 637                 rcu_read_lock();
 638 
 639                 if (READ_ONCE(page->mapping) != mapping) {
 640                         rcu_read_unlock();
 641                         put_user_page(page);
 642 
 643                         goto again;
 644                 }
 645 
 646                 inode = READ_ONCE(mapping->host);
 647                 if (!inode) {
 648                         rcu_read_unlock();
 649                         put_user_page(page);
 650 
 651                         goto again;
 652                 }
 653 
 654                 key->both.offset |= FUT_OFF_INODE; /* inode-based key */
 655                 key->shared.i_seq = get_inode_sequence_number(inode);
 656                 key->shared.pgoff = basepage_index(tail);
 657                 rcu_read_unlock();
 658         }
 659 
 660 out:
 661         put_user_page(page);
 662         return err;
 663 }

 828                  * refcount is at 0 - put it back to 1.

 843 static void exit_pi_state_list(struct task_struct *curr)
     /*   */
 844 {
 845         struct list_head *next, *head = &curr->pi_state_list;
 846         struct futex_pi_state *pi_state;
 847         struct futex_hash_bucket *hb;
 848         union futex_key key = FUTEX_KEY_INIT;

 850         if (!futex_cmpxchg_enabled)

 853          * We are a ZOMBIE and nobody can enqueue itself on
 854          * pi_state_list anymore, but we have to be careful
 855          * versus waiters unqueueing themselves:
 856          */
 857         raw_spin_lock_irq(&curr->pi_lock);
 858         while (!list_empty(head)) {
 859                 next = head->next;
 860                 pi_state = list_entry(next, struct futex_pi_state, list);
 861                 key = pi_state->key;
 862                 hb = hash_futex(&key);

 865                  * We can race against put_pi_state() removing itself from the
 866                  * list (a waiter going away). put_pi_state() will first
 867                  * decrement the reference count and then modify the list, so
 868                  * its possible to see the list entry but fail this reference
 869                  * acquire.
 870                  *
 871                  * In that case; drop the locks to let put_pi_state() make
 872                  * progress and retry the loop.
 873                  */
 874                 if (!refcount_inc_not_zero(&pi_state->refcount)) {
 875                         raw_spin_unlock_irq(&curr->pi_lock);
 876                         cpu_relax();
 877                         raw_spin_lock_irq(&curr->pi_lock);

 879                 }

 887                  * task still owns the PI-state:

 889                 if (head->next != next) {
 890                         /* retain curr->pi_lock for the loop invariant */
 891                         raw_spin_unlock(&pi_state->pi_mutex.wait_lock);
 892                         spin_unlock(&hb->lock);
 893                         put_pi_state(pi_state);

 903                 raw_spin_unlock_irq(&pi_state->pi_mutex.wait_lock);
 904                 spin_unlock(&hb->lock);
 905 
 906                 rt_mutex_futex_unlock(&pi_state->pi_mutex);
 907                 put_pi_state(pi_state);
 908 
 909                 raw_spin_lock_irq(&curr->pi_lock);
 910         }

 917 /*
 918  * We need to check the following states:
 919  *
 920  *      Waiter | pi_state | pi->owner | uTID      | uODIED | ?
 921  *
 922  * [1]  NULL   | ---      | ---       | 0         | 0/1    | Valid
 923  * [2]  NULL   | ---      | ---       | >0        | 0/1    | Valid
 924  *
 925  * [3]  Found  | NULL     | --        | Any       | 0/1    | Invalid
 926  *
 927  * [4]  Found  | Found    | NULL      | 0         | 1      | Valid
 928  * [5]  Found  | Found    | NULL      | >0        | 1      | Invalid
 929  *
 930  * [6]  Found  | Found    | task      | 0         | 1      | Valid
 931  *
 932  * [7]  Found  | Found    | NULL      | Any       | 0      | Invalid
 933  *
 934  * [8]  Found  | Found    | task      | ==taskTID | 0/1    | Valid
 935  * [9]  Found  | Found    | task      | 0         | 0      | Invalid
 936  * [10] Found  | Found    | task      | !=taskTID | 0/1    | Invalid
 937  *
 938  * [1]  Indicates that the kernel can acquire the futex atomically. We
 939  *      came here due to a stale FUTEX_WAITERS/FUTEX_OWNER_DIED bit.
 940  *
 941  * [2]  Valid, if TID does not belong to a kernel thread. If no matching
 942  *      thread is found then it indicates that the owner TID has died.
 943  *
 944  * [3]  Invalid. The waiter is queued on a non PI futex
 945  *
 946  * [4]  Valid state after exit_robust_list(), which sets the user space
 947  *      value to FUTEX_WAITERS | FUTEX_OWNER_DIED.
 948  *
 949  * [5]  The user space value got manipulated between exit_robust_list()
 950  *      and exit_pi_state_list()
 951  *
 952  * [6]  Valid state after exit_pi_state_list() which sets the new owner in
 953  *      the pi_state but cannot access the user space value.
 954  *
 955  * [7]  pi_state->owner can only be NULL when the OWNER_DIED bit is set.
 956  *
 957  * [8]  Owner and user space value match
 958  *
 959  * [9]  There is no transient state which sets the user space TID to 0
 960  *      except exit_robust_list(), but this is indicated by the
 961  *      FUTEX_OWNER_DIED bit. See [4]
 962  *
 963  * [10] There is no transient state which leaves owner and user space
 964  *      TID out of sync. Except one error case where the kernel is denied
 965  *      write access to the user address, see fixup_pi_state_owner().
 966  *
 967  *
 968  * Serialization and lifetime rules:
 969  *
 970  * hb->lock:
 971  *
 972  *      hb -> futex_q, relation
 973  *      futex_q -> pi_state, relation
 974  *
 975  *      (cannot be raw because hb can contain arbitrary amount
 976  *       of futex_q's)
 977  *
 978  * pi_mutex->wait_lock:
 979  *
 980  *      {uval, pi_state}
 981  *
 982  *      (and pi_mutex 'obviously')
 983  *
 984  * p->pi_lock:
 985  *
 986  *      p->pi_state_list -> pi_state->list, relation
 987  *
 988  * pi_state->refcount:
 989  *
 990  *      pi_state lifetime
 991  *
 992  *
 993  * Lock order:
 994  *
 995  *   hb->lock
 996  *     pi_mutex->wait_lock
 997  *       p->pi_lock
 998  *
 999  */

1001 /*
1002  * Validate that the existing waiter has a pi_state and sanity check
1003  * the pi_state against the user space value. If correct, attach to
1004  * it.
1005  */
1006 static int attach_to_pi_state(u32 __user *uaddr, u32 uval,
     /*   */
1007                               struct futex_pi_state *pi_state,
1008                               struct futex_pi_state **ps)
1009 {
1010         pid_t pid = uval & FUTEX_TID_MASK;
1011         u32 uval2;
1012         int ret;
1013 
1014         /*
1015          * Userspace might have messed up non-PI and PI futexes [3]
1016          */
1017         if (unlikely(!pi_state))
1018                 return -EINVAL;
1019 
1020         /*
1021          * We get here with hb->lock held, and having found a
1022          * futex_top_waiter(). This means that futex_lock_pi() of said futex_q
1023          * has dropped the hb->lock in between queue_me() and unqueue_me_pi(),
1024          * which in turn means that futex_lock_pi() still has a reference on
1025          * our pi_state.
1026          *
1027          * The waiter holding a reference on @pi_state also protects against
1028          * the unlocked put_pi_state() in futex_unlock_pi(), futex_lock_pi()
1029          * and futex_wait_requeue_pi() as it cannot go to 0 and consequently
1030          * free pi_state before we can take a reference ourselves.
1031          */
1032         WARN_ON(!refcount_read(&pi_state->refcount));
1033 
1034         /*
1035          * Now that we have a pi_state, we can acquire wait_lock
1036          * and do the state validation.
1037          */
1038         raw_spin_lock_irq(&pi_state->pi_mutex.wait_lock);
1039 
1040         /*
1041          * Since {uval, pi_state} is serialized by wait_lock, and our current
1042          * uval was read without holding it, it can have changed. Verify it
1043          * still is what we expect it to be, otherwise retry the entire
1044          * operation.
1045          */
1046         if (get_futex_value_locked(&uval2, uaddr))
1047                 goto out_efault;
1048 
1049         if (uval != uval2)
1050                 goto out_eagain;
1051 
1052         /*
1053          * Handle the owner died case:
1054          */
1055         if (uval & FUTEX_OWNER_DIED) {
1056                 /*
1057                  * exit_pi_state_list sets owner to NULL and wakes the
1058                  * topmost waiter. The task which acquires the
1059                  * pi_state->rt_mutex will fixup owner.
1060                  */
1061                 if (!pi_state->owner) {
1062                         /*
1063                          * No pi state owner, but the user space TID
1064                          * is not 0. Inconsistent state. [5]
1065                          */
1066                         if (pid)
1067                                 goto out_einval;
1068                         /*
1069                          * Take a ref on the state and return success. [4]
1070                          */
1071                         goto out_attach;
1072                 }
1073 
1074                 /*
1075                  * If TID is 0, then either the dying owner has not
1076                  * yet executed exit_pi_state_list() or some waiter
1077                  * acquired the rtmutex in the pi state, but did not
1078                  * yet fixup the TID in user space.
1079                  *
1080                  * Take a ref on the state and return success. [6]
1081                  */
1082                 if (!pid)
1083                         goto out_attach;
1084         } else {
1085                 /*
1086                  * If the owner died bit is not set, then the pi_state
1087                  * must have an owner. [7]
1088                  */
1089                 if (!pi_state->owner)
1090                         goto out_einval;
1091         }
1092 
1093         /*
1094          * Bail out if user space manipulated the futex value. If pi
1095          * state exists then the owner TID must be the same as the
1096          * user space TID. [9/10]
1097          */
1098         if (pid != task_pid_vnr(pi_state->owner))
1099                 goto out_einval;
1100 
1101 out_attach:
1102         get_pi_state(pi_state);
1103         raw_spin_unlock_irq(&pi_state->pi_mutex.wait_lock);
1104         *ps = pi_state;
1105         return 0;
1106 
1107 out_einval:
1108         ret = -EINVAL;
1109         goto out_error;
1110 
1111 out_eagain:
1112         ret = -EAGAIN;
1113         goto out_error;
1114 
1115 out_efault:
1116         ret = -EFAULT;
1117         goto out_error;
1118 
1119 out_error:
1120         raw_spin_unlock_irq(&pi_state->pi_mutex.wait_lock);
1121         return ret;
1122 }

1155 static int handle_exit_race(u32 __user *uaddr, u32 uval,
     /*   */
1156                             struct task_struct *tsk)
1157 {
1158         u32 uval2;
1159 
1160         /*
1161          * If the futex exit state is not yet FUTEX_STATE_DEAD, tell the
1162          * caller that the alleged owner is busy.
1163          */
1164         if (tsk && tsk->futex_state != FUTEX_STATE_DEAD)
1165                 return -EBUSY;
1166 
1167         /*
1168          * Reread the user space value to handle the following situation:
1169          *
1170          * CPU0                         CPU1
1171          *
1172          * sys_exit()                   sys_futex()
1173          *  do_exit()                    futex_lock_pi()
1174          *                                futex_lock_pi_atomic()
1175          *   exit_signals(tsk)              No waiters:
1176          *    tsk->flags |= PF_EXITING;     *uaddr == 0x00000PID
1177          *  mm_release(tsk)                 Set waiter bit
1178          *   exit_robust_list(tsk) {        *uaddr = 0x80000PID;
1179          *      Set owner died              attach_to_pi_owner() {
1180          *    *uaddr = 0xC0000000;           tsk = get_task(PID);
1181          *   }                               if (!tsk->flags & PF_EXITING) {
1182          *  ...                                attach();
1183          *  tsk->futex_state =               } else {
1184          *      FUTEX_STATE_DEAD;              if (tsk->futex_state !=
1185          *                                        FUTEX_STATE_DEAD)
1186          *                                       return -EAGAIN;
1187          *                                     return -ESRCH; <--- FAIL
1188          *                                   }
1189          *
1190          * Returning ESRCH unconditionally is wrong here because the
1191          * user space value has been changed by the exiting task.
1192          *
1193          * The same logic applies to the case where the exiting task is
1194          * already gone.
1195          */
1196         if (get_futex_value_locked(&uval2, uaddr))
1197                 return -EFAULT;
1198 
1199         /* If the user space value has changed, try again. */
1200         if (uval2 != uval)
1201                 return -EAGAIN;
1202 
1203         /*
1204          * The exiting task did not have a robust list, the robust list was
1205          * corrupted or the user space value in *uaddr is simply bogus.
1206          * Give up and tell user space.
1207          */
1208         return -ESRCH;
1209 }

1211 /*
1212  * Lookup the task for the TID provided from user space and attach to
1213  * it after doing proper sanity checks.
1214  */
1215 static int attach_to_pi_owner(u32 __user *uaddr, u32 uval, union futex_key *key,
     /*   */
1216                               struct futex_pi_state **ps,
1217                               struct task_struct **exiting)
1218 {
1219         pid_t pid = uval & FUTEX_TID_MASK;
1220         struct futex_pi_state *pi_state;
1221         struct task_struct *p;
1222 
1223         /*
1224          * We are the first waiter - try to look up the real owner and attach
1225          * the new pi_state to it, but bail out when TID = 0 [1]
1226          *
1227          * The !pid check is paranoid. None of the call sites should end up
1228          * with pid == 0, but better safe than sorry. Let the caller retry
1229          */
1230         if (!pid)
1231                 return -EAGAIN;
1232         p = find_get_task_by_vpid(pid);
1233         if (!p)
1234                 return handle_exit_race(uaddr, uval, NULL);
1235 
1236         if (unlikely(p->flags & PF_KTHREAD)) {
1237                 put_task_struct(p);
1238                 return -EPERM;
1239         }
1240 
1241         /*
1242          * We need to look at the task state to figure out, whether the
1243          * task is exiting. To protect against the change of the task state
1244          * in futex_exit_release(), we do this protected by p->pi_lock:
1245          */
1246         raw_spin_lock_irq(&p->pi_lock);
1247         if (unlikely(p->futex_state != FUTEX_STATE_OK)) {
1248                 /*
1249                  * The task is on the way out. When the futex state is
1250                  * FUTEX_STATE_DEAD, we know that the task has finished
1251                  * the cleanup:
1252                  */
1253                 int ret = handle_exit_race(uaddr, uval, p);
1254 
1255                 raw_spin_unlock_irq(&p->pi_lock);
1256                 /*
1257                  * If the owner task is between FUTEX_STATE_EXITING and
1258                  * FUTEX_STATE_DEAD then store the task pointer and keep
1259                  * the reference on the task struct. The calling code will
1260                  * drop all locks, wait for the task to reach
1261                  * FUTEX_STATE_DEAD and then drop the refcount. This is
1262                  * required to prevent a live lock when the current task
1263                  * preempted the exiting task between the two states.
1264                  */
1265                 if (ret == -EBUSY)
1266                         *exiting = p;
1267                 else
1268                         put_task_struct(p);
1269                 return ret;
1270         }
1271 
1272         /*
1273          * No existing pi state. First waiter. [2]
1274          *
1275          * This creates pi_state, we have hb->lock held, this means nothing can
1276          * observe this state, wait_lock is irrelevant.
1277          */
1278         pi_state = alloc_pi_state();
1279 
1280         /*
1281          * Initialize the pi_mutex in locked state and make @p
1282          * the owner of it:
1283          */
1284         rt_mutex_init_proxy_locked(&pi_state->pi_mutex, p);
1285 
1286         /* Store the key for possible exit cleanups: */
1287         pi_state->key = *key;
1288 
1289         WARN_ON(!list_empty(&pi_state->list));
1290         list_add(&pi_state->list, &p->pi_state_list);
1291         /*
1292          * Assignment without holding pi_state->pi_mutex.wait_lock is safe
1293          * because there is no concurrency as the object is not published yet.
1294          */
1295         pi_state->owner = p;
1296         raw_spin_unlock_irq(&p->pi_lock);
1297 
1298         put_task_struct(p);
1299 
1300         *ps = pi_state;
1301 
1302         return 0;
1303 }

1305 static int lookup_pi_state(u32 __user *uaddr, u32 uval,
     /*   */
1306                            struct futex_hash_bucket *hb,
1307                            union futex_key *key, struct futex_pi_state **ps,
1308                            struct task_struct **exiting)
1309 {
1310         struct futex_q *top_waiter = futex_top_waiter(hb, key);
1311 
1312         /*
1313          * If there is a waiter on that futex, validate it and
1314          * attach to the pi_state when the validation succeeds.
1315          */
1316         if (top_waiter)
1317                 return attach_to_pi_state(uaddr, uval, top_waiter->pi_state, ps);

1320          * We are the first waiter - try to look up the owner based on
1321          * @uval and attach to it.
1322          */
1323         return attach_to_pi_owner(uaddr, uval, key, ps, exiting);

1326 static int lock_pi_update_atomic(u32 __user *uaddr, u32 uval, u32 newval)
     /*   */
1327 {
1328         int err;
1329         u32 curval;
1330 
1331         if (unlikely(should_fail_futex(true)))
1332                 return -EFAULT;
1333 
1334         err = cmpxchg_futex_value_locked(&curval, uaddr, uval, newval);
1335         if (unlikely(err))
1336                 return err;
1337 
1338         /* If user space value changed, let the caller retry */
1339         return curval != uval ? -EAGAIN : 0;

1342 /**
1343  * futex_lock_pi_atomic() - Atomic work required to acquire a pi aware futex
1344  * @uaddr:              the pi futex user address
1345  * @hb:                 the pi futex hash bucket
1346  * @key:                the futex key associated with uaddr and hb
1347  * @ps:                 the pi_state pointer where we store the result of the
1348  *                      lookup
1349  * @task:               the task to perform the atomic lock work for.  This will
1350  *                      be "current" except in the case of requeue pi.
1351  * @exiting:            Pointer to store the task pointer of the owner task
1352  *                      which is in the middle of exiting
1353  * @set_waiters:        force setting the FUTEX_WAITERS bit (1) or not (0)
1354  *
1355  * Return:
1356  *  -  0 - ready to wait;
1357  *  -  1 - acquired the lock;
1358  *  - <0 - error
1359  *
1360  * The hb->lock and futex_key refs shall be held by the caller.
1361  *
1362  * @exiting is only set when the return value is -EBUSY. If so, this holds
1363  * a refcount on the exiting task on return and the caller needs to drop it
1364  * after waiting for the exit to complete.
1365  */
1366 static int futex_lock_pi_atomic(u32 __user *uaddr, struct futex_hash_bucket *hb,
     /*   */
1367                                 union futex_key *key,
1368                                 struct futex_pi_state **ps,
1369                                 struct task_struct *task,
1370                                 struct task_struct **exiting,
1371                                 int set_waiters)
1372 {
1373         u32 uval, newval, vpid = task_pid_vnr(task);
1374         struct futex_q *top_waiter;
1375         int ret;
1376 
1377         /*
1378          * Read the user space value first so we can validate a few
1379          * things before proceeding further.
1380          */
1381         if (get_futex_value_locked(&uval, uaddr))
1382                 return -EFAULT;
1383 
1384         if (unlikely(should_fail_futex(true)))
1385                 return -EFAULT;
1386 
1387         /*
1388          * Detect deadlocks.
1389          */
1390         if ((unlikely((uval & FUTEX_TID_MASK) == vpid)))
1391                 return -EDEADLK;
1392 
1393         if ((unlikely(should_fail_futex(true))))
1394                 return -EDEADLK;
1395 
1396         /*
1397          * Lookup existing state first. If it exists, try to attach to
1398          * its pi_state.
1399          */
1400         top_waiter = futex_top_waiter(hb, key);
1401         if (top_waiter)
1402                 return attach_to_pi_state(uaddr, uval, top_waiter->pi_state, ps);
1403 
1404         /*
1405          * No waiter and user TID is 0. We are here because the
1406          * waiters or the owner died bit is set or called from
1407          * requeue_cmp_pi or for whatever reason something took the
1408          * syscall.
1409          */
1410         if (!(uval & FUTEX_TID_MASK)) {
1411                 /*
1412                  * We take over the futex. No other waiters and the user space
1413                  * TID is 0. We preserve the owner died bit.
1414                  */
1415                 newval = uval & FUTEX_OWNER_DIED;
1416                 newval |= vpid;
1417 
1418                 /* The futex requeue_pi code can enforce the waiters bit */
1419                 if (set_waiters)
1420                         newval |= FUTEX_WAITERS;
1421 
1422                 ret = lock_pi_update_atomic(uaddr, uval, newval);
1423                 /* If the take over worked, return 1 */
1424                 return ret < 0 ? ret : 1;
1425         }
1426 
1427         /*
1428          * First waiter. Set the waiters bit before attaching ourself to
1429          * the owner. If owner tries to unlock, it will be forced into
1430          * the kernel and blocked on hb->lock.
1431          */
1432         newval = uval | FUTEX_WAITERS;
1433         ret = lock_pi_update_atomic(uaddr, uval, newval);
1434         if (ret)
1435                 return ret;
1436         /*
1437          * If the update of the user space value succeeded, we try to
1438          * attach to the owner. If that fails, no harm done, we only
1439          * set the FUTEX_WAITERS bit in the user space variable.
1440          */
1441         return attach_to_pi_owner(uaddr, newval, key, ps, exiting);
1442 }

1444 /**
1445  * __unqueue_futex() - Remove the futex_q from its futex_hash_bucket
1446  * @q:  The futex_q to unqueue
1447  *
1448  * The q->lock_ptr must not be NULL and must be held by the caller.
1449  */
1450 static void __unqueue_futex(struct futex_q *q)
     /*   */
1451 {
1452         struct futex_hash_bucket *hb;
1453 
1454         if (WARN_ON_SMP(!q->lock_ptr) || WARN_ON(plist_node_empty(&q->list)))
1455                 return;
1456         lockdep_assert_held(q->lock_ptr);
1457 
1458         hb = container_of(q->lock_ptr, struct futex_hash_bucket, lock);
1459         plist_del(&q->list, &hb->chain);
1460         hb_waiters_dec(hb);
1461 }
1462 
1463 /*
1464  * The hash bucket lock must be held when this is called.
1465  * Afterwards, the futex_q must not be accessed. Callers
1466  * must ensure to later call wake_up_q() for the actual
1467  * wakeups to occur.
1468  */
1469 static void mark_wake_futex(struct wake_q_head *wake_q, struct futex_q *q)
     /*   */
1470 {
1471         struct task_struct *p = q->task;
1472 
1473         if (WARN(q->pi_state || q->rt_waiter, "refusing to wake PI futex\n"))
1474                 return;
1475 
1476         get_task_struct(p);
1477         __unqueue_futex(q);
1478         /*
1479          * The waiting task can free the futex_q as soon as q->lock_ptr = NULL
1480          * is written, without taking any locks. This is possible in the event
1481          * of a spurious wakeup, for example. A memory barrier is required here
1482          * to prevent the following store to lock_ptr from getting ahead of the
1483          * plist_del in __unqueue_futex().
1484          */
1485         smp_store_release(&q->lock_ptr, NULL);
1486 
1487         /*
1488          * Queue the task for later wakeup for after we've released
1489          * the hb->lock.
1490          */
1491         wake_q_add_safe(wake_q, p);
1492 }

1494 /*
1495  * Caller must hold a reference on @pi_state.
1496  */
1497 static int wake_futex_pi(u32 __user *uaddr, u32 uval, struct futex_pi_state *pi_state)
     /*   */
1498 {
1499         u32 curval, newval;
1500         struct task_struct *new_owner;
1501         bool postunlock = false;
1502         DEFINE_WAKE_Q(wake_q);
1503         int ret = 0;
1504 
1505         new_owner = rt_mutex_next_owner(&pi_state->pi_mutex);
1506         if (WARN_ON_ONCE(!new_owner)) {
1507                 /*
1508                  * As per the comment in futex_unlock_pi() this should not happen.
1509                  *
1510                  * When this happens, give up our locks and try again, giving
1511                  * the futex_lock_pi() instance time to complete, either by
1512                  * waiting on the rtmutex or removing itself from the futex
1513                  * queue.
1514                  */
1515                 ret = -EAGAIN;
1516                 goto out_unlock;
1517         }
1518 
1519         /*
1520          * We pass it to the next owner. The WAITERS bit is always kept
1521          * enabled while there is PI state around. We cleanup the owner
1522          * died bit, because we are the owner.
1523          */
1524         newval = FUTEX_WAITERS | task_pid_vnr(new_owner);
1525 
1526         if (unlikely(should_fail_futex(true))) {
1527                 ret = -EFAULT;
1528                 goto out_unlock;
1529         }
1530 
1531         ret = cmpxchg_futex_value_locked(&curval, uaddr, uval, newval);
1532         if (!ret && (curval != uval)) {
1533                 /*
1534                  * If a unconditional UNLOCK_PI operation (user space did not
1535                  * try the TID->0 transition) raced with a waiter setting the
1536                  * FUTEX_WAITERS flag between get_user() and locking the hash
1537                  * bucket lock, retry the operation.
1538                  */
1539                 if ((FUTEX_TID_MASK & curval) == uval)
1540                         ret = -EAGAIN;
1541                 else
1542                         ret = -EINVAL;
1543         }
1544 
1545         if (!ret) {
1546                 /*
1547                  * This is a point of no return; once we modified the uval
1548                  * there is no going back and subsequent operations must
1549                  * not fail.
1550                  */
1551                 pi_state_update_owner(pi_state, new_owner);
1552                 postunlock = __rt_mutex_futex_unlock(&pi_state->pi_mutex, &wake_q);
1553         }
1554 
1555 out_unlock:
1556         raw_spin_unlock_irq(&pi_state->pi_mutex.wait_lock);
1557 
1558         if (postunlock)
1559                 rt_mutex_postunlock(&wake_q);
1560 
1561         return ret;
1562 }

1564 /*
1565  * Express the locking dependencies for lockdep:
1566  */
1567 static inline void
1568 double_lock_hb(struct futex_hash_bucket *hb1, struct futex_hash_bucket *hb2)
     /*   */
1569 {
1570         if (hb1 <= hb2) {
1571                 spin_lock(&hb1->lock);
1572                 if (hb1 < hb2)
1573                         spin_lock_nested(&hb2->lock, SINGLE_DEPTH_NESTING);
1574         } else { /* hb1 > hb2 */
1575                 spin_lock(&hb2->lock);
1576                 spin_lock_nested(&hb1->lock, SINGLE_DEPTH_NESTING);
1577         }
1578 }

1580 static inline void
1581 double_unlock_hb(struct futex_hash_bucket *hb1, struct futex_hash_bucket *hb2)
     /*   */
1582 {
1583         spin_unlock(&hb1->lock);
1584         if (hb1 != hb2)
1585                 spin_unlock(&hb2->lock);
1586 }

1588 /*
1589  * Wake up waiters matching bitset queued on this futex (uaddr).
1590  */
1591 static int
1592 futex_wake(u32 __user *uaddr, unsigned int flags, int nr_wake, u32 bitset)
     /*   */
1593 {
1594         struct futex_hash_bucket *hb;
1595         struct futex_q *this, *next;
1596         union futex_key key = FUTEX_KEY_INIT;
1597         int ret;
1598         DEFINE_WAKE_Q(wake_q);
1599 
1600         if (!bitset)
1601                 return -EINVAL;
1602 
1603         ret = get_futex_key(uaddr, flags & FLAGS_SHARED, &key, FUTEX_READ);
1604         if (unlikely(ret != 0))
1605                 return ret;
1606 
1607         hb = hash_futex(&key);
1608 
1609         /* Make sure we really have tasks to wakeup */
1610         if (!hb_waiters_pending(hb))
1611                 return ret;
1612 
1613         spin_lock(&hb->lock);
1614 
1615         plist_for_each_entry_safe(this, next, &hb->chain, list) {
1616                 if (match_futex (&this->key, &key)) {
1617                         if (this->pi_state || this->rt_waiter) {
1618                                 ret = -EINVAL;
1619                                 break;
1620                         }
1621 
1622                         /* Check if one of the bits is set in both bitsets */
1623                         if (!(this->bitset & bitset))
1624                                 continue;
1625 
1626                         mark_wake_futex(&wake_q, this);
1627                         if (++ret >= nr_wake)
1628                                 break;
1629                 }
1630         }
1631 
1632         spin_unlock(&hb->lock);
1633         wake_up_q(&wake_q);
1634         return ret;
1635 }

1637 static int futex_atomic_op_inuser(unsigned int encoded_op, u32 __user *uaddr)
     /*   */
1638 {
1639         unsigned int op =         (encoded_op & 0x70000000) >> 28;
1640         unsigned int cmp =        (encoded_op & 0x0f000000) >> 24;
1641         int oparg = sign_extend32((encoded_op & 0x00fff000) >> 12, 11);
1642         int cmparg = sign_extend32(encoded_op & 0x00000fff, 11);
1643         int oldval, ret;
1644 
1645         if (encoded_op & (FUTEX_OP_OPARG_SHIFT << 28)) {
1646                 if (oparg < 0 || oparg > 31) {
1647                         char comm[sizeof(current->comm)];
1648                         /*
1649                          * kill this print and return -EINVAL when userspace
1650                          * is sane again
1651                          */
1652                         pr_info_ratelimited("futex_wake_op: %s tries to shift op by %d; fix this program\n",
1653                                         get_task_comm(comm, current), oparg);
1654                         oparg &= 31;
1655                 }
1656                 oparg = 1 << oparg;
1657         }
1658 
1659         pagefault_disable();
1660         ret = arch_futex_atomic_op_inuser(op, oparg, &oldval, uaddr);
1661         pagefault_enable();
1662         if (ret)
1663                 return ret;
1664 
1665         switch (cmp) {
1666         case FUTEX_OP_CMP_EQ:
1667                 return oldval == cmparg;
1668         case FUTEX_OP_CMP_NE:
1669                 return oldval != cmparg;
1670         case FUTEX_OP_CMP_LT:
1671                 return oldval < cmparg;
1672         case FUTEX_OP_CMP_GE:
1673                 return oldval >= cmparg;
1674         case FUTEX_OP_CMP_LE:
1675                 return oldval <= cmparg;
1676         case FUTEX_OP_CMP_GT:
1677                 return oldval > cmparg;
1678         default:
1679                 return -ENOSYS;
1680         }
1681 }

1683 /*
1684  * Wake up all waiters hashed on the physical page that is mapped
1685  * to this virtual address:
1686  */
1687 static int
1688 futex_wake_op(u32 __user *uaddr1, unsigned int flags, u32 __user *uaddr2,
     /*   */
1689               int nr_wake, int nr_wake2, int op)
1690 {
1691         union futex_key key1 = FUTEX_KEY_INIT, key2 = FUTEX_KEY_INIT;
1692         struct futex_hash_bucket *hb1, *hb2;
1693         struct futex_q *this, *next;
1694         int ret, op_ret;
1695         DEFINE_WAKE_Q(wake_q);
1696 
1697 retry:
1698         ret = get_futex_key(uaddr1, flags & FLAGS_SHARED, &key1, FUTEX_READ);
1699         if (unlikely(ret != 0))
1700                 return ret;
1701         ret = get_futex_key(uaddr2, flags & FLAGS_SHARED, &key2, FUTEX_WRITE);
1702         if (unlikely(ret != 0))
1703                 return ret;
1704 
1705         hb1 = hash_futex(&key1);
1706         hb2 = hash_futex(&key2);
1707 
1708 retry_private:
1709         double_lock_hb(hb1, hb2);
1710         op_ret = futex_atomic_op_inuser(op, uaddr2);
1711         if (unlikely(op_ret < 0)) {
1712                 double_unlock_hb(hb1, hb2);
1713 
1714                 if (!IS_ENABLED(CONFIG_MMU) ||
1715                     unlikely(op_ret != -EFAULT && op_ret != -EAGAIN)) {
1716                         /*
1717                          * we don't get EFAULT from MMU faults if we don't have
1718                          * an MMU, but we might get them from range checking
1719                          */
1720                         ret = op_ret;
1721                         return ret;
1722                 }
1723 
1724                 if (op_ret == -EFAULT) {
1725                         ret = fault_in_user_writeable(uaddr2);
1726                         if (ret)
1727                                 return ret;
1728                 }
1729 
1730                 if (!(flags & FLAGS_SHARED)) {
1731                         cond_resched();
1732                         goto retry_private;
1733                 }
1734 
1735                 cond_resched();
1736                 goto retry;
1737         }
1738 
1739         plist_for_each_entry_safe(this, next, &hb1->chain, list) {
1740                 if (match_futex (&this->key, &key1)) {
1741                         if (this->pi_state || this->rt_waiter) {
1742                                 ret = -EINVAL;
1743                                 goto out_unlock;
1744                         }
1745                         mark_wake_futex(&wake_q, this);
1746                         if (++ret >= nr_wake)
1747                                 break;
1748                 }
1749         }
1750 
1751         if (op_ret > 0) {
1752                 op_ret = 0;
1753                 plist_for_each_entry_safe(this, next, &hb2->chain, list) {
1754                         if (match_futex (&this->key, &key2)) {
1755                                 if (this->pi_state || this->rt_waiter) {
1756                                         ret = -EINVAL;
1757                                         goto out_unlock;
1758                                 }
1759                                 mark_wake_futex(&wake_q, this);
1760                                 if (++op_ret >= nr_wake2)
1761                                         break;
1762                         }
1763                 }
1764                 ret += op_ret;
1765         }
1766 
1767 out_unlock:
1768         double_unlock_hb(hb1, hb2);
1769         wake_up_q(&wake_q);
1770         return ret;
1771 }

1773 /**
1774  * requeue_futex() - Requeue a futex_q from one hb to another
1775  * @q:          the futex_q to requeue
1776  * @hb1:        the source hash_bucket
1777  * @hb2:        the target hash_bucket
1778  * @key2:       the new key for the requeued futex_q
1779  */
1780 static inline
1781 void requeue_futex(struct futex_q *q, struct futex_hash_bucket *hb1,
     /*   */
1782                    struct futex_hash_bucket *hb2, union futex_key *key2)
1783 {
1784 
1785         /*
1786          * If key1 and key2 hash to the same bucket, no need to
1787          * requeue.
1788          */
1789         if (likely(&hb1->chain != &hb2->chain)) {
1790                 plist_del(&q->list, &hb1->chain);
1791                 hb_waiters_dec(hb1);
1792                 hb_waiters_inc(hb2);
1793                 plist_add(&q->list, &hb2->chain);
1794                 q->lock_ptr = &hb2->lock;
1795         }
1796         q->key = *key2;
1797 }

1799 /**
1800  * requeue_pi_wake_futex() - Wake a task that acquired the lock during requeue
1801  * @q:          the futex_q
1802  * @key:        the key of the requeue target futex
1803  * @hb:         the hash_bucket of the requeue target futex
1804  *
1805  * During futex_requeue, with requeue_pi=1, it is possible to acquire the
1806  * target futex if it is uncontended or via a lock steal.  Set the futex_q key
1807  * to the requeue target futex so the waiter can detect the wakeup on the right
1808  * futex, but remove it from the hb and NULL the rt_waiter so it can detect
1809  * atomic lock acquisition.  Set the q->lock_ptr to the requeue target hb->lock
1810  * to protect access to the pi_state to fixup the owner later.  Must be called
1811  * with both q->lock_ptr and hb->lock held.
1812  */
1813 static inline
1814 void requeue_pi_wake_futex(struct futex_q *q, union futex_key *key,
     /*   */
1815                            struct futex_hash_bucket *hb)
1816 {
1817         q->key = *key;
1818

1829 /**
1830  * futex_proxy_trylock_atomic() - Attempt an atomic lock for the top waiter
1831  * @pifutex:            the user address of the to futex
1832  * @hb1:                the from futex hash bucket, must be locked by the caller
1833  * @hb2:                the to futex hash bucket, must be locked by the caller
1834  * @key1:               the from futex key
1835  * @key2:               the to futex key
1836  * @ps:                 address to store the pi_state pointer
1837  * @exiting:            Pointer to store the task pointer of the owner task
1838  *                      which is in the middle of exiting
1839  * @set_waiters:        force setting the FUTEX_WAITERS bit (1) or not (0)
1840  *
1841  * Try and get the lock on behalf of the top waiter if we can do it atomically.
1842  * Wake the top waiter if we succeed.  If the caller specified set_waiters,
1843  * then direct futex_lock_pi_atomic() to force setting the FUTEX_WAITERS bit.
1844  * hb1 and hb2 must be held by the caller.
1845  *
1846  * @exiting is only set when the return value is -EBUSY. If so, this holds
1847  * a refcount on the exiting task on return and the caller needs to drop it
1848  * after waiting for the exit to complete.
1849  *
1850  * Return:
1851  *  -  0 - failed to acquire the lock atomically;
1852  *  - >0 - acquired the lock, return value is vpid of the top_waiter
1853  *  - <0 - error
1854  */
1855 static int
1856 futex_proxy_trylock_atomic(u32 __user *pifutex, struct futex_hash_bucket *hb1,
     /*   */
1857                            struct futex_hash_bucket *hb2, union futex_key *key1,
1858                            union futex_key *key2, struct futex_pi_state **ps,
1859                            struct task_struct **exiting, int set_waiters)
1860 {
1861         struct futex_q *top_waiter = NULL;
1862         u32 curval;
1863         int ret, vpid;
1864 
1865         if (get_futex_value_locked(&curval, pifutex))
1866                 return -EFAULT;
1867 
1868         if (unlikely(should_fail_futex(true)))
1869                 return -EFAULT;
1870 
1871         /*
1872          * Find the top_waiter and determine if there are additional waiters.
1873          * If the caller intends to requeue more than 1 waiter to pifutex,
1874          * force futex_lock_pi_atomic() to set the FUTEX_WAITERS bit now,
1875          * as we have means to handle the possible fault.  If not, don't set
1876          * the bit unecessarily as it will force the subsequent unlock to enter
1877          * the kernel.
1878          */
1879         top_waiter = futex_top_waiter(hb1, key1);
1880 
1881         /* There are no waiters, nothing for us to do. */
1882         if (!top_waiter)
1883                 return 0;
1884 
1885         /* Ensure we requeue to the expected futex. */
1886         if (!match_futex(top_waiter->requeue_pi_key, key2))
1887                 return -EINVAL;
1888 
1889         /*
1890          * Try to take the lock for top_waiter.  Set the FUTEX_WAITERS bit in
1891          * the contended case or if set_waiters is 1.  The pi_state is returned
1892          * in ps in contended cases.
1893          */
1894         vpid = task_pid_vnr(top_waiter->task);
1895         ret = futex_lock_pi_atomic(pifutex, hb2, key2, ps, top_waiter->task,
1896                                    exiting, set_waiters);
1897         if (ret == 1) {
1898                 requeue_pi_wake_futex(top_waiter, key2, hb2);
1899                 return vpid;
1900         }
1901         return ret;
1902 }

1904 /**

1907  * @flags:      futex flags (FLAGS_SHARED, etc.)

1909  * @nr_wake:    number of waiters to wake (must be 1 for requeue_pi)
1910  * @nr_requeue: number of waiters to requeue (0-INT_MAX)
1911  * @cmpval:     @uaddr1 expected value (or %NULL)

1913  *              pi futex (pi to pi requeue is not supported)

1915  * Requeue waiters on uaddr1 to uaddr2. In the requeue_pi case, try to acquire
1916  * uaddr2 atomically on behalf of the top waiter.

1918  * Return:
1919  *  - >=0 - on success, the number of tasks requeued or woken;
1920  *  -  <0 - on error

     /*   */
1923                          u32 __user *uaddr2, int nr_wake, int nr_requeue,
1924                          u32 *cmpval, int requeue_pi)

1939          * this point which will optimize away all the conditional code
1940          * further down.

1948                  * check is only valid for private futexes. See below.
1949                  */

1955                  * without any locks in case it fails.
1956                  */

1962                  * returning to userspace, so as to not leave the rt_mutex with
1963                  * waiters and no owner.  However, second and third wake-ups
1964                  * cannot be predicted as they involve race conditions with the

1975         if (unlikely(ret != 0))

1978                             requeue_pi ? FUTEX_WRITE : FUTEX_READ);

2007                                 return ret;

2010                                 goto retry_private;

2014                 if (curval != *cmpval) {
2015                         ret = -EAGAIN;
2016                         goto out_unlock;
2017                 }
2018         }

2023                 /*
2024                  * Attempt to acquire uaddr2 and wake the top waiter. If we
2025                  * intend to requeue waiters, force setting the FUTEX_WAITERS
2026                  * bit.  We force this here where we are able to easily handle
2027                  * faults rather in the requeue loop below.
2028                  */

2030                                                  &key2, &pi_state,

2033                 /*

2035                  * waiting on it.  If the former, then the pi_state will not
2036                  * exist yet, look it up one more time to ensure we have a
2037                  * reference to it. If the lock was taken, ret contains the
2038                  * vpid of the top waiter task.
2039                  * If the lock was not taken, we have pi_state and an initial
2040                  * refcount on it. In case of an error we have nothing.
2041                  */

2043                         WARN_ON(pi_state);
2044                         task_count++;
2045                         /*
2046                          * If we acquired the lock, then the user space value
2047                          * of uaddr2 should be vpid. It cannot be changed by
2048                          * the top waiter as it is blocked on hb2 lock if it
2049                          * tries to do so. If something fiddled with it behind
2050                          * our back the pi state lookup might unearth it. So
2051                          * we rather use the known value than rereading and
2052                          * handing potential crap to lookup_pi_state.
2053                          *
2054                          * If that call succeeds then we have pi_state and an
2055                          * initial refcount on it.

2069                         hb_waiters_dec(hb2);
2070                         ret = fault_in_user_writeable(uaddr2);
2071                         if (!ret)
2072                                 goto retry;
2073                         return ret;
2074                 case -EBUSY:
2075                 case -EAGAIN:
2076                         /*

2078                          * - EBUSY: Owner is exiting and we just wait for the
2079                          *   exit to complete.
2080                          * - EAGAIN: The user space value changed.
2081                          */

2083                         hb_waiters_dec(hb2);
2084                         /*
2085                          * Handle the case where the owner is in the middle of
2086                          * exiting. Wait for the exit to complete otherwise
2087                          * this task might loop forever, aka. live lock.

2092                 default:
2093                         goto out_unlock;
2094                 }

2099                         break;

2101                 if (!match_futex(&this->key, &key1))
2102                         continue;
2103 
2104                 /*
2105                  * FUTEX_WAIT_REQEUE_PI and FUTEX_CMP_REQUEUE_PI should always
2106                  * be paired with each other and no other futex ops.
2107                  *
2108                  * We should never be requeueing a futex_q with a pi_state,
2109                  * which is awaiting a futex_unlock_pi().

2112                     (!requeue_pi && this->rt_waiter) ||
2113                     this->pi_state) {
2114                         ret = -EINVAL;
2115                         break;
2116                 }
2117 
2118                 /*
2119                  * Wake nr_wake waiters.  For requeue_pi, if we acquired the
2120                  * lock, we already woke the top_waiter.  If not, it will be
2121                  * woken by futex_unlock_pi().
2122                  */
2123                 if (++task_count <= nr_wake && !requeue_pi) {
2124                         mark_wake_futex(&wake_q, this);
2125                         continue;
2126                 }
2127 
2128                 /* Ensure we requeue to the expected futex for requeue_pi. */
2129                 if (requeue_pi && !match_futex(this->requeue_pi_key, &key2)) {
2130                         ret = -EINVAL;
2131                         break;
2132                 }
2133 
2134                 /*
2135                  * Requeue nr_requeue waiters and possibly one more in the case
2136                  * of requeue_pi if we couldn't acquire the lock atomically.
2137                  */
2138                 if (requeue_pi) {
2139                         /*
2140                          * Prepare the waiter to take the rt_mutex. Take a
2141                          * refcount on the pi_state and store the pointer in
2142                          * the futex_q object of the waiter.
2143                          */
2144                         get_pi_state(pi_state);
2145                         this->pi_state = pi_state;
2146                         ret = rt_mutex_start_proxy_lock(&pi_state->pi_mutex,
2147                                                         this->rt_waiter,
2148                                                         this->task);
2149                         if (ret == 1) {
2150                                 /*
2151                                  * We got the lock. We do neither drop the
2152                                  * refcount on pi_state nor clear
2153                                  * this->pi_state because the waiter needs the
2154                                  * pi_state for cleaning up the user space
2155                                  * value. It will drop the refcount after
2156                                  * doing so.
2157                                  */
2158                                 requeue_pi_wake_futex(this, &key2, hb2);
2159                                 continue;
2160                         } else if (ret) {
2161                                 /*
2162                                  * rt_mutex_start_proxy_lock() detected a
2163                                  * potential deadlock when we tried to queue
2164                                  * that waiter. Drop the pi_state reference
2165                                  * which we took above and remove the pointer
2166                                  * to the state from the waiters futex_q
2167                                  * object.
2168                                  */
2169                                 this->pi_state = NULL;
2170                                 put_pi_state(pi_state);
2171                                 /*
2172                                  * We stop queueing more waiters and let user
2173                                  * space deal with the mess.
2174                                  */
2175                                 break;
2176                         }
2177                 }
2178                 requeue_futex(this, hb1, hb2, &key2);
2179         }
2180 
2181         /*
2182          * We took an extra initial reference to the pi_state either
2183          * in futex_proxy_trylock_atomic() or in lookup_pi_state(). We
2184          * need to drop it here again.
2185          */
2186         put_pi_state(pi_state);

2188 out_unlock:
2189         double_unlock_hb(hb1, hb2);
2190         wake_up_q(&wake_q);
2191         hb_waiters_dec(hb2);
2192         return ret ? ret : task_count;
2193 }

2273  * be paired with exactly one earlier call to queue_me().

2275  * Return:
2276  *  - 1 - if the futex_q was still queued (and we removed unqueued it);
2277  *  - 0 - if the futex_q was already removed by the waking thread

     /*   */

2339 static int __fixup_pi_state_owner(u32 __user *uaddr, struct futex_q *q,
     /*   */
2340                                   struct task_struct *argowner)
2341 {
2342         struct futex_pi_state *pi_state = q->pi_state;
2343         struct task_struct *oldowner, *newowner;
2344         u32 uval, curval, newval, newtid;
2345         int err = 0;
2346 
2347         oldowner = pi_state->owner;
2348 
2349         /*
2350          * We are here because either:
2351          *
2352          *  - we stole the lock and pi_state->owner needs updating to reflect
2353          *    that (@argowner == current),
2354          *
2355          * or:
2356          *
2357          *  - someone stole our lock and we need to fix things to point to the
2358          *    new owner (@argowner == NULL).
2359          *
2360          * Either way, we have to replace the TID in the user space variable.
2361          * This must be atomic as we have to preserve the owner died bit here.
2362          *
2363          * Note: We write the user space value _before_ changing the pi_state
2364          * because we can fault here. Imagine swapped out pages or a fork
2365          * that marked all the anonymous memory readonly for cow.
2366          *
2367          * Modifying pi_state _before_ the user space value would leave the
2368          * pi_state in an inconsistent state when we fault here, because we
2369          * need to drop the locks to handle the fault. This might be observed
2370          * in the PID check in lookup_pi_state.
2371          */
2372 retry:
2373         if (!argowner) {
2374                 if (oldowner != current) {
2375                         /*
2376                          * We raced against a concurrent self; things are
2377                          * already fixed up. Nothing to do.
2378                          */
2379                         return 0;
2380                 }
2381 
2382                 if (__rt_mutex_futex_trylock(&pi_state->pi_mutex)) {
2383                         /* We got the lock. pi_state is correct. Tell caller. */
2384                         return 1;
2385                 }
2386 
2387                 /*
2388                  * The trylock just failed, so either there is an owner or
2389                  * there is a higher priority waiter than this one.
2390                  */
2391                 newowner = rt_mutex_owner(&pi_state->pi_mutex);
2392                 /*
2393                  * If the higher priority waiter has not yet taken over the
2394                  * rtmutex then newowner is NULL. We can't return here with
2395                  * that state because it's inconsistent vs. the user space
2396                  * state. So drop the locks and try again. It's a valid
2397                  * situation and not any different from the other retry
2398                  * conditions.
2399                  */
2400                 if (unlikely(!newowner)) {
2401                         err = -EAGAIN;
2402                         goto handle_err;
2403                 }
2404         } else {
2405                 WARN_ON_ONCE(argowner != current);
2406                 if (oldowner == current) {
2407                         /*
2408                          * We raced against a concurrent self; things are
2409                          * already fixed up. Nothing to do.
2410                          */
2411                         return 1;
2412                 }
2413                 newowner = argowner;
2414         }
2415 
2416         newtid = task_pid_vnr(newowner) | FUTEX_WAITERS;
2417         /* Owner died? */
2418         if (!pi_state->owner)
2419                 newtid |= FUTEX_OWNER_DIED;
2420 
2421         err = get_futex_value_locked(&uval, uaddr);
2422         if (err)
2423                 goto handle_err;
2424 
2425         for (;;) {
2426                 newval = (uval & FUTEX_OWNER_DIED) | newtid;
2427 
2428                 err = cmpxchg_futex_value_locked(&curval, uaddr, uval, newval);
2429                 if (err)
2430                         goto handle_err;
2431 
2432                 if (curval == uval)
2433                         break;
2434                 uval = curval;
2435         }
2436 
2437         /*
2438          * We fixed up user space. Now we need to fix the pi_state
2439          * itself.
2440          */
2441         pi_state_update_owner(pi_state, newowner);
2442 
2443         return argowner == current;
2444 
2445         /*
2446          * In order to reschedule or handle a page fault, we need to drop the
2447          * locks here. In the case of a fault, this gives the other task
2448          * (either the highest priority waiter itself or the task which stole
2449          * the rtmutex) the chance to try the fixup of the pi_state. So once we
2450          * are back from handling the fault we need to check the pi_state after
2451          * reacquiring the locks and before trying to do another fixup. When
2452          * the fixup has been done already we simply return.
2453          *
2454          * Note: we hold both hb->lock and pi_mutex->wait_lock. We can safely
2455          * drop hb->lock since the caller owns the hb -> futex_q relation.
2456          * Dropping the pi_mutex->wait_lock requires the state revalidate.
2457          */
2458 handle_err:
2459         raw_spin_unlock_irq(&pi_state->pi_mutex.wait_lock);
2460         spin_unlock(q->lock_ptr);
2461 
2462         switch (err) {
2463         case -EFAULT:
2464                 err = fault_in_user_writeable(uaddr);
2465                 break;
2466 
2467         case -EAGAIN:
2468                 cond_resched();
2469                 err = 0;
2470                 break;
2471 
2472         default:
2473                 WARN_ON_ONCE(1);
2474                 break;
2475         }
2476 
2477         spin_lock(q->lock_ptr);
2478         raw_spin_lock_irq(&pi_state->pi_mutex.wait_lock);
2479 
2480         /*
2481          * Check if someone else fixed it for us:
2482          */
2483         if (pi_state->owner != oldowner)
2484                 return argowner == current;
2485 
2486         /* Retry if err was -EAGAIN or the fault in succeeded */
2487         if (!err)
2488                 goto retry;
2489 
2490         /*
2491          * fault_in_user_writeable() failed so user state is immutable. At
2492          * best we can make the kernel state consistent but user state will
2493          * be most likely hosed and any subsequent unlock operation will be
2494          * rejected due to PI futex rule [10].
2495          *
2496          * Ensure that the rtmutex owner is also the pi_state owner despite
2497          * the user space value claiming something different. There is no
2498          * point in unlocking the rtmutex if current is the owner as it
2499          * would need to wait until the next waiter has taken the rtmutex
2500          * to guarantee consistent state. Keep it simple. Userspace asked
2501          * for this wreckaged state.
2502          *
2503          * The rtmutex has an owner - either current or some other
2504          * task. See the EAGAIN loop above.
2505          */
2506         pi_state_update_owner(pi_state, rt_mutex_owner(&pi_state->pi_mutex));
2507 
2508         return err;
2509 }

2527 /**
2528  * fixup_owner() - Post lock pi_state and corner case management
2529  * @uaddr:      user address of the futex
2530  * @q:          futex_q (contains pi_state and access to the rt_mutex)
2531  * @locked:     if the attempt to take the rt_mutex succeeded (1) or not (0)
2532  *
2533  * After attempting to lock an rt_mutex, this function is called to cleanup
2534  * the pi_state owner as well as handle race conditions that may allow us to
2535  * acquire the lock. Must be called with the hb lock held.
2536  *
2537  * Return:
2538  *  -  1 - success, lock taken;
2539  *  -  0 - success, lock not taken;
2540  *  - <0 - on error (-EFAULT)
2541  */
2542 static int fixup_owner(u32 __user *uaddr, struct futex_q *q, int locked)
     /*   */
2543 {
2544         if (locked) {
2545                 /*
2546                  * Got the lock. We might not be the anticipated owner if we
2547                  * did a lock-steal - fix up the PI-state in that case:
2548                  *
2549                  * Speculative pi_state->owner read (we don't hold wait_lock);
2550                  * since we own the lock pi_state->owner == current is the
2551                  * stable state, anything else needs more attention.
2552                  */
2553                 if (q->pi_state->owner != current)
2554                         return fixup_pi_state_owner(uaddr, q, current);
2555                 return 1;
2556         }
2557 
2558         /*
2559          * If we didn't get the lock; check if anybody stole it from us. In
2560          * that case, we need to fix up the uval to point to them instead of
2561          * us, otherwise bad things happen. [10]
2562          *
2563          * Another speculative read; pi_state->owner == current is unstable
2564          * but needs our attention.
2565          */
2566         if (q->pi_state->owner == current)
2567                 return fixup_pi_state_owner(uaddr, q, NULL);
2568 
2569         /*
2570          * Paranoia check. If we did not take the lock, then we should not be
2571          * the owner of the rt_mutex. Warn and establish consistent state.
2572          */
2573         if (WARN_ON_ONCE(rt_mutex_owner(&q->pi_state->pi_mutex) == current))
2574                 return fixup_pi_state_owner(uaddr, q, current);
2575 
2576         return 0;
2577 }

2579 /**
2580  * futex_wait_queue_me() - queue_me() and wait for wakeup, timeout, or signal
2581  * @hb:         the futex hash bucket, must be locked by the caller
2582  * @q:          the futex_q to queue up on
2583  * @timeout:    the prepared hrtimer_sleeper, or null for no timeout
2584  */
2585 static void futex_wait_queue_me(struct futex_hash_bucket *hb, struct futex_q *q,
     /*   */
2586                                 struct hrtimer_sleeper *timeout)
2587 {
2588         /*
2589          * The task state is guaranteed to be set before another task can
2590          * wake it. set_current_state() is implemented using smp_store_mb() and
2591          * queue_me() calls spin_unlock() upon completion, both serializing
2592          * access to the hash list and forcing another memory barrier.
2593          */
2594         set_current_state(TASK_INTERRUPTIBLE);
2595         queue_me(q, hb);
2596 
2597         /* Arm the timer */
2598         if (timeout)
2599                 hrtimer_sleeper_start_expires(timeout, HRTIMER_MODE_ABS);
2600 
2601         /*
2602          * If we have been removed from the hash list, then another task
2603          * has tried to wake us, and we can skip the call to schedule().
2604          */
2605         if (likely(!plist_node_empty(&q->list))) {
2606                 /*
2607                  * If the timer has already expired, current will already be
2608                  * flagged for rescheduling. Only call schedule if there
2609                  * is no timeout, or if it has yet to expire.
2610                  */
2611                 if (!timeout || timeout->task)
2612                         freezable_schedule();
2613         }
2614         __set_current_state(TASK_RUNNING);
2615 }

2617 /**
2618  * futex_wait_setup() - Prepare to wait on a futex
2619  * @uaddr:      the futex userspace address
2620  * @val:        the expected value
2621  * @flags:      futex flags (FLAGS_SHARED, etc.)
2622  * @q:          the associated futex_q
2623  * @hb:         storage for hash_bucket pointer to be returned to caller
2624  *
2625  * Setup the futex_q and locate the hash_bucket.  Get the futex value and
2626  * compare it with the expected value.  Handle atomic faults internally.
2627  * Return with the hb lock held and a q.key reference on success, and unlocked
2628  * with no q.key reference on failure.
2629  *
2630  * Return:
2631  *  -  0 - uaddr contains val and hb has been locked;
2632  *  - <1 - -EFAULT or -EWOULDBLOCK (uaddr does not contain val) and hb is unlocked
2633  */
2634 static int futex_wait_setup(u32 __user *uaddr, u32 val, unsigned int flags,
     /*   */
2635                            struct futex_q *q, struct futex_hash_bucket **hb)
2636 {
2637         u32 uval;
2638         int ret;
2639 
2640         /*
2641          * Access the page AFTER the hash-bucket is locked.
2642          * Order is important:
2643          *
2644          *   Userspace waiter: val = var; if (cond(val)) futex_wait(&var, val);
2645          *   Userspace waker:  if (cond(var)) { var = new; futex_wake(&var); }
2646          *
2647          * The basic logical guarantee of a futex is that it blocks ONLY
2648          * if cond(var) is known to be true at the time of blocking, for
2649          * any cond.  If we locked the hash-bucket after testing *uaddr, that
2650          * would open a race condition where we could block indefinitely with
2651          * cond(var) false, which would violate the guarantee.
2652          *
2653          * On the other hand, we insert q and release the hash-bucket only
2654          * after testing *uaddr.  This guarantees that futex_wait() will NOT
2655          * absorb a wakeup if *uaddr does not match the desired values
2656          * while the syscall executes.
2657          */
2658 retry:
2659         ret = get_futex_key(uaddr, flags & FLAGS_SHARED, &q->key, FUTEX_READ);
2660         if (unlikely(ret != 0))
2661                 return ret;
2662 
2663 retry_private:
2664         *hb = queue_lock(q);
2665 
2666         ret = get_futex_value_locked(&uval, uaddr);
2667 
2668         if (ret) {
2669                 queue_unlock(*hb);
2670 
2671                 ret = get_user(uval, uaddr);
2672                 if (ret)
2673                         return ret;
2674 
2675                 if (!(flags & FLAGS_SHARED))
2676                         goto retry_private;
2677 
2678                 goto retry;
2679         }
2680 
2681         if (uval != val) {
2682                 queue_unlock(*hb);
2683                 ret = -EWOULDBLOCK;
2684         }
2685 
2686         return ret;
2687 }

2689 static int futex_wait(u32 __user *uaddr, unsigned int flags, u32 val,
     /*   */
2690                       ktime_t *abs_time, u32 bitset)
2691 {
2692         struct hrtimer_sleeper timeout, *to;
2693         struct restart_block *restart;
2694         struct futex_hash_bucket *hb;
2695         struct futex_q q = futex_q_init;
2696         int ret;
2697 
2698         if (!bitset)
2699                 return -EINVAL;
2700         q.bitset = bitset;
2701 
2702         to = futex_setup_timer(abs_time, &timeout, flags,
2703                                current->timer_slack_ns);
2704 retry:
2705         /*
2706          * Prepare to wait on uaddr. On success, holds hb lock and increments
2707          * q.key refs.
2708          */
2709         ret = futex_wait_setup(uaddr, val, flags, &q, &hb);
2710         if (ret)
2711                 goto out;
2712 
2713         /* queue_me and wait for wakeup, timeout, or a signal. */
2714         futex_wait_queue_me(hb, &q, to);
2715 
2716         /* If we were woken (and unqueued), we succeeded, whatever. */
2717         ret = 0;
2718         /* unqueue_me() drops q.key ref */
2719         if (!unqueue_me(&q))
2720                 goto out;
2721         ret = -ETIMEDOUT;
2722         if (to && !to->task)
2723                 goto out;
2724 
2725         /*
2726          * We expect signal_pending(current), but we might be the
2727          * victim of a spurious wakeup as well.
2728          */
2729         if (!signal_pending(current))
2730                 goto retry;
2731 
2732         ret = -ERESTARTSYS;
2733         if (!abs_time)
2734                 goto out;
2735 
2736         restart = &current->restart_block;
2737         restart->futex.uaddr = uaddr;
2738         restart->futex.val = val;
2739         restart->futex.time = *abs_time;
2740         restart->futex.bitset = bitset;
2741         restart->futex.flags = flags | FLAGS_HAS_TIMEOUT;
2742 
2743         ret = set_restart_fn(restart, futex_wait_restart);
2744 
2745 out:
2746         if (to) {
2747                 hrtimer_cancel(&to->timer);
2748                 destroy_hrtimer_on_stack(&to->timer);
2749         }
2750         return ret;
2751 }

2770 /*
2771  * Userspace tried a 0 -> TID atomic transition of the futex value
2772  * and failed. The kernel side here does the whole locking operation:
2773  * if there are waiters then it will block as a consequence of relying
2774  * on rt-mutexes, it does PI, etc. (Due to races the kernel might see
2775  * a 0 value of the futex too.).
2776  *
2777  * Also serves as futex trylock_pi()'ing, and due semantics.
2778  */
2779 static int futex_lock_pi(u32 __user *uaddr, unsigned int flags,
     /*   */
2780                          ktime_t *time, int trylock)
2781 {
2782         struct hrtimer_sleeper timeout, *to;
2783         struct task_struct *exiting = NULL;
2784         struct rt_mutex_waiter rt_waiter;
2785         struct futex_hash_bucket *hb;
2786         struct futex_q q = futex_q_init;
2787         int res, ret;
2788 
2789         if (!IS_ENABLED(CONFIG_FUTEX_PI))
2790                 return -ENOSYS;
2791 
2792         if (refill_pi_state_cache())
2793                 return -ENOMEM;
2794 
2795         to = futex_setup_timer(time, &timeout, FLAGS_CLOCKRT, 0);
2796 
2797 retry:
2798         ret = get_futex_key(uaddr, flags & FLAGS_SHARED, &q.key, FUTEX_WRITE);
2799         if (unlikely(ret != 0))
2800                 goto out;
2801 
2802 retry_private:
2803         hb = queue_lock(&q);
2804 
2805         ret = futex_lock_pi_atomic(uaddr, hb, &q.key, &q.pi_state, current,
2806                                    &exiting, 0);
2807         if (unlikely(ret)) {
2808                 /*
2809                  * Atomic work succeeded and we got the lock,
2810                  * or failed. Either way, we do _not_ block.
2811                  */
2812                 switch (ret) {
2813                 case 1:
2814                         /* We got the lock. */
2815                         ret = 0;
2816                         goto out_unlock_put_key;
2817                 case -EFAULT:
2818                         goto uaddr_faulted;
2819                 case -EBUSY:
2820                 case -EAGAIN:
2821                         /*
2822                          * Two reasons for this:
2823                          * - EBUSY: Task is exiting and we just wait for the
2824                          *   exit to complete.
2825                          * - EAGAIN: The user space value changed.
2826                          */
2827                         queue_unlock(hb);
2828                         /*
2829                          * Handle the case where the owner is in the middle of
2830                          * exiting. Wait for the exit to complete otherwise
2831                          * this task might loop forever, aka. live lock.
2832                          */
2833                         wait_for_owner_exiting(ret, exiting);
2834                         cond_resched();
2835                         goto retry;
2836                 default:
2837                         goto out_unlock_put_key;
2838                 }
2839         }
2840 
2841         WARN_ON(!q.pi_state);
2842 
2843         /*
2844          * Only actually queue now that the atomic ops are done:
2845          */
2846         __queue_me(&q, hb);
2847 
2848         if (trylock) {
2849                 ret = rt_mutex_futex_trylock(&q.pi_state->pi_mutex);
2850                 /* Fixup the trylock return value: */
2851                 ret = ret ? 0 : -EWOULDBLOCK;
2852                 goto no_block;
2853         }
2854 
2855         rt_mutex_init_waiter(&rt_waiter);
2856 
2857         /*
2858          * On PREEMPT_RT_FULL, when hb->lock becomes an rt_mutex, we must not
2859          * hold it while doing rt_mutex_start_proxy(), because then it will
2860          * include hb->lock in the blocking chain, even through we'll not in
2861          * fact hold it while blocking. This will lead it to report -EDEADLK
2862          * and BUG when futex_unlock_pi() interleaves with this.
2863          *
2864          * Therefore acquire wait_lock while holding hb->lock, but drop the
2865          * latter before calling __rt_mutex_start_proxy_lock(). This
2866          * interleaves with futex_unlock_pi() -- which does a similar lock
2867          * handoff -- such that the latter can observe the futex_q::pi_state
2868          * before __rt_mutex_start_proxy_lock() is done.
2869          */
2870         raw_spin_lock_irq(&q.pi_state->pi_mutex.wait_lock);
2871         spin_unlock(q.lock_ptr);
2872         /*
2873          * __rt_mutex_start_proxy_lock() unconditionally enqueues the @rt_waiter
2874          * such that futex_unlock_pi() is guaranteed to observe the waiter when
2875          * it sees the futex_q::pi_state.
2876          */
2877         ret = __rt_mutex_start_proxy_lock(&q.pi_state->pi_mutex, &rt_waiter, current);
2878         raw_spin_unlock_irq(&q.pi_state->pi_mutex.wait_lock);
2879 
2880         if (ret) {
2881                 if (ret == 1)
2882                         ret = 0;
2883                 goto cleanup;
2884         }
2885 
2886         if (unlikely(to))
2887                 hrtimer_sleeper_start_expires(to, HRTIMER_MODE_ABS);
2888 
2889         ret = rt_mutex_wait_proxy_lock(&q.pi_state->pi_mutex, to, &rt_waiter);
2890 
2891 cleanup:
2892         spin_lock(q.lock_ptr);
2893         /*
2894          * If we failed to acquire the lock (deadlock/signal/timeout), we must
2895          * first acquire the hb->lock before removing the lock from the
2896          * rt_mutex waitqueue, such that we can keep the hb and rt_mutex wait
2897          * lists consistent.
2898          *
2899          * In particular; it is important that futex_unlock_pi() can not
2900          * observe this inconsistency.
2901          */
2902         if (ret && !rt_mutex_cleanup_proxy_lock(&q.pi_state->pi_mutex, &rt_waiter))
2903                 ret = 0;
2904 
2905 no_block:
2906         /*
2907          * Fixup the pi_state owner and possibly acquire the lock if we
2908          * haven't already.
2909          */
2910         res = fixup_owner(uaddr, &q, !ret);
2911         /*
2912          * If fixup_owner() returned an error, proprogate that.  If it acquired
2913          * the lock, clear our -ETIMEDOUT or -EINTR.
2914          */
2915         if (res)
2916                 ret = (res < 0) ? res : 0;
2917 
2918         /* Unqueue and drop the lock */
2919         unqueue_me_pi(&q);
2920         goto out;
2921 
2922 out_unlock_put_key:
2923         queue_unlock(hb);
2924 
2925 out:
2926         if (to) {
2927                 hrtimer_cancel(&to->timer);
2928                 destroy_hrtimer_on_stack(&to->timer);
2929         }
2930         return ret != -EINTR ? ret : -ERESTARTNOINTR;
2931 
2932 uaddr_faulted:
2933         queue_unlock(hb);
2934 
2935         ret = fault_in_user_writeable(uaddr);
2936         if (ret)
2937                 goto out;
2938 
2939         if (!(flags & FLAGS_SHARED))
2940                 goto retry_private;
2941 
2942         goto retry;
2943 }

2945 /*
2946  * Userspace attempted a TID -> 0 atomic transition, and failed.
2947  * This is the in-kernel slowpath: we look up the PI state (if any),
2948  * and do the rt-mutex unlock.
2949  */
2950 static int futex_unlock_pi(u32 __user *uaddr, unsigned int flags)
     /*   */
2951 {
2952         u32 curval, uval, vpid = task_pid_vnr(current);
2953         union futex_key key = FUTEX_KEY_INIT;
2954         struct futex_hash_bucket *hb;
2955         struct futex_q *top_waiter;
2956         int ret;
2957 
2958         if (!IS_ENABLED(CONFIG_FUTEX_PI))
2959                 return -ENOSYS;
2960 
2961 retry:
2962         if (get_user(uval, uaddr))
2963                 return -EFAULT;
2964         /*
2965          * We release only a lock we actually own:
2966          */
2967         if ((uval & FUTEX_TID_MASK) != vpid)
2968                 return -EPERM;
2969 
2970         ret = get_futex_key(uaddr, flags & FLAGS_SHARED, &key, FUTEX_WRITE);
2971         if (ret)
2972                 return ret;
2973 
2974         hb = hash_futex(&key);
2975         spin_lock(&hb->lock);
2976 
2977         /*
2978          * Check waiters first. We do not trust user space values at
2979          * all and we at least want to know if user space fiddled
2980          * with the futex value instead of blindly unlocking.
2981          */
2982         top_waiter = futex_top_waiter(hb, &key);
2983         if (top_waiter) {
2984                 struct futex_pi_state *pi_state = top_waiter->pi_state;
2985 
2986                 ret = -EINVAL;
2987                 if (!pi_state)
2988                         goto out_unlock;
2989 
2990                 /*
2991                  * If current does not own the pi_state then the futex is
2992                  * inconsistent and user space fiddled with the futex value.
2993                  */
2994                 if (pi_state->owner != current)
2995                         goto out_unlock;
2996 
2997                 get_pi_state(pi_state);
2998                 /*
2999                  * By taking wait_lock while still holding hb->lock, we ensure
3000                  * there is no point where we hold neither; and therefore
3001                  * wake_futex_pi() must observe a state consistent with what we
3002                  * observed.
3003                  *
3004                  * In particular; this forces __rt_mutex_start_proxy() to
3005                  * complete such that we're guaranteed to observe the
3006                  * rt_waiter. Also see the WARN in wake_futex_pi().
3007                  */
3008                 raw_spin_lock_irq(&pi_state->pi_mutex.wait_lock);
3009                 spin_unlock(&hb->lock);
3010 
3011                 /* drops pi_state->pi_mutex.wait_lock */
3012                 ret = wake_futex_pi(uaddr, uval, pi_state);
3013 
3014                 put_pi_state(pi_state);
3015 
3016                 /*
3017                  * Success, we're done! No tricky corner cases.
3018                  */
3019                 if (!ret)
3020                         goto out_putkey;
3021                 /*
3022                  * The atomic access to the futex value generated a
3023                  * pagefault, so retry the user-access and the wakeup:
3024                  */
3025                 if (ret == -EFAULT)
3026                         goto pi_faulted;
3027                 /*
3028                  * A unconditional UNLOCK_PI op raced against a waiter
3029                  * setting the FUTEX_WAITERS bit. Try again.
3030                  */
3031                 if (ret == -EAGAIN)
3032                         goto pi_retry;
3033                 /*
3034                  * wake_futex_pi has detected invalid state. Tell user
3035                  * space.
3036                  */
3037                 goto out_putkey;
3038         }
3039 
3040         /*
3041          * We have no kernel internal state, i.e. no waiters in the
3042          * kernel. Waiters which are about to queue themselves are stuck
3043          * on hb->lock. So we can safely ignore them. We do neither

3405 /*
3406  * Process a futex-list entry, check whether it's owned by the
3407  * dying task, and do notification if so:
3408  */
3409 static int handle_futex_death(u32 __user *uaddr, struct task_struct *curr,
     /*   */
3410                               bool pi, bool pending_op)
3411 {
3412         u32 uval, nval, mval;
3413         int err;
3414 
3415         /* Futex address must be 32bit aligned */
3416         if ((((unsigned long)uaddr) % sizeof(*uaddr)) != 0)
3417                 return -1;
3418 
3419 retry:
3420         if (get_user(uval, uaddr))
3421                 return -1;
3422 
3423         /*
3424          * Special case for regular (non PI) futexes. The unlock path in
3425          * user space has two race scenarios:
3426          *
3427          * 1. The unlock path releases the user space futex value and
3428          *    before it can execute the futex() syscall to wake up
3429          *    waiters it is killed.
3430          *
3431          * 2. A woken up waiter is killed before it can acquire the
3432          *    futex in user space.
3433          *
3434          * In both cases the TID validation below prevents a wakeup of
3435          * potential waiters which can cause these waiters to block
3436          * forever.
3437          *
3438          * In both cases the following conditions are met:
3439          *
3440          *      1) task->robust_list->list_op_pending != NULL
3441          *         @pending_op == true
3442          *      2) User space futex value == 0
3443          *      3) Regular futex: @pi == false
3444          *
3445          * If these conditions are met, it is safe to attempt waking up a
3446          * potential waiter without touching the user space futex value and
3447          * trying to set the OWNER_DIED bit. The user space futex value is
3448          * uncontended and the rest of the user space mutex state is
3449          * consistent, so a woken waiter will just take over the
3450          * uncontended futex. Setting the OWNER_DIED bit would create
3451          * inconsistent state and malfunction of the user space owner died
3452          * handling.
3453          */
3454         if (pending_op && !pi && !uval) {
3455                 futex_wake(uaddr, 1, 1, FUTEX_BITSET_MATCH_ANY);
3456                 return 0;
3457         }
3458 
3459         if ((uval & FUTEX_TID_MASK) != task_pid_vnr(curr))
3460                 return 0;
3461 
3462         /*
3463          * Ok, this dying thread is truly holding a futex
3464          * of interest. Set the OWNER_DIED bit atomically
3465          * via cmpxchg, and if the value had FUTEX_WAITERS
3466          * set, wake up a waiter (if any). (We have to do a
3467          * futex_wake() even if OWNER_DIED is already set -
3468          * to handle the rare but possible case of recursive
3469          * thread-death.) The rest of the cleanup is done in
3470          * userspace.
3471          */
3472         mval = (uval & FUTEX_WAITERS) | FUTEX_OWNER_DIED;
3473 
3474         /*
3475          * We are not holding a lock here, but we want to have
3476          * the pagefault_disable/enable() protection because
3477          * we want to handle the fault gracefully. If the
3478          * access fails we try to fault in the futex with R/W
3479          * verification via get_user_pages. get_user() above
3480          * does not guarantee R/W access. If that fails we
3481          * give up and leave the futex locked.
3482          */
3483         if ((err = cmpxchg_futex_value_locked(&nval, uaddr, uval, mval))) {
3484                 switch (err) {
3485                 case -EFAULT:
3486                         if (fault_in_user_writeable(uaddr))
3487                                 return -1;
3488                         goto retry;
3489 
3490                 case -EAGAIN:
3491                         cond_resched();
3492                         goto retry;
3493 
3494                 default:
3495                         WARN_ON_ONCE(1);
3496                         return err;
3497                 }
3498         }
3499 
3500         if (nval != uval)
3501                 goto retry;
3502 
3503         /*
3504          * Wake robust non-PI futexes here. The wakeup of
3505          * PI futexes happens in exit_pi_state():
3506          */
3507         if (!pi && (uval & FUTEX_WAITERS))
3508                 futex_wake(uaddr, 1, 1, FUTEX_BITSET_MATCH_ANY);
3509 
3510         return 0;
3511 }

3533  * and mark any locks found there dead, and notify any waiters.
3534  *
3535  * We silently return on any sign of list-walking problem.
3536  */
3537 static void exit_robust_list(struct task_struct *curr)
     /*   */
3538 {
3539         struct robust_list_head __user *head = curr->robust_list;
3540         struct robust_list __user *entry, *next_entry, *pending;
3541         unsigned int limit = ROBUST_LIST_LIMIT, pi, pip;

3551          * sys_set_robust_list()):

3554                 return;
3555         /*
3556          * Fetch the relative futex offset:
3557          */
3558         if (get_user(futex_offset, &head->futex_offset))
3559                 return;
3560         /*

3562          * if it exists:

3564         if (fetch_robust_entry(&pending, &head->list_op_pending, &pip))
3565                 return;

3571                  * handle_futex_death:

3574                 /*
3575                  * A pending lock might already be on the list, so
3576                  * don't process it twice:
3577                  */

3579                         if (handle_futex_death((void __user *)entry + futex_offset,
3580                                                 curr, pi, HANDLE_DEATH_LIST))

3586                 pi = next_pi;

3588                  * Avoid excessively long or circular lists:
3589                  */
3590                 if (!--limit)
3591                         break;
3592 
3593                 cond_resched();
3594         }
3595 
3596         if (pending) {
3597                 handle_futex_death((void __user *)pending + futex_offset,
3598                                    curr, pip, HANDLE_DEATH_PENDING);

3660          * tsk->pi_lock held.
3661          *
3662          * It guarantees also that a pi_state which was queued right before
3663          * the state change under tsk->pi_lock by a concurrent waiter must
3664          * be observed in exit_pi_state_list().
3665          */
3666         raw_spin_lock_irq(&tsk->pi_lock);
3667         tsk->futex_state = FUTEX_STATE_EXITING;
3668         raw_spin_unlock_irq(&tsk->pi_lock);
3669 }
3670 
3671 static void futex_cleanup_end(struct task_struct *tsk, int state)
     /*   */
3672 {
3673         /*
3674          * Lockless store. The only side effect is that an observer might
3675          * take another loop until it becomes visible.
3676          */
3677         tsk->futex_state = state;
3678         /*
3679          * Drop the exit protection. This unblocks waiters which observed
3680          * FUTEX_STATE_EXITING to reevaluate the state.
3681          */
3682         mutex_unlock(&tsk->futex_exit_mutex);
3683 }
3684 
3685 void futex_exec_release(struct task_struct *tsk)
     /*   */
3686 {
3687         /*
3688          * The state handling is done for consistency, but in the case of
3689          * exec() there is no way to prevent futher damage as the PID stays
3690          * the same. But for the unlikely and arguably buggy case that a
3691          * futex is held on exec(), this provides at least as much state
3692          * consistency protection which is possible.
3693          */
3694         futex_cleanup_begin(tsk);
3695         futex_cleanup(tsk);
3696         /*
3697          * Reset the state to FUTEX_STATE_OK. The task is alive and about
3698          * exec a new binary.
3699          */
3700         futex_cleanup_end(tsk, FUTEX_STATE_OK);
3701 }
3702 
3703 void futex_exit_release(struct task_struct *tsk)
     /*   */
3704 {
3705         futex_cleanup_begin(tsk);
3706         futex_cleanup(tsk);
3707         futex_cleanup_end(tsk, FUTEX_STATE_DEAD);
3708 }

3710 long do_futex(u32 __user *uaddr, int op, u32 val, ktime_t *timeout,
     /*   */
3711                 u32 __user *uaddr2, u32 val2, u32 val3)
3712 {
3713         int cmd = op & FUTEX_CMD_MASK;
3714         unsigned int flags = 0;
3715 
3716         if (!(op & FUTEX_PRIVATE_FLAG))
3717                 flags |= FLAGS_SHARED;
3718 
3719         if (op & FUTEX_CLOCK_REALTIME) {
3720                 flags |= FLAGS_CLOCKRT;
3721                 if (cmd != FUTEX_WAIT_BITSET && cmd != FUTEX_WAIT_REQUEUE_PI)
3722                         return -ENOSYS;
3723         }
3724 
3725         switch (cmd) {
3726         case FUTEX_LOCK_PI:
3727         case FUTEX_UNLOCK_PI:
3728         case FUTEX_TRYLOCK_PI:
3729         case FUTEX_WAIT_REQUEUE_PI:
3730         case FUTEX_CMP_REQUEUE_PI:
3731                 if (!futex_cmpxchg_enabled)
3732                         return -ENOSYS;
3733         }
3734 
3735         switch (cmd) {
3736         case FUTEX_WAIT:
3737                 val3 = FUTEX_BITSET_MATCH_ANY;
3738                 fallthrough;
3739         case FUTEX_WAIT_BITSET:
3740                 return futex_wait(uaddr, flags, val, timeout, val3);
3741         case FUTEX_WAKE:
3742                 val3 = FUTEX_BITSET_MATCH_ANY;
3743                 fallthrough;
3744         case FUTEX_WAKE_BITSET:
3745                 return futex_wake(uaddr, flags, val, val3);
3746         case FUTEX_REQUEUE:
3747                 return futex_requeue(uaddr, flags, uaddr2, val, val2, NULL, 0);
3748         case FUTEX_CMP_REQUEUE:
3749                 return futex_requeue(uaddr, flags, uaddr2, val, val2, &val3, 0);
3750         case FUTEX_WAKE_OP:
3751                 return futex_wake_op(uaddr, flags, uaddr2, val, val2, val3);
3752         case FUTEX_LOCK_PI:
3753                 return futex_lock_pi(uaddr, flags, timeout, 0);
3754         case FUTEX_UNLOCK_PI:
3755                 return futex_unlock_pi(uaddr, flags);
3756         case FUTEX_TRYLOCK_PI:
3757                 return futex_lock_pi(uaddr, flags, NULL, 1);
3758         case FUTEX_WAIT_REQUEUE_PI:
3759                 val3 = FUTEX_BITSET_MATCH_ANY;
3760                 return futex_wait_requeue_pi(uaddr, flags, val, timeout, val3,
3761                                              uaddr2);
3762         case FUTEX_CMP_REQUEUE_PI:
3763                 return futex_requeue(uaddr, flags, uaddr2, val, val2, &val3, 1);
3764         }
3765         return -ENOSYS;
3766 }

     /*   */
3770                 struct __kernel_timespec __user *, utime, u32 __user *, uaddr2,
3771                 u32, val3)
3772 {
3773         struct timespec64 ts;
3774         ktime_t t, *tp = NULL;
3775         u32 val2 = 0;
3776         int cmd = op & FUTEX_CMD_MASK;
3777 
3778         if (utime && (cmd == FUTEX_WAIT || cmd == FUTEX_LOCK_PI ||
3779                       cmd == FUTEX_WAIT_BITSET ||
3780                       cmd == FUTEX_WAIT_REQUEUE_PI)) {
3781                 if (unlikely(should_fail_futex(!(op & FUTEX_PRIVATE_FLAG))))
3782                         return -EFAULT;
3783                 if (get_timespec64(&ts, utime))
3784                         return -EFAULT;
3785                 if (!timespec64_valid(&ts))
3786                         return -EINVAL;
3787 
3788                 t = timespec64_to_ktime(ts);
3789                 if (cmd == FUTEX_WAIT)
3790                         t = ktime_add_safe(ktime_get(), t);
3791                 else if (cmd != FUTEX_LOCK_PI && !(op & FUTEX_CLOCK_REALTIME))
3792                         t = timens_ktime_to_host(CLOCK_MONOTONIC, t);
3793                 tp = &t;
3794         }
3795         /*
3796          * requeue parameter in 'utime' if cmd == FUTEX_*_REQUEUE_*.
3797          * number of waiters to wake in 'utime' if cmd == FUTEX_WAKE_OP.
3798          */
3799         if (cmd == FUTEX_REQUEUE || cmd == FUTEX_CMP_REQUEUE ||
3800             cmd == FUTEX_CMP_REQUEUE_PI || cmd == FUTEX_WAKE_OP)
3801                 val2 = (u32) (unsigned long) utime;
3802 
3803         return do_futex(uaddr, op, val, tp, uaddr2, val2, val3);
3804 }

     /*   */
3824                                 compat_long_t futex_offset)
3825 {

3827         void __user *uaddr = compat_ptr(base + futex_offset);

3832 /*
3833  * Walk curr->robust_list (very carefully, it's a userspace list!)
3834  * and mark any locks found there dead, and notify any waiters.
3835  *
3836  * We silently return on any sign of list-walking problem.
3837  */
3838 static void compat_exit_robust_list(struct task_struct *curr)
     /*   */
3839 {
3840         struct compat_robust_list_head __user *head = curr->compat_robust_list;
3841         struct robust_list __user *entry, *next_entry, *pending;
3842         unsigned int limit = ROBUST_LIST_LIMIT, pi, pip;
3843         unsigned int next_pi;
3844         compat_uptr_t uentry, next_uentry, upending;
3845         compat_long_t futex_offset;
3846         int rc;
3847 
3848         if (!futex_cmpxchg_enabled)
3849                 return;
3850 
3851         /*
3852          * Fetch the list head (which was registered earlier, via
3853          * sys_set_robust_list()):
3854          */
3855         if (compat_fetch_robust_entry(&uentry, &entry, &head->list.next, &pi))
3856                 return;
3857         /*
3858          * Fetch the relative futex offset:
3859          */
3860         if (get_user(futex_offset, &head->futex_offset))
3861                 return;
3862         /*
3863          * Fetch any possibly pending lock-add first, and handle it
3864          * if it exists:
3865          */
3866         if (compat_fetch_robust_entry(&upending, &pending,
3867                                &head->list_op_pending, &pip))
3868                 return;
3869 
3870         next_entry = NULL;      /* avoid warning with gcc */
3871         while (entry != (struct robust_list __user *) &head->list) {
3872                 /*
3873                  * Fetch the next entry in the list before calling
3874                  * handle_futex_death:
3875                  */
3876                 rc = compat_fetch_robust_entry(&next_uentry, &next_entry,
3877                         (compat_uptr_t __user *)&entry->next, &next_pi);
3878                 /*
3879                  * A pending lock might already be on the list, so
3880                  * dont process it twice:
3881                  */
3882                 if (entry != pending) {
3883                         void __user *uaddr = futex_uaddr(entry, futex_offset);
3884 
3885                         if (handle_futex_death(uaddr, curr, pi,
3886                                                HANDLE_DEATH_LIST))
3887                                 return;
3888                 }
3889                 if (rc)
3890                         return;
3891                 uentry = next_uentry;
3892                 entry = next_entry;
3893                 pi = next_pi;
3894                 /*
3895                  * Avoid excessively long or circular lists:
3896                  */
3897                 if (!--limit)
3898                         break;
3899 
3900                 cond_resched();
3901         }
3902         if (pending) {
3903                 void __user *uaddr = futex_uaddr(pending, futex_offset);
3904 
3905                 handle_futex_death(uaddr, curr, pip, HANDLE_DEATH_PENDING);
3906         }
3907 }
