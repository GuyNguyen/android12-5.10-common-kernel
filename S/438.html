  51 /*
  52  * Scan threshold to break wait queue walk.
  53  * This allows a waker to take a break from holding the
  54  * wait queue lock during the wait queue walk.
  55  */
  56 #define WAITQUEUE_WALK_BREAK_CNT 64
  57 
  58 /*
  59  * The core wakeup function. Non-exclusive wakeups (nr_exclusive == 0) just
  60  * wake everything up. If it's an exclusive wakeup (nr_exclusive == small +ve
  61  * number) then we wake all the non-exclusive tasks and one exclusive task.
  62  *
  63  * There are circumstances in which we can try to wake a task which has already
  64  * started to run but is not in state TASK_RUNNING. try_to_wake_up() returns
  65  * zero in this (rare) case, and we handle it by continuing to scan the queue.
  66  */
  67 static int __wake_up_common(struct wait_queue_head *wq_head, unsigned int mode,
  68                         int nr_exclusive, int wake_flags, void *key,
  69                         wait_queue_entry_t *bookmark)
  70 {
  71         wait_queue_entry_t *curr, *next;
  72         int cnt = 0;
  73 
  74         lockdep_assert_held(&wq_head->lock);
  75 
  76         if (bookmark && (bookmark->flags & WQ_FLAG_BOOKMARK)) {
  77                 curr = list_next_entry(bookmark, entry);
  78 
  79                 list_del(&bookmark->entry);
  80                 bookmark->flags = 0;
  81         } else
  82                 curr = list_first_entry(&wq_head->head, wait_queue_entry_t, entry);
  83 
  84         if (&curr->entry == &wq_head->head)
  85                 return nr_exclusive;
  86 
  87         list_for_each_entry_safe_from(curr, next, &wq_head->head, entry) {
  88                 unsigned flags = curr->flags;
  89                 int ret;
  90 
  91                 if (flags & WQ_FLAG_BOOKMARK)
  92                         continue;
  93 
  94                 ret = curr->func(curr, mode, wake_flags, key);
  95                 if (ret < 0)
  96                         break;
  97                 if (ret && (flags & WQ_FLAG_EXCLUSIVE) && !--nr_exclusive)
  98                         break;
  99 
 100                 if (bookmark && (++cnt > WAITQUEUE_WALK_BREAK_CNT) &&
 101                                 (&next->entry != &wq_head->head)) {
 102                         bookmark->flags = WQ_FLAG_BOOKMARK;
 103                         list_add_tail(&bookmark->entry, &next->entry);
 104                         break;
 105                 }
 106         }
 107 
 108         return nr_exclusive;
 109 }

 111 static void __wake_up_common_lock(struct wait_queue_head *wq_head, unsigned int mode,
 112                         int nr_exclusive, int wake_flags, void *key)
 113 {
 114         unsigned long flags;
 115         wait_queue_entry_t bookmark;
 116 
 117         bookmark.flags = 0;
 118         bookmark.private = NULL;
 119         bookmark.func = NULL;
 120         INIT_LIST_HEAD(&bookmark.entry);
 121 
 122         do {
 123                 spin_lock_irqsave(&wq_head->lock, flags);
 124                 nr_exclusive = __wake_up_common(wq_head, mode, nr_exclusive,
 125                                                 wake_flags, key, &bookmark);
 126                 spin_unlock_irqrestore(&wq_head->lock, flags);
 127         } while (bookmark.flags & WQ_FLAG_BOOKMARK);
 128 }

 171  * @wq_head: the waitqueue
 172  * @mode: which threads
 173  * @key: opaque value to be passed to wakeup targets
 174  *
 175  * The sync wakeup differs that the waker knows that it will schedule

 230 /*
 231  * Note: we use "set_current_state()" _after_ the wait-queue add,
 232  * because we need a memory barrier there on SMP, so that any
 233  * wake-function that tests for the wait-queue being active
 234  * will be guaranteed to see waitqueue addition _or_ subsequent
 235  * tests in this thread will see the wakeup having taken place.
 236  *
 237  * The spin_unlock() itself is semi-permeable and only protects
 238  * one way (it only protects stuff inside the critical region and
 239  * stops them from bleeding out - it would still allow subsequent
 240  * loads to move into the critical region).
 241  */
 242 void
 243 prepare_to_wait(struct wait_queue_head *wq_head, struct wait_queue_entry *wq_entry, int state)
 244 {
 245         unsigned long flags;
 246 
 247         wq_entry->flags &= ~WQ_FLAG_EXCLUSIVE;
 248         spin_lock_irqsave(&wq_head->lock, flags);
 249         if (list_empty(&wq_entry->entry))
 250                 __add_wait_queue(wq_head, wq_entry);
 251         set_current_state(state);
 252         spin_unlock_irqrestore(&wq_head->lock, flags);
 253 }
 254 EXPORT_SYMBOL(prepare_to_wait);

 279 long prepare_to_wait_event(struct wait_queue_head *wq_head, struct wait_queue_entry *wq_entry, int state)
 280 {
 281         unsigned long flags;
 282         long ret = 0;
 283 
 284         spin_lock_irqsave(&wq_head->lock, flags);
 285         if (signal_pending_state(state, current)) {
 286                 /*
 287                  * Exclusive waiter must not fail if it was selected by wakeup,
 288                  * it should "consume" the condition we were waiting for.
 289                  *
 290                  * The caller will recheck the condition and return success if
 291                  * we were already woken up, we can not miss the event because
 292                  * wakeup locks/unlocks the same wq_head->lock.
 293                  *
 294                  * But we need to ensure that set-condition + wakeup after that
 295                  * can't see us, it should wake up another exclusive waiter if
 296                  * we fail.
 297                  */
 298                 list_del_init(&wq_entry->entry);
 299                 ret = -ERESTARTSYS;
 300         } else {
 301                 if (list_empty(&wq_entry->entry)) {
 302                         if (wq_entry->flags & WQ_FLAG_EXCLUSIVE)
 303                                 __add_wait_queue_entry_tail(wq_head, wq_entry);
 304                         else
 305                                 __add_wait_queue(wq_head, wq_entry);
 306                 }
 307                 set_current_state(state);
 308         }
 309         spin_unlock_irqrestore(&wq_head->lock, flags);
 310 
 311         return ret;
 312 }
 313 EXPORT_SYMBOL(prepare_to_wait_event);

 315 /*
 316  * Note! These two wait functions are entered with the
 317  * wait-queue lock held (and interrupts off in the _irq
 318  * case), so there is no race with testing the wakeup
 319  * condition in the caller before they add the wait
 320  * entry to the wake queue.
 321  */
 322 int do_wait_intr(wait_queue_head_t *wq, wait_queue_entry_t *wait)
 323 {
 324         if (likely(list_empty(&wait->entry)))
 325                 __add_wait_queue_entry_tail(wq, wait);
 326 
 327         set_current_state(TASK_INTERRUPTIBLE);
 328         if (signal_pending(current))
 329                 return -ERESTARTSYS;
 330 
 331         spin_unlock(&wq->lock);
 332         schedule();
 333         spin_lock(&wq->lock);
 334 
 335         return 0;
 336 }
 337 EXPORT_SYMBOL(do_wait_intr);

 356 /**
 357  * finish_wait - clean up after waiting in a queue
 358  * @wq_head: waitqueue waited on
 359  * @wq_entry: wait descriptor
 360  *
 361  * Sets current thread back to running state and removes
 362  * the wait descriptor from the given waitqueue if still
 363  * queued.
 364  */
 365 void finish_wait(struct wait_queue_head *wq_head, struct wait_queue_entry *wq_entry)
 366 {
 367         unsigned long flags;
 368 
 369         __set_current_state(TASK_RUNNING);
 370         /*
 371          * We can check for list emptiness outside the lock
 372          * IFF:
 373          *  - we use the "careful" check that verifies both
 374          *    the next and prev pointers, so that there cannot
 375          *    be any half-pending updates in progress on other
 376          *    CPU's that we haven't seen yet (and that might
 377          *    still change the stack area.
 378          * and
 379          *  - all other users take the lock (ie we can only
 380          *    have _one_ other CPU that looks at or modifies
 381          *    the list).
 382          */
 383         if (!list_empty_careful(&wq_entry->entry)) {
 384                 spin_lock_irqsave(&wq_head->lock, flags);
 385                 list_del_init(&wq_entry->entry);
 386                 spin_unlock_irqrestore(&wq_head->lock, flags);
 387         }
 388 }
 389 EXPORT_SYMBOL(finish_wait);

 408 /*
 409  * DEFINE_WAIT_FUNC(wait, woken_wake_func);
 410  *
 411  * add_wait_queue(&wq_head, &wait);
 412  * for (;;) {
 413  *     if (condition)
 414  *         break;
 415  *
 416  *     // in wait_woken()                       // in woken_wake_function()
 417  *
 418  *     p->state = mode;                         wq_entry->flags |= WQ_FLAG_WOKEN;
 419  *     smp_mb(); // A                           try_to_wake_up():
 420  *     if (!(wq_entry->flags & WQ_FLAG_WOKEN))     <full barrier>
 421  *         schedule()                              if (p->state & mode)
 422  *     p->state = TASK_RUNNING;                       p->state = TASK_RUNNING;
 423  *     wq_entry->flags &= ~WQ_FLAG_WOKEN;       ~~~~~~~~~~~~~~~~~~
 424  *     smp_mb(); // B                           condition = true;
 425  * }                                            smp_mb(); // C
 426  * remove_wait_queue(&wq_head, &wait);          wq_entry->flags |= WQ_FLAG_WOKEN;
 427  */
 428 __sched long wait_woken(struct wait_queue_entry *wq_entry, unsigned int mode, long timeout)
 429 {
 430         /*
 431          * The below executes an smp_mb(), which matches with the full barrier
 432          * executed by the try_to_wake_up() in woken_wake_function() such that
 433          * either we see the store to wq_entry->flags in woken_wake_function()
 434          * or woken_wake_function() sees our store to current->state.
 435          */
 436         set_current_state(mode); /* A */
 437         if (!(wq_entry->flags & WQ_FLAG_WOKEN) && !is_kthread_should_stop())
 438                 timeout = schedule_timeout(timeout);
 439         __set_current_state(TASK_RUNNING);
 440 
 441         /*
 442          * The below executes an smp_mb(), which matches with the smp_mb() (C)
 443          * in woken_wake_function() such that either we see the wait condition
 444          * being true or the store to wq_entry->flags in woken_wake_function()
 445          * follows ours in the coherence order.
 446          */
 447         smp_store_mb(wq_entry->flags, wq_entry->flags & ~WQ_FLAG_WOKEN); /* B */
 448 
 449         return timeout;
 450 }
 451 EXPORT_SYMBOL(wait_woken);
