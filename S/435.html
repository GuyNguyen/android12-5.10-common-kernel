 209 static void __kthread_parkme(struct kthread *self)
     /*   */
 210 {
 211         for (;;) {
 212                 /*
 213                  * TASK_PARKED is a special state; we must serialize against
 214                  * possible pending wakeups to avoid store-store collisions on
 215                  * task->state.
 216                  *
 217                  * Such a collision might possibly result in the task state
 218                  * changin from TASK_PARKED and us failing the
 219                  * wait_task_inactive() in kthread_park().
 220                  */
 221                 set_special_state(TASK_PARKED);
 222                 if (!test_bit(KTHREAD_SHOULD_PARK, &self->flags))
 223                         break;
 224 
 225                 /*
 226                  * Thread is going to call schedule(), do not preempt it,
 227                  * or the caller of kthread_park() may spend more time in
 228                  * wait_task_inactive().
 229                  */
 230                 preempt_disable();
 231                 complete(&self->parked);
 232                 schedule_preempt_disabled();
 233                 preempt_enable();
 234         }
 235         __set_current_state(TASK_RUNNING);
 236 }

 244 static int kthread(void *_create)
     /*   */
 245 {
 246         /* Copy data: it's on kthread's stack */
 247         struct kthread_create_info *create = _create;
 248         int (*threadfn)(void *data) = create->threadfn;
 249         void *data = create->data;
 250         struct completion *done;
 251         struct kthread *self;
 252         int ret;
 253 
 254         self = kzalloc(sizeof(*self), GFP_KERNEL);
 255         set_kthread_struct(self);
 256 
 257         /* If user was SIGKILLed, I release the structure. */
 258         done = xchg(&create->done, NULL);
 259         if (!done) {
 260                 kfree(create);
 261                 do_exit(-EINTR);
 262         }
 263 
 264         if (!self) {
 265                 create->result = ERR_PTR(-ENOMEM);
 266                 complete(done);
 267                 do_exit(-ENOMEM);
 268         }
 269 
 270         self->threadfn = threadfn;
 271         self->data = data;
 272         init_completion(&self->exited);
 273         init_completion(&self->parked);
 274         current->vfork_done = &self->exited;
 275 
 276         /* OK, tell user we're spawned, wait for stop or wakeup */
 277         __set_current_state(TASK_UNINTERRUPTIBLE);
 278         create->result = current;
 279         /*
 280          * Thread is going to call schedule(), do not preempt it,
 281          * or the creator may spend more time in wait_task_inactive().
 282          */
 283         preempt_disable();
 284         complete(done);
 285         schedule_preempt_disabled();
 286         preempt_enable();
 287 
 288         ret = -EINTR;
 289         if (!test_bit(KTHREAD_SHOULD_STOP, &self->flags)) {
 290                 cgroup_kthread_ready();
 291                 __kthread_parkme(self);
 292                 ret = threadfn(data);
 293         }
 294         do_exit(ret);
 295 }

 307 static void create_kthread(struct kthread_create_info *create)
     /*   */
 308 {
 309         int pid;
 310 
 311 #ifdef CONFIG_NUMA
 312         current->pref_node_fork = create->node;
 313 #endif
 314         /* We want our own signal handler (we take no signals by default). */
 315         pid = kernel_thread(kthread, create, CLONE_FS | CLONE_FILES | SIGCHLD);
 316         if (pid < 0) {
 317                 /* If user was SIGKILLed, I release the structure. */
 318                 struct completion *done = xchg(&create->done, NULL);
 319 
 320                 if (!done) {
 321                         kfree(create);
 322                         return;
 323                 }
 324                 create->result = ERR_PTR(pid);
 325                 complete(done);
 326         }
 327 }

 329 static __printf(4, 0)
     /*   */
 330 struct task_struct *__kthread_create_on_node(int (*threadfn)(void *data),
 331                                                     void *data, int node,
 332                                                     const char namefmt[],
 333                                                     va_list args)
 334 {
 335         DECLARE_COMPLETION_ONSTACK(done);
 336         struct task_struct *task;
 337         struct kthread_create_info *create = kmalloc(sizeof(*create),
 338                                                      GFP_KERNEL);
 339 
 340         if (!create)
 341                 return ERR_PTR(-ENOMEM);
 342         create->threadfn = threadfn;
 343         create->data = data;
 344         create->node = node;
 345         create->done = &done;
 346 
 347         spin_lock(&kthread_create_lock);
 348         list_add_tail(&create->list, &kthread_create_list);
 349         spin_unlock(&kthread_create_lock);
 350 
 351         wake_up_process(kthreadd_task);
 352         /*
 353          * Wait for completion in killable state, for I might be chosen by
 354          * the OOM killer while kthreadd is trying to allocate memory for
 355          * new kernel thread.
 356          */
 357         if (unlikely(wait_for_completion_killable(&done))) {
 358                 /*
 359                  * If I was SIGKILLed before kthreadd (or new kernel thread)
 360                  * calls complete(), leave the cleanup of this structure to
 361                  * that thread.
 362                  */
 363                 if (xchg(&create->done, NULL))
 364                         return ERR_PTR(-EINTR);
 365                 /*
 366                  * kthreadd (or new kernel thread) will call complete()
 367                  * shortly.
 368                  */
 369                 wait_for_completion(&done);
 370         }
 371         task = create->result;
 372         if (!IS_ERR(task)) {
 373                 static const struct sched_param param = { .sched_priority = 0 };
 374                 char name[TASK_COMM_LEN];
 375 
 376                 /*
 377                  * task is already visible to other tasks, so updating
 378                  * COMM must be protected.
 379                  */
 380                 vsnprintf(name, sizeof(name), namefmt, args);
 381                 set_task_comm(task, name);
 382                 /*
 383                  * root may have changed our (kthreadd's) priority or CPU mask.
 384                  * The kernel thread should not inherit these properties.
 385                  */
 386                 sched_setscheduler_nocheck(task, SCHED_NORMAL, &param);
 387                 set_cpus_allowed_ptr(task,
 388                                      housekeeping_cpumask(HK_FLAG_KTHREAD));
 389         }
 390         kfree(create);
 391         return task;
 392 }

 394 /**
 395  * kthread_create_on_node - create a kthread.
 396  * @threadfn: the function to run until signal_pending(current).
 397  * @data: data ptr for @threadfn.
 398  * @node: task and thread structures for the thread are allocated on this node
 399  * @namefmt: printf-style name for the thread.
 400  *
 401  * Description: This helper function creates and names a kernel
 402  * thread.  The thread will be stopped: use wake_up_process() to start
 403  * it.  See also kthread_run().  The new thread has SCHED_NORMAL policy and
 404  * is affine to all CPUs.
 405  *
 406  * If thread is going to be bound on a particular cpu, give its node
 407  * in @node, to get NUMA affinity for kthread stack, or else give NUMA_NO_NODE.
 408  * When woken, the thread will run @threadfn() with @data as its
 409  * argument. @threadfn() can either call do_exit() directly if it is a
 410  * standalone thread for which no one will call kthread_stop(), or
 411  * return when 'kthread_should_stop()' is true (which means
 412  * kthread_stop() has been called).  The return value should be zero
 413  * or a negative error number; it will be passed to kthread_stop().
 414  *
 415  * Returns a task_struct or ERR_PTR(-ENOMEM) or ERR_PTR(-EINTR).
 416  */
 417 struct task_struct *kthread_create_on_node(int (*threadfn)(void *data),
     /*   */
 418                                            void *data, int node,
 419                                            const char namefmt[],
 420                                            ...)
 421 {
 422         struct task_struct *task;
 423         va_list args;
 424 
 425         va_start(args, namefmt);
 426         task = __kthread_create_on_node(threadfn, data, node, namefmt, args);
 427         va_end(args);
 428 
 429         return task;
 430 }

 527 /**
 528  * kthread_unpark - unpark a thread created by kthread_create().
 529  * @k:          thread created by kthread_create().
 530  *
 531  * Sets kthread_should_park() for @k to return false, wakes it, and
 532  * waits for it to return. If the thread is marked percpu then its
 533  * bound to the cpu again.
 534  */
 535 void kthread_unpark(struct task_struct *k)
     /*   */
 536 {
 537         struct kthread *kthread = to_kthread(k);
 538 
 539         /*
 540          * Newly created kthread was parked when the CPU was offline.
 541          * The binding was lost and we need to set it again.
 542          */
 543         if (test_bit(KTHREAD_IS_PER_CPU, &kthread->flags))
 544                 __kthread_bind(k, kthread->cpu, TASK_PARKED);
 545 
 546         clear_bit(KTHREAD_SHOULD_PARK, &kthread->flags);
 547         /*
 548          * __kthread_parkme() will either see !SHOULD_PARK or get the wakeup.
 549          */
 550         wake_up_state(k, TASK_PARKED);
 551 }

 554 /**
 555  * kthread_park - park a thread created by kthread_create().
 556  * @k: thread created by kthread_create().
 557  *
 558  * Sets kthread_should_park() for @k to return true, wakes it, and
 559  * waits for it to return. This can also be called after kthread_create()
 560  * instead of calling wake_up_process(): the thread will park without
 561  * calling threadfn().
 562  *
 563  * Returns 0 if the thread is parked, -ENOSYS if the thread exited.
 564  * If called by the kthread itself just the park bit is set.
 565  */
 566 int kthread_park(struct task_struct *k)
     /*   */
 567 {
 568         struct kthread *kthread = to_kthread(k);
 569 
 570         if (WARN_ON(k->flags & PF_EXITING))
 571                 return -ENOSYS;
 572 
 573         if (WARN_ON_ONCE(test_bit(KTHREAD_SHOULD_PARK, &kthread->flags)))
 574                 return -EBUSY;
 575 
 576         set_bit(KTHREAD_SHOULD_PARK, &kthread->flags);
 577         if (k != current) {
 578                 wake_up_process(k);
 579                 /*
 580                  * Wait for __kthread_parkme() to complete(), this means we
 581                  * _will_ have TASK_PARKED and are about to call schedule().
 582                  */
 583                 wait_for_completion(&kthread->parked);
 584                 /*
 585                  * Now wait for that schedule() to complete and the task to
 586                  * get scheduled out.
 587                  */
 588                 WARN_ON_ONCE(!wait_task_inactive(k, TASK_PARKED));
 589         }
 590 
 591         return 0;
 592 }
 593 EXPORT_SYMBOL_GPL(kthread_park);

 595 /**
 596  * kthread_stop - stop a thread created by kthread_create().
 597  * @k: thread created by kthread_create().
 598  *
 599  * Sets kthread_should_stop() for @k to return true, wakes it, and
 600  * waits for it to exit. This can also be called after kthread_create()
 601  * instead of calling wake_up_process(): the thread will exit without
 602  * calling threadfn().
 603  *
 604  * If threadfn() may call do_exit() itself, the caller must ensure
 605  * task_struct can't go away.
 606  *
 607  * Returns the result of threadfn(), or %-EINTR if wake_up_process()
 608  * was never called.
 609  */
 610 int kthread_stop(struct task_struct *k)
     /*   */
 611 {
 612         struct kthread *kthread;
 613         int ret;
 614 
 615         trace_sched_kthread_stop(k);
 616 
 617         get_task_struct(k);
 618         kthread = to_kthread(k);
 619         set_bit(KTHREAD_SHOULD_STOP, &kthread->flags);
 620         kthread_unpark(k);
 621         wake_up_process(k);
 622         wait_for_completion(&kthread->exited);
 623         ret = k->exit_code;
 624         put_task_struct(k);
 625 
 626         trace_sched_kthread_stop_ret(ret);
 627         return ret;
 628 }

 631 int kthreadd(void *unused)
     /*   */
 632 {
 633         struct task_struct *tsk = current;
 634 
 635         /* Setup a clean context for our children to inherit. */
 636         set_task_comm(tsk, "kthreadd");
 637         ignore_signals(tsk);
 638         set_cpus_allowed_ptr(tsk, housekeeping_cpumask(HK_FLAG_KTHREAD));
 639         set_mems_allowed(node_states[N_MEMORY]);
 640 
 641         current->flags |= PF_NOFREEZE;
 642         cgroup_init_kthreadd();
 643 
 644         for (;;) {
 645                 set_current_state(TASK_INTERRUPTIBLE);
 646                 if (list_empty(&kthread_create_list))
 647                         schedule();
 648                 __set_current_state(TASK_RUNNING);
 649 
 650                 spin_lock(&kthread_create_lock);
 651                 while (!list_empty(&kthread_create_list)) {
 652                         struct kthread_create_info *create;
 653 
 654                         create = list_entry(kthread_create_list.next,
 655                                             struct kthread_create_info, list);
 656                         list_del_init(&create->list);
 657                         spin_unlock(&kthread_create_lock);
 658 
 659                         create_kthread(create);
 660 
 661                         spin_lock(&kthread_create_lock);
 662                 }
 663                 spin_unlock(&kthread_create_lock);
 664         }
 665 
 666         return 0;
 667 }

 681 /**
 682  * kthread_worker_fn - kthread function to process kthread_worker
 683  * @worker_ptr: pointer to initialized kthread_worker
 684  *
 685  * This function implements the main cycle of kthread worker. It processes
 686  * work_list until it is stopped with kthread_stop(). It sleeps when the queue
 687  * is empty.
 688  *
 689  * The works are not allowed to keep any locks, disable preemption or interrupts
 690  * when they finish. There is defined a safe point for freezing when one work
 691  * finishes and before a new one is started.
 692  *
 693  * Also the works must not be handled by more than one worker at the same time,
 694  * see also kthread_queue_work().
 695  */
 696 int kthread_worker_fn(void *worker_ptr)
     /*   */
 697 {
 698         struct kthread_worker *worker = worker_ptr;
 699         struct kthread_work *work;
 700 
 701         /*
 702          * FIXME: Update the check and remove the assignment when all kthread
 703          * worker users are created using kthread_create_worker*() functions.
 704          */
 705         WARN_ON(worker->task && worker->task != current);
 706         worker->task = current;
 707 
 708         if (worker->flags & KTW_FREEZABLE)
 709                 set_freezable();
 710 
 711 repeat:
 712         set_current_state(TASK_INTERRUPTIBLE);  /* mb paired w/ kthread_stop */
 713 
 714         if (kthread_should_stop()) {
 715                 __set_current_state(TASK_RUNNING);
 716                 raw_spin_lock_irq(&worker->lock);
 717                 worker->task = NULL;
 718                 raw_spin_unlock_irq(&worker->lock);
 719                 return 0;
 720         }
 721 
 722         work = NULL;
 723         raw_spin_lock_irq(&worker->lock);
 724         if (!list_empty(&worker->work_list)) {
 725                 work = list_first_entry(&worker->work_list,
 726                                         struct kthread_work, node);
 727                 list_del_init(&work->node);
 728         }
 729         worker->current_work = work;
 730         raw_spin_unlock_irq(&worker->lock);
 731 
 732         if (work) {
 733                 __set_current_state(TASK_RUNNING);
 734                 work->func(work);
 735         } else if (!freezing(current))
 736                 schedule();
 737 
 738         try_to_freeze();
 739         cond_resched();
 740         goto repeat;
 741 }
 742 EXPORT_SYMBOL_GPL(kthread_worker_fn);

 744 static __printf(3, 0) struct kthread_worker *
     /*   */
 745 __kthread_create_worker(int cpu, unsigned int flags,
 746                         const char namefmt[], va_list args)
 747 {
 748         struct kthread_worker *worker;
 749         struct task_struct *task;
 750         int node = NUMA_NO_NODE;
 751 
 752         worker = kzalloc(sizeof(*worker), GFP_KERNEL);
 753         if (!worker)
 754                 return ERR_PTR(-ENOMEM);
 755 
 756         kthread_init_worker(worker);
 757 
 758         if (cpu >= 0)
 759                 node = cpu_to_node(cpu);
 760 
 761         task = __kthread_create_on_node(kthread_worker_fn, worker,
 762                                                 node, namefmt, args);
 763         if (IS_ERR(task))
 764                 goto fail_task;
 765 
 766         if (cpu >= 0)
 767                 kthread_bind(task, cpu);
 768 
 769         worker->flags = flags;
 770         worker->task = task;
 771         wake_up_process(task);
 772         return worker;
 773 
 774 fail_task:
 775         kfree(worker);
 776         return ERR_CAST(task);
 777 }

 834 /*
 835  * Returns true when the work could not be queued at the moment.
 836  * It happens when it is already pending in a worker list
 837  * or when it is being cancelled.
 838  */
 839 static inline bool queuing_blocked(struct kthread_worker *worker,
     /*   */
 840                                    struct kthread_work *work)
 841 {
 842         lockdep_assert_held(&worker->lock);
 843 
 844         return !list_empty(&work->node) || work->canceling;

 847 static void kthread_insert_work_sanity_check(struct kthread_worker *worker,
     /*   */
 848                                              struct kthread_work *work)
 849 {
 850         lockdep_assert_held(&worker->lock);
 851         WARN_ON_ONCE(!list_empty(&work->node));
 852         /* Do not use a work with >1 worker, see kthread_queue_work() */
 853         WARN_ON_ONCE(work->worker && work->worker != worker);
 854 }

 856 /* insert @work before @pos in @worker */
 857 static void kthread_insert_work(struct kthread_worker *worker,
     /*   */
 858                                 struct kthread_work *work,
 859                                 struct list_head *pos)
 860 {
 861         kthread_insert_work_sanity_check(worker, work);
 862 
 863         list_add_tail(&work->node, pos);
 864         work->worker = worker;
 865         if (!worker->current_work && likely(worker->task))
 866                 wake_up_process(worker->task);
 867 }

 869 /**
 870  * kthread_queue_work - queue a kthread_work
 871  * @worker: target kthread_worker
 872  * @work: kthread_work to queue
 873  *
 874  * Queue @work to work processor @task for async execution.  @task
 875  * must have been created with kthread_worker_create().  Returns %true
 876  * if @work was successfully queued, %false if it was already pending.
 877  *
 878  * Reinitialize the work if it needs to be used by another worker.
 879  * For example, when the worker was stopped and started again.
 880  */
 881 bool kthread_queue_work(struct kthread_worker *worker,
     /*   */
 882                         struct kthread_work *work)
 883 {
 884         bool ret = false;
 885         unsigned long flags;
 886 
 887         raw_spin_lock_irqsave(&worker->lock, flags);
 888         if (!queuing_blocked(worker, work)) {
 889                 kthread_insert_work(worker, work, &worker->work_list);
 890                 ret = true;
 891         }
 892         raw_spin_unlock_irqrestore(&worker->lock, flags);
 893         return ret;
 894 }
 895 EXPORT_SYMBOL_GPL(kthread_queue_work);

 897 /**
 898  * kthread_delayed_work_timer_fn - callback that queues the associated kthread
 899  *      delayed work when the timer expires.
 900  * @t: pointer to the expired timer
 901  *
 902  * The format of the function is defined by struct timer_list.
 903  * It should have been called from irqsafe timer with irq already off.
 904  */
 905 void kthread_delayed_work_timer_fn(struct timer_list *t)
     /*   */
 906 {
 907         struct kthread_delayed_work *dwork = from_timer(dwork, t, timer);
 908         struct kthread_work *work = &dwork->work;
 909         struct kthread_worker *worker = work->worker;
 910         unsigned long flags;
 911 
 912         /*
 913          * This might happen when a pending work is reinitialized.
 914          * It means that it is used a wrong way.
 915          */
 916         if (WARN_ON_ONCE(!worker))
 917                 return;
 918 
 919         raw_spin_lock_irqsave(&worker->lock, flags);
 920         /* Work must not be used with >1 worker, see kthread_queue_work(). */
 921         WARN_ON_ONCE(work->worker != worker);
 922 
 923         /* Move the work from worker->delayed_work_list. */
 924         WARN_ON_ONCE(list_empty(&work->node));
 925         list_del_init(&work->node);
 926         if (!work->canceling)
 927                 kthread_insert_work(worker, work, &worker->work_list);
 928 
 929         raw_spin_unlock_irqrestore(&worker->lock, flags);
 930 }
 931 EXPORT_SYMBOL(kthread_delayed_work_timer_fn);

 933 static void __kthread_queue_delayed_work(struct kthread_worker *worker,
     /*   */
 934                                          struct kthread_delayed_work *dwork,
 935                                          unsigned long delay)
 936 {
 937         struct timer_list *timer = &dwork->timer;
 938         struct kthread_work *work = &dwork->work;
 939 
 940         /*
 941          * With CFI, timer->function can point to a jump table entry in a module,
 942          * which fails the comparison. Disable the warning if CFI and modules are
 943          * both enabled.
 944          */
 945         if (!IS_ENABLED(CONFIG_CFI_CLANG) || !IS_ENABLED(CONFIG_MODULES))
 946                 WARN_ON_ONCE(timer->function != kthread_delayed_work_timer_fn);
 947 
 948         /*
 949          * If @delay is 0, queue @dwork->work immediately.  This is for
 950          * both optimization and correctness.  The earliest @timer can
 951          * expire is on the closest next tick and delayed_work users depend
 952          * on that there's no such delay when @delay is 0.
 953          */
 954         if (!delay) {
 955                 kthread_insert_work(worker, work, &worker->work_list);
 956                 return;
 957         }
 958 
 959         /* Be paranoid and try to detect possible races already now. */
 960         kthread_insert_work_sanity_check(worker, work);
 961 
 962         list_add(&work->node, &worker->delayed_work_list);
 963         work->worker = worker;
 964         timer->expires = jiffies + delay;
 965         add_timer(timer);
 966 }

 968 /**
 969  * kthread_queue_delayed_work - queue the associated kthread work
 970  *      after a delay.
 971  * @worker: target kthread_worker
 972  * @dwork: kthread_delayed_work to queue
 973  * @delay: number of jiffies to wait before queuing
 974  *
 975  * If the work has not been pending it starts a timer that will queue
 976  * the work after the given @delay. If @delay is zero, it queues the
 977  * work immediately.
 978  *
 979  * Return: %false if the @work has already been pending. It means that
 980  * either the timer was running or the work was queued. It returns %true
 981  * otherwise.
 982  */
 983 bool kthread_queue_delayed_work(struct kthread_worker *worker,
     /*   */
 984                                 struct kthread_delayed_work *dwork,
 985                                 unsigned long delay)
 986 {
 987         struct kthread_work *work = &dwork->work;
 988         unsigned long flags;
 989         bool ret = false;
 990 
 991         raw_spin_lock_irqsave(&worker->lock, flags);
 992 
 993         if (!queuing_blocked(worker, work)) {
 994                 __kthread_queue_delayed_work(worker, dwork, delay);
 995                 ret = true;
 996         }
 997 
 998         raw_spin_unlock_irqrestore(&worker->lock, flags);
 999         return ret;
1000 }

1015 /**
1016  * kthread_flush_work - flush a kthread_work
1017  * @work: work to flush
1018  *
1019  * If @work is queued or executing, wait for it to finish execution.
1020  */
1021 void kthread_flush_work(struct kthread_work *work)
     /*   */
1022 {
1023         struct kthread_flush_work fwork = {
1024                 KTHREAD_WORK_INIT(fwork.work, kthread_flush_work_fn),
1025                 COMPLETION_INITIALIZER_ONSTACK(fwork.done),
1026         };
1027         struct kthread_worker *worker;
1028         bool noop = false;
1029 
1030         worker = work->worker;
1031         if (!worker)
1032                 return;
1033 
1034         raw_spin_lock_irq(&worker->lock);
1035         /* Work must not be used with >1 worker, see kthread_queue_work(). */
1036         WARN_ON_ONCE(work->worker != worker);
1037 
1038         if (!list_empty(&work->node))
1039                 kthread_insert_work(worker, &fwork.work, work->node.next);
1040         else if (worker->current_work == work)
1041                 kthread_insert_work(worker, &fwork.work,
1042                                     worker->work_list.next);
1043         else
1044                 noop = true;
1045 
1046         raw_spin_unlock_irq(&worker->lock);
1047 
1048         if (!noop)
1049                 wait_for_completion(&fwork.done);
1050 }

1053 /*
1054  * This function removes the work from the worker queue. Also it makes sure
1055  * that it won't get queued later via the delayed work's timer.
1056  *
1057  * The work might still be in use when this function finishes. See the
1058  * current_work proceed by the worker.
1059  *
1060  * Return: %true if @work was pending and successfully canceled,
1061  *      %false if @work was not pending
1062  */
1063 static bool __kthread_cancel_work(struct kthread_work *work, bool is_dwork,
     /*   */
1064                                   unsigned long *flags)
1065 {
1066         /* Try to cancel the timer if exists. */
1067         if (is_dwork) {
1068                 struct kthread_delayed_work *dwork =
1069                         container_of(work, struct kthread_delayed_work, work);
1070                 struct kthread_worker *worker = work->worker;
1071 
1072                 /*
1073                  * del_timer_sync() must be called to make sure that the timer
1074                  * callback is not running. The lock must be temporary released
1075                  * to avoid a deadlock with the callback. In the meantime,
1076                  * any queuing is blocked by setting the canceling counter.
1077                  */
1078                 work->canceling++;
1079                 raw_spin_unlock_irqrestore(&worker->lock, *flags);
1080                 del_timer_sync(&dwork->timer);
1081                 raw_spin_lock_irqsave(&worker->lock, *flags);
1082                 work->canceling--;
1083         }
1084 
1085         /*
1086          * Try to remove the work from a worker list. It might either
1087          * be from worker->work_list or from worker->delayed_work_list.
1088          */
1089         if (!list_empty(&work->node)) {
1090                 list_del_init(&work->node);
1091                 return true;
1092         }
1093 
1094         return false;
1095 }

1097 /**
1098  * kthread_mod_delayed_work - modify delay of or queue a kthread delayed work
1099  * @worker: kthread worker to use
1100  * @dwork: kthread delayed work to queue
1101  * @delay: number of jiffies to wait before queuing
1102  *
1103  * If @dwork is idle, equivalent to kthread_queue_delayed_work(). Otherwise,
1104  * modify @dwork's timer so that it expires after @delay. If @delay is zero,
1105  * @work is guaranteed to be queued immediately.
1106  *
1107  * Return: %true if @dwork was pending and its timer was modified,
1108  * %false otherwise.
1109  *
1110  * A special case is when the work is being canceled in parallel.
1111  * It might be caused either by the real kthread_cancel_delayed_work_sync()
1112  * or yet another kthread_mod_delayed_work() call. We let the other command
1113  * win and return %false here. The caller is supposed to synchronize these
1114  * operations a reasonable way.
1115  *
1116  * This function is safe to call from any context including IRQ handler.
1117  * See __kthread_cancel_work() and kthread_delayed_work_timer_fn()
1118  * for details.
1119  */
1120 bool kthread_mod_delayed_work(struct kthread_worker *worker,
     /*   */
1121                               struct kthread_delayed_work *dwork,
1122                               unsigned long delay)
1123 {
1124         struct kthread_work *work = &dwork->work;
1125         unsigned long flags;
1126         int ret = false;
1127 
1128         raw_spin_lock_irqsave(&worker->lock, flags);
1129 
1130         /* Do not bother with canceling when never queued. */
1131         if (!work->worker)
1132                 goto fast_queue;
1133 
1134         /* Work must not be used with >1 worker, see kthread_queue_work() */
1135         WARN_ON_ONCE(work->worker != worker);
1136 
1137         /* Do not fight with another command that is canceling this work. */
1138         if (work->canceling)
1139                 goto out;
1140 
1141         ret = __kthread_cancel_work(work, true, &flags);
1142 
1143         /*
1144          * Canceling could run in parallel from kthread_cancel_delayed_work_sync
1145          * and change work's canceling count as the spinlock is released and regain
1146          * in __kthread_cancel_work so we need to check the count again. Otherwise,
1147          * we might incorrectly queue the dwork and further cause
1148          * cancel_delayed_work_sync thread waiting for flush dwork endlessly.
1149          */
1150         if (work->canceling) {
1151                 ret = false;
1152                 goto out;
1153         }
1154 
1155 fast_queue:
1156         __kthread_queue_delayed_work(worker, dwork, delay);
1157 out:
1158         raw_spin_unlock_irqrestore(&worker->lock, flags);
1159         return ret;
1160 }
1161 EXPORT_SYMBOL_GPL(kthread_mod_delayed_work);

1163 static bool __kthread_cancel_work_sync(struct kthread_work *work, bool is_dwork)
     /*   */
1164 {
1165         struct kthread_worker *worker = work->worker;
1166         unsigned long flags;
1167         int ret = false;
1168 
1169         if (!worker)
1170                 goto out;
1171 
1172         raw_spin_lock_irqsave(&worker->lock, flags);
1173         /* Work must not be used with >1 worker, see kthread_queue_work(). */
1174         WARN_ON_ONCE(work->worker != worker);
1175 
1176         ret = __kthread_cancel_work(work, is_dwork, &flags);
1177 
1178         if (worker->current_work != work)
1179                 goto out_fast;
1180 
1181         /*
1182          * The work is in progress and we need to wait with the lock released.
1183          * In the meantime, block any queuing by setting the canceling counter.
1184          */
1185         work->canceling++;
1186         raw_spin_unlock_irqrestore(&worker->lock, flags);
1187         kthread_flush_work(work);
1188         raw_spin_lock_irqsave(&worker->lock, flags);
1189         work->canceling--;
1190 
1191 out_fast:
1192         raw_spin_unlock_irqrestore(&worker->lock, flags);
1193 out:
1194         return ret;
1195 }

1234 /**

1236  * @worker: worker to flush
1237  *
1238  * Wait until all currently executing or pending works on @worker are
1239  * finished.

1241 void kthread_flush_worker(struct kthread_worker *worker)
     /*   */
1242 {
1243         struct kthread_flush_work fwork = {
1244                 KTHREAD_WORK_INIT(fwork.work, kthread_flush_work_fn),
1245                 COMPLETION_INITIALIZER_ONSTACK(fwork.done),
1246         };
1247 
1248         kthread_queue_work(worker, &fwork.work);
1249         wait_for_completion(&fwork.done);
1250 }
1251 EXPORT_SYMBOL_GPL(kthread_flush_worker);

1253 /**
1254  * kthread_destroy_worker - destroy a kthread worker
1255  * @worker: worker to be destroyed
1256  *
1257  * Flush and destroy @worker.  The simple flush is enough because the kthread
1258  * worker API is used only in trivial scenarios.  There are no multi-step state
1259  * machines needed.
1260  */
1261 void kthread_destroy_worker(struct kthread_worker *worker)
     /*   */
1262 {
1263         struct task_struct *task;
1264 
1265         task = worker->task;
1266         if (WARN_ON(!task))
1267                 return;
1268 
1269         kthread_flush_worker(worker);
1270         kthread_stop(task);
1271         WARN_ON(!list_empty(&worker->work_list));
1272         kfree(worker);
1273 }

1276 /**
1277  * kthread_use_mm - make the calling kthread operate on an address space
1278  * @mm: address space to operate on
1279  */
1280 void kthread_use_mm(struct mm_struct *mm)
     /*   */
1281 {
1282         struct mm_struct *active_mm;
1283         struct task_struct *tsk = current;
1284 
1285         WARN_ON_ONCE(!(tsk->flags & PF_KTHREAD));
1286         WARN_ON_ONCE(tsk->mm);
1287 
1288         task_lock(tsk);
1289         /* Hold off tlb flush IPIs while switching mm's */
1290         local_irq_disable();
1291         active_mm = tsk->active_mm;
1292         if (active_mm != mm) {
1293                 mmgrab(mm);
1294                 tsk->active_mm = mm;
1295         }
1296         tsk->mm = mm;
1297         switch_mm_irqs_off(active_mm, mm, tsk);
1298         local_irq_enable();
1299         task_unlock(tsk);
1300 #ifdef finish_arch_post_lock_switch
1301         finish_arch_post_lock_switch();
1302 #endif
1303 
1304         if (active_mm != mm)
1305                 mmdrop(active_mm);
1306 
1307         to_kthread(tsk)->oldfs = force_uaccess_begin();
1308 }
1309 EXPORT_SYMBOL_GPL(kthread_use_mm);
1310 
1311 /**
1312  * kthread_unuse_mm - reverse the effect of kthread_use_mm()
1313  * @mm: address space to operate on
1314  */
1315 void kthread_unuse_mm(struct mm_struct *mm)
     /*   */
1316 {
1317         struct task_struct *tsk = current;
1318 
1319         WARN_ON_ONCE(!(tsk->flags & PF_KTHREAD));
1320         WARN_ON_ONCE(!tsk->mm);
1321 
1322         force_uaccess_end(to_kthread(tsk)->oldfs);
1323 
1324         task_lock(tsk);
1325         sync_mm_rss(mm);
1326         local_irq_disable();
1327         tsk->mm = NULL;
1328         /* active_mm is still 'mm' */
1329         enter_lazy_tlb(mm, tsk);
1330         local_irq_enable();
1331         task_unlock(tsk);
1332 }
1333 EXPORT_SYMBOL_GPL(kthread_unuse_mm);
