  39 static void __synchronize_hardirq(struct irq_desc *desc, bool sync_chip)
  40 {
  41         struct irq_data *irqd = irq_desc_get_irq_data(desc);
  42         bool inprogress;
  43 
  44         do {
  45                 unsigned long flags;
  46 
  47                 /*
  48                  * Wait until we're out of the critical section.  This might
  49                  * give the wrong answer due to the lack of memory barriers.
  50                  */
  51                 while (irqd_irq_inprogress(&desc->irq_data))
  52                         cpu_relax();
  53 
  54                 /* Ok, that indicated we're done: double-check carefully. */
  55                 raw_spin_lock_irqsave(&desc->lock, flags);
  56                 inprogress = irqd_irq_inprogress(&desc->irq_data);
  57 
  58                 /*
  59                  * If requested and supported, check at the chip whether it
  60                  * is in flight at the hardware level, i.e. already pending
  61                  * in a CPU and waiting for service and acknowledge.
  62                  */
  63                 if (!inprogress && sync_chip) {
  64                         /*
  65                          * Ignore the return code. inprogress is only updated
  66                          * when the chip supports it.
  67                          */
  68                         __irq_get_irqchip_state(irqd, IRQCHIP_STATE_ACTIVE,
  69                                                 &inprogress);
  70                 }
  71                 raw_spin_unlock_irqrestore(&desc->lock, flags);
  72 
  73                 /* Oops, that failed? */
  74         } while (inprogress);
  75 }

  77 /**
  78  *      synchronize_hardirq - wait for pending hard IRQ handlers (on other CPUs)
  79  *      @irq: interrupt number to wait for
  80  *
  81  *      This function waits for any pending hard IRQ handlers for this
  82  *      interrupt to complete before returning. If you use this
  83  *      function while holding a resource the IRQ handler may need you
  84  *      will deadlock. It does not take associated threaded handlers
  85  *      into account.
  86  *
  87  *      Do not use this for shutdown scenarios where you must be sure
  88  *      that all parts (hardirq and threaded handler) have completed.
  89  *
  90  *      Returns: false if a threaded handler is active.
  91  *
  92  *      This function may be called - with care - from IRQ context.
  93  *
  94  *      It does not check whether there is an interrupt in flight at the
  95  *      hardware level, but not serviced yet, as this might deadlock when
  96  *      called with interrupts disabled and the target CPU of the interrupt
  97  *      is the current CPU.
  98  */
  99 bool synchronize_hardirq(unsigned int irq)
 100 {
 101         struct irq_desc *desc = irq_to_desc(irq);
 102 
 103         if (desc) {
 104                 __synchronize_hardirq(desc, false);
 105                 return !atomic_read(&desc->threads_active);
 106         }
 107 
 108         return true;
 109 }
 110 EXPORT_SYMBOL(synchronize_hardirq);

 112 /**
 113  *      synchronize_irq - wait for pending IRQ handlers (on other CPUs)
 114  *      @irq: interrupt number to wait for
 115  *
 116  *      This function waits for any pending IRQ handlers for this interrupt
 117  *      to complete before returning. If you use this function while
 118  *      holding a resource the IRQ handler may need you will deadlock.
 119  *
 120  *      Can only be called from preemptible code as it might sleep when
 121  *      an interrupt thread is associated to @irq.
 122  *
 123  *      It optionally makes sure (when the irq chip supports that method)
 124  *      that the interrupt is not pending in any CPU and waiting for
 125  *      service.
 126  */
 127 void synchronize_irq(unsigned int irq)
 128 {
 129         struct irq_desc *desc = irq_to_desc(irq);
 130 
 131         if (desc) {
 132                 __synchronize_hardirq(desc, true);
 133                 /*
 134                  * We made sure that no hardirq handler is
 135                  * running. Now verify that no threaded handlers are
 136                  * active.
 137                  */
 138                 wait_event(desc->wait_for_threads,
 139                            !atomic_read(&desc->threads_active));
 140         }
 141 }
 142 EXPORT_SYMBOL(synchronize_irq);

 221 int irq_do_set_affinity(struct irq_data *data, const struct cpumask *mask,
 222                         bool force)
 223 {
 224         struct irq_desc *desc = irq_data_to_desc(data);
 225         struct irq_chip *chip = irq_data_get_irq_chip(data);
 226         int ret;
 227 
 228         if (!chip || !chip->irq_set_affinity)
 229                 return -EINVAL;
 230 
 231         /*
 232          * If this is a managed interrupt and housekeeping is enabled on
 233          * it check whether the requested affinity mask intersects with
 234          * a housekeeping CPU. If so, then remove the isolated CPUs from
 235          * the mask and just keep the housekeeping CPU(s). This prevents
 236          * the affinity setter from routing the interrupt to an isolated
 237          * CPU to avoid that I/O submitted from a housekeeping CPU causes
 238          * interrupts on an isolated one.
 239          *
 240          * If the masks do not intersect or include online CPU(s) then
 241          * keep the requested mask. The isolated target CPUs are only
 242          * receiving interrupts when the I/O operation was submitted
 243          * directly from them.
 244          *
 245          * If all housekeeping CPUs in the affinity mask are offline, the
 246          * interrupt will be migrated by the CPU hotplug code once a
 247          * housekeeping CPU which belongs to the affinity mask comes
 248          * online.
 249          */
 250         if (irqd_affinity_is_managed(data) &&
 251             housekeeping_enabled(HK_FLAG_MANAGED_IRQ)) {
 252                 const struct cpumask *hk_mask, *prog_mask;
 253 
 254                 static DEFINE_RAW_SPINLOCK(tmp_mask_lock);
 255                 static struct cpumask tmp_mask;
 256 
 257                 hk_mask = housekeeping_cpumask(HK_FLAG_MANAGED_IRQ);
 258 
 259                 raw_spin_lock(&tmp_mask_lock);
 260                 cpumask_and(&tmp_mask, mask, hk_mask);
 261                 if (!cpumask_intersects(&tmp_mask, cpu_online_mask))
 262                         prog_mask = mask;
 263                 else
 264                         prog_mask = &tmp_mask;
 265                 ret = chip->irq_set_affinity(data, prog_mask, force);
 266                 raw_spin_unlock(&tmp_mask_lock);
 267         } else {
 268                 ret = chip->irq_set_affinity(data, mask, force);
 269         }
 270         switch (ret) {
 271         case IRQ_SET_MASK_OK:
 272         case IRQ_SET_MASK_OK_DONE:
 273                 cpumask_copy(desc->irq_common_data.affinity, mask);
 274                 fallthrough;
 275         case IRQ_SET_MASK_OK_NOCOPY:
 276                 irq_validate_effective_affinity(data);
 277                 irq_set_thread_affinity(desc);
 278                 ret = 0;
 279         }
 280 
 281         return ret;
 282 }
 283 EXPORT_SYMBOL_GPL(irq_do_set_affinity);

 342 int irq_set_affinity_locked(struct irq_data *data, const struct cpumask *mask,
 343                             bool force)
 344 {
 345         struct irq_chip *chip = irq_data_get_irq_chip(data);
 346         struct irq_desc *desc = irq_data_to_desc(data);
 347         int ret = 0;
 348 
 349         if (!chip || !chip->irq_set_affinity)
 350                 return -EINVAL;
 351 
 352         if (irq_set_affinity_deactivated(data, mask, force))
 353                 return 0;
 354 
 355         if (irq_can_move_pcntxt(data) && !irqd_is_setaffinity_pending(data)) {
 356                 ret = irq_try_set_affinity(data, mask, force);
 357         } else {
 358                 irqd_set_move_pending(data);
 359                 irq_copy_pending(desc, mask);
 360         }
 361 
 362         if (desc->affinity_notify) {
 363                 kref_get(&desc->affinity_notify->kref);
 364                 if (!schedule_work(&desc->affinity_notify->work)) {
 365                         /* Work was already scheduled, drop our extra ref */
 366                         kref_put(&desc->affinity_notify->kref,
 367                                  desc->affinity_notify->release);
 368                 }
 369         }
 370         irqd_set(data, IRQD_AFFINITY_SET);
 371 
 372         return ret;
 373 }

 406 static void irq_affinity_notify(struct work_struct *work)
 407 {
 408         struct irq_affinity_notify *notify =
 409                 container_of(work, struct irq_affinity_notify, work);
 410         struct irq_desc *desc = irq_to_desc(notify->irq);
 411         cpumask_var_t cpumask;
 412         unsigned long flags;
 413 
 414         if (!desc || !alloc_cpumask_var(&cpumask, GFP_KERNEL))
 415                 goto out;
 416 
 417         raw_spin_lock_irqsave(&desc->lock, flags);
 418         if (irq_move_pending(&desc->irq_data))
 419                 irq_get_pending(cpumask, desc);
 420         else
 421                 cpumask_copy(cpumask, desc->irq_common_data.affinity);
 422         raw_spin_unlock_irqrestore(&desc->lock, flags);
 423 
 424         notify->notify(notify, cpumask);
 425 
 426         free_cpumask_var(cpumask);
 427 out:
 428         kref_put(&notify->kref, notify->release);
 429 }

 431 /**
 432  *      irq_set_affinity_notifier - control notification of IRQ affinity changes
 433  *      @irq:           Interrupt for which to enable/disable notification
 434  *      @notify:        Context for notification, or %NULL to disable
 435  *                      notification.  Function pointers must be initialised;
 436  *                      the other fields will be initialised by this function.
 437  *
 438  *      Must be called in process context.  Notification may only be enabled
 439  *      after the IRQ is allocated and must be disabled before the IRQ is
 440  *      freed using free_irq().
 441  */
 442 int
 443 irq_set_affinity_notifier(unsigned int irq, struct irq_affinity_notify *notify)
 444 {
 445         struct irq_desc *desc = irq_to_desc(irq);
 446         struct irq_affinity_notify *old_notify;
 447         unsigned long flags;
 448 
 449         /* The release function is promised process context */
 450         might_sleep();
 451 
 452         if (!desc || desc->istate & IRQS_NMI)
 453                 return -EINVAL;
 454 
 455         /* Complete initialisation of *notify */
 456         if (notify) {
 457                 notify->irq = irq;
 458                 kref_init(&notify->kref);
 459                 INIT_WORK(&notify->work, irq_affinity_notify);
 460         }
 461 
 462         raw_spin_lock_irqsave(&desc->lock, flags);
 463         old_notify = desc->affinity_notify;
 464         desc->affinity_notify = notify;
 465         raw_spin_unlock_irqrestore(&desc->lock, flags);
 466 
 467         if (old_notify) {
 468                 if (cancel_work_sync(&old_notify->work)) {
 469                         /* Pending work had a ref, put that one too */
 470                         kref_put(&old_notify->kref, old_notify->release);
 471                 }
 472                 kref_put(&old_notify->kref, old_notify->release);
 473         }
 474 
 475         return 0;
 476 }

 480 /*
 481  * Generic version of the affinity autoselector.

 483 int irq_setup_affinity(struct irq_desc *desc)
 484 {
 485         struct cpumask *set = irq_default_affinity;
 486         int ret, node = irq_desc_get_node(desc);
 487         static DEFINE_RAW_SPINLOCK(mask_lock);
 488         static struct cpumask mask;
 489 
 490         /* Excludes PER_CPU and NO_BALANCE interrupts */
 491         if (!__irq_can_set_affinity(desc))
 492                 return 0;
 493 
 494         raw_spin_lock(&mask_lock);
 495         /*
 496          * Preserve the managed affinity setting and a userspace affinity
 497          * setup, but make sure that one of the targets is online.
 498          */
 499         if (irqd_affinity_is_managed(&desc->irq_data) ||
 500             irqd_has_set(&desc->irq_data, IRQD_AFFINITY_SET)) {
 501                 if (cpumask_intersects(desc->irq_common_data.affinity,
 502                                        cpu_online_mask))
 503                         set = desc->irq_common_data.affinity;
 504                 else
 505                         irqd_clear(&desc->irq_data, IRQD_AFFINITY_SET);
 506         }
 507 
 508         cpumask_and(&mask, cpu_online_mask, set);
 509         if (cpumask_empty(&mask))
 510                 cpumask_copy(&mask, cpu_online_mask);
 511 
 512         if (node != NUMA_NO_NODE) {
 513                 const struct cpumask *nodemask = cpumask_of_node(node);
 514 
 515                 /* make sure at least one of the cpus in nodemask is online */
 516                 if (cpumask_intersects(&mask, nodemask))
 517                         cpumask_and(&mask, &mask, nodemask);
 518         }
 519         ret = irq_do_set_affinity(&desc->irq_data, &mask, false);
 520         raw_spin_unlock(&mask_lock);
 521         return ret;
 522 }

 671 {

 691                 break;
 692         }
 693         default:
 694                 desc->depth--;
 695         }

 754 /**
 755  *      irq_set_irq_wake - control irq power management wakeup
 756  *      @irq:   interrupt to control
 757  *      @on:    enable/disable power management wakeup
 758  *
 759  *      Enable/disable power management wakeup mode, which is
 760  *      disabled by default.  Enables and disables must match,
 761  *      just as they match for non-wakeup mode support.
 762  *
 763  *      Wakeup mode lets this IRQ wake the system from sleep
 764  *      states like "suspend to RAM".
 765  *
 766  *      Note: irq enable/disable state is completely orthogonal
 767  *      to the enable/disable state of irq wake. An irq can be
 768  *      disabled with disable_irq() and still wake the system as
 769  *      long as the irq has wake enabled. If this does not hold,
 770  *      then the underlying irq chip and the related driver need
 771  *      to be investigated.
 772  */
 773 int irq_set_irq_wake(unsigned int irq, unsigned int on)
 774 {
 775         unsigned long flags;
 776         struct irq_desc *desc = irq_get_desc_buslock(irq, &flags, IRQ_GET_DESC_CHECK_GLOBAL);
 777         int ret = 0;
 778 
 779         if (!desc)
 780                 return -EINVAL;
 781 
 782         /* Don't use NMIs as wake up interrupts please */
 783         if (desc->istate & IRQS_NMI) {
 784                 ret = -EINVAL;
 785                 goto out_unlock;
 786         }
 787 
 788         /* wakeup-capable irqs can be shared between drivers that
 789          * don't need to have the same sleep mode behaviors.
 790          */
 791         if (on) {
 792                 if (desc->wake_depth++ == 0) {
 793                         ret = set_irq_wake_real(irq, on);
 794                         if (ret)
 795                                 desc->wake_depth = 0;
 796                         else
 797                                 irqd_set(&desc->irq_data, IRQD_WAKEUP_STATE);
 798                 }
 799         } else {
 800                 if (desc->wake_depth == 0) {
 801                         WARN(1, "Unbalanced IRQ %d wake disable\n", irq);
 802                 } else if (--desc->wake_depth == 0) {
 803                         ret = set_irq_wake_real(irq, on);
 804                         if (ret)
 805                                 desc->wake_depth = 1;
 806                         else
 807                                 irqd_clear(&desc->irq_data, IRQD_WAKEUP_STATE);
 808                 }
 809         }
 810 
 811 out_unlock:
 812         irq_put_desc_busunlock(desc, flags);
 813         return ret;
 814 }
 815 EXPORT_SYMBOL(irq_set_irq_wake);

 840 int __irq_set_trigger(struct irq_desc *desc, unsigned long flags)
 841 {
 842         struct irq_chip *chip = desc->irq_data.chip;
 843         int ret, unmask = 0;
 844 
 845         if (!chip || !chip->irq_set_type) {
 846                 /*
 847                  * IRQF_TRIGGER_* but the PIC does not support multiple
 848                  * flow-types?
 849                  */
 850                 pr_debug("No set_type function for IRQ %d (%s)\n",
 851                          irq_desc_get_irq(desc),
 852                          chip ? (chip->name ? : "unknown") : "unknown");
 853                 return 0;
 854         }
 855 
 856         if (chip->flags & IRQCHIP_SET_TYPE_MASKED) {
 857                 if (!irqd_irq_masked(&desc->irq_data))
 858                         mask_irq(desc);
 859                 if (!irqd_irq_disabled(&desc->irq_data))
 860                         unmask = 1;
 861         }
 862 
 863         /* Mask all flags except trigger mode */
 864         flags &= IRQ_TYPE_SENSE_MASK;
 865         ret = chip->irq_set_type(&desc->irq_data, flags);
 866 
 867         switch (ret) {
 868         case IRQ_SET_MASK_OK:
 869         case IRQ_SET_MASK_OK_DONE:
 870                 irqd_clear(&desc->irq_data, IRQD_TRIGGER_MASK);
 871                 irqd_set(&desc->irq_data, flags);
 872                 fallthrough;
 873 
 874         case IRQ_SET_MASK_OK_NOCOPY:
 875                 flags = irqd_get_trigger_type(&desc->irq_data);
 876                 irq_settings_set_trigger_mask(desc, flags);
 877                 irqd_clear(&desc->irq_data, IRQD_LEVEL);
 878                 irq_settings_clr_level(desc);
 879                 if (flags & IRQ_TYPE_LEVEL_MASK) {
 880                         irq_settings_set_level(desc);
 881                         irqd_set(&desc->irq_data, IRQD_LEVEL);
 882                 }
 883 
 884                 ret = 0;
 885                 break;
 886         default:
 887                 pr_err("Setting trigger mode %lu for irq %u failed (%pS)\n",
 888                        flags, irq_desc_get_irq(desc), chip->irq_set_type);
 889         }
 890         if (unmask)
 891                 unmask_irq(desc);
 892         return ret;
 893 }

 938 static int irq_wait_for_interrupt(struct irqaction *action)
 939 {
 940         for (;;) {
 941                 set_current_state(TASK_INTERRUPTIBLE);
 942 
 943                 if (kthread_should_stop()) {
 944                         /* may need to run one last time */
 945                         if (test_and_clear_bit(IRQTF_RUNTHREAD,
 946                                                &action->thread_flags)) {
 947                                 __set_current_state(TASK_RUNNING);
 948                                 return 0;
 949                         }
 950                         __set_current_state(TASK_RUNNING);
 951                         return -1;
 952                 }
 953 
 954                 if (test_and_clear_bit(IRQTF_RUNTHREAD,
 955                                        &action->thread_flags)) {
 956                         __set_current_state(TASK_RUNNING);
 957                         return 0;
 958                 }
 959                 schedule();
 960         }
 961 }

 963 /*
 964  * Oneshot interrupts keep the irq line masked until the threaded
 965  * handler finished. unmask if the interrupt has not been disabled and
 966  * is marked MASKED.
 967  */
 968 static void irq_finalize_oneshot(struct irq_desc *desc,
 969                                  struct irqaction *action)
 970 {
 971         if (!(desc->istate & IRQS_ONESHOT) ||
 972             action->handler == irq_forced_secondary_handler)
 973                 return;
 974 again:
 975         chip_bus_lock(desc);
 976         raw_spin_lock_irq(&desc->lock);
 977 
 978         /*
 979          * Implausible though it may be we need to protect us against
 980          * the following scenario:
 981          *
 982          * The thread is faster done than the hard interrupt handler
 983          * on the other CPU. If we unmask the irq line then the
 984          * interrupt can come in again and masks the line, leaves due
 985          * to IRQS_INPROGRESS and the irq line is masked forever.
 986          *
 987          * This also serializes the state of shared oneshot handlers
 988          * versus "desc->threads_onehsot |= action->thread_mask;" in
 989          * irq_wake_thread(). See the comment there which explains the
 990          * serialization.
 991          */
 992         if (unlikely(irqd_irq_inprogress(&desc->irq_data))) {
 993                 raw_spin_unlock_irq(&desc->lock);
 994                 chip_bus_sync_unlock(desc);
 995                 cpu_relax();
 996                 goto again;
 997         }
 998 
 999         /*
1000          * Now check again, whether the thread should run. Otherwise
1001          * we would clear the threads_oneshot bit of this thread which
1002          * was just set.
1003          */
1004         if (test_bit(IRQTF_RUNTHREAD, &action->thread_flags))
1005                 goto out_unlock;
1006 
1007         desc->threads_oneshot &= ~action->thread_mask;
1008 
1009         if (!desc->threads_oneshot && !irqd_irq_disabled(&desc->irq_data) &&
1010             irqd_irq_masked(&desc->irq_data))
1011                 unmask_threaded_irq(desc);
1012 
1013 out_unlock:
1014         raw_spin_unlock_irq(&desc->lock);
1015         chip_bus_sync_unlock(desc);
1016 }

1152 /*
1153  * Interrupt handler thread
1154  */
1155 static int irq_thread(void *data)
1156 {
1157         struct callback_head on_exit_work;
1158         struct irqaction *action = data;
1159         struct irq_desc *desc = irq_to_desc(action->irq);
1160         irqreturn_t (*handler_fn)(struct irq_desc *desc,
1161                         struct irqaction *action);
1162 
1163         if (force_irqthreads && test_bit(IRQTF_FORCED_THREAD,
1164                                         &action->thread_flags))
1165                 handler_fn = irq_forced_thread_fn;
1166         else
1167                 handler_fn = irq_thread_fn;
1168 
1169         init_task_work(&on_exit_work, irq_thread_dtor);
1170         task_work_add(current, &on_exit_work, TWA_NONE);
1171 
1172         irq_thread_check_affinity(desc, action);
1173 
1174         while (!irq_wait_for_interrupt(action)) {
1175                 irqreturn_t action_ret;
1176 
1177                 irq_thread_check_affinity(desc, action);
1178 
1179                 action_ret = handler_fn(desc, action);
1180                 if (action_ret == IRQ_WAKE_THREAD)
1181                         irq_wake_secondary(desc, action);
1182 
1183                 wake_threads_waitq(desc);
1184         }
1185 
1186         /*
1187          * This is the regular exit path. __free_irq() is stopping the
1188          * thread via kthread_stop() after calling
1189          * synchronize_hardirq(). So neither IRQTF_RUNTHREAD nor the
1190          * oneshot mask bit can be set.
1191          */
1192         task_work_cancel(current, irq_thread_dtor);
1193         return 0;
1194 }

1223 static int irq_setup_forced_threading(struct irqaction *new)
1224 {
1225         if (!force_irqthreads)
1226                 return 0;
1227         if (new->flags & (IRQF_NO_THREAD | IRQF_PERCPU | IRQF_ONESHOT))
1228                 return 0;
1229 
1230         /*
1231          * No further action required for interrupts which are requested as
1232          * threaded interrupts already
1233          */
1234         if (new->handler == irq_default_primary_handler)
1235                 return 0;
1236 
1237         new->flags |= IRQF_ONESHOT;
1238 
1239         /*
1240          * Handle the case where we have a real primary handler and a
1241          * thread handler. We force thread them as well by creating a
1242          * secondary action.
1243          */
1244         if (new->handler && new->thread_fn) {
1245                 /* Allocate the secondary action */
1246                 new->secondary = kzalloc(sizeof(struct irqaction), GFP_KERNEL);
1247                 if (!new->secondary)
1248                         return -ENOMEM;
1249                 new->secondary->handler = irq_forced_secondary_handler;
1250                 new->secondary->thread_fn = new->thread_fn;
1251                 new->secondary->dev_id = new->dev_id;
1252                 new->secondary->irq = new->irq;
1253                 new->secondary->name = new->name;
1254         }
1255         /* Deal with the primary handler */
1256         set_bit(IRQTF_FORCED_THREAD, &new->thread_flags);
1257         new->thread_fn = new->handler;
1258         new->handler = irq_default_primary_handler;
1259         return 0;
1260 }

1314 {
1315         struct task_struct *t;

1317         if (!secondary) {

1319                                    new->name);

1321                 t = kthread_create(irq_thread, new, "irq/%d-s-%s", irq,
1322                                    new->name);
1323         }
1324

1326                 return PTR_ERR(t);

1331          * We keep the reference to the task struct even if
1332          * the thread dies to avoid that the interrupt code

1349 /*
1350  * Internal function to register an irqaction - typically used to
1351  * allocate special interrupts that are part of the architecture.
1352  *
1353  * Locking rules:
1354  *
1355  * desc->request_mutex  Provides serialization against a concurrent free_irq()
1356  *   chip_bus_lock      Provides serialization for slow bus operations
1357  *     desc->lock       Provides serialization against hard interrupts
1358  *
1359  * chip_bus_lock and desc->lock are sufficient for all other management and
1360  * interrupt related functions. desc->request_mutex solely serializes
1361  * request/free_irq().
1362  */
1363 static int
1364 __setup_irq(unsigned int irq, struct irq_desc *desc, struct irqaction *new)
1365 {
1366         struct irqaction *old, **old_ptr;
1367         unsigned long flags, thread_mask = 0;
1368         int ret, nested, shared = 0;
1369 
1370         if (!desc)
1371                 return -EINVAL;
1372 
1373         if (desc->irq_data.chip == &no_irq_chip)
1374                 return -ENOSYS;
1375         if (!try_module_get(desc->owner))
1376                 return -ENODEV;
1377 
1378         new->irq = irq;
1379 
1380         /*
1381          * If the trigger type is not specified by the caller,
1382          * then use the default for this interrupt.
1383          */
1384         if (!(new->flags & IRQF_TRIGGER_MASK))
1385                 new->flags |= irqd_get_trigger_type(&desc->irq_data);
1386 
1387         /*
1388          * Check whether the interrupt nests into another interrupt
1389          * thread.
1390          */
1391         nested = irq_settings_is_nested_thread(desc);
1392         if (nested) {
1393                 if (!new->thread_fn) {
1394                         ret = -EINVAL;
1395                         goto out_mput;
1396                 }
1397                 /*
1398                  * Replace the primary handler which was provided from
1399                  * the driver for non nested interrupt handling by the
1400                  * dummy function which warns when called.
1401                  */
1402                 new->handler = irq_nested_primary_handler;
1403         } else {
1404                 if (irq_settings_can_thread(desc)) {
1405                         ret = irq_setup_forced_threading(new);
1406                         if (ret)
1407                                 goto out_mput;
1408                 }
1409         }
1410 
1411         /*
1412          * Create a handler thread when a thread function is supplied
1413          * and the interrupt does not nest into another interrupt
1414          * thread.
1415          */
1416         if (new->thread_fn && !nested) {
1417                 ret = setup_irq_thread(new, irq, false);
1418                 if (ret)
1419                         goto out_mput;
1420                 if (new->secondary) {
1421                         ret = setup_irq_thread(new->secondary, irq, true);
1422                         if (ret)
1423                                 goto out_thread;
1424                 }
1425         }
1426 
1427         /*
1428          * Drivers are often written to work w/o knowledge about the
1429          * underlying irq chip implementation, so a request for a
1430          * threaded irq without a primary hard irq context handler
1431          * requires the ONESHOT flag to be set. Some irq chips like
1432          * MSI based interrupts are per se one shot safe. Check the
1433          * chip flags, so we can avoid the unmask dance at the end of
1434          * the threaded handler for those.
1435          */
1436         if (desc->irq_data.chip->flags & IRQCHIP_ONESHOT_SAFE)
1437                 new->flags &= ~IRQF_ONESHOT;
1438 
1439         /*
1440          * Protects against a concurrent __free_irq() call which might wait
1441          * for synchronize_hardirq() to complete without holding the optional
1442          * chip bus lock and desc->lock. Also protects against handing out
1443          * a recycled oneshot thread_mask bit while it's still in use by
1444          * its previous owner.
1445          */
1446         mutex_lock(&desc->request_mutex);
1447 
1448         /*
1449          * Acquire bus lock as the irq_request_resources() callback below
1450          * might rely on the serialization or the magic power management
1451          * functions which are abusing the irq_bus_lock() callback,
1452          */
1453         chip_bus_lock(desc);
1454 
1455         /* First installed action requests resources. */
1456         if (!desc->action) {
1457                 ret = irq_request_resources(desc);
1458                 if (ret) {
1459                         pr_err("Failed to request resources for %s (irq %d) on irqchip %s\n",
1460                                new->name, irq, desc->irq_data.chip->name);
1461                         goto out_bus_unlock;
1462                 }
1463         }
1464 
1465         /*
1466          * The following block of code has to be executed atomically
1467          * protected against a concurrent interrupt and any of the other
1468          * management calls which are not serialized via
1469          * desc->request_mutex or the optional bus lock.
1470          */
1471         raw_spin_lock_irqsave(&desc->lock, flags);
1472         old_ptr = &desc->action;
1473         old = *old_ptr;
1474         if (old) {
1475                 /*
1476                  * Can't share interrupts unless both agree to and are
1477                  * the same type (level, edge, polarity). So both flag
1478                  * fields must have IRQF_SHARED set and the bits which
1479                  * set the trigger type must match. Also all must
1480                  * agree on ONESHOT.
1481                  * Interrupt lines used for NMIs cannot be shared.
1482                  */
1483                 unsigned int oldtype;
1484 
1485                 if (desc->istate & IRQS_NMI) {
1486                         pr_err("Invalid attempt to share NMI for %s (irq %d) on irqchip %s.\n",
1487                                 new->name, irq, desc->irq_data.chip->name);
1488                         ret = -EINVAL;
1489                         goto out_unlock;
1490                 }
1491 
1492                 /*
1493                  * If nobody did set the configuration before, inherit
1494                  * the one provided by the requester.
1495                  */
1496                 if (irqd_trigger_type_was_set(&desc->irq_data)) {
1497                         oldtype = irqd_get_trigger_type(&desc->irq_data);
1498                 } else {
1499                         oldtype = new->flags & IRQF_TRIGGER_MASK;
1500                         irqd_set_trigger_type(&desc->irq_data, oldtype);
1501                 }
1502 
1503                 if (!((old->flags & new->flags) & IRQF_SHARED) ||
1504                     (oldtype != (new->flags & IRQF_TRIGGER_MASK)) ||
1505                     ((old->flags ^ new->flags) & IRQF_ONESHOT))
1506                         goto mismatch;
1507 
1508                 /* All handlers must agree on per-cpuness */
1509                 if ((old->flags & IRQF_PERCPU) !=
1510                     (new->flags & IRQF_PERCPU))
1511                         goto mismatch;
1512 
1513                 /* add new interrupt at end of irq queue */
1514                 do {
1515                         /*
1516                          * Or all existing action->thread_mask bits,
1517                          * so we can find the next zero bit for this
1518                          * new action.
1519                          */
1520                         thread_mask |= old->thread_mask;
1521                         old_ptr = &old->next;
1522                         old = *old_ptr;
1523                 } while (old);
1524                 shared = 1;
1525         }
1526 
1527         /*
1528          * Setup the thread mask for this irqaction for ONESHOT. For
1529          * !ONESHOT irqs the thread mask is 0 so we can avoid a
1530          * conditional in irq_wake_thread().
1531          */
1532         if (new->flags & IRQF_ONESHOT) {
1533                 /*
1534                  * Unlikely to have 32 resp 64 irqs sharing one line,
1535                  * but who knows.
1536                  */
1537                 if (thread_mask == ~0UL) {
1538                         ret = -EBUSY;
1539                         goto out_unlock;
1540                 }
1541                 /*
1542                  * The thread_mask for the action is or'ed to
1543                  * desc->thread_active to indicate that the
1544                  * IRQF_ONESHOT thread handler has been woken, but not
1545                  * yet finished. The bit is cleared when a thread
1546                  * completes. When all threads of a shared interrupt
1547                  * line have completed desc->threads_active becomes
1548                  * zero and the interrupt line is unmasked. See
1549                  * handle.c:irq_wake_thread() for further information.
1550                  *
1551                  * If no thread is woken by primary (hard irq context)
1552                  * interrupt handlers, then desc->threads_active is
1553                  * also checked for zero to unmask the irq line in the
1554                  * affected hard irq flow handlers
1555                  * (handle_[fasteoi|level]_irq).
1556                  *
1557                  * The new action gets the first zero bit of
1558                  * thread_mask assigned. See the loop above which or's
1559                  * all existing action->thread_mask bits.
1560                  */
1561                 new->thread_mask = 1UL << ffz(thread_mask);
1562 
1563         } else if (new->handler == irq_default_primary_handler &&
1564                    !(desc->irq_data.chip->flags & IRQCHIP_ONESHOT_SAFE)) {
1565                 /*
1566                  * The interrupt was requested with handler = NULL, so
1567                  * we use the default primary handler for it. But it
1568                  * does not have the oneshot flag set. In combination
1569                  * with level interrupts this is deadly, because the
1570                  * default primary handler just wakes the thread, then
1571                  * the irq lines is reenabled, but the device still
1572                  * has the level irq asserted. Rinse and repeat....
1573                  *
1574                  * While this works for edge type interrupts, we play
1575                  * it safe and reject unconditionally because we can't
1576                  * say for sure which type this interrupt really
1577                  * has. The type flags are unreliable as the
1578                  * underlying chip implementation can override them.
1579                  */
1580                 pr_err("Threaded irq requested with handler=NULL and !ONESHOT for %s (irq %d)\n",
1581                        new->name, irq);
1582                 ret = -EINVAL;
1583                 goto out_unlock;
1584         }
1585 
1586         if (!shared) {
1587                 init_waitqueue_head(&desc->wait_for_threads);
1588 
1589                 /* Setup the type (level, edge polarity) if configured: */
1590                 if (new->flags & IRQF_TRIGGER_MASK) {
1591                         ret = __irq_set_trigger(desc,
1592                                                 new->flags & IRQF_TRIGGER_MASK);
1593 
1594                         if (ret)
1595                                 goto out_unlock;
1596                 }
1597 
1598                 /*
1599                  * Activate the interrupt. That activation must happen
1600                  * independently of IRQ_NOAUTOEN. request_irq() can fail
1601                  * and the callers are supposed to handle
1602                  * that. enable_irq() of an interrupt requested with
1603                  * IRQ_NOAUTOEN is not supposed to fail. The activation
1604                  * keeps it in shutdown mode, it merily associates
1605                  * resources if necessary and if that's not possible it
1606                  * fails. Interrupts which are in managed shutdown mode
1607                  * will simply ignore that activation request.
1608                  */
1609                 ret = irq_activate(desc);
1610                 if (ret)
1611                         goto out_unlock;
1612 
1613                 desc->istate &= ~(IRQS_AUTODETECT | IRQS_SPURIOUS_DISABLED | \
1614                                   IRQS_ONESHOT | IRQS_WAITING);
1615                 irqd_clear(&desc->irq_data, IRQD_IRQ_INPROGRESS);
1616 
1617                 if (new->flags & IRQF_PERCPU) {
1618                         irqd_set(&desc->irq_data, IRQD_PER_CPU);
1619                         irq_settings_set_per_cpu(desc);
1620                 }
1621 
1622                 if (new->flags & IRQF_ONESHOT)
1623                         desc->istate |= IRQS_ONESHOT;
1624 
1625                 /* Exclude IRQ from balancing if requested */
1626                 if (new->flags & IRQF_NOBALANCING) {
1627                         irq_settings_set_no_balancing(desc);
1628                         irqd_set(&desc->irq_data, IRQD_NO_BALANCING);
1629                 }
1630 
1631                 if (irq_settings_can_autoenable(desc)) {
1632                         irq_startup(desc, IRQ_RESEND, IRQ_START_COND);
1633                 } else {
1634                         /*
1635                          * Shared interrupts do not go well with disabling
1636                          * auto enable. The sharing interrupt might request
1637                          * it while it's still disabled and then wait for
1638                          * interrupts forever.
1639                          */
1640                         WARN_ON_ONCE(new->flags & IRQF_SHARED);
1641                         /* Undo nested disables: */
1642                         desc->depth = 1;
1643                 }
1644 
1645         } else if (new->flags & IRQF_TRIGGER_MASK) {
1646                 unsigned int nmsk = new->flags & IRQF_TRIGGER_MASK;
1647                 unsigned int omsk = irqd_get_trigger_type(&desc->irq_data);
1648 
1649                 if (nmsk != omsk)
1650                         /* hope the handler works with current  trigger mode */
1651                         pr_warn("irq %d uses trigger mode %u; requested %u\n",
1652                                 irq, omsk, nmsk);
1653         }
1654 
1655         *old_ptr = new;
1656 
1657         irq_pm_install_action(desc, new);
1658 
1659         /* Reset broken irq detection when installing new handler */
1660         desc->irq_count = 0;
1661         desc->irqs_unhandled = 0;
1662 
1663         /*
1664          * Check whether we disabled the irq via the spurious handler
1665          * before. Reenable it and give it another chance.
1666          */
1667         if (shared && (desc->istate & IRQS_SPURIOUS_DISABLED)) {
1668                 desc->istate &= ~IRQS_SPURIOUS_DISABLED;
1669                 __enable_irq(desc);
1670         }
1671 
1672         raw_spin_unlock_irqrestore(&desc->lock, flags);
1673         chip_bus_sync_unlock(desc);
1674         mutex_unlock(&desc->request_mutex);
1675 
1676         irq_setup_timings(desc, new);
1677 
1678         /*
1679          * Strictly no need to wake it up, but hung_task complains
1680          * when no hard interrupt wakes the thread up.
1681          */
1682         if (new->thread)
1683                 wake_up_process(new->thread);
1684         if (new->secondary)
1685                 wake_up_process(new->secondary->thread);
1686 
1687         register_irq_proc(irq, desc);
1688         new->dir = NULL;
1689         register_handler_proc(irq, new);
1690         return 0;
1691 
1692 mismatch:
1693         if (!(new->flags & IRQF_PROBE_SHARED)) {
1694                 pr_err("Flags mismatch irq %d. %08x (%s) vs. %08x (%s)\n",
1695                        irq, new->flags, new->name, old->flags, old->name);
1696 #ifdef CONFIG_DEBUG_SHIRQ
1697                 dump_stack();
1698 #endif
1699         }
1700         ret = -EBUSY;
1701 
1702 out_unlock:
1703         raw_spin_unlock_irqrestore(&desc->lock, flags);
1704 
1705         if (!desc->action)
1706                 irq_release_resources(desc);
1707 out_bus_unlock:
1708         chip_bus_sync_unlock(desc);
1709         mutex_unlock(&desc->request_mutex);
1710 
1711 out_thread:
1712         if (new->thread) {
1713                 struct task_struct *t = new->thread;
1714 
1715                 new->thread = NULL;
1716                 kthread_stop(t);
1717                 put_task_struct(t);
1718         }
1719         if (new->secondary && new->secondary->thread) {
1720                 struct task_struct *t = new->secondary->thread;
1721 
1722                 new->secondary->thread = NULL;
1723                 kthread_stop(t);
1724                 put_task_struct(t);
1725         }
1726 out_mput:
1727         module_put(desc->owner);
1728         return ret;
1729 }

1731 /*
1732  * Internal function to unregister an irqaction - used to free
1733  * regular and special interrupts that are part of the architecture.
1734  */
1735 static struct irqaction *__free_irq(struct irq_desc *desc, void *dev_id)
1736 {
1737         unsigned irq = desc->irq_data.irq;
1738         struct irqaction *action, **action_ptr;
1739         unsigned long flags;
1740 
1741         WARN(in_interrupt(), "Trying to free IRQ %d from IRQ context!\n", irq);
1742 
1743         mutex_lock(&desc->request_mutex);
1744         chip_bus_lock(desc);
1745         raw_spin_lock_irqsave(&desc->lock, flags);
1746 
1747         /*
1748          * There can be multiple actions per IRQ descriptor, find the right
1749          * one based on the dev_id:
1750          */
1751         action_ptr = &desc->action;
1752         for (;;) {
1753                 action = *action_ptr;
1754 
1755                 if (!action) {
1756                         WARN(1, "Trying to free already-free IRQ %d\n", irq);
1757                         raw_spin_unlock_irqrestore(&desc->lock, flags);
1758                         chip_bus_sync_unlock(desc);
1759                         mutex_unlock(&desc->request_mutex);
1760                         return NULL;
1761                 }
1762 
1763                 if (action->dev_id == dev_id)
1764                         break;
1765                 action_ptr = &action->next;
1766         }
1767 
1768         /* Found it - now remove it from the list of entries: */
1769         *action_ptr = action->next;
1770 
1771         irq_pm_remove_action(desc, action);
1772 
1773         /* If this was the last handler, shut down the IRQ line: */
1774         if (!desc->action) {
1775                 irq_settings_clr_disable_unlazy(desc);
1776                 /* Only shutdown. Deactivate after synchronize_hardirq() */
1777                 irq_shutdown(desc);
1778         }
1779 
1780 #ifdef CONFIG_SMP
1781         /* make sure affinity_hint is cleaned up */
1782         if (WARN_ON_ONCE(desc->affinity_hint))
1783                 desc->affinity_hint = NULL;
1784 #endif
1785 
1786         raw_spin_unlock_irqrestore(&desc->lock, flags);
1787         /*
1788          * Drop bus_lock here so the changes which were done in the chip
1789          * callbacks above are synced out to the irq chips which hang
1790          * behind a slow bus (I2C, SPI) before calling synchronize_hardirq().
1791          *
1792          * Aside of that the bus_lock can also be taken from the threaded
1793          * handler in irq_finalize_oneshot() which results in a deadlock
1794          * because kthread_stop() would wait forever for the thread to
1795          * complete, which is blocked on the bus lock.
1796          *
1797          * The still held desc->request_mutex() protects against a
1798          * concurrent request_irq() of this irq so the release of resources
1799          * and timing data is properly serialized.
1800          */
1801         chip_bus_sync_unlock(desc);
1802 
1803         unregister_handler_proc(irq, action);
1804 
1805         /*
1806          * Make sure it's not being used on another CPU and if the chip
1807          * supports it also make sure that there is no (not yet serviced)
1808          * interrupt in flight at the hardware level.
1809          */
1810         __synchronize_hardirq(desc, true);
1811 
1812 #ifdef CONFIG_DEBUG_SHIRQ
1813         /*
1814          * It's a shared IRQ -- the driver ought to be prepared for an IRQ
1815          * event to happen even now it's being freed, so let's make sure that
1816          * is so by doing an extra call to the handler ....
1817          *
1818          * ( We do this after actually deregistering it, to make sure that a
1819          *   'real' IRQ doesn't run in parallel with our fake. )
1820          */
1821         if (action->flags & IRQF_SHARED) {
1822                 local_irq_save(flags);
1823                 action->handler(irq, dev_id);
1824                 local_irq_restore(flags);
1825         }
1826 #endif
1827 
1828         /*
1829          * The action has already been removed above, but the thread writes
1830          * its oneshot mask bit when it completes. Though request_mutex is
1831          * held across this which prevents __setup_irq() from handing out
1832          * the same bit to a newly requested action.
1833          */
1834         if (action->thread) {
1835                 kthread_stop(action->thread);
1836                 put_task_struct(action->thread);
1837                 if (action->secondary && action->secondary->thread) {
1838                         kthread_stop(action->secondary->thread);
1839                         put_task_struct(action->secondary->thread);
1840                 }
1841         }
1842 
1843         /* Last action releases resources */
1844         if (!desc->action) {
1845                 /*
1846                  * Reaquire bus lock as irq_release_resources() might
1847                  * require it to deallocate resources over the slow bus.
1848                  */
1849                 chip_bus_lock(desc);
1850                 /*
1851                  * There is no interrupt on the fly anymore. Deactivate it
1852                  * completely.
1853                  */
1854                 raw_spin_lock_irqsave(&desc->lock, flags);
1855                 irq_domain_deactivate_irq(&desc->irq_data);
1856                 raw_spin_unlock_irqrestore(&desc->lock, flags);
1857 
1858                 irq_release_resources(desc);
1859                 chip_bus_sync_unlock(desc);
1860                 irq_remove_timings(desc);
1861         }
1862 
1863         mutex_unlock(&desc->request_mutex);
1864 
1865         irq_chip_pm_put(&desc->irq_data);
1866         module_put(desc->owner);
1867         kfree(action->secondary);
1868         return action;
1869 }

1871 /**
1872  *      free_irq - free an interrupt allocated with request_irq
1873  *      @irq: Interrupt line to free
1874  *      @dev_id: Device identity to free
1875  *
1876  *      Remove an interrupt handler. The handler is removed and if the
1877  *      interrupt line is no longer in use by any driver it is disabled.
1878  *      On a shared IRQ the caller must ensure the interrupt is disabled

1965 /**

1970  *                If NULL and thread_fn != NULL the default
1971  *                primary handler is installed

1973  *                  If NULL, no irq thread is created
1974  *      @irqflags: Interrupt type flags
1975  *      @devname: An ascii name for the claiming device
1976  *      @dev_id: A cookie passed back to the handler function

1978  *      This call allocates interrupt resources and enables the
1979  *      interrupt line and IRQ handling. From the point this
1980  *      call is made your handler function may be invoked. Since
1981  *      your handler function must clear any interrupt the board
1982  *      raises, you must take care both to initialise your hardware
1983  *      and to set up the interrupt handler in the right order.
1984  *

1995  *      device data structure is used as the cookie. Since the handler
1996  *      receives this value it makes sense to use it.

1999  *      as this is required when freeing the interrupt.
2000  *
2001  *      Flags:
2002  *

2004  *      IRQF_TRIGGER_*          Specify active edge(s) or level

2010 {

2012         struct irq_desc *desc;
2013         int retval;
2014 
2015         if (irq == IRQ_NOTCONNECTED)
2016                 return -ENOTCONN;
2017

2020          * otherwise we'll have trouble later trying to figure out
2021          * which interrupt is which (messes up the interrupt freeing
2022          * logic etc).
2023          *
2024          * Also IRQF_COND_SUSPEND only makes sense for shared interrupts and
2025          * it cannot be set along with IRQF_NO_SUSPEND.
2026          */
2027         if (((irqflags & IRQF_SHARED) && !dev_id) ||
2028             (!(irqflags & IRQF_SHARED) && (irqflags & IRQF_COND_SUSPEND)) ||

2030                 return -EINVAL;
2031

2033         if (!desc)
2034                 return -EINVAL;
2035 
2036         if (!irq_settings_can_request(desc) ||
2037             WARN_ON(irq_settings_is_per_cpu_devid(desc)))
2038                 return -EINVAL;
2039 
2040         if (!handler) {
2041                 if (!thread_fn)
2042                         return -EINVAL;
2043                 handler = irq_default_primary_handler;
2044         }
2045 
2046         action = kzalloc(sizeof(struct irqaction), GFP_KERNEL);
2047         if (!action)
2048                 return -ENOMEM;

2051         action->thread_fn = thread_fn;
2052         action->flags = irqflags;
2053         action->name = devname;
2054         action->dev_id = dev_id;
2055 
2056         retval = irq_chip_pm_get(&desc->irq_data);
2057         if (retval < 0) {
2058                 kfree(action);
2059                 return retval;
2060         }
2061 
2062         retval = __setup_irq(irq, desc, action);
2063 
2064         if (retval) {
2065                 irq_chip_pm_put(&desc->irq_data);
2066                 kfree(action->secondary);
2067                 kfree(action);
2068         }
2069 
2070 #ifdef CONFIG_DEBUG_SHIRQ_FIXME

2076                  * run in parallel with our fake.

2080                 disable_irq(irq);

2082

2084
