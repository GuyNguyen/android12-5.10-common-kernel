 127 static __always_inline bool
 128 mod_tree_less(struct latch_tree_node *a, struct latch_tree_node *b)
 129 {
 130         return __mod_tree_val(a) < __mod_tree_val(b);
 131 }

 133 static __always_inline int
 134 mod_tree_comp(void *key, struct latch_tree_node *n)
 135 {
 136         unsigned long val = (unsigned long)key;
 137         unsigned long start, end;
 138 
 139         start = __mod_tree_val(n);
 140         if (val < start)
 141                 return -1;
 142 
 143         end = start + __mod_tree_size(n);
 144         if (val >= end)
 145                 return 1;
 146 
 147         return 0;
 148 }

 351 /* Find a module section: 0 means not found. */
 352 static unsigned int find_sec(const struct load_info *info, const char *name)
 353 {
 354         unsigned int i;
 355 
 356         for (i = 1; i < info->hdr->e_shnum; i++) {
 357                 Elf_Shdr *shdr = &info->sechdrs[i];
 358                 /* Alloc bit cleared means "ignore it." */
 359                 if ((shdr->sh_flags & SHF_ALLOC)
 360                     && strcmp(info->secstrings + shdr->sh_name, name) == 0)
 361                         return i;
 362         }
 363         return 0;
 364 }

 932 /* Try to release refcount of module, 0 means success. */
 933 static int try_release_module_ref(struct module *mod)
 934 {
 935         int ret;
 936 
 937         /* Try to decrement refcnt which we set at loading */
 938         ret = atomic_sub_return(MODULE_REF_BASE, &mod->refcnt);
 939         BUG_ON(ret < 0);
 940         if (ret)
 941                 /* Someone can put this right now, recover with checking */
 942                 ret = atomic_add_unless(&mod->refcnt, MODULE_REF_BASE, 0);
 943 
 944         return ret;
 945 }

 947 static int try_stop_module(struct module *mod, int flags, int *forced)
 948 {
 949         /* If it's not unused, quit unless we're forcing. */
 950         if (try_release_module_ref(mod) != 0) {
 951                 *forced = try_force_unload(flags);
 952                 if (!(*forced))
 953                         return -EWOULDBLOCK;
 954         }
 955 
 956         /* Mark it as dying. */
 957         mod->state = MODULE_STATE_GOING;
 958 
 959         return 0;
 960 }

 980 SYSCALL_DEFINE2(delete_module, const char __user *, name_user,
 981                 unsigned int, flags)
 982 {
 983         struct module *mod;
 984         char name[MODULE_NAME_LEN];
 985         int ret, forced = 0;
 986 
 987         if (!capable(CAP_SYS_MODULE) || modules_disabled)
 988                 return -EPERM;
 989 
 990         if (strncpy_from_user(name, name_user, MODULE_NAME_LEN-1) < 0)
 991                 return -EFAULT;
 992         name[MODULE_NAME_LEN-1] = '\0';
 993 
 994         audit_log_kern_module(name);
 995 
 996         if (mutex_lock_interruptible(&module_mutex) != 0)
 997                 return -EINTR;
 998 
 999         mod = find_module(name);
1000         if (!mod) {
1001                 ret = -ENOENT;
1002                 goto out;
1003         }
1004 
1005         if (!list_empty(&mod->source_list)) {
1006                 /* Other modules depend on us: get rid of them first. */
1007                 ret = -EWOULDBLOCK;
1008                 goto out;
1009         }
1010 
1011         /* Doing init or already dying? */
1012         if (mod->state != MODULE_STATE_LIVE) {
1013                 /* FIXME: if (force), slam module count damn the torpedoes */
1014                 pr_debug("%s already dying\n", mod->name);
1015                 ret = -EBUSY;
1016                 goto out;
1017         }
1018 
1019         /* If it has an init func, it must have an exit func to unload */
1020         if (mod->init && !mod->exit) {
1021                 forced = try_force_unload(flags);
1022                 if (!forced) {
1023                         /* This module can't be removed */
1024                         ret = -EBUSY;
1025                         goto out;
1026                 }
1027         }
1028 
1029         /* Stop the machine so refcounts can't move and disable module. */
1030         ret = try_stop_module(mod, flags, &forced);
1031         if (ret != 0)
1032                 goto out;
1033 
1034         mutex_unlock(&module_mutex);
1035         /* Final destruction now no one is using it. */
1036         if (mod->exit != NULL)
1037                 mod->exit();
1038         blocking_notifier_call_chain(&module_notify_list,
1039                                      MODULE_STATE_GOING, mod);
1040         klp_module_going(mod);
1041         ftrace_release_mod(mod);
1042 
1043         async_synchronize_full();
1044 
1045         /* Store the name of the last unloaded module for diagnostic purposes */
1046         strlcpy(last_unloaded_module, mod->name, sizeof(last_unloaded_module));
1047 
1048         free_module(mod);
1049         /* someone could wait for the module in add_unformed_module() */
1050         wake_up_all(&module_wq);
1051         return 0;
1052 out:
1053         mutex_unlock(&module_mutex);
1054         return ret;
1055 }

1307 static int check_version(const struct load_info *info,
1308                          const char *symname,
1309                          struct module *mod,
1310                          const s32 *crc)
1311 {
1312         Elf_Shdr *sechdrs = info->sechdrs;
1313         unsigned int versindex = info->index.vers;
1314         unsigned int i, num_versions;
1315         struct modversion_info *versions;
1316 
1317         /* Exporting module didn't supply crcs?  OK, we're already tainted. */
1318         if (!crc)
1319                 return 1;
1320 
1321         /* No versions at all?  modprobe --force does this. */
1322         if (versindex == 0)
1323                 return try_to_force_load(mod, symname) == 0;

1326         num_versions = sechdrs[versindex].sh_size
1327                 / sizeof(struct modversion_info);

1342                          crcval, versions[i].crc);

1345 
1346         /* Broken toolchain. Warn once, then let it go.. */
1347         pr_warn_once("%s: no symbol version for %s\n", info->name, symname);

1352                info->name, symname);

1459 /* Resolve a symbol for this module.  I.e. if we find one, record usage. */
1460 static const struct kernel_symbol *resolve_symbol(struct module *mod,
1461                                                   const struct load_info *info,
1462                                                   const char *name,
1463                                                   char ownername[])
1464 {
1465         struct module *owner;
1466         const struct kernel_symbol *sym;
1467         const s32 *crc;
1468         enum mod_license license;
1469         int err;
1470 
1471         /*
1472          * The module_mutex should not be a heavily contended lock;
1473          * if we get the occasional sleep here, we'll go an extra iteration
1474          * in the wait_event_interruptible(), which is harmless.
1475          */
1476         sched_annotate_sleep();
1477         mutex_lock(&module_mutex);
1478         sym = find_symbol(name, &owner, &crc, &license,
1479                           !(mod->taints & (1 << TAINT_PROPRIETARY_MODULE)), true);
1480         if (!sym)
1481                 goto unlock;
1482 
1483         if (license == GPL_ONLY)
1484                 mod->using_gplonly_symbols = true;
1485 
1486         if (!inherit_taint(mod, owner)) {
1487                 sym = NULL;
1488                 goto getname;
1489         }
1490 
1491         if (!check_version(info, name, mod, crc)) {
1492                 sym = ERR_PTR(-EINVAL);
1493                 goto getname;
1494         }
1495 
1496         err = verify_namespace_is_imported(info, sym, mod);
1497         if (err) {
1498                 sym = ERR_PTR(err);
1499                 goto getname;
1500         }
1501 
1502         err = ref_module(mod, owner);
1503         if (err) {
1504                 sym = ERR_PTR(err);
1505                 goto getname;
1506         }
1507 
1508 getname:
1509         /* We must make copy under the lock if we failed to get ref. */
1510         strncpy(ownername, module_name(owner), MODULE_NAME_LEN);
1511 unlock:
1512         mutex_unlock(&module_mutex);
1513         return sym;
1514 }

1557 #define MODULE_SECT_READ_SIZE (3 /* "0x", "\n" */ + (BITS_PER_LONG / 4))
1558 static ssize_t module_sect_read(struct file *file, struct kobject *kobj,
1559                                 struct bin_attribute *battr,
1560                                 char *buf, loff_t pos, size_t count)
1561 {
1562         struct module_sect_attr *sattr =
1563                 container_of(battr, struct module_sect_attr, battr);
1564         char bounce[MODULE_SECT_READ_SIZE + 1];
1565         size_t wrote;
1566 
1567         if (pos != 0)
1568                 return -EINVAL;
1569 
1570         /*
1571          * Since we're a binary read handler, we must account for the
1572          * trailing NUL byte that sprintf will write: if "buf" is
1573          * too small to hold the NUL, or the NUL is exactly the last
1574          * byte, the read will look like it got truncated by one byte.
1575          * Since there is no way to ask sprintf nicely to not write
1576          * the NUL, we have to use a bounce buffer.
1577          */
1578         wrote = scnprintf(bounce, sizeof(bounce), "0x%px\n",
1579                          kallsyms_show_value(file->f_cred)
1580                                 ? (void *)sattr->address : NULL);
1581         count = min(count, wrote);
1582         memcpy(buf, bounce, count);
1583 
1584         return count;
1585 }

2488                 { SHF_WRITE | SHF_ALLOC, ARCH_SHF_SMALL },

2490         };
2491         unsigned int m, i;

2493         for (i = 0; i < info->hdr->e_shnum; i++)

2499                         Elf_Shdr *s = &info->sechdrs[i];
2500                         const char *sname = info->secstrings + s->sh_name;
2501 
2502                         if ((s->sh_flags & masks[m][0]) != masks[m][0]
2503                             || (s->sh_flags & masks[m][1])
2504                             || s->sh_entsize != ~0UL
2505                             || module_init_section(sname))

2513                         mod->core_layout.text_size = mod->core_layout.size;
2514                         break;
2515                 case 1: /* RO: text and ro-data */
2516                         mod->core_layout.size = debug_align(mod->core_layout.size);
2517                         mod->core_layout.ro_size = mod->core_layout.size;
2518                         break;
2519                 case 2: /* RO after init */
2520                         mod->core_layout.size = debug_align(mod->core_layout.size);
2521                         mod->core_layout.ro_after_init_size = mod->core_layout.size;
2522                         break;
2523                 case 4: /* whole core */
2524                         mod->core_layout.size = debug_align(mod->core_layout.size);
2525                         break;
2526                 }
2527         }
2528 
2529         pr_debug("Init section allocation order:\n");
2530         for (m = 0; m < ARRAY_SIZE(masks); ++m) {
2531                 for (i = 0; i < info->hdr->e_shnum; ++i) {
2532                         Elf_Shdr *s = &info->sechdrs[i];
2533                         const char *sname = info->secstrings + s->sh_name;
2534 
2535                         if ((s->sh_flags & masks[m][0]) != masks[m][0]
2536                             || (s->sh_flags & masks[m][1])
2537                             || s->sh_entsize != ~0UL
2538                             || !module_init_section(sname))
2539                                 continue;
2540                         s->sh_entsize = (get_offset(mod, &mod->init_layout.size, s, i)
2541                                          | INIT_OFFSET_MASK);
2542                         pr_debug("\t%s\n", sname);
2543                 }
2544                 switch (m) {
2545                 case 0: /* executable */
2546                         mod->init_layout.size = debug_align(mod->init_layout.size);
2547                         mod->init_layout.text_size = mod->init_layout.size;
2548                         break;
2549                 case 1: /* RO: text and ro-data */
2550                         mod->init_layout.size = debug_align(mod->init_layout.size);
2551                         mod->init_layout.ro_size = mod->init_layout.size;
2552                         break;
2553                 case 2:
2554                         /*
2555                          * RO after init doesn't apply to init_layout (only
2556                          * core_layout), so it just takes the value of ro_size.
2557                          */
2558                         mod->init_layout.ro_after_init_size = mod->init_layout.ro_size;
2559                         break;
2560                 case 4: /* whole init */
2561                         mod->init_layout.size = debug_align(mod->init_layout.size);

2677 static char elf_type(const Elf_Sym *sym, const struct load_info *info)
2678 {
2679         const Elf_Shdr *sechdrs = info->sechdrs;
2680 
2681         if (ELF_ST_BIND(sym->st_info) == STB_WEAK) {
2682                 if (ELF_ST_TYPE(sym->st_info) == STT_OBJECT)
2683                         return 'v';

2698                         return 'r';

2700                         return 'g';
2701                 else
2702                         return 'd';

2704         if (sechdrs[sym->st_shndx].sh_type == SHT_NOBITS) {
2705                 if (sechdrs[sym->st_shndx].sh_flags & ARCH_SHF_SMALL)
2706                         return 's';
2707                 else
2708                         return 'b';
2709         }
2710         if (strstarts(info->secstrings + sechdrs[sym->st_shndx].sh_name,
2711                       ".debug")) {

2713         }

2751 {

2757         /* Put symbol section at end of init part of module. */

2787                                          info->index.str) | INIT_OFFSET_MASK;

2988 /*
2989  * Sanity checks against invalid binaries, wrong arch, weird elf version.
2990  *
2991  * Also do basic validity checks against section offsets and sizes, the
2992  * section name string table, and the indices used for it (sh_name).
2993  */
2994 static int elf_validity_check(struct load_info *info)
2995 {
2996         unsigned int i;
2997         Elf_Shdr *shdr, *strhdr;
2998         int err;
2999 
3000         if (info->len < sizeof(*(info->hdr)))
3001                 return -ENOEXEC;
3002 
3003         if (memcmp(info->hdr->e_ident, ELFMAG, SELFMAG) != 0
3004             || info->hdr->e_type != ET_REL
3005             || !elf_check_arch(info->hdr)
3006             || info->hdr->e_shentsize != sizeof(Elf_Shdr))
3007                 return -ENOEXEC;
3008 
3009         /*
3010          * e_shnum is 16 bits, and sizeof(Elf_Shdr) is
3011          * known and small. So e_shnum * sizeof(Elf_Shdr)
3012          * will not overflow unsigned long on any platform.
3013          */
3014         if (info->hdr->e_shoff >= info->len
3015             || (info->hdr->e_shnum * sizeof(Elf_Shdr) >
3016                 info->len - info->hdr->e_shoff))
3017                 return -ENOEXEC;
3018 
3019         info->sechdrs = (void *)info->hdr + info->hdr->e_shoff;
3020 
3021         /*
3022          * Verify if the section name table index is valid.
3023          */
3024         if (info->hdr->e_shstrndx == SHN_UNDEF
3025             || info->hdr->e_shstrndx >= info->hdr->e_shnum)
3026                 return -ENOEXEC;
3027 
3028         strhdr = &info->sechdrs[info->hdr->e_shstrndx];
3029         err = validate_section_offset(info, strhdr);
3030         if (err < 0)
3031                 return err;
3032 
3033         /*
3034          * The section name table must be NUL-terminated, as required
3035          * by the spec. This makes strcmp and pr_* calls that access
3036          * strings in the section safe.
3037          */
3038         info->secstrings = (void *)info->hdr + strhdr->sh_offset;
3039         if (info->secstrings[strhdr->sh_size - 1] != '\0')
3040                 return -ENOEXEC;
3041 
3042         /*
3043          * The code assumes that section 0 has a length of zero and
3044          * an addr of zero, so check for it.
3045          */
3046         if (info->sechdrs[0].sh_type != SHT_NULL
3047             || info->sechdrs[0].sh_size != 0
3048             || info->sechdrs[0].sh_addr != 0)
3049                 return -ENOEXEC;
3050 
3051         for (i = 1; i < info->hdr->e_shnum; i++) {
3052                 shdr = &info->sechdrs[i];
3053                 switch (shdr->sh_type) {
3054                 case SHT_NULL:
3055                 case SHT_NOBITS:
3056                         continue;
3057                 case SHT_SYMTAB:
3058                         if (shdr->sh_link == SHN_UNDEF
3059                             || shdr->sh_link >= info->hdr->e_shnum)
3060                                 return -ENOEXEC;
3061                         fallthrough;
3062                 default:
3063                         err = validate_section_offset(info, shdr);
3064                         if (err < 0) {

3420 static int move_module(struct module *mod, struct load_info *info)
3421 {
3422         int i;
3423         void *ptr;
3424 
3425         /* Do the allocs. */
3426         ptr = module_alloc(mod->core_layout.size);
3427         /*
3428          * The pointer to this block is stored in the module structure
3429          * which is inside the block. Just mark it as not being a
3430          * leak.
3431          */
3432         kmemleak_not_leak(ptr);
3433         if (!ptr)
3434                 return -ENOMEM;
3435 
3436         memset(ptr, 0, mod->core_layout.size);
3437         mod->core_layout.base = ptr;
3438 
3439         if (mod->init_layout.size) {
3440                 ptr = module_alloc(mod->init_layout.size);
3441                 /*
3442                  * The pointer to this block is stored in the module structure
3443                  * which is inside the block. This block doesn't need to be
3444                  * scanned as it contains data and code that will be freed
3445                  * after the module is initialized.
3446                  */
3447                 kmemleak_ignore(ptr);
3448                 if (!ptr) {
3449                         module_memfree(mod->core_layout.base);
3450                         return -ENOMEM;
3451                 }
3452                 memset(ptr, 0, mod->init_layout.size);
3453                 mod->init_layout.base = ptr;
3454         } else
3455                 mod->init_layout.base = NULL;
3456 
3457         /* Transfer each section which specifies SHF_ALLOC */
3458         pr_debug("final section addresses:\n");
3459         for (i = 0; i < info->hdr->e_shnum; i++) {
3460                 void *dest;
3461                 Elf_Shdr *shdr = &info->sechdrs[i];
3462 
3463                 if (!(shdr->sh_flags & SHF_ALLOC))
3464                         continue;
3465 
3466                 if (shdr->sh_entsize & INIT_OFFSET_MASK)
3467                         dest = mod->init_layout.base
3468                                 + (shdr->sh_entsize & ~INIT_OFFSET_MASK);
3469                 else
3470                         dest = mod->core_layout.base + shdr->sh_entsize;
3471 
3472                 if (shdr->sh_type != SHT_NOBITS)
3473                         memcpy(dest, (void *)shdr->sh_addr, shdr->sh_size);

3475                 shdr->sh_addr = (unsigned long)dest;

3477                          (long)shdr->sh_addr, info->secstrings + shdr->sh_name);

3720 /*
3721  * This is where the real work happens.
3722  *
3723  * Keep it uninlined to provide a reliable breakpoint target, e.g. for the gdb
3724  * helper command 'lx-symbols'.
3725  */
3726 static noinline int do_init_module(struct module *mod)
3727 {
3728         int ret = 0;
3729         struct mod_initfree *freeinit;
3730 
3731         freeinit = kmalloc(sizeof(*freeinit), GFP_KERNEL);
3732         if (!freeinit) {
3733                 ret = -ENOMEM;
3734                 goto fail;
3735         }
3736         freeinit->module_init = mod->init_layout.base;
3737 
3738         /*
3739          * We want to find out whether @mod uses async during init.  Clear
3740          * PF_USED_ASYNC.  async_schedule*() will set it.
3741          */
3742         current->flags &= ~PF_USED_ASYNC;
3743 
3744         do_mod_ctors(mod);
3745         /* Start the module */
3746         if (mod->init != NULL)
3747                 ret = do_one_initcall(mod->init);
3748         if (ret < 0) {
3749                 goto fail_free_freeinit;
3750         }
3751         if (ret > 0) {
3752                 pr_warn("%s: '%s'->init suspiciously returned %d, it should "
3753                         "follow 0/-E convention\n"
3754                         "%s: loading module anyway...\n",
3755                         __func__, mod->name, ret, __func__);
3756                 dump_stack();
3757         }
3758 
3759         /* Now it's a first class citizen! */
3760         mod->state = MODULE_STATE_LIVE;
3761         blocking_notifier_call_chain(&module_notify_list,
3762                                      MODULE_STATE_LIVE, mod);
3763 
3764         /* Delay uevent until module has finished its init routine */
3765         kobject_uevent(&mod->mkobj.kobj, KOBJ_ADD);
3766 
3767         /*
3768          * We need to finish all async code before the module init sequence
3769          * is done.  This has potential to deadlock.  For example, a newly
3770          * detected block device can trigger request_module() of the
3771          * default iosched from async probing task.  Once userland helper
3772          * reaches here, async_synchronize_full() will wait on the async
3773          * task waiting on request_module() and deadlock.
3774          *
3775          * This deadlock is avoided by perfomring async_synchronize_full()
3776          * iff module init queued any async jobs.  This isn't a full
3777          * solution as it will deadlock the same if module loading from
3778          * async jobs nests more than once; however, due to the various
3779          * constraints, this hack seems to be the best option for now.
3780          * Please refer to the following thread for details.
3781          *
3782          * http://thread.gmane.org/gmane.linux.kernel/1420814
3783          */
3784         if (!mod->async_probe_requested && (current->flags & PF_USED_ASYNC))
3785                 async_synchronize_full();
3786 
3787         ftrace_free_mem(mod, mod->init_layout.base, mod->init_layout.base +
3788                         mod->init_layout.size);
3789         mutex_lock(&module_mutex);
3790         /* Drop initial reference. */
3791         module_put(mod);
3792         trim_init_extable(mod);
3793 #ifdef CONFIG_KALLSYMS
3794         /* Switch to core kallsyms now init is done: kallsyms may be walking! */
3795         rcu_assign_pointer(mod->kallsyms, &mod->core_kallsyms);
3796 #endif
3797         module_enable_ro(mod, true);
3798         trace_android_vh_set_module_permit_after_init(mod);
3799         mod_tree_remove_init(mod);
3800         module_arch_freeing_init(mod);
3801         trace_android_vh_set_memory_rw((unsigned long)mod->init_layout.base,
3802                 (mod->init_layout.size)>>PAGE_SHIFT);
3803         trace_android_vh_set_memory_nx((unsigned long)mod->init_layout.base,
3804                 (mod->init_layout.size)>>PAGE_SHIFT);
3805         mod->init_layout.base = NULL;
3806         mod->init_layout.size = 0;
3807         mod->init_layout.ro_size = 0;
3808         mod->init_layout.ro_after_init_size = 0;
3809         mod->init_layout.text_size = 0;
3810         /*
3811          * We want to free module_init, but be aware that kallsyms may be
3812          * walking this with preempt disabled.  In all the failure paths, we
3813          * call synchronize_rcu(), but we don't want to slow down the success
3814          * path. module_memfree() cannot be called in an interrupt, so do the
3815          * work and call synchronize_rcu() in a work queue.
3816          *
3817          * Note that module_alloc() on most architectures creates W+X page
3818          * mappings which won't be cleaned up until do_free_init() runs.  Any
3819          * code such as mark_rodata_ro() which depends on those mappings to
3820          * be cleaned up needs to sync with the queued work - ie
3821          * rcu_barrier()
3822          */
3823         if (llist_add(&freeinit->node, &init_free_list))
3824                 schedule_work(&init_free_wq);
3825 
3826         mutex_unlock(&module_mutex);
3827         wake_up_all(&module_wq);
3828 
3829         return 0;
3830 
3831 fail_free_freeinit:
3832         kfree(freeinit);
3833 fail:
3834         /* Try to protect us from buggy refcounters. */
3835         mod->state = MODULE_STATE_GOING;
3836         synchronize_rcu();
3837         module_put(mod);
3838         blocking_notifier_call_chain(&module_notify_list,
3839                                      MODULE_STATE_GOING, mod);
3840         klp_module_going(mod);
3841         ftrace_release_mod(mod);
3842         free_module(mod);
3843         wake_up_all(&module_wq);
3844         return ret;
3845 }

4265 /*
4266  * Given a module and address, find the corresponding symbol and return its name
4267  * while providing its size and offset if needed.
4268  */
4269 static const char *find_kallsyms_symbol(struct module *mod,
4270                                         unsigned long addr,
4271                                         unsigned long *size,
4272                                         unsigned long *offset)
4273 {
4274         unsigned int i, best = 0;
4275         unsigned long nextval, bestval;
4276         struct mod_kallsyms *kallsyms = rcu_dereference_sched(mod->kallsyms);
4277 
4278         /* At worse, next value is at end of module */
4279         if (within_module_init(addr, mod))
4280                 nextval = (unsigned long)mod->init_layout.base+mod->init_layout.text_size;
4281         else
4282                 nextval = (unsigned long)mod->core_layout.base+mod->core_layout.text_size;
4283 
4284         bestval = kallsyms_symbol_value(&kallsyms->symtab[best]);
4285 
4286         /* Scan for closest preceding symbol, and next symbol. (ELF
4287            starts real symbols at 1). */
4288         for (i = 1; i < kallsyms->num_symtab; i++) {
4289                 const Elf_Sym *sym = &kallsyms->symtab[i];
4290                 unsigned long thisval = kallsyms_symbol_value(sym);
4291 
4292                 if (sym->st_shndx == SHN_UNDEF)
4293                         continue;
4294 
4295                 /* We ignore unnamed symbols: they're uninformative
4296                  * and inserted at a whim. */
4297                 if (*kallsyms_symbol_name(kallsyms, i) == '\0'
4298                     || is_arm_mapping_symbol(kallsyms_symbol_name(kallsyms, i))
4299                     || is_cfi_typeid_symbol(kallsyms_symbol_name(kallsyms, i)))
4300                         continue;
